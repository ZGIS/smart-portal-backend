39   /**
40     * quick and dirty bbox from wkt stored in DB
41     *
42     * @param bboxAsWkt
43     * @return
44     */
45   def createOptionalBbox(bboxAsWkt: Option[String]): Option[Rectangle] = {
46     if (bboxAsWkt.isDefined) {
47       Try {
48         wktReader.read(bboxAsWkt.get).asInstanceOf[Rectangle]
49       }.toOption
50     } else {
51       None
52     }
53   }

58     * @param rect
59     * @return
60     */
61   def rectToWkt(rect: Rectangle): String = {
62     val shpWriter = ctx.getFormats().getWriter(ShapeIO.WKT)
63     shpWriter.toString(rect)
64   }

66   /**
67     *
68     * @param dateStringOption
69     * @return
70     */
71   def parseOffsetDateString(dateStringOption: Option[String]): Option[List[OffsetDateTime]] = {
72     if (dateStringOption.isDefined) {
73       val isoTemporalString = dateStringOption.get
74       if (isoTemporalString.contains("/")) {
75         parseDateStringAsOffsetInterval(isoTemporalString).toOption
76       } else {
77         parseDateStringAsOffsetDateTime(isoTemporalString).toOption
78       }
79     } else {
80       None
81     }
82   }
83 
84   /**
85     *
86     * @param isoTemporalString
87     * @return
88     */
89   private def parseDateStringAsOffsetDateTime(isoTemporalString: String): Try[List[OffsetDateTime]] = {
90     Try {
91       val date = OffsetDateTime.parse(isoTemporalString)
92       List(date)
93     }
94   }
95 
96   /**
97     *
98     * @param isoTemporalString
99     * @return
100     */
101   private def parseDateStringAsOffsetInterval(isoTemporalString: String): Try[List[OffsetDateTime]] = {
102     Try {
103       val dateStrings = isoTemporalString.replace("\"", "").trim.split("/").toList
104       val date1 = OffsetDateTime.parse(dateStrings.head)
105       val date2 = OffsetDateTime.parse(dateStrings.last)
106       List(date1, date2)
107     }
108   }
109 
110   /**
111     * String serialiser to store either single offset date or the interval in DB
112     *
113     * @param dateRange
114     * @return
115     */
116   def writeOffsetDatesAsDateString(dateRange: List[OffsetDateTime]): Option[String] = {
117     lazy val formatter = DateTimeFormatter.ISO_OFFSET_DATE_TIME
118     val dates = dateRange.sortWith((a, b) => a.isBefore(b))
119     if (dates.head == dates.last) {
120       Try(dates.head.format(formatter)).toOption
121     } else {
122       Try {
123         val d1 = dates.head.format(formatter)
124         val d2 = dates.last.format(formatter)
125         d1 + "/" + d2
126       }.toOption
127     }
128   }
