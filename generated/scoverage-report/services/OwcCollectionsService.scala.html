62   def getOwcDocumentsForUserAndId(authUserOption: Option[String], idOption: Option[String]): Seq[OwcDocument] = {
63 
64     authUserOption.fold {
65       // no email provided
66       idOption.fold {
67         // TODO docs for anonymous, no id provided => all public docs (implies docs must be public)
68         owcDocumentDAO.getAllPublicOwcDocuments
69       } {
70         // TODO find doc by id for anonymous, only one doc if available (implies doc must be public)
71         id => {
72           owcDocumentDAO.findPublicOwcDocumentsById(id).toSeq
73         }
74       }
75     } { authUser => {
76       // trying to find a user from provided authuser option
77       userDAO.findUserByEmail(authUser).fold {
78         logger.warn("Provided user not found.")
79         idOption.fold {
80           // TODO docs for anonymous, no id provided => all public docs (later maybe check if public)
81           owcDocumentDAO.getAllPublicOwcDocuments
82         } {
83           // docs for anonymous, but id provided only one doc if available (and only if public)
84           id => {
85             owcDocumentDAO.findPublicOwcDocumentsById(id).toSeq
86           }
87         }
88       } { user =>
89         // we have a distinct ok user here
90         idOption.fold {
91           // docs for user, no id provided => all user visible docs
92           // TODO technically would be more than "only" publicly visible at some point
93           val publicDocs = owcDocumentDAO.getAllPublicOwcDocuments
94           val userDocs = owcDocumentDAO.findOwcDocumentByUser(user.email)
95 
96           publicDocs ++ userDocs
97         } {
98           // TODO find doc by id for provided user if visible/available (later maybe check constraint)
99           id => {
100             owcDocumentDAO.findOwcDocumentByIdAndUser(id, user.email).toSeq
101           }
102         }
103       }
104     }
105     }
106   }
