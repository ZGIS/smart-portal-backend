103       errors => {

121             ZonedDateTime.now.withZoneSameInstant(ZoneId.of(appTimeZone)))

124             logger.error("User create error.")
125             val error = ErrorResult("User create error.", None)
126             BadRequest(Json.toJson(error)).as(JSON)
127           } { user =>
128             val emailWentOut = emailService.sendRegistrationEmail(jsuser.email, "Please confirm your GW HUB account", jsuser.firstname, regLinkId)

130             // creating default collection only after registration, account is only barely usable here

134           // found user with that email already

370               }
