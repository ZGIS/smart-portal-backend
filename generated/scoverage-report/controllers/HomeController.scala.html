101   /**
102     * login with JSON (from Angular / form)
103     * {"email":"akmoch","password":"testpass123"}
104     *
105     * Create the authentication token and sets the cookie [[AuthTokenCookieKey]]
106     * for AngularJS to use also in X-XSRF-TOKEN in HTTP header.
107     *
108     * @return
109     */
110   def login: Action[JsValue] = Action(parse.json) { implicit request =>
111     request.body.validate[LoginCredentials].fold(
112       errors => {
113         logger.error(JsError.toJson(errors).toString())
114         BadRequest(Json.obj("status" -> "ERR", "message" -> JsError.toJson(errors)))
115       },
116       valid = credentials => {
117         // find user in db and compare password stuff
118         userDAO.authenticate(credentials.email, credentials.password).fold {
119           logger.error("User email or password wrong.")
120           Unauthorized(Json.obj("status" -> "ERR", "message" -> "User email or password wrong."))
121         } { user =>
122           val uaIdentifier: String = request.headers.get(UserAgentHeader).getOrElse(UserAgentHeaderDefault)
123           // logger.debug(s"Logging in email from $uaIdentifier")
124           val token = passwordHashing.createSessionCookie(user.email, uaIdentifier)
125           cache.set(token, user.email)
126           Ok(Json.obj("status" -> "OK", "token" -> token, "email" -> user.email, "userprofile" -> user.asProfileJs()))
127             .withCookies(Cookie(AuthTokenCookieKey, token, None, httpOnly = false))
128         }
129       })
130   }

132   /**
133     * https://developers.google.com/recaptcha/docs/verify
134     *
135     * @param recaptcaChallenge
136     * @return
137     */
138   def recaptchaValidate(recaptcaChallenge: String): Action[AnyContent] = Action.async { implicit request =>
139     // TODO check from where URL referer comes from
140     ws.url(recaptcaVerifyUrl)
141       .withHeaders("Accept" -> "application/json")
142       .withRequestTimeout(10000.millis)
143       .withQueryString("secret" -> reCaptchaSecret,
144         "response" -> recaptcaChallenge).get().map {
145       response =>
146         val success = (response.json \ "success").as[Boolean]
147         if (success) {
148           Ok(Json.obj("status" -> "OK", "message" -> "granted", "success" -> JsBoolean(true)))
149         }
150         else {
151           val errors = (response.json \ "error-codes")
152           val jsErrors = errors.getOrElse(JsString("No further errors"))
153           BadRequest(Json.obj("status" -> "OK", "message" -> jsErrors, "success" -> JsBoolean(false)))
154         }
155     }.recover {
156       case e: Exception =>
157         val exceptionData = Map("error" -> Seq(e.getMessage))
158         Ok(Json.obj("status" -> "ERR", "message" -> exceptionData, "success" -> JsBoolean(false)))
159     }
160   }

169     email =>
170       implicit request =>
171         cache.remove(token)
172         Ok.discardingCookies(DiscardingCookie(name = AuthTokenCookieKey))
173   }
