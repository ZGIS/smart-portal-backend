353   def updatePassword(): Action[JsValue] = HasToken(parse.json) {
354     token =>
355       cachedSecUserEmail =>
356         implicit request =>
357           request.body.validate[PasswordUpdateCredentials].fold(
358             errors => {
359               logger.error(JsError.toJson(errors).toString())
360               val error = ErrorResult("Could not validate request.", Some(JsError.toJson(errors).toString()))
361               BadRequest(Json.toJson(error)).as(JSON)
362             },
363             valid = credentials => {
364               // find user in db
365               userService.findUserByEmailAddress(credentials.email).fold {
366                 logger.error("User not found.")
367                 val error = ErrorResult("User not found", None)
368                 BadRequest(Json.toJson(error)).as(JSON)
369               } { dbuser =>
370                 // compare password
371                 if (passwordHashing.validatePassword(credentials.oldPassword, dbuser.password)) {
372                   val newCryptPass = passwordHashing.createHash(credentials.newPassword)
373                   val updateUser = dbuser.copy(
374                     password = newCryptPass,
375                     laststatustoken = s"${StatusToken.ACTIVE}:PASSWORDUPDATE",
376                     laststatuschange = ZonedDateTime.now.withZoneSameInstant(ZoneId.of(appTimeZone)))
377 
378                   userService.updatePassword(updateUser).fold {
379                     logger.error("Password update error.")
380                     val error = ErrorResult("Password update error.", None)
381                     BadRequest(Json.toJson(error)).as(JSON)
382                   } { user =>
383                     val uaIdentifier: String = request.headers.get(UserAgentHeader).getOrElse(UserAgentHeaderDefault)
384                     // cache.remove(token)
385                     userService.removeUserSessionCache(cachedSecUserEmail, token)
386                     emailService.sendPasswordUpdateEmail(user.email, "Password Update on GW HUB", user.firstname)
387                     // val newtoken = passwordHashing.createSessionCookie(user.email, uaIdentifier)
388                     // cache.set(newtoken, user.email.value)
389                     val newtoken = userService.upsertUserSessionCache(user.email.value, uaIdentifier)
390                     Ok(Json.obj("status" -> "OK", "token" -> newtoken, "email" -> user.email.value))
391                       .withCookies(Cookie(AuthTokenCookieKey, newtoken, None, httpOnly = false))
392                   }
393                 } else {
394                   logger.error("Password update error, not matching.")
395                   val error = ErrorResult("Password update error, not matching.", None)
396                   BadRequest(Json.toJson(error)).as(JSON)
397                 }
398               }
399 
400             })

503   def gconnect: Action[JsValue] = Action(parse.json) { implicit request =>
504     request.body.validate[GAuthCredentials].fold(
505       errors => {
506         logger.error(JsError.toJson(errors).toString())
507         val error = ErrorResult("Could not validate request.", Some(JsError.toJson(errors).toString()))
508         BadRequest(Json.toJson(error)).as(JSON)
509       },
510       valid = gAuthCredentials => {
511         val accessTokenResult = googleService.getGoogleAuthorization(gAuthCredentials)
512         accessTokenResult match {
513           case Left(errorResult) => {
514             logger.error(errorResult.message + errorResult.details.mkString)
515             InternalServerError(Json.toJson(errorResult)).as(JSON)
516           }
517           case Right(googleTokenResponse) => {
518             // Get profile info from ID token
519             val gAuthPayload = googleTokenResponse.parseIdToken().getPayload
520             // Use this value as a key to identify a user.
521             val userId = gAuthPayload.getSubject()
522             val email = gAuthPayload.getEmail()
523             val emailVerified = gAuthPayload.getEmailVerified()
524             // get further payload things that is easy in java but needs to work in scala too
525             val name = Option(gAuthPayload.get("name").asInstanceOf[String])
526             val familyName = Option(gAuthPayload.get("family_name").asInstanceOf[String])
527             val givenName = Option(gAuthPayload.get("given_name").asInstanceOf[String])
528             if ((gAuthCredentials.accesstype.equalsIgnoreCase("LOGIN") || gAuthCredentials.accesstype.equalsIgnoreCase("REGISTER")) &&
529               emailVerified && EmailAddress.isValid(email)) {
530 
531               val emailAddress = EmailAddress(email)
532               // viaConnection lookup email block
533               val firstUserLookup = userService.findUserByEmailAddress(emailAddress)
534 
535               firstUserLookup.fold {
536                 // 'error, no user': well, must be a new user coming through from Google
537                 val cryptPass = passwordHashing.createHash(UUID.randomUUID().toString)
538 
539                 // Use access token to call API for refresh or check-up?
540                 // val googleAccessToken = googleTokenResponse.getAccessToken()
541                 // val googleCredential = new GoogleCredential().setAccessToken(googleAccessToken)
542                 val newUser = User(emailAddress,
543                   s"google:${userId}",
544                   givenName.getOrElse(name.getOrElse(emailAddress.mailbox.value)),
545                   familyName.getOrElse(""),
546                   cryptPass,
547                   s"${StatusToken.ACTIVE}:REGCONFIRMED",
548                   ZonedDateTime.now.withZoneSameInstant(ZoneId.of(appTimeZone)))
549 
550                 val createUserCall = userService.createUser(newUser)
551 
552                 createUserCall.fold {
553                   logger.error("User create error.")
554                   val error = ErrorResult("Error while creating user", None)
555                   InternalServerError(Json.toJson(error)).as(JSON)
556                 } { createdUser =>
557                   val emailWentOut = emailService.sendConfirmationEmail(createdUser.email, "Your GW HUB account is now active", createdUser.firstname)
558                   // all good here, creating default collection (Unit)
559                   val userOwcDocTransaction = collectionsService.createUserDefaultCollection(createdUser)
560 
561                   userOwcDocTransaction.fold {
562                     logger.error(s"Couldn't create user default collection for ${createdUser.accountSubject}.")
563                     val error = ErrorResult("Error while creating user default collection.", None)
564                     InternalServerError(Json.toJson(error)).as(JSON)
565                   } { owcDoc =>
566                     logger.info(s"New user registered ${createdUser.accountSubject}. Email went out $emailWentOut. " +
567                       s"User Collection ${owcDoc.id.toString}")
568                     val uaIdentifier: String = request.headers.get(UserAgentHeader).getOrElse(UserAgentHeaderDefault)
569                     // logger.debug(s"Logging in email from $uaIdentifier")
570                     // val token = passwordHashing.createSessionCookie(createdUser.email, uaIdentifier)
571                     // cache.set(token, createdUser.email.value)
572                     val token = userService.upsertUserSessionCache(createdUser.email.value, uaIdentifier)
573                     Ok(Json.obj("status" -> "OK", "token" -> token, "email" -> createdUser.email.value, "userprofile" -> createdUser.asProfileJs))
574                       .withCookies(Cookie(AuthTokenCookieKey, token, None, httpOnly = false))
575                   }
576                 }
577               } { existingUser =>
578                 val uaIdentifier: String = request.headers.get(UserAgentHeader).getOrElse(UserAgentHeaderDefault)
579                 // logger.debug(s"Logging in email from $uaIdentifier")
580                 // val token = passwordHashing.createSessionCookie(existingUser.email.value, uaIdentifier)
581                 // cache.set(token, existingUser.email.value)
582                 val token = userService.upsertUserSessionCache(existingUser.email.value, uaIdentifier)
583                 Ok(Json.obj("status" -> "OK", "token" -> token, "email" -> existingUser.email.value, "userprofile" -> existingUser.asProfileJs))
584                   .withCookies(Cookie(AuthTokenCookieKey, token, None, httpOnly = false))
585               }
586

591             }
592           }

597           }
598         }
