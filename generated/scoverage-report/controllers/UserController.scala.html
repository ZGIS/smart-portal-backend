320           BadRequest(Json.obj("status" -> "ERR", "message" -> "User not found."))
321         } { dbuser =>

323           val updateUser = User(dbuser.email,
324             dbuser.username,
325             dbuser.firstname,
326             dbuser.lastname,
327             cryptPass,
328             "ACTIVE:PASSWORDUPDATE",
329             ZonedDateTime.now.withZoneSameInstant(ZoneId.of(appTimeZone)))
330 
331           userDAO.updatePassword(updateUser).fold {
332             logger.error("Password update error.")
333             BadRequest(Json.obj("status" -> "ERR", "message" -> "Password update error."))
334           } { user =>
335             val uaIdentifier: String = request.headers.get(UserAgentHeader).getOrElse(UserAgentHeaderDefault)
336             // logger.info(s"Logging in username from $uaIdentifier")
337             cache.remove(token)
338             val newtoken = passwordHashing.createSessionCookie(user.username, uaIdentifier)
339             cache.set(newtoken, user.username)
340             Ok(Json.obj("status" -> "OK", "token" -> newtoken, "username" -> user.username))
341               .withCookies(Cookie(AuthTokenCookieKey, newtoken, None, httpOnly = false))
342           }
