20 import com.google.inject.AbstractModule
21 import java.time.Clock
22 
23 import services.{ApplicationTimer, AtomicCounter, Counter}
24 
25 /**
26  * This class is a Guice module that tells Guice how to bind several
27  * different types. This Guice module is created when the Play
28  * application starts.
29 
30  * Play will automatically use any class called `Module` that is in
31  * the root package. You can create modules in other locations by
32  * adding `play.modules.enabled` settings to the `application.conf`
33  * configuration file.
34  */
35 class Module extends AbstractModule {
36 
37   override def configure() = {
38     // Use the system clock as the default implementation of Clock
39     bind(classOf[Clock]).toInstance(Clock.systemDefaultZone)
40     // Ask Guice to create an instance of ApplicationTimer when the
41     // application starts.
42     bind(classOf[ApplicationTimer]).asEagerSingleton()
43     // Set AtomicCounter as the implementation for Counter.
44     bind(classOf[Counter]).to(classOf[AtomicCounter])
45   }
46 
47 }
