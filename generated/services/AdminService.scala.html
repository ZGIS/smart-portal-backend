99   def queryActiveSessions(token: Option[String], max: Option[Int], email: Option[String]): Seq[UserSession] = {
100     dbSession.viaConnection{ implicit connection =>
101 
102       val count = max.getOrElse(100)
103       (token, email) match {
104         case (Some(to), Some(em)) =>
105           val viaToken = UserSession.findUserSessionByToken(to, count).toSeq
106           val viaEmail = UserSession.findUserSessionByEmail(em, count)
107           (viaToken ++ viaEmail).filter(ul => ul.token.contains(to) && ul.email.contains(em))
108         case (Some(to), None) =>
109           UserSession.findUserSessionByToken(to, count).toSeq
110         case (None, Some(em)) =>
111           UserSession.findUserSessionByEmail(em, count)
112         case (None, None) =>
113           UserSession.getAllUserSessions(count)
114       }
115     }
116   }

118   /**
119     * block and Unblock Users by setting status token
120     * @param command
121     * @param email
122     * @return
123     */
124   def blockUnblockUsers(command: String, email: String): Option[User] = {
125     val userOpt = dbSession.viaConnection( implicit connection => {
126       UserDAO.findUserByEmailAsString(email)
127     })
128     userOpt.fold[Option[User]]{
129       None
130     } {
131       user =>
132         val statusToken = if (command.equalsIgnoreCase("BLOCK")) StatusToken.BLOCKED else StatusToken.ACTIVE
133         val updateUser = User(
134           user.email,
135           user.accountSubject,
136           user.firstname,
137           user.lastname,
138           "***",
139           s"${statusToken.value}:BY-ADMIN",
140           ZonedDateTime.now.withZoneSameInstant(ZoneId.of(appTimeZone)))
141 
142         // remove any latent active sessions
143         val sessions = dbSession.viaConnection { implicit connection =>
144           UserSession.deleteUserSessionByEmail(user.email.value)
145         }
146         if (!sessions) {
147           logger.warn(s"There was a not completely successful sessions delete for ${user.email.value} when setting status to ${statusToken.value}?")
148         }
149 
150         // result of Option can be evaluated upstream in controller
151         dbSession.viaTransaction { implicit connection =>
152           UserDAO.updateNoPass(updateUser)
153         }
154 
155     }
156   }
