37   /**
38     * check if user can edit this particular group
39     *
40     * @param user
41     * @param userGroup
42     * @return
43     */
44   def amIPowerUserForGroup(user: User, userGroup: UserGroup): Boolean = {
45     dbSession.viaConnection(implicit connection => {
46       UserGroup.findUserGroupsForUser(user)
47         .find(ug => ug.uuid.equals(userGroup.uuid))
48         .exists(ug => ug.hasUsersLevel.exists(ul => ul.users_accountsubject.equals(user.accountSubject) && ul.userlevel >= 1))
49     })
50   }

91   /**
92     * will update a group if the calling user is poweruser
93     *
94     * @param user
95     * @param userGroup
96     * @return
97     */
98   def updateUsersOwnUserGroup(user: User, userGroup: UserGroup): Option[UserGroup] = {
99     val edit = amIPowerUserForGroup(user, userGroup)
100     if (edit) {
101       dbSession.viaTransaction(implicit connection => {
102         UserGroup.updateUserGroup(userGroup: UserGroup)
103       })
104     } else {
105       logger.warn(s"user ${user.accountSubject} illegally attempted to edit group ${userGroup.uuid}")
106       None
107     }
108   }

110   /**
111     * will delete a group if the calling user is poweruser
112     *
113     * @param user
114     * @param uuid
115     * @return
116     */
117   def deleteUsersOwnUserGroup(user: User, uuid: UUID): Boolean = {
118     val userGroupOption = dbSession.viaConnection(implicit connection => UserGroup.findUserGroupsById(uuid))
119     userGroupOption.exists {
120       userGroup =>
121         val edit = amIPowerUserForGroup(user, userGroup)
122         if (edit) {
123           dbSession.viaTransaction(implicit connection => {
124             UserGroup.deleteUserGroup(userGroup: UserGroup)
125           })
126         } else {
127           logger.warn(s"user ${user.accountSubject} illegally attempted to edit group ${userGroup.uuid}")
128           false
129         }
130     }
131   }

133   def getOwcContextsRightsMatrixForUser(user: User): Seq[OwcContextsRightsMatrix] = {
134     val userGroupsList = getUsersOwnUserGroups(user)
135 
136     val originalOwnContextsNativeRights = dbSession.viaConnection(implicit connection => {
137       OwcContextDAO.findOwcContextsByNativeOwnerBrief(user).distinct.map(uco =>
138         OwcContextsRightsMatrix(
139           owcContextId = uco.owcContextId,
140           queryingUserAccountSubject = user.accountSubject,
141           origOwnerAccountSubject = uco.origOwnerAccountSubject,
142           viaGroups = Seq(),
143           contextIntrinsicVisibility = uco.contextOwnersVisibility,
144           queryingUserAccessLevel = 2)
145       )
146     })
147 
148     val nativeRightsPerContextInGroups = dbSession.viaConnection(implicit connection => {
149       userGroupsList.flatMap {
150         g =>
151           val viaGroup = g.name
152           val ownLevel = g.hasUsersLevel.find(_.users_accountsubject.contentEquals(user.accountSubject)).map(_.userlevel).getOrElse(-1)
153 
154           val contextsRights: Seq[OwcContextsRightsMatrix] = g.hasOwcContextsVisibility.flatMap {
155             o =>
156               val groupContextsBriefTuples: Seq[UserHasOwcRightsNative] = OwcContextDAO.findOwcContextsByContextBrief(o.owc_context_id)
157 
158               val rightsBundle: Seq[OwcContextsRightsMatrix] = groupContextsBriefTuples.map {
159                 tup =>
160                   OwcContextsRightsMatrix(
161                     owcContextId = tup.owcContextId,
162                     queryingUserAccountSubject = user.accountSubject,
163                     origOwnerAccountSubject = tup.origOwnerAccountSubject,
164                     viaGroups = Seq(viaGroup),
165                     contextIntrinsicVisibility = tup.contextOwnersVisibility,
166                     queryingUserAccessLevel = ownLevel)
167               }
168               rightsBundle
169           }
170           contextsRights
171       }
172     })
173 
174     originalOwnContextsNativeRights ++ nativeRightsPerContextInGroups
175   }
