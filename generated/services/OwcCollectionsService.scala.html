273   def generateMdResource(externalCatalogUrl: String, mdMetadata: MdMetadata, userMetaEntry: UserMetaRecord): OwcResource = {
274 
275     lazy val bboxFormat = new BboxArrayFormat
276 
277     val updatedTime = Try(OffsetDateTime.of(mdMetadata.citation.ciDate, LocalTime.of(12, 0), ZoneOffset.UTC))
278       .getOrElse(OffsetDateTime.now(ZoneId.systemDefault()))
279 
280     val baseLink = new URL(s"${portalConfig.portalExternalBaseLink}/context/resource/${userMetaEntry.uuid.toString}")
281 
282     val cswGetCapaOps = OwcOperation(
283       code = "GetCapabilities",
284       method = "GET",
285       mimeType = Some("application/xml"),
286       requestUrl = new URL(s"${portalConfig.portalExternalBaseLink}/pycsw/csw?SERVICE=CSW&VERSION=2.0.2&REQUEST=GetCapabilities"))

288     val cswGetRecordOps = OwcOperation(
289       code = "GetRecordById",
290       method = "GET",
291       mimeType = Some("application/xml"),
292       requestUrl = new URL(s"${portalConfig.portalExternalBaseLink}/pycsw/csw?request=GetRecordById&service=CSW&version=2.0.2&elementSetName=full&outputSchema=http://www.isotc211.org/2005/gmd&id=${mdMetadata.fileIdentifier}"))
293 
294     val cswOffering = OwcOffering(

301       mimeType = Some("application/json"),

304     val cswLink = OwcLink(

306       mimeType = cswGetRecordOps.mimeType,

308 
309     val owcResource = OwcResource(

312       title = mdMetadata.title,
313       subtitle = mdMetadata.abstrakt.toOption(),
314       updateDate = updatedTime,
315       author = List(OwcAuthor(name = mdMetadata.responsibleParty.individualName.toOption(),
316         email = parseEmailStringtoEmailAddress(mdMetadata.responsibleParty.email),
317         uri = Some(new URL(mdMetadata.responsibleParty.orgWebLinkage)))),
318       publisher = mdMetadata.responsibleParty.pointOfContact.toOption(),
319       rights = mdMetadata.distribution.useLimitation.toOption(),
320       temporalExtent = GeoDateParserUtils.parseOffsetDateString(Some(mdMetadata.extent.temporalExtent)),
321 
322       // links.alternates[] and rel=alternate
323       contentDescription = List(),
324 
325       // aka links.previews[] and rel=icon (atom)
326       preview = List(),
327 
328       // aka links.data[] and rel=enclosure (atom)
329       contentByRef = List(),
330 
331       // aka links.via[] & rel=via
332       resourceMetadata = List(viaLink, cswLink),
333       offering = List(cswOffering),
334       minScaleDenominator = Try(mdMetadata.scale.toDouble).toOption,
335       maxScaleDenominator = None,
336       active = None,
337       keyword = mdMetadata.keywords.map(w => OwcCategory(term = w, scheme = None, label = None)),
338       folder = None)
339 
340     logger.trace(Json.prettyPrint(owcResource.toJson))

343   }

602   def parseEmailStringtoEmailAddress(emailString: String): Option[EmailAddress] = {
603 
604     if (EmailAddress.isValid(emailString)) {
605       Some(EmailAddress(emailString))
606     } else {

609

622   def calculateBBoxForCollection(owcContext: OwcContext): Option[Rectangle] = {
623 
624     def computeJointRect(leftRo: Option[Rectangle], rightRo: Option[Rectangle]): Option[Rectangle] = {
625 
626       (leftRo, rightRo) match {
627         case (l, r) if l.isDefined && r.isEmpty => l
628         case (l, r) if l.isEmpty && r.isDefined => r
629         case (l, r) => l.flatMap(rectL => r.map { rectR =>
630           var bboxCalc = new BBoxCalculator(JtsSpatialContext.GEO)
631           bboxCalc.expandRange(rectL)
632           bboxCalc.expandRange(rectR)
633           bboxCalc.getBoundary
634         })
635       }
636     }
637 
638     val geoms: List[Option[Rectangle]] = owcContext.resource.map(_.geospatialExtent).filter(_.isDefined)
639     geoms.foldLeft(owcContext.areaOfInterest)((leftRo, rightRo) => computeJointRect(leftRo, rightRo))
640   }
