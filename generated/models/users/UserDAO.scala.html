39   val userParser = {
40     get[String]("email") ~

52   /**
53     * Create a User with a NamedParameterSet (supposedly more typesafe?)
54     *
55     * new java.time. API only partially implemented in Anorm Type mapping
56     *
57     * @param user
58     */
59   def createUser(user: User)(implicit connection: Connection): Option[User] = {
60     val nps = Seq[NamedParameter](// Tuples as NamedParameter
61       "email" -> user.email.value,
62       "accountsubject" -> user.accountSubject,
63       "firstname" -> user.firstname,
64       "lastname" -> user.lastname,
65       "laststatustoken" -> user.laststatustoken,
66       "password" -> user.password,
67       "laststatuschange" -> user.laststatuschange)
68 
69     val rowCount = SQL(
70       s"""
71           insert into $table_users values (
72             {email}, {accountsubject}, {firstname}, {lastname}, {password}, {laststatustoken}, {laststatuschange}
73           )
74         """).on(nps: _*).executeUpdate()
75 
76     rowCount match {
77       case 1 => Some(user)
78       case _ => None
79     }
80   }

144   def deleteUser(email: String)(implicit connection: Connection): Boolean = {
145     val u = findUserByEmailAsString(email)
