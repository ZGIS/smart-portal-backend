113     * trying to generalise for all transaction types
114     *
115     * @param transaction
116     * @return

118   def initiateTransactionRequest(transaction: CswTransactionWithIndexUpdate): Future[(WSResponse, WSResponse)] = {
119     //FIXME SR I find that pretty hard to read. Is there a better way of chaining WS calls?
120     // see https://www.playframework.com/documentation/2.5.x/ScalaWS#chaining-ws-calls
121     for {
122       getCapaResponse <- wsClient.url(CSW_OPERATIONS_METADATA_URL).get()
123       transactionResponse <- {
124         //check if getCapaResponse indicates, that the CSW can perform Transactions
125         val operation = (getCapaResponse.xml \\ "OperationsMetadata" \\ "Operation").filter(node => {
126           logger.debug(s"Attribute 'name'=${node.attribute("name").get.text.toString}")
127           node.attribute("name").get.text.equals("Transaction")
128         })
129         logger.debug(s"operation = ${operation.toString()}")
130         if (operation.isEmpty) {
131           throw new UnsupportedOperationException("CSW does not support Transaction.")
132         }
133 
134         // prepare transaction template
135         val finalXML = transaction.transform
136 
137         logger.debug(s"finalXml: ${finalXML.toString()}")
138         wsClient.url(CSW_URL).post(finalXML.toString())
139       }
140 
141       updateIngesterIndexResponse <- {
142         // if (transactionResponse.status == 200) {
143         val responseTuple = if (transaction.transactionType.equals(CswTransctionType.DELETE)) {
144           (wsClient.url(INGESTER_DELETE_FROM_INDEX_URL).withQueryString("fileIdentifier" -> transaction.fileIdentifier)get(), INGESTER_DELETE_FROM_INDEX_URL)
145         } else {
146           (wsClient.url(INGESTER_UPDATE_INDEX_URL).get(), INGESTER_UPDATE_INDEX_URL)
147         }
148         responseTuple._1.onSuccess {
149           case response if response.status == 200 => {
150             logger.info(
151               s"Successfully called '${responseTuple._2}' (${response.status} - ${response.statusText})")
152             logger.info(s"response: ${response.body}")
153           }
154           case response => {
155             logger.warn(
156               s"Call to '${responseTuple._2}' returned error: ${response.status} - ${response.statusText}")
157             logger.warn(s"response: ${response.body}")
158           }
159         }
160         responseTuple._1
161 
162       }
163     } yield (transactionResponse, updateIngesterIndexResponse)
164   }

167     * check the response of the tansaction
168     *
169     * @param response
170     * @param transaction
171     * @return
172     */
173   def evaluateTransactionResponse(response: WSResponse, transaction: CswTransactionWithIndexUpdate): Either[ErrorResult, UUID] = {
174     logger.debug(response.xml.toString())
175     response.xml match {
176       case e: Elem if e.label == "TransactionResponse" =>
177 
178         if (transaction.responsePositive(e)) {
179           Right(UUID.fromString(transaction.fileIdentifier))
180         } else {
181           Left(ErrorResult("Identifier not clear in TransactionResponse", None))
182         }
183 
184       case e: Elem if e.label == "ExceptionReport" =>
185         val message = (e \\ "Exception" \\ "ExceptionText").text
186         val error = ErrorResult(message, None)
187         Left(error)
188 
189       case _ =>
190         val error = ErrorResult(s"Unexpected Response from CSW: ${response.status} - ${response.statusText}",
191           Some(response.body))
192         Left(error)
193 
194     }
195   }

202   def insertMetadataRecord: Action[JsValue] = (authenticationAction andThen userAction).async(parse.json) {
203     request =>
204 
205       logger.debug(request.toString)
206       (request.body.as[JsObject] \ "metadata").validate[MdMetadata].fold(
207         errors => {
208           logger.error(JsError.toJson(errors).toString())
209           val error = ErrorResult("Could not validate metadata from your request.", Some(JsError.toJson(errors).toString()))
210           Future {
211             BadRequest(Json.toJson(error)).as(JSON)
212           }
213         },
214         mdMetadata => {
215           logger.trace(Json.prettyPrint(mdMetadata.toJson()))
216           logger.debug(s"MD_Metadata XML: ${mdMetadata.toXml()}")
217 
218           val insertTransaction = CswInsertRequest(cswtInsertXml, mdMetadata)
219           val futureResponse: Future[(WSResponse, WSResponse)] = initiateTransactionRequest(insertTransaction)
220 
221           futureResponse.recover {
222             case e: Exception =>
223               logger.error("Insert CSW threw exception", e)
224               val error = ErrorResult(s"Exception (${e.getClass.getCanonicalName}) on CSW insert: ${e.getMessage}",
225                 None)
226               InternalServerError(Json.toJson(error)).as(JSON)
227           }
228 
229           futureResponse.map { response =>
230             logger.debug("Ingester Update: " + response._2.body)
231             val responseEval = evaluateTransactionResponse(response._1, insertTransaction)
232             responseEval.fold[Result](
233               error => {
234                 logger.warn(error.message)
235                 InternalServerError(Json.toJson(error)).as(JSON)
236               },
237               uuid => {
238                 logger.debug(s"Adding ${mdMetadata.fileIdentifier} to default collection of ${request.user.email}.")
239                 val userMetaEntryOk = userService.insertUserMetaRecordEntry(CSW_URL, mdMetadata.fileIdentifier, request.user.email)
240                 userMetaEntryOk.fold {
241                   val error = ErrorResult(s"Could not add ${mdMetadata.fileIdentifier} to your collection of ${request.user.email}.", None)
242                   logger.warn(error.message)
243                   BadRequest(Json.toJson(error)).as(JSON)
244                 } {
245                   userMetaEntry =>
246                     val owcresource = collectionsService.generateMdResource(CSW_URL, mdMetadata, userMetaEntry)
247                     val added = collectionsService.addMdResourceToUserDefaultCollection(CSW_URL, owcresource, userMetaEntry)
248                     if (added) {
249                       Ok(Json.obj("type" -> "success", "fileIdentifier" -> userMetaEntry.originaluuid.toString, "userMetaRecord" -> Json.toJson(userMetaEntry),
250                         "message" -> s"Inserted as ${uuid.toString} and reference entry with with refUuid ${userMetaEntry.uuid} added to your data collection."))
251                     } else {
252                       val error = ErrorResult(s"Could not add ${mdMetadata.fileIdentifier} to your collection of ${request.user.email}.", None)
253                       logger.warn(error.message)
254                       BadRequest(Json.toJson(error)).as(JSON)
255                     }
256                 }
257               })
258           }
259         }
260       )
261   }
