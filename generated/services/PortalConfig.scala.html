57 /**
58   * provides reasonably easy central place for the variable config values from application.conf
59   * for emailing and app defaults, if any of these fails, we don't want the program to start
60   *
61   * @param configuration
62   */
63 @Singleton
64 class PortalConfig @Inject()(configuration: Configuration) extends PortalConfigHolder with ClassnameLogger {
65 
66   implicit class OptionOps[A](opt: Option[A]) {
67     def check(msg: String): Option[A] = {
68       if (opt.isEmpty) {
69         new NoSuchElementException("Config not found for " + msg)
70       }
71       opt
72     }
73   }
74 
75   /**
76     *
77     * fail early
78     *
79     * @param a
80     * @param path
81     * @return
82     */
83   private def getOrReportStr(path: String): Option[String] = {
84     val a = configuration.getString(path)
85     a.check(path)
86   }
87 
88   private def getOrReportEmpty[A](a: Option[A], path: String): Option[A] = {
89     a.fold({
90       throw new NoSuchElementException("Config not found for " + path)
91     })({
92       lst =>
93         lst match {
94           case x: JListConfiguration =>
95             if (x.size() <= 0) {
96               logger.error("Config is empty list for " + path)
97             }
98             Some(lst)
99           case x: JListString  =>
100             if (x.size() <= 0) {
101               logger.error("Config is empty list for " + path)
102             } else {
103               x.asScala.foreach(s => logger.info(path + " value: " + s))
104             }
105             Some(lst)
106           case _ => Some(lst)
107         }
108     })
109   }
