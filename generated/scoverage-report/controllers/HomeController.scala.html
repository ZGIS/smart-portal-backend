95   /**
96     * login with JSON (from Angular / form)
97     * {"username":"akmoch","password":"testpass123"}
98     *
99     * Create the authentication token and sets the cookie [[AuthTokenCookieKey]]
100     * for AngularJS to use also in X-XSRF-TOKEN in HTTP header.
101     *
102     * @return
103     */
104   def login = Action(parse.json) { implicit request =>
105     request.body.validate[LoginCredentials].fold(
106       errors => {
107         logger.error(JsError.toJson(errors).toString())
108         BadRequest(Json.obj("status" -> "ERR", "message" -> JsError.toJson(errors)))
109       },
110       valid = credentials => {
111         // find user in db and compare password stuff
112         userDAO.authenticate(credentials.username, credentials.password).fold {
113           logger.error("User or password wrong.")
114           Unauthorized(Json.obj("status" -> "ERR", "message" -> "Username or password wrong."))
115         } { user =>
116           val uaIdentifier: String = request.headers.get(UserAgentHeader).getOrElse(UserAgentHeaderDefault)
117           // logger.debug(s"Logging in username from $uaIdentifier")
118           val token = passwordHashing.createSessionCookie(user.username, uaIdentifier)
119           cache.set(token, user.username)
120           Ok(Json.obj("status" -> "OK", "token" -> token, "username" -> user.username, "userprofile" -> user.asProfileJs()))
121             .withCookies(Cookie(AuthTokenCookieKey, token, None, httpOnly = false))
122         }
123       })
124   }
125 
126   /**
127     * Log-out a user. Invalidates the authentication token.
128     *
129     * Discard the cookie [[AuthTokenCookieKey]] to have AngularJS no longer set the
130     * X-XSRF-TOKEN in HTTP header.
131     */
132   def logout = HasToken(parse.empty) { token =>
133     username =>
134       implicit request =>
135         cache.remove(token)
136         Ok.discardingCookies(DiscardingCookie(name = AuthTokenCookieKey))
137   }

162   }
