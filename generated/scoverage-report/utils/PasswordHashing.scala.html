33   // This Scala implementation of password hashing was inspired by:

35   val PBKDF2_ALGORITHM = "PBKDF2WithHmacSHA1"

37   val SALT_BYTE_SIZE = 24
38   val HASH_BYTE_SIZE = 24
39   val PBKDF2_ITERATIONS = 1000

41   val ITERATION_INDEX = 0
42   val SALT_INDEX = 1
43   val PBKDF2_INDEX = 2

45   val DEFAULT_SECRET = "insecure"

47   lazy private val appSecret = configuration.getString("application.secret").getOrElse(DEFAULT_SECRET)

49   def createSessionCookie(username: String, useragent: String): String = {
50     val in = appSecret + username + useragent.toUpperCase
51     val digest = MessageDigest.getInstance("SHA-512").digest(in.getBytes("UTF-8"))
52     toHex(digest)
53   }
54 
55   def testSessionCookie(cookieHash: String, username: String, useragent: String): Boolean = {
56     val in = appSecret + username + useragent.toUpperCase
57     val digestHash = MessageDigest.getInstance("SHA-512").digest(in.getBytes("UTF-8"))
58     val testHash = fromHex(cookieHash)
59     slowEquals(digestHash, testHash)
60   }
61 
62   private def toHex(bytes: Array[Byte]): String = {
63     bytes.map("%02X" format _).mkString
64   }
65 
66   def createHash(password: String): String = {
67     createHash(password.toCharArray())
68   }

70   def createHash(password: Array[Char]): String = {
71     val salt = nextRandomSalt
72     val hash = pbkdf2(password, salt, PBKDF2_ITERATIONS, HASH_BYTE_SIZE)
73     PBKDF2_ITERATIONS + ":" + toHex(salt) + ":" + toHex(hash)
74   }
75 
76   private def nextRandomSalt(): Array[Byte] = {
77     val random = new SecureRandom()
78     val salt = Array.ofDim[Byte](SALT_BYTE_SIZE)
79     random.nextBytes(salt)
80     salt
81   }
82 
83   def validatePassword(password: String, correctHash: String): Boolean = {
84     validatePassword(password.toCharArray(), correctHash)
85   }
86 
87   def validatePassword(password: Array[Char], correctHash: String): Boolean = {
88     val params = correctHash.split(":")
89     val iterations = Integer.parseInt(params(ITERATION_INDEX))
90     val salt = fromHex(params(SALT_INDEX))
91     val hash = fromHex(params(PBKDF2_INDEX))
92     val testHash = pbkdf2(password, salt, iterations, hash.length)
93     slowEquals(hash, testHash)
94   }
95 
96   private def slowEquals(a: Array[Byte], b: Array[Byte]): Boolean = {
97     val range = 0 until scala.math.min(a.length, b.length)
98     val diff = range.foldLeft(a.length ^ b.length) {
99       case (acc, i) => acc | a(i) ^ b(i)
100     }
101     diff == 0
102   }
103 
104   private def pbkdf2(password: Array[Char], salt: Array[Byte], iterations: Int, bytes: Int): Array[Byte] = {
105     val spec = new PBEKeySpec(password, salt, iterations, bytes * 8)
106     val skf = SecretKeyFactory.getInstance(PBKDF2_ALGORITHM)
107     skf.generateSecret(spec).getEncoded()
108   }
109 
110   private def fromHex(hex: String): Array[Byte] = {
111     hex.sliding(2, 2).toArray.map(Integer.parseInt(_, 16).toByte)
112   }
