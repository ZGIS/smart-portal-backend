82   /**
83     * login with JSON (from Angular / form)
84     * {"username":"akmoch","password":"testpass123"}
85     *
86     * Create the authentication token and sets the cookie [[AuthTokenCookieKey]]
87     * for AngularJS to use also in X-XSRF-TOKEN in HTTP header.
88     *
89     * @return
90     */
91   def login = Action(parse.json) { implicit request =>
92     request.body.validate[LoginCredentials].fold(
93       errors => {
94         logger.error(JsError.toJson(errors).toString())
95         BadRequest(Json.obj("status" -> "ERR", "message" -> JsError.toJson(errors)))
96       },
97       valid = credentials => {
98         // find user in db and compare password stuff
99         userDAO.authenticate(credentials.username, credentials.password).fold {
100           logger.error("User or password wrong.")
101           BadRequest(Json.obj("status" -> "ERR", "message" -> "Username or password wrong."))
102         } { user =>
103           val uaIdentifier: String = request.headers.get(UserAgentHeader).getOrElse(UserAgentHeaderDefault)
104           // logger.debug(s"Logging in username from $uaIdentifier")
105           val token = passwordHashing.createSessionCookie(user.username, uaIdentifier)
106           cache.set(token, user.username)
107           Ok(Json.obj("status" -> "OK", "token" -> token, "username" -> user.username))
108             .withCookies(Cookie(AuthTokenCookieKey, token, None, httpOnly = false))
109         }
110       })
111   }
112 
113   /**
114     * Log-out a user. Invalidates the authentication token.
115     *
116     * Discard the cookie [[AuthTokenCookieKey]] to have AngularJS no longer set the
117     * X-XSRF-TOKEN in HTTP header.
118     */
119   def logout = HasToken(parse.empty) { token =>
120     username => implicit request =>
121       cache.remove(token)
122       Ok.discardingCookies(DiscardingCookie(name = AuthTokenCookieKey))
123   }
124

135   }

143     username => implicit request =>
144       cache.remove(token)
145       // do some Google OAuth2 unregisteR
146       Ok.discardingCookies(DiscardingCookie(name = AuthTokenCookieKey))
147   }
