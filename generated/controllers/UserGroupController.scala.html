95   /**
96     * create a new user group. user must already be in the sent object rights list as power user
97     *
98     * @return
99     */
100   def createUsersOwnUserGroup: Action[JsValue] = defaultAuthAction(parse.json) {
101     request =>
102       request.body.validate[UserGroup].fold(
103         errors => {
104           logger.error(JsError.toJson(errors).toString())
105           val error: ErrorResult = ErrorResult("Usergroup format could not be read.",
106             Some(StringEscapeUtils.escapeJson(errors.mkString("; "))))
107           BadRequest(Json.toJson(error)).as(JSON)
108         },
109         userGroup => {
110           // check if creating user is in the member list and has power user right
111           val isPresent = userGroup.hasUsersLevel.exists(ul => ul.users_accountsubject.equals(request.user.accountSubject) && ul.userlevel >= 2)
112           if (!isPresent) {
113             logger.error("User not listed with adequate rights. We would create a group that you can't edit, aborting.")
114             val error = ErrorResult("User not listed with adequate rights. We would create a group that you can't edit, aborting.", None)
115             BadRequest(Json.toJson(error)).as(JSON)
116           } else {
117             userGroupService.createUsersOwnUserGroup(request.user, userGroup).fold {
118               logger.error("Error creating the user group.")
119               val error = ErrorResult("Error creating the user group.", None)
120               BadRequest(Json.toJson(error)).as(JSON)
121             } {
122               ugroup =>
123                 Ok(Json.obj("status" -> "OK", "usergroup" -> Json.toJson(ugroup)))
124             }
125           }
126         })
127   }

198   def updateOwcContextVisibility(owcContextId: String, visibility: Int): Action[Unit] = defaultAuthAction(parse.empty) {
199     request =>
200       // does collection exist, user exists because must be authenticated
201       val ugList = userGroupService.getOwcContextsRightsMatrixForUser(request.user)
202 
203       val areYouPowerUserForThisContext = ugList.filter(_.owcContextId.contentEquals(owcContextId)).exists(_.queryingUserAccessLevel >= 2)
204 
205       if (areYouPowerUserForThisContext) {
206         val visOk = collectionsService.updateOwcContextVisibility(owcContextId, request.user, visibility)
207         if (visOk) {
208           Ok(Json.obj("status" -> "OK", "owcContext" -> owcContextId, "user" -> request.user.accountSubject, "visibility" -> visibility))
209         } else {
210           logger.error("Error changing the visibility of the context, database error.")
211           val error = ErrorResult("Error changing the visibility of the context, database error.", None)
212           BadRequest(Json.toJson(error)).as(JSON)
213         }
214       } else {
215         logger.error("Error changing the visibility of the context, not allowed.")
216         val error = ErrorResult("Error changing the visibility of the context, not allowed.", None)
217         BadRequest(Json.toJson(error)).as(JSON)
218       }
