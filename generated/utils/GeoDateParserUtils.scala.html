72   def parseOffsetDateString(dateStringOption: Option[String]): Option[List[OffsetDateTime]] = {
73     if (dateStringOption.isDefined) {
74       val isoTemporalString = dateStringOption.get
75       if (isoTemporalString.contains("/")) {
76         parseDateStringAsOffsetInterval(isoTemporalString).toOption
77       } else {

79       }
80     } else {

82     }
83   }

103   /**
104     *
105     * @param isoTemporalString
106     * @return
107     */
108   private def parseDateStringAsOffsetInterval(isoTemporalString: String): Try[List[OffsetDateTime]] = {
109     Try {
110       val dateStrings = isoTemporalString.replace("\"", "").trim.split("/").toList
111       val date1 = OffsetDateTime.parse(dateStrings.head)
112       val date2 = OffsetDateTime.parse(dateStrings.last)
113       List(date1, date2)
114     }
115   }
116 
117   /**
118     * String serialiser to store either single offset date or the interval in DB
119     *
120     * @param dateRange
121     * @return
122     */
123   def writeOffsetDatesAsDateString(dateRange: List[OffsetDateTime]): Option[String] = {
124     lazy val formatter = DateTimeFormatter.ISO_OFFSET_DATE_TIME
125     val dates = dateRange.sortWith((a, b) => a.isBefore(b))
126     if (dates.head == dates.last) {
127       Try(dates.head.format(formatter)).toOption
128     } else {
129       Try {
130         val d1 = dates.head.format(formatter)
131         val d2 = dates.last.format(formatter)
132         d1 + "/" + d2
133       }.toOption
134     }
135   }
136 
137   def timeLineOrder: Comparator[OffsetDateTime] = OFFSET_INSTANT_COMPARATOR

139   private val OFFSET_INSTANT_COMPARATOR: Comparator[OffsetDateTime] =

141       override def compare(datetime1: OffsetDateTime, datetime2: OffsetDateTime): Int = {
142         var cmp: Int = java.lang.Long.compare(datetime1.toEpochSecond, datetime2.toEpochSecond)
143         if (cmp == 0)
144           cmp = java.lang.Long.compare(datetime1.getNano, datetime2.getNano)
145         cmp
146       }
147     }
