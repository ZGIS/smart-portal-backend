54   def apply(v: String) : StatusToken = {
55     v match {
56       case "REGISTERED" => REGISTERED
57       case "ACTIVE" => ACTIVE
58       case "PASSWORDRESET" => PASSWORDRESET
59       case "EMAILVALIDATION" => EMAILVALIDATION
60       case "BLOCKED" => BLOCKED
61       case "DELETED" => DELETED
62       case _ => throw new IllegalArgumentException(s"Value $v is not a valid StatusToken value")
63     }
64   }
65 
66   val activatedTokens: List[String] = List(ACTIVE.value, PASSWORDRESET.value, EMAILVALIDATION.value)
67 
68   val deactivatedTokens: List[String] = List(BLOCKED.value, DELETED.value)
69 
70   def isValid(v: String) : Boolean = {
71     Try (apply(v)).isSuccess
72   }
73 
74   def isActive(token: StatusToken): Boolean = {
75     activatedTokens.contains(token.value)
76   }
77 
78   def isBlocked(token: StatusToken): Boolean = {
79     deactivatedTokens.contains(token.value)
