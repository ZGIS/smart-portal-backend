56 /**
57   * provides reasonably easy central place for the variable config values from application.conf
58   * for emailing and app defaults, if any of these fails, we don't want the program to start
59   *
60   * @param configuration
61   */
62 @Singleton
63 class PortalConfig @Inject()(configuration: Configuration) extends PortalConfigHolder with ClassnameLogger {
64 
65   implicit class OptionOps[A](opt: Option[A]) {
66     def check(msg: String): Option[A] = {
67       if (opt.isEmpty) {
68         new NoSuchElementException("Config not found for " + msg)
69       }
70       opt
71     }
72   }
73 
74   /**
75     *
76     * fail early
77     *
78     * @param a
79     * @param path
80     * @return
81     */
82   private def getOrReportStr(path: String): Option[String] = {
83     val a = configuration.getString(path)
84     a.check(path)
85   }
86 
87   private def getOrReportEmpty[A](a: Option[A], path: String): Option[A] = {
88     a.fold({
89       throw new NoSuchElementException("Config not found for " + path)
90     })({
91       lst =>
92         lst match {
93           case x: JListConfiguration =>
94             if (x.size() <= 0) {
95               logger.error("Config is empty list for " + path)
96             }
97             Some(lst)
98           case x: JListString  =>
99             if (x.size() <= 0) {
100               logger.error("Config is empty list for " + path)
101             } else {
102               x.asScala.foreach(s => logger.info(path + " value: " + s))
103             }
104             Some(lst)
105           case _ => Some(lst)
106         }
107     })
108   }
