20 import com.google.inject.AbstractModule
21 import java.time.Clock
22 
23 import models.UserDAO
24 import services.{ApplicationTimer, AtomicCounter, Counter, EmailService}
25 import utils.PasswordHashing
26 
27 /**
28  * This class is a Guice module that tells Guice how to bind several
29  * different types. This Guice module is created when the Play
30  * application starts.
31 
32  * Play will automatically use any class called `Module` that is in
33  * the root package. You can create modules in other locations by
34  * adding `play.modules.enabled` settings to the `application.conf`
35  * configuration file.
36  */
37 class Module extends AbstractModule {
38 
39   override def configure() = {
40     // Use the system clock as the default implementation of Clock
41     bind(classOf[Clock]).toInstance(Clock.systemDefaultZone)
42     // Ask Guice to create an instance of ApplicationTimer when the
43     // application starts.
44     bind(classOf[ApplicationTimer]).asEagerSingleton()
45     // Set AtomicCounter as the implementation for Counter.
46     bind(classOf[Counter]).to(classOf[AtomicCounter])
47 
48     bind(classOf[PasswordHashing]).asEagerSingleton()
49     bind(classOf[EmailService]).asEagerSingleton()
50     bind(classOf[UserDAO]).asEagerSingleton()
51   }
52 
53 }
