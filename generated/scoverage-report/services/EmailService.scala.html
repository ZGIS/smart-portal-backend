54 
55     val emailText =
56       """Hello %s,
57         |thank you for registering on the GW HUB.
58         |Please click on the following link to confirm your registration:

61         |
62         |If you have any question please email us to %s.
63         |
64         |Your GW HUB Team
65       """.format(usernameTo, portalApiHost, linkId, emailFrom).stripMargin
66 
67     val from = new Email(emailFrom)
68     val to = new Email(emailTo)
69     val content = new Content("text/plain", emailText)
70     val mail = new Mail(from, subject, to, content)
71 
72     try {
73 
74       val request = new Request()
75       request.method = Method.POST
76       request.endpoint = "mail/send"
77       request.body = mail.build()
78       val response = sg.api(request)
79       logger.trace(s"mail api response status: ${response.statusCode}")
80       logger.trace(s"mail api response.body: ${response.body}")
81       logger.trace(s"response.headers: ${response.headers}")
82       logger.trace(s"mail api response status: ${response.statusCode}")
83 
84       true
85     } catch {
86       case ioex: IOException => {
87         logger.error("IO Messaging exception: " + ioex.getLocalizedMessage)
88         false
89       }
90       case e: Exception => {
91         logger.error("Other email problem: " + e.getLocalizedMessage)

93       }
