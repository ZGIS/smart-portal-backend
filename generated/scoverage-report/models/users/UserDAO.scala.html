42     */
43   val userParser = {
44     get[String]("email") ~
45       get[String]("username") ~
46       get[String]("firstname") ~
47       get[String]("lastname") ~
48       get[String]("password") ~
49       get[String]("laststatustoken") ~
50       get[ZonedDateTime]("laststatuschange") map {
51       case email~username~firstname~lastname~password~laststatustoken~laststatuschange =>
52         User(email, username, firstname, lastname, password, laststatustoken, laststatuschange)

54   }

125       val rowCount = SQL(
126         """
127           insert into users values (
128             {email}, {username}, {firstname}, {lastname}, {password}, {laststatustoken}, {laststatuschange}

130         """).on(nps: _*).executeUpdate()

132       rowCount match {
