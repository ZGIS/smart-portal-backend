643   /**
644     * create owc properties in the database, title has additionally unique constraint, uuid will be unique and primary key
645     *
646     * @param owcProperties
647     * @return
648     */
649   def createOwcProperties(owcProperties: OwcProperties): Option[OwcProperties] = {
650 
651     db.withTransaction {
652       implicit connection => {
653 
654         val rowCount = SQL(
655           s"""
656           insert into $tableOwcProperties values (
657             {uuid}, {language}, {title}, {subtitle}, {updated}, {generator}, {rights}, {creator}, {publisher}
658           )
659         """).on(
660           'uuid -> owcProperties.uuid.toString,
661           'language -> owcProperties.language,
662           'title -> owcProperties.title,
663           'subtitle -> owcProperties.subtitle,
664           'updated -> owcProperties.updated,
665           'generator -> owcProperties.generator,
666           'rights -> owcProperties.rights,
667           'creator -> owcProperties.creator,
668           'publisher -> owcProperties.publisher
669         ).executeUpdate()
670 
671         owcProperties.authors.foreach {
672           author => {
673             if (findOwcAuthorByUuid(author.uuid).isEmpty) {
674               createOwcAuthor(author)
675             }
676 
677             SQL(
678               s"""insert into $tableOwcPropertiesHasOwcAuthors values (
679                  |{owc_properties_uuid}, {owc_authors_uuid}
680                  |)
681                """.stripMargin).on(
682               'owc_properties_uuid -> owcProperties.uuid.toString,
683               'owc_authors_uuid -> author.uuid.toString
684             ).executeUpdate()
685           }
686         }
687 
688         owcProperties.contributors.foreach {
689           authorAsContributor => {
690             if (findOwcAuthorByUuid(authorAsContributor.uuid).isEmpty) {
691               createOwcAuthor(authorAsContributor)
692             }
693 
694             SQL(
695               s"""insert into $tableOwcPropertiesHasOwcAuthorsAsContributors values (
696                  |{owc_properties_uuid}, {owc_authors_as_contributors_uuid}
697                  |)
698                """.stripMargin).on(
699               'owc_properties_uuid -> owcProperties.uuid.toString,
700               'owc_authors_as_contributors_uuid -> authorAsContributor.uuid.toString
701             ).executeUpdate()
702           }
703         }
704 
705         owcProperties.categories.foreach {
706           category => {
707             if (findOwcCategoriesByUuid(category.uuid).isEmpty) {
708               createOwcCategory(category)
709             }
710 
711             SQL(
712               s"""insert into $tableOwcPropertiesHasOwcCategories  values (
713                  |{owc_properties_uuid}, {owc_categories_uuid}
714                  |)
715                """.stripMargin).on(
716               'owc_properties_uuid -> owcProperties.uuid.toString,
717               'owc_categories_uuid -> category.uuid.toString
718             ).executeUpdate()
719           }
720         }
721 
722         owcProperties.links.foreach {
723           owcLink => {
724             if (findOwcLinksByUuid(owcLink.uuid).isEmpty) {
725               createOwcLink(owcLink)
726             }
727 
728             SQL(
729               s"""insert into $tableOwcPropertiesHasOwcLinks  values (
730                  |{owc_properties_uuid}, {owc_links_uuid}
731                  |)
732                """.stripMargin).on(
733               'owc_properties_uuid -> owcProperties.uuid.toString,
734               'owc_links_uuid -> owcLink.uuid.toString
735             ).executeUpdate()
736           }
737         }
738 
739         rowCount match {
740           case 1 => Some(owcProperties)
741           case _ => None
742         }
743       }
744     }
745   }
