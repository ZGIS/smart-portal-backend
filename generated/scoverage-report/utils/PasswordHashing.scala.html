50   def createSessionCookie(username: String, useragent: String): String = {
51     val in = appSecret + username + useragent.toUpperCase
52     val digest = MessageDigest.getInstance("SHA-512").digest(in.getBytes("UTF-8"))
53     toHex(digest)
54   }

56   def testSessionCookie(cookieHash: String, username: String, useragent: String): Boolean = {
57     val in = appSecret + username + useragent.toUpperCase
58     val digestHash = MessageDigest.getInstance("SHA-512").digest(in.getBytes("UTF-8"))
59     val testHash = fromHex(cookieHash)
60     slowEquals(digestHash, testHash)
61   }

63   private def toHex(bytes: Array[Byte]): String = {
64     bytes.map("%02X" format _).mkString
65   }

71   def createHash(password: Array[Char]): String = {
72     val salt = nextRandomSalt
73     val hash = pbkdf2(password, salt, PBKDF2_ITERATIONS, HASH_BYTE_SIZE)
74     PBKDF2_ITERATIONS + ":" + toHex(salt) + ":" + toHex(hash)
75   }
76 
77   private def nextRandomSalt(): Array[Byte] = {
78     val random = new SecureRandom()
79     val salt = Array.ofDim[Byte](SALT_BYTE_SIZE)
80     random.nextBytes(salt)
81     salt
82   }
83 
84   def validatePassword(password: String, correctHash: String): Boolean = {
85     validatePassword(password.toCharArray(), correctHash)
86   }
87 
88   def validatePassword(password: Array[Char], correctHash: String): Boolean = {
89     val params = correctHash.split(":")
90     val iterations = Integer.parseInt(params(ITERATION_INDEX))
91     val salt = fromHex(params(SALT_INDEX))
92     val hash = fromHex(params(PBKDF2_INDEX))
93     val testHash = pbkdf2(password, salt, iterations, hash.length)
94     slowEquals(hash, testHash)
95   }
96 
97   private def slowEquals(a: Array[Byte], b: Array[Byte]): Boolean = {
98     val range = 0 until scala.math.min(a.length, b.length)
99     val diff = range.foldLeft(a.length ^ b.length) {
100       case (acc, i) => acc | a(i) ^ b(i)
101     }
102     diff == 0
103   }
104 
105   private def pbkdf2(password: Array[Char], salt: Array[Byte], iterations: Int, bytes: Int): Array[Byte] = {
106     val spec = new PBEKeySpec(password, salt, iterations, bytes * 8)
107     val skf = SecretKeyFactory.getInstance(PBKDF2_ALGORITHM)
108     skf.generateSecret(spec).getEncoded()
109   }
110 
111   private def fromHex(hex: String): Array[Byte] = {
112     hex.sliding(2, 2).toArray.map(Integer.parseInt(_, 16).toByte)
113   }
