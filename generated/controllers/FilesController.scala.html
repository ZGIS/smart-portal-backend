60   def uploadMultipartForm: Action[MultipartFormData[TemporaryFile]] = HasToken(parse.multipartFormData) {
61     token =>
62       authUser =>
63         implicit request =>
64 
65           request.body.file("file").map { theFile =>
66 
67             val filename = theFile.filename
68             val contentType = theFile.contentType
69             val pathOfUploadTmp = Paths.get(uploadDataPath)
70             val intermTempDir = Files.createTempDirectory(pathOfUploadTmp, "sac-upload-")
71             val tmpFile = new File(intermTempDir.resolve(filename).toAbsolutePath.toString)
72             val handle = theFile.ref.moveTo(tmpFile)
73             val kilobytes = handle.length / 1024
74 
75             googleService.uploadFileGoogleBucket(handle) match {
76               case Left(errorResult) =>
77                 logger.error(errorResult.message + errorResult.details.mkString)
78                 InternalServerError(Json.toJson(errorResult)).as(JSON)
79               case Right(blob) =>
80                 // return the public download link
81                 val owcResource = collectionsService.fileMetadata(filename, contentType, authUser, blob.getMediaLink(), Try(kilobytes.toInt).toOption)
82                 logger.trace(Json.prettyPrint(owcResource.toJson))
83 
84                 val userFileEntryOk = userService.insertUserFileEntry(filename, authUser, blob.getMediaLink())
85                 val insertOk = collectionsService.addPlainFileResourceToUserDefaultCollection(owcResource, authUser)
86 
87                 if (insertOk && userFileEntryOk.isDefined) {
88                   logger.debug(s"file upload $filename to ${blob.getMediaLink}")
89                   Try(handle.delete()).failed.map(ex => logger.error(ex.getLocalizedMessage))
90                   Try(Files.delete(intermTempDir)).failed.map(ex => logger.error(ex.getLocalizedMessage))
91                   //FIXME SR do we also want to have a general "return" object? Status is always in the respone so it does not need to be in here
92                   Ok(Json.obj("status" -> "OK", "message" -> s"file uploaded $filename.", "file" -> blob.getMediaLink(), "entry" -> owcResource.toJson))
93                 } else {
94                   logger.error("file metadata insert failed.")
95                   val error = ErrorResult("File metadata insert failed.", None)
96                   BadRequest(Json.toJson(error)).as(JSON)
97                 }
98               case _ =>
99                 logger.error("file upload to cloud storage failed")
100                 val error = ErrorResult("File upload to cloud storage failed.", None)
101                 BadRequest(Json.toJson(error)).as(JSON)
102             }
103 
104           }.getOrElse {
105             logger.error("file upload from client failed")
106             val error = ErrorResult("File upload from client failed.", None)
107             BadRequest(Json.toJson(error)).as(JSON)
108           }
109   }
