58   def uploadMultipartForm: Action[MultipartFormData[TemporaryFile]] = (authenticationAction andThen userAction) (parse.multipartFormData) {
59     request =>
60       request.body.file("file").map { theFile =>
61 
62         val filename = theFile.filename
63         val contentType = theFile.contentType
64         val pathOfUploadTmp = Paths.get(uploadDataPath)
65         val intermTempDir = Files.createTempDirectory(pathOfUploadTmp, "sac-upload-")
66         val tmpFile = new File(intermTempDir.resolve(filename).toAbsolutePath.toString)
67         val handle = theFile.ref.moveTo(tmpFile)
68         val kilobytes = handle.length / 1024
69 
70         googleService.uploadFileGoogleBucket(handle) match {
71           case Left(errorResult) =>
72             logger.error(errorResult.message + errorResult.details.mkString)
73             InternalServerError(Json.toJson(errorResult)).as(JSON)
74           case Right(blob) =>
75             // return the public download link
76             val userFileEntryOk = userService.insertUserFileEntry(filename, request.user.email, blob.getMediaLink())
77             userFileEntryOk.fold {
78               logger.error("file metadata insert failed.")
79               val error = ErrorResult("File metadata insert failed.", None)
80               BadRequest(Json.toJson(error)).as(JSON)
81             } { userFile =>
82               val owcResource = collectionsService.fileMetadata(userFile, contentType, Try(kilobytes.toInt).toOption)
83               logger.trace(Json.prettyPrint(owcResource.toJson))
84               logger.debug(s"file upload $filename to ${blob.getMediaLink} with reference ${userFile.linkreference}")
85               Try(handle.delete()).failed.map(ex => logger.error(ex.getLocalizedMessage))
86               Try(Files.delete(intermTempDir)).failed.map(ex => logger.error(ex.getLocalizedMessage))
87 
88               //FIXME SR do we also want to have a general "return" object? Status is always in the respone so it does not need to be in here
89               val added = collectionsService.addPlainFileResourceToUserDefaultCollection(owcResource, request.user)
90               if (added) {
91                 Ok(Json.obj("status" -> "OK", "message" -> s"file uploaded $filename.", "file" -> userFile.linkreference, "entry" -> owcResource.toJson))
92               } else {
93                 logger.error("Insert into own collection failed.")
94                 val error = ErrorResult("Insert into own collection failed.", None)
95                 BadRequest(Json.toJson(error)).as(JSON)
96               }
97             }
98           case _ =>
99             logger.error("file upload to cloud storage failed")
100             val error = ErrorResult("File upload to cloud storage failed.", None)
101             BadRequest(Json.toJson(error)).as(JSON)
102         }
103       }.getOrElse {
104         logger.error("file upload failed, no file found for upload")
105         val error = ErrorResult("file upload failed, no file found for upload", None)
106         BadRequest(Json.toJson(error)).as(JSON)
107       }
108   }

123           val logRequest = UserLinkLogging(id = None,
124             timestamp = ZonedDateTime.now.withZoneSameInstant(ZoneId.of(appTimeZone)),
125             ipaddress = Some(request.remoteAddress),
126             useragent = request.headers.get(UserAgentHeader),
127             email = request.optionalSession.map(_.email),
128             link = userFile.linkreference,
129             referer = request.headers.get(RefererHeader))
130 
131           val updated = userService.logLinkInfo(logRequest)
132           logger.trace(logRequest.toString)
133           Ok(Json.obj("status" -> "OK", "linkreference" -> userFile.linkreference, "originalfilename" -> userFile.originalfilename))
134       }
135   }
