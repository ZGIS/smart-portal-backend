76   /**
77     * instantiate an OwcFeatureType either OwcEntry, or OwcDocument
78     *
79     * @param id
80     * @param featureType
81     * @param bboxAsWkt
82     * @return
83     */
84   def instantiateOwcFeatureType(id: String, featureType: String, bboxAsWkt: Option[String]): OwcFeatureType = {
85 
86     val emptyProfile = OwcLink(UUID.randomUUID(), "profile", None, "http://www.opengis.net/spec/owc-atom/1.0/req/core", None)
87     val emptySelf = OwcLink(UUID.randomUUID(), "self", Some("application/json"), "http://example.com/empty", None)
88     val now = ZonedDateTime.now.withZoneSameInstant(ZoneId.systemDefault())
89     logger.debug(s"instatiate for $featureType - $id")
90 
91     val emptyDefaultProps = OwcProperties(
92       UUID.randomUUID(),
93       "en",
94       "empty properties",
95       Some("no property information was found"),
96       Some(now),
97       Some("automated system response"),
98       None,
99       List(),
100       List(),
101       None,
102       None,
103       List(),
104       List(emptyProfile, emptySelf)
105     )
106 
107     val rectOpt = createOptionalBbox(bboxAsWkt)
108     val props = owcPropertiesDAO.findOwcPropertiesForOwcFeatureType(id).getOrElse(emptyDefaultProps)
109 
110     featureType match {
111       case "OwcEntry" => {
112         logger.debug(s"found OwcEntry ${id}")
113         val offerings = owcOfferingDAO.findOwcOfferingsForOwcEntry(id).toList
114         OwcEntry(id, rectOpt, props, offerings)
115       }
116       case "OwcDocument" => {
117         logger.debug(s"found OwcDocument ${id}")
118         val entries = findOwcEntriesForOwcDocument(id).toList
119         OwcDocument(id, rectOpt, props, entries)
120       }
121       case _ => throw new InvalidClassException(featureType, s"Unknown feature type $featureType")
122     }
123   }

344   /**
345     * create/insert into db Owc Document
346     *
347     * @param owcDocument
348     * @param email
349     * @param visibility
350     * @param collectionType
351     * @return
352     */
353   def createOwcDocument(owcDocument: OwcDocument, email: String, visibility: Int, collectionType: String): Option[OwcDocument] = {
354     db.withTransaction {
355       implicit connection => {
356 
357         val rowCount = SQL(
358           s"""
359           insert into $tableOwcFeatureTypes values (
360             {id}, {feature_type}, {bbox}
361           )
362         """).on(
363           'id -> owcDocument.id,
364           'feature_type -> "OwcDocument",
365           'bbox -> owcDocument.bbox.map(rect => rectToWkt(rect))
366         ).executeUpdate()
367 
368         owcDocument.features.foreach {
369           owcEntry => {
370             if (findOwcEntriesById(owcEntry.id).isEmpty) {
371               createOwcEntry(owcEntry)
372             }
373 
374             SQL(
375               s"""insert into $tableOwcDocumentsHasOwcEntries  values (
376                  |{owc_feature_types_as_document_id}, {owc_feature_types_as_entry_id}
377                  |)
378                """.stripMargin).on(
379               'owc_feature_types_as_document_id -> owcDocument.id,
380               'owc_feature_types_as_entry_id -> owcEntry.id
381             ).executeUpdate()
382           }
383         }
384 
385         // the single dependent owc properties object
386         if (owcPropertiesDAO.findOwcPropertiesByUuid(owcDocument.properties.uuid).isEmpty) {
387           owcPropertiesDAO.createOwcProperties(owcDocument.properties)
388         }
389 
390         SQL(
391           s"""insert into $tableOwcFeatureTypesHasOwcProperties  values (
392              |{owc_feature_types_id}, {owc_properties_uuid}
393              |)
394                """.stripMargin).on(
395           'owc_feature_types_id -> owcDocument.id,
396           'owc_properties_uuid -> owcDocument.properties.uuid.toString
397         ).executeUpdate()
398 
399         SQL(
400           s"""insert into $tableUserHasOwcDocuments values (
401              |{email}, {owcDocumentId}, {collection_type}, {visibility}
402              |)""".stripMargin).on(
403           'email -> email,
404           'owcDocumentId -> owcDocument.id,
405           'collection_type -> collectionType,
406           'visibility -> visibility
407         ).executeUpdate()
408 
409         rowCount match {
410           case 1 => Some(owcDocument)
411           case _ => None
412         }
413       }
414     }
415   }

435     * create an owc entry with dependent offerings and properties
436     *
437     * @param owcEntry
438     * @return
439     */
440   def createOwcEntry(owcEntry: OwcEntry): Option[OwcEntry] = {
441 
442     db.withTransaction {
443       implicit connection => {
444 
445         val sql = SQL(
446           s"""
447           insert into $tableOwcFeatureTypes values (
448             {id}, {feature_type}, {bbox}
449           )
450         """).on(
451           'id -> owcEntry.id,
452           'feature_type -> "OwcEntry",
453           'bbox -> owcEntry.bbox.map(rect => rectToWkt(rect))
454         )
455 
456         val rowCount = sql.executeUpdate()
457 
458         logger.debug(s"inserted ${owcEntry.getClass.getSimpleName} $rowCount ${sql.toString}")
459 
460         owcEntry.offerings.foreach {
461           owcOffering => {
462             if (owcOfferingDAO.findOwcOfferingByUuid(owcOffering.uuid).isEmpty) {
463               owcOfferingDAO.createOwcOffering(owcOffering)
464             }
465 
466             SQL(
467               s"""insert into $tableOwcEntriesHasOwcOfferings  values (
468                  |{owc_feature_types_as_entry_id}, {owc_offerings_uuid}
469                  |)
470                """.stripMargin).on(
471               'owc_feature_types_as_entry_id -> owcEntry.id,
472               'owc_offerings_uuid -> owcOffering.uuid.toString
473             ).executeUpdate()
474           }
475         }
476 
477         // the single dependent owc properties object
478         if (owcPropertiesDAO.findOwcPropertiesByUuid(owcEntry.properties.uuid).isEmpty) {
479           owcPropertiesDAO.createOwcProperties(owcEntry.properties)
480         }
481 
482         SQL(
483           s"""insert into $tableOwcFeatureTypesHasOwcProperties  values (
484              |{owc_feature_types_id}, {owc_properties_uuid}
485              |)
486                """.stripMargin).on(
487           'owc_feature_types_id -> owcEntry.id,
488           'owc_properties_uuid -> owcEntry.properties.uuid.toString
489         ).executeUpdate()
490 
491         rowCount match {
492           case 1 => Some(owcEntry)
493           case _ => None
494         }
495       }
496     }
497   }

594   /**
595     *
596     * @param owcDocument
597     * @param email
598     * @return
599     */
600   def replaceEntryInCollection(owcDocument: OwcDocument, owcEntry: OwcEntry, email: String): Option[OwcDocument] = {
601 
602     val updateCount = db.withTransaction {
603       implicit connection => {
604 
605         // update / replace the entry, due to SQL foreign key reference, we can only update if noone else has a ref on it
606         val entryToReplace = findOwcEntriesById(owcEntry.id)
607 
608         // if uuid of provided entry is not found in DB create new entry (upsert)
609         if (entryToReplace.isDefined) {
610 
611           // and now delete our own ref from specified relation to owc doc
612           SQL(
613             s"""delete from $tableOwcDocumentsHasOwcEntries where owc_feature_types_as_document_id = {doc_id}
614                |AND owc_feature_types_as_entry_id = {entry_id}
615              """.stripMargin).on(
616             'doc_id -> owcDocument.id,
617             'entry_id -> owcEntry.id
618           ).executeUpdate()
619 
620         }
621 
622         val isItOrphan = SQL(
623           s"""select owc_feature_types_as_entry_id from
624              |$tableOwcDocumentsHasOwcEntries where owc_feature_types_as_entry_id = {id}""".stripMargin).on(
625           'id -> owcEntry.id
626         ).as(SqlParser.str("owc_feature_types_as_entry_id") *).isEmpty
627 
628         val replacementStep = if (isItOrphan) {
629           if (entryToReplace.isDefined) {
630             if (deleteOwcEntry(owcEntry)) {
631               createOwcEntry(owcEntry).isDefined
632             } else {
633               logger.error(s"owcentry with entry id ${owcEntry.id} scheduled for replacement in owcdoc ${owcDocument.id} " +
634                 "could not be deleted/replaced properly")
635               false
636             }
637           } else {
638             logger.warn(s"owcentry with entry id ${owcEntry.id} scheduled for replacement in owcdoc ${owcDocument.id} " +
639               "did not exist before (doing upsert")
640             createOwcEntry(owcEntry).isDefined
641           }
642         } else {
643           logger.error(s"owcentry with entry id ${owcEntry.id} scheduled for replacement in owcdoc ${owcDocument.id} " +
644             "is referenced in other collections and cannot be touched.")
645           false
646         }
647 
648         // add relation to owc doc
649         val reAddRelations = SQL(
650           s"""insert into $tableOwcDocumentsHasOwcEntries  values (
651              |{owc_feature_types_as_document_id}, {owc_feature_types_as_entry_id}
652              |)
653                """.stripMargin).on(
654           'owc_feature_types_as_document_id -> owcDocument.id,
655           'owc_feature_types_as_entry_id -> owcEntry.id
656         ).executeUpdate()
657 
658         val yesOrNo = if (replacementStep) 1 else 0
659         reAddRelations + yesOrNo
660       }
661     }
662 
663     updateCount match {
664       case 2 => Some(owcDocument) // the chained new document
665       case 1 => findOwcDocumentByIdAndUser(owcDocument.id, email) // the old document
666       case _ => None
667     }
668   }

676   def deleteOwcEntryFromOwcDocument(owcDocument: OwcDocument, owcEntryId: String, email: String): Option[OwcDocument] = {
