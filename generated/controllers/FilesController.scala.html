61   def uploadMultipartForm: Action[MultipartFormData[TemporaryFile]] = (authenticationAction andThen userAction) (parse.multipartFormData) {
62     request =>
63       request.body.file("file").map { theFile =>
64 
65         val filename = theFile.filename
66         val contentType = theFile.contentType
67         val pathOfUploadTmp = Paths.get(portalConfig.uploadDataPath)
68         val intermTempDir = Files.createTempDirectory(pathOfUploadTmp, "sac-upload-")
69         val tmpFile = new File(intermTempDir.resolve(filename).toAbsolutePath.toString)
70         val handle = theFile.ref.moveTo(tmpFile)
71         val kilobytes = handle.length / 1024
72 
73         googleService.uploadFileGoogleBucket(handle) match {
74           case Left(errorResult) =>
75             logger.error(errorResult.message + errorResult.details.mkString)
76             InternalServerError(Json.toJson(errorResult)).as(JSON)
77           case Right(blobInfo) =>
78             // return the public download link
79             val userFileEntryOk = userService.insertUserFileEntry(filename, request.user.email, blobInfo.mediaLink)
80             userFileEntryOk.fold {
81               logger.error("file metadata insert failed.")
82               val error = ErrorResult("File metadata insert failed.", None)
83               BadRequest(Json.toJson(error)).as(JSON)
84             } { userFile =>
85               val owcResource = collectionsService.fileMetadata(userFile, contentType, Try(kilobytes.toInt).toOption)
86               logger.trace(Json.prettyPrint(owcResource.toJson))
87               logger.debug(s"file upload $filename to ${blobInfo.mediaLink} with reference ${userFile.linkreference}; bob details: ${blobInfo.name} ${blobInfo.bucket} ${blobInfo.contentType} ${blobInfo.contentEncoding} ${blobInfo.size} ${blobInfo.md5}")
88               Try(handle.delete()).failed.map(ex => logger.error(ex.getLocalizedMessage))
89               Try(Files.delete(intermTempDir)).failed.map(ex => logger.error(ex.getLocalizedMessage))
90 
91               //FIXME SR do we also want to have a general "return" object? Status is always in the respone so it does not need to be in here
92               val added = collectionsService.addPlainFileResourceToUserDefaultCollection(owcResource, request.user)
93               if (added) {
94                 Ok(Json.obj("status" -> "OK", "message" -> s"file uploaded $filename.", "file" -> userFile.linkreference, "entry" -> owcResource.toJson))
95               } else {
96                 logger.error("Insert into own collection failed.")
97                 val error = ErrorResult("Insert into own collection failed.", None)
98                 BadRequest(Json.toJson(error)).as(JSON)
99               }
100             }
101           case _ =>
102             logger.error("file upload to cloud storage failed")
103             val error = ErrorResult("File upload to cloud storage failed.", None)
104             BadRequest(Json.toJson(error)).as(JSON)
105         }
106       }.getOrElse {
107         logger.error("file upload failed, no file found for upload")
108         val error = ErrorResult("file upload failed, no file found for upload", None)
109         BadRequest(Json.toJson(error)).as(JSON)
110       }
111   }
