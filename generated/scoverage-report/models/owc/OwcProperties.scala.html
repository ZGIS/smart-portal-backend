29 /**
30   * Model of OWS Context Documents and provide GeoJson encoding thereof (and maybe AtomXML)
31   * An OWC document is an extended FeatureCollection, where the features (aka entries) hold a variety of metadata
32   * about the things they provide the context for (i.e. other data sets, services, metadata records)
33   * OWC documents do not duplicate a CSW MD_Metadata record, but a collection of referenced resources;
34   *
35   * http://www.opengeospatial.org/standards/owc
36   *
37   * Classically, the WMC documents (Web Map Context documents) were a list of WMS layers for a web map viewer
38   * with a certain context, i.e. title, Bounding Box and a few visualisation properties like scale/zoom,
39   * OWC has superseded that concept into a generic collection of resources:
40   *
41   * We use OWC primarily in the form of collections of case studies, preferably with at least two offerings per entry:
42   * 1) a web visualisable form, e.g.WMS, WFS, SOS ...
43   * 2) a CSW addressable MD_Metadata record according to the resource
44   *
45   * The OWC JSON Encoding is a profile of GeoJSON FeatureCollection, the XML encoding is a profile of Atom/GeoRSS Feed
46   */

340     * @return
341     */
342   def parseJson(json: JsValue): Option[OwcProperties] = {
343     val resultFromJson: JsResult[OwcProperties] = Json.fromJson[OwcProperties](json)
344     resultFromJson match {
345       case JsSuccess(r: OwcProperties, path: JsPath) => Some(r)
346       case e: JsError => {
347         val lines = e.errors.map { tupleAction =>
348           val jsPath = tupleAction._1
349           val valErrors = tupleAction._2.map( valErr => valErr.message ).toList.mkString(" ; ")
350           jsPath.toJsonString + " >> " +  valErrors
351         }
352 
353         logger.error(s"JsError info  ${lines.mkString(" | ")}")
354         None
355       }
356     }
