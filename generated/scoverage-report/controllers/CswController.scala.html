42 private class AddMDMetadataToInsert(xml: Node) extends RewriteRule {
43   override def transform(n: Node): Seq[Node] = n match {
44     case e: Elem if e.label == "Insert" => {

48   }

82   def getValidValuesFor(topic: String): Action[AnyContent] = Action { request =>
83     import scala.collection.JavaConversions._
84 
85     logger.debug(s"returning valid values for '${topic}'")
86     val configurations = metadataValidValues.get
87     val validValuesConf = configurations.filter(conf => conf.getString("for").getOrElse("").equals(topic))
88     if (validValuesConf.isEmpty()) {
89       //TODO SR should we have a generic error return object that we return as JSON?
90       BadRequest(s"There are not valid values for '${topic}'")
91     }
92     else {
93       val conf = validValuesConf.head
94       val result = ValidValues.parseConfiguration(validValuesConf.head)
95       Ok(Json.toJson(result))
96     }

99   /** calls CSW:Transaction -> Insert
100     * FIXME should implement HasToken(parse.json) instead of Action.async, to only allow authenticated users
101     */
102   def insert: Action[AnyContent] = Action.async { request =>
103     logger.debug(request.body.asJson.toString)
104 
105     //TODO parse JSON to MDMetadataSet and convert that to XML
106 
107     val futureResponse: Future[WSResponse] = for {
108       getCapaResponse <- wsClient.url(CSW_OPERATIONS_METADATA_URL).get()
109       insertResponse <- {
110         //check if getCapaResponse indicates, that the CSW can perform Transactions
111         val operation = (getCapaResponse.xml \\ "OperationsMetadata" \\ "Operation").filter(node => {
112           logger.debug(s"Attribute 'name'=${node.attribute("name").get.text.toString}")
113           node.attribute("name").get.text.equals("Transaction")
114         })
115         logger.debug(s"operation = ${operation.toString()}")
116         if (operation.isEmpty) {
117           throw new UnsupportedOperationException("CSW does not support Transaction.")
118         }
119 
120         //insert MDMEtadata in insert template
121         val rule = new RuleTransformer(new AddMDMetadataToInsert(mdMetadata))
122         val finalXML = rule.transform(cswtInsertXml)
123         logger.error(finalXML.toString())
124         wsClient.url(CSW_URL).post(finalXML.toString())
125       }
126     } yield insertResponse
127 
128     futureResponse.recover {
129       case e: Exception =>
130         logger.error("Insert CSW threw exception", e)
131         InternalServerError(e.getMessage)
132     }
133 
134     futureResponse.map(response => {
135       logger.error(response.xml.toString())
136       response.xml match {
137         case e: Elem if e.label == "TransactionResponse" => {
138           val fileIdentifier = (e \\ "InsertResult" \\ "BriefRecord" \\ "identifier").text
139           Ok(Json.obj("type" -> "success", "fileIdentifier" -> fileIdentifier,
140             "message" -> s"Inserted as ${fileIdentifier}."))
141         }
142         case e: Elem if e.label == "ExceptionReport" => {
143           val message = (e \\ "Exception" \\ "ExceptionText").text
144           //TODO SR make this InternalServerError
145           Ok(Json.obj("type" -> "danger", "message" -> message))
146         }
147         case _ => InternalServerError(s"Unexpected Response: ${response}")
148       }
149     })
150   }
