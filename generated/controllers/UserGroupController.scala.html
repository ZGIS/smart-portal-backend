78   /**
79     * create a new user group. user must already be in the sent object rights list as power user
80     *
81     * @return
82     */
83   def createUsersOwnUserGroup: Action[JsValue] = defaultAuthAction(parse.json) {
84     request =>
85       request.body.validate[UserGroup].fold(
86         errors => {
87           logger.error(JsError.toJson(errors).toString())
88           val error: ErrorResult = ErrorResult("Usergroup format could not be read.",
89             Some(StringEscapeUtils.escapeJson(errors.mkString("; "))))
90           BadRequest(Json.toJson(error)).as(JSON)
91         },
92         userGroup => {
93           // check if creating user is in the member list and has power user right
94           val isPresent = userGroup.hasUsersLevel.exists(ul => ul.users_accountsubject.equals(request.user.accountSubject) && ul.userlevel >= 1)
95           if (!isPresent) {
96             logger.error("User not listed with adequate rights. We would create a group that you can't edit, aborting.")
97             val error = ErrorResult("User not listed with adequate rights. We would create a group that you can't edit, aborting.", None)
98             BadRequest(Json.toJson(error)).as(JSON)
99           } else {
100             userGroupService.createUsersOwnUserGroup(request.user, userGroup).fold {
101               logger.error("Error creating the user group.")
102               val error = ErrorResult("Error creating the user group.", None)
103               BadRequest(Json.toJson(error)).as(JSON)
104             } {
105               ugroup =>
106                 Ok(Json.obj("status" -> "OK", "usergroup" -> Json.toJson(ugroup)))
107             }
108           }
109         })
110   }

115     * @return

117   def updateUsersOwnUserGroup: Action[JsValue] = defaultAuthAction(parse.json) {

123             Some(StringEscapeUtils.escapeJson(errors.mkString("; "))))
124           BadRequest(Json.toJson(error)).as(JSON)
125         },
126         userGroup => {
127           // check if creating user is in the member list and has power user rights from the database happens in underlying service
128           // (so that you cannot smuggle yourself in)
129           // however, as a poweruser you can remove yourself from the group or degrade your rights (subsequently you won't be able to edit anymore)
130           userGroupService.updateUsersOwnUserGroup(request.user, userGroup).fold {
131             logger.error("Error updating the user group.")
132             val error = ErrorResult("Error updating the user group.", None)
133             BadRequest(Json.toJson(error)).as(JSON)
134           } {

136               Ok(Json.obj("status" -> "OK", "usergroup" -> Json.toJson(ugroup)))
