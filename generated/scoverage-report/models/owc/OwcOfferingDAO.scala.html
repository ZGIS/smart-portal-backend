237   /**
238     * instantiates the concrete type of the offering,
239     * I believe there is possibly also something from Ratiopharm, like Class.forName, reflection etc,
240     * I am not even sure right now if we need the type
241     *
242     * @param uuid
243     * @param offeringType
244     * @param code
245     * @param content
246     * @return
247     */
248   def instantiateOffering(uuid: String, offeringType: String, code: String, content: Option[String]): OwcOffering = {
249     val uuidObj = UUID.fromString(uuid)
250     val ops = findOwcOperationsByOfferingsUUID(UUID.fromString(uuid)).toList
251     val contentList = content.map(text => List(text)).getOrElse(List())
252 
253     offeringType match {
254       case "WmsOffering" => WmsOffering(uuidObj, code, ops, contentList)
255       case "WmtsOffering" => WmtsOffering(uuidObj, code, ops, contentList)
256       case "WfsOffering" => WfsOffering(uuidObj, code, ops, contentList)
257       case "WcsOffering" => WcsOffering(uuidObj, code, ops, contentList)
258       case "CswOffering" => CswOffering(uuidObj, code, ops, contentList)
259       case "WpsOffering" => WpsOffering(uuidObj, code, ops, contentList)
260       case "GmlOffering" => GmlOffering(uuidObj, code, ops, contentList)
261       case "KmlOffering" => KmlOffering(uuidObj, code, ops, contentList)
262       case "GeoTiffOffering" => GeoTiffOffering(uuidObj, code, ops, contentList)
263       case "SosOffering" => SosOffering(uuidObj, code, ops, contentList)
264       case "NetCdfOffering" => NetCdfOffering(uuidObj, code, ops, contentList)
265       case "HttpLinkOffering" => HttpLinkOffering(uuidObj, code, ops, contentList)
266       case _ => throw new InvalidClassException(offeringType, s"Unknown Offering type $offeringType")
267     }
268   }

312   def createOwcOffering(owcOffering: OwcOffering): Option[OwcOffering] = {
313     db.withTransaction {
314       implicit connection => {
315 
316         val rowCount = SQL(
317           s"""
318           insert into $tableOwcOfferings values (
319             {uuid}, {offering_type}, {code}, {content}
320           )
321         """).on(
322           'uuid -> owcOffering.uuid.toString,
323           'offering_type -> owcOffering.getClass.getSimpleName,
324           'code -> owcOffering.code,
325           'content -> owcOffering.content.headOption
326         ).executeUpdate()
327 
328         owcOffering.operations.foreach {
329           owcOperation => {
330             if (findOwcOperationByUuid(owcOperation.uuid).isEmpty) {
331               createOwcOperation(owcOperation)
332             }
333 
334             SQL(
335               s"""insert into $tableOwcOfferingsHasOwcOperations  values (
336                  |{owc_offerings_uuid}, {owc_operations_uuid}
337                  |)
338                """.stripMargin).on(
339               'owc_offerings_uuid -> owcOffering.uuid.toString,
340               'owc_operations_uuid -> owcOperation.uuid.toString
341             ).executeUpdate()
342           }
343         }
344 
345         rowCount match {
346           case 1 => Some(owcOffering)
347           case _ => None
348         }
349       }
