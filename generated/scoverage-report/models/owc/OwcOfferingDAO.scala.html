256     * instantiates the concrete type of the offering,
257     * I believe there is possibly also something from Ratiopharm, like Class.forName, reflection etc,

264     * @return

266   def instantiateOwcOffering(uuid: String, offeringType: String, code: String, content: Option[String]): OwcOffering = {
267     val uuidObj = UUID.fromString(uuid)
268     val ops = findOwcOperationsByOfferingUUID(UUID.fromString(uuid)).toList
269     val contentList = content.map(text => List(text)).getOrElse(List())
270 
271     offeringType match {
272       case "WmsOffering" => WmsOffering(uuidObj, code, ops, contentList)
273       case "WmtsOffering" => WmtsOffering(uuidObj, code, ops, contentList)
274       case "WfsOffering" => WfsOffering(uuidObj, code, ops, contentList)
275       case "WcsOffering" => WcsOffering(uuidObj, code, ops, contentList)
276       case "CswOffering" => CswOffering(uuidObj, code, ops, contentList)
277       case "WpsOffering" => WpsOffering(uuidObj, code, ops, contentList)
278       case "GmlOffering" => GmlOffering(uuidObj, code, ops, contentList)
279       case "KmlOffering" => KmlOffering(uuidObj, code, ops, contentList)
280       case "GeoTiffOffering" => GeoTiffOffering(uuidObj, code, ops, contentList)
281       case "SosOffering" => SosOffering(uuidObj, code, ops, contentList)
282       case "NetCdfOffering" => NetCdfOffering(uuidObj, code, ops, contentList)
283       case "HttpLinkOffering" => HttpLinkOffering(uuidObj, code, ops, contentList)
284       case _ => throw new InvalidClassException(offeringType, s"Unknown Offering type $offeringType")
285     }

350   def createOwcOffering(owcOffering: OwcOffering): Option[OwcOffering] = {

356           insert into $tableOwcOfferings values (

362           'code -> owcOffering.code,
363           'content -> owcOffering.content.headOption

366         owcOffering.operations.foreach {
367           owcOperation => {

374                  |{owc_offerings_uuid}, {owc_operations_uuid}
375                  |)
376                """.stripMargin).on(
377               'owc_offerings_uuid -> owcOffering.uuid.toString,
378               'owc_operations_uuid -> owcOperation.uuid.toString

380           }
