93   /**
94     * login with JSON (from Angular / form)
95     * {"username":"akmoch","password":"testpass123"}
96     *
97     * Create the authentication token and sets the cookie [[AuthTokenCookieKey]]
98     * for AngularJS to use also in X-XSRF-TOKEN in HTTP header.
99     *
100     * @return
101     */
102   def login = Action(parse.json) { implicit request =>
103     request.body.validate[LoginCredentials].fold(
104       errors => {
105         logger.error(JsError.toJson(errors).toString())
106         BadRequest(Json.obj("status" -> "ERR", "message" -> JsError.toJson(errors)))
107       },
108       valid = credentials => {
109         // find user in db and compare password stuff
110         userDAO.authenticate(credentials.username, credentials.password).fold {
111           logger.error("User or password wrong.")
112           Unauthorized(Json.obj("status" -> "ERR", "message" -> "Username or password wrong."))
113         } { user =>
114           val uaIdentifier: String = request.headers.get(UserAgentHeader).getOrElse(UserAgentHeaderDefault)
115           // logger.debug(s"Logging in username from $uaIdentifier")
116           val token = passwordHashing.createSessionCookie(user.username, uaIdentifier)
117           cache.set(token, user.username)
118           Ok(Json.obj("status" -> "OK", "token" -> token, "username" -> user.username, "userprofile" -> user.asProfileJs()))
119             .withCookies(Cookie(AuthTokenCookieKey, token, None, httpOnly = false))
120         }
121       })
122   }
123 
124   /**
125     * Log-out a user. Invalidates the authentication token.
126     *
127     * Discard the cookie [[AuthTokenCookieKey]] to have AngularJS no longer set the
128     * X-XSRF-TOKEN in HTTP header.
129     */
130   def logout = HasToken(parse.empty) { token =>
131     username =>
132       implicit request =>
133         cache.remove(token)
134         Ok.discardingCookies(DiscardingCookie(name = AuthTokenCookieKey))
135   }

160   }
