80   /**
81     * instantiate an OwcFeatureType either OwcEntry, or OwcDocument
82     *
83     * @param id
84     * @param featureType
85     * @param bboxAsWkt
86     * @return
87     */
88   def instantiateOwcFeatureType(id: String, featureType: String, bboxAsWkt: Option[String]): OwcFeatureType = {
89 
90     val emptyProfile = OwcLink(UUID.randomUUID(), "profile", None, "http://www.opengis.net/spec/owc-atom/1.0/req/core", None)
91     val emptySelf = OwcLink(UUID.randomUUID(), "self", Some("application/json"), "http://example.com/empty", None)
92     val now = ZonedDateTime.now.withZoneSameInstant(ZoneId.systemDefault())
93     logger.debug(s"instatiate for $featureType - $id")
94 
95     val emptyDefaultProps = OwcProperties(
96       UUID.randomUUID(),
97       "en",
98       "empty properties",
99       Some("no property information was found"),
100       Some(now),
101       Some("automated system response"),
102       None,
103       List(),
104       List(),
105       None,
106       None,
107       List(),
108       List(emptyProfile, emptySelf)
109     )
110 
111     val rectOpt = createOptionalBbox(bboxAsWkt)
112     val props = owcPropertiesDAO.findOwcPropertiesForOwcFeatureType(id).getOrElse(emptyDefaultProps)
113 
114     featureType match {
115       case "OwcEntry" => {
116         logger.debug(s"found OwcEntry ${id}")
117         val offerings = owcOfferingDAO.findOwcOfferingsForOwcEntry(id).toList
118         OwcEntry(id, rectOpt, props, offerings)
119       }
120       case "OwcDocument" => {
121         logger.debug(s"found OwcDocument ${id}")
122         val entries = findOwcEntriesForOwcDocument(id).toList
123         OwcDocument(id, rectOpt, props, entries)
124       }
125       case _ => throw new InvalidClassException(featureType, s"Unknown feature type $featureType")
126     }
127   }

209   /**
210     * create an owc entry with dependent offerings and properties
211     *
212     * @param owcEntry
213     * @return
214     */
215   def createOwcEntry(owcEntry: OwcEntry): Option[OwcEntry] = {
216 
217     db.withTransaction {
218       implicit connection => {
219 
220         val sql = SQL(
221           s"""
222           insert into $tableOwcFeatureTypes values (
223             {id}, {feature_type}, {bbox}
224           )
225         """).on(
226           'id -> owcEntry.id,
227           'feature_type -> "OwcEntry",
228           'bbox -> owcEntry.bbox.map(rect => rectToWkt(rect))
229         )
230 
231         val rowCount = sql.executeUpdate()
232 
233         logger.debug(s"inserted ${owcEntry.getClass.getSimpleName} $rowCount ${sql.toString}")
234 
235         owcEntry.offerings.foreach {
236           owcOffering => {
237             if (owcOfferingDAO.findOwcOfferingByUuid(owcOffering.uuid).isEmpty) {
238               owcOfferingDAO.createOwcOffering(owcOffering)
239             }
240 
241             SQL(
242               s"""insert into $tableOwcEntriesHasOwcOfferings  values (
243                  |{owc_feature_types_as_entry_id}, {owc_offerings_uuid}
244                  |)
245                """.stripMargin).on(
246               'owc_feature_types_as_entry_id -> owcEntry.id,
247               'owc_offerings_uuid -> owcOffering.uuid.toString
248             ).executeUpdate()
249           }
250         }
251 
252         // the single dependent owc properties object
253         if (owcPropertiesDAO.findOwcPropertiesByUuid(owcEntry.properties.uuid).isEmpty) {
254           owcPropertiesDAO.createOwcProperties(owcEntry.properties)
255         }
256 
257         SQL(
258           s"""insert into $tableOwcFeatureTypesHasOwcProperties  values (
259              |{owc_feature_types_id}, {owc_properties_uuid}
260              |)
261                """.stripMargin).on(
262           'owc_feature_types_id -> owcEntry.id,
263           'owc_properties_uuid -> owcEntry.properties.uuid.toString
264         ).executeUpdate()
265 
266         rowCount match {
267           case 1 => Some(owcEntry)
268           case _ => None
269         }
270       }
271     }
272   }
