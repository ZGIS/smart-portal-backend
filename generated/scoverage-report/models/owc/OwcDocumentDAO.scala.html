54   val owcEntryParser: RowParser[OwcEntry] = {
55     str("id") ~

57       get[Option[String]]("bbox") map {
58       case id ~ featureType ~ bboxAsWkt => {
59         logger.debug(s"owcEntryParser $id, $featureType, $bboxAsWkt")
60         instantiateOwcFeatureType(id, featureType, bboxAsWkt).asInstanceOf[OwcEntry]
61       }
62     }

76   /**
77     * instantiate an OwcFeatureType either OwcEntry, or OwcDocument
78     *
79     * @param id
80     * @param featureType
81     * @param bboxAsWkt
82     * @return
83     */
84   def instantiateOwcFeatureType(id: String, featureType: String, bboxAsWkt: Option[String]): OwcFeatureType = {
85 
86     val emptyProfile = OwcLink(UUID.randomUUID(), "profile", None, "http://www.opengis.net/spec/owc-atom/1.0/req/core", None)
87     val emptySelf = OwcLink(UUID.randomUUID(), "self", Some("application/json"), "http://example.com/empty", None)
88     val now = ZonedDateTime.now.withZoneSameInstant(ZoneId.systemDefault())
89     logger.debug(s"instatiate for $featureType - $id")
90 
91     val emptyDefaultProps = OwcProperties(
92       UUID.randomUUID(),
93       "en",
94       "empty properties",
95       Some("no property information was found"),
96       Some(now),
97       Some("automated system response"),
98       None,
99       List(),
100       List(),
101       None,
102       None,
103       List(),
104       List(emptyProfile, emptySelf)
105     )
106 
107     val rectOpt = createOptionalBbox(bboxAsWkt)
108     val props = owcPropertiesDAO.findOwcPropertiesForOwcFeatureType(id).getOrElse(emptyDefaultProps)
109 
110     featureType match {
111       case "OwcEntry" => {
112         logger.debug(s"found OwcEntry ${id}")
113         val offerings = owcOfferingDAO.findOwcOfferingsForOwcEntry(id).toList
114         OwcEntry(id, rectOpt, props, offerings)
115       }
116       case "OwcDocument" => {
117         logger.debug(s"found OwcDocument ${id}")
118         val entries = findOwcEntriesForOwcDocument(id).toList
119         OwcDocument(id, rectOpt, props, entries)
120       }
121       case _ => throw new InvalidClassException(featureType, s"Unknown feature type $featureType")
122     }
123   }

396   /**
397     * create/insert into db Owc Document
398     *
399     * @param owcDocument
400     * @param email
401     * @param visibility
402     * @param collectionType
403     * @return
404     */
405   def createOwcDocument(owcDocument: OwcDocument, email: String, visibility: Int, collectionType: String): Option[OwcDocument] = {
406     db.withTransaction {
407       implicit connection => {
408 
409         val rowCount = SQL(
410           s"""
411           insert into $tableOwcFeatureTypes values (
412             {id}, {feature_type}, {bbox}
413           )
414         """).on(
415           'id -> owcDocument.id,
416           'feature_type -> "OwcDocument",
417           'bbox -> owcDocument.bbox.map(rect => rectToWkt(rect))
418         ).executeUpdate()
419 
420         owcDocument.features.foreach {
421           owcEntry => {
422             if (findOwcEntriesById(owcEntry.id).isEmpty) {
423               createOwcEntry(owcEntry)
424             }
425 
426             SQL(
427               s"""insert into $tableOwcDocumentsHasOwcEntries  values (
428                  |{owc_feature_types_as_document_id}, {owc_feature_types_as_entry_id}
429                  |)
430                """.stripMargin).on(
431               'owc_feature_types_as_document_id -> owcDocument.id,
432               'owc_feature_types_as_entry_id -> owcEntry.id
433             ).executeUpdate()
434           }
435         }
436 
437         // the single dependent owc properties object
438         if (owcPropertiesDAO.findOwcPropertiesByUuid(owcDocument.properties.uuid).isEmpty) {
439           owcPropertiesDAO.createOwcProperties(owcDocument.properties)
440         }
441 
442         SQL(
443           s"""insert into $tableOwcFeatureTypesHasOwcProperties  values (
444              |{owc_feature_types_id}, {owc_properties_uuid}
445              |)
446                """.stripMargin).on(
447           'owc_feature_types_id -> owcDocument.id,
448           'owc_properties_uuid -> owcDocument.properties.uuid.toString
449         ).executeUpdate()
450 
451         SQL(
452           s"""insert into $tableUserHasOwcDocuments values (
453              |{email}, {owcDocumentId}, {collection_type}, {visibility}
454              |)""".stripMargin).on(
455           'email -> email,
456           'owcDocumentId -> owcDocument.id,
457           'collection_type -> collectionType,
458           'visibility -> visibility
459         ).executeUpdate()
460 
461         rowCount match {
462           case 1 => Some(owcDocument)
463           case _ => None
464         }
465       }
466     }
467   }

486   /**
487     * create an owc entry with dependent offerings and properties
488     *
489     * @param owcEntry
490     * @return
491     */
492   def createOwcEntry(owcEntry: OwcEntry): Option[OwcEntry] = {
493 
494     db.withTransaction {
495       implicit connection => {
496 
497         val sql = SQL(
498           s"""
499           insert into $tableOwcFeatureTypes values (
500             {id}, {feature_type}, {bbox}
501           )
502         """).on(
503           'id -> owcEntry.id,
504           'feature_type -> "OwcEntry",
505           'bbox -> owcEntry.bbox.map(rect => rectToWkt(rect))
506         )
507 
508         val rowCount = sql.executeUpdate()
509 
510         logger.debug(s"inserted ${owcEntry.getClass.getSimpleName} $rowCount ${sql.toString}")
511 
512         owcEntry.offerings.foreach {
513           owcOffering => {
514             if (owcOfferingDAO.findOwcOfferingByUuid(owcOffering.uuid).isEmpty) {
515               owcOfferingDAO.createOwcOffering(owcOffering)
516             }
517 
518             SQL(
519               s"""insert into $tableOwcEntriesHasOwcOfferings  values (
520                  |{owc_feature_types_as_entry_id}, {owc_offerings_uuid}
521                  |)
522                """.stripMargin).on(
523               'owc_feature_types_as_entry_id -> owcEntry.id,
524               'owc_offerings_uuid -> owcOffering.uuid.toString
525             ).executeUpdate()
526           }
527         }
528 
529         // the single dependent owc properties object
530         if (owcPropertiesDAO.findOwcPropertiesByUuid(owcEntry.properties.uuid).isEmpty) {
531           owcPropertiesDAO.createOwcProperties(owcEntry.properties)
532         }
533 
534         SQL(
535           s"""insert into $tableOwcFeatureTypesHasOwcProperties  values (
536              |{owc_feature_types_id}, {owc_properties_uuid}
537              |)
538                """.stripMargin).on(
539           'owc_feature_types_id -> owcEntry.id,
540           'owc_properties_uuid -> owcEntry.properties.uuid.toString
541         ).executeUpdate()
542 
543         rowCount match {
544           case 1 => Some(owcEntry)
545           case _ => None
546         }
547       }
548     }
549   }

652   def replaceEntryInCollection(owcDocument: OwcDocument, owcEntry: OwcEntry, email: String): Option[OwcDocument] = {
653 
654     val updateCount = db.withTransaction {
655       implicit connection => {
656 
657         // update / replace the entry, due to SQL foreign key reference, we can only update if noone else has a ref on it
658         val entryToReplace = findOwcEntriesById(owcEntry.id)
659 
660         // if uuid of provided entry is not found in DB create new entry (upsert)
661         if (entryToReplace.isDefined) {
662 
663           // and now delete our own ref from specified relation to owc doc
664           SQL(
665             s"""delete from $tableOwcDocumentsHasOwcEntries where owc_feature_types_as_document_id = {doc_id}
666                |AND owc_feature_types_as_entry_id = {entry_id}
