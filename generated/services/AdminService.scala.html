108           val viaToken = UserSession.findUserSessionByToken(to, count).toSeq
109           val viaEmail = UserSession.findUserSessionByEmail(em, count)
110           (viaToken ++ viaEmail).filter(ul => ul.token.contains(to) && ul.email.contains(em))
111         case (Some(to), None) =>
112           UserSession.findUserSessionByToken(to, count).toSeq
113         case (None, Some(em)) =>
114           UserSession.findUserSessionByEmail(em, count)
115         case (None, None) =>
116           UserSession.getAllUserSessions(count)
117       }
118     }

121   /**
122     * block and Unblock Users by setting status token
123     * @param command
124     * @param email
125     * @return
126     */
127   def blockUnblockUsers(command: String, email: String): Option[User] = {
128     val userOpt = dbSession.viaConnection( implicit connection => {
129       UserDAO.findUserByEmailAsString(email)
130     })
131     userOpt.fold[Option[User]]{
132       None
133     } {
134       user =>
135         val statusToken = if (command.equalsIgnoreCase("BLOCK")) StatusToken.BLOCKED else StatusToken.ACTIVE
136         val updateUser = User(
137           user.email,
138           user.accountSubject,
139           user.firstname,
140           user.lastname,
141           "***",
142           s"${statusToken.value}:BY-ADMIN",
143           ZonedDateTime.now.withZoneSameInstant(ZoneId.of(appTimeZone)))
144 
145         // remove any latent active sessions
146         val sessions = dbSession.viaConnection { implicit connection =>
147           UserSession.deleteUserSessionByEmail(user.email.value)
148         }
149         if (!sessions) {
150           logger.warn(s"There was a not completely successful sessions delete for ${user.email.value} when setting status to ${statusToken.value}?")
151         }
152 
153         // result of Option can be evaluated upstream in controller
154         dbSession.viaTransaction { implicit connection =>
155           UserDAO.updateNoPass(updateUser)
156         }
157 
158     }
159   }
