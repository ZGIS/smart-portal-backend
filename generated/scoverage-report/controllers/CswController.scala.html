108     token =>
109       authUser =>
110         implicit request => {
111           logger.debug(request.toString)
112 
113           //TODO parse JSON to MDMetadataSet and convert that to XML
114           val mdMetadata = MdMetadata.fromJson(((request.body.as[JsObject]) \ "metadata").get)
115 
116           //FIXME SR I find that pretty hard to read. Is there a better way of chaining WS calls?
117           // see https://www.playframework.com/documentation/2.5.x/ScalaWS#chaining-ws-calls
118           val futureResponse: Future[WSResponse] = for {
119             getCapaResponse <- wsClient.url(CSW_OPERATIONS_METADATA_URL).get()
120             insertResponse <- {
121               //check if getCapaResponse indicates, that the CSW can perform Transactions
122               val operation = (getCapaResponse.xml \\ "OperationsMetadata" \\ "Operation").filter(node => {
123                 logger.debug(s"Attribute 'name'=${node.attribute("name").get.text.toString}")
124                 node.attribute("name").get.text.equals("Transaction")
125               })
126               logger.debug(s"operation = ${operation.toString()}")
127               if (operation.isEmpty) {
128                 throw new UnsupportedOperationException("CSW does not support Transaction.")
129               }
130 
131               //insert MDMEtadata in insert template
132               logger.debug(s"MD_MetadataXML: ${mdMetadata.get.toXml().toString()}")
133               val rule = new RuleTransformer(new AddMDMetadataToInsert(mdMetadata.get.toXml()))
134               val finalXML = rule.transform(cswtInsertXml)
135               logger.debug(s"finalXml: ${finalXML.toString()}")
136               wsClient.url(CSW_URL).post(finalXML.toString())
137             }
138             updateIngesterIndexResponse <- {
139               wsClient.url(INGESTER_UPDATE_INDEX_URL).get()
140               /*
141               .onSuccess {
142                 case response if response.status == 200 => {
143                   logger.info(

145                   logger.info(s"response: ${response.body}");
146                 }
147                 case response => {
148                   logger.warn(
149                     s"Call to '$INGESTER_UPDATE_INDEX_URL' returned error: ${response.status} - ${response.statusText}");
150                   logger.warn(s"response: ${response.body}");
151                 }
152               }
153               */
154             }
155           } yield insertResponse
156 
157           futureResponse.recover {
158             case e: Exception =>
159               logger.error("Insert CSW threw exception", e)
160               InternalServerError(e.getMessage)
161           }
162 
163           futureResponse.map(response => {
164             logger.debug(response.xml.toString())
165             response.xml match {
166               case e: Elem if e.label == "TransactionResponse" => {
167                 val fileIdentifier = (e \\ "InsertResult" \\ "BriefRecord" \\ "identifier").text
168                 logger.debug(s"Adding ${mdMetadata.get.fileIdentifier} to default collection of $authUser.")
169                 val added = collectionsService.addMdEntryToUserDefaultCollection(mdMetadata.get, authUser)
170                 if (!added) {
171                   logger.warn(s"Could not add ${mdMetadata.get.fileIdentifier} to default collection of $authUser.")
172                 }
173                 Ok(Json.obj("type" -> "success", "fileIdentifier" -> fileIdentifier,
174                   "message" -> s"Inserted as ${fileIdentifier}."))
175               }
176               case e: Elem if e.label == "ExceptionReport" => {
177                 val message = (e \\ "Exception" \\ "ExceptionText").text
178                 //TODO SR make this InternalServerError
179                 Ok(Json.obj("type" -> "danger", "message" -> message))
180               }
181               case _ => InternalServerError(s"Unexpected Response: ${response}")
182             }
183           })
184         }
185   }
