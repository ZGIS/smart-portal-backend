42 /**
43   * the action checks if the incoming requets is for a valid server-side session (our original HasToken),
44   * right for an incoming UserRequest, for direct securing of a controller function (compose per function as required)
45   *
46   * @param userService
47   */
48 @Singleton
49 class AuthenticationAction @Inject()(val userService: UserService) extends ActionBuilder[AuthenticatedRequest]
50   with ActionRefiner[Request, AuthenticatedRequest] with ClassnameLogger {
51 
52   def refine[A](request: Request[A]): Future[Either[Result, AuthenticatedRequest[A]]] = {
53 
54     val result = request.cookies.get(AuthTokenCookieKey).fold[Either[Result, AuthenticatedRequest[A]]] {
55       Left(Results.Unauthorized(Json.obj("status" -> "ERR", "message" -> "Invalid XSRF Token cookie")))
56     } { xsrfTokenCookie =>
57       logger.trace(s"cookie ${xsrfTokenCookie.value}")
58       request.headers.get(AuthTokenHeader).orElse(request.getQueryString(AuthTokenUrlKey)).map {
59         headerToken =>
60           // ua needed to differentiate between different devices/sessions
61           val uaIdentifier: String = request.headers.get(UserAgentHeader).getOrElse(UserAgentHeaderDefault)
62           logger.trace(s"headerToken: $headerToken")
63           logger.trace(s"ua: $uaIdentifier")
64           userService.getUserSessionByToken(headerToken, xsrfTokenCookie.value, uaIdentifier).fold[Either[Result, AuthenticatedRequest[A]]] {
65             Left(Results.Unauthorized(Json.obj("status" -> "ERR", "message" -> "No server-side session"))
66               .discardingCookies(DiscardingCookie(name = AuthTokenCookieKey)))
67           } {
68             userSession =>
69               Right(new AuthenticatedRequest[A](userSession, request))
70           }
71 
72       }.getOrElse(Left(Results.Unauthorized(Json.obj("status" -> "ERR", "message" -> "No token header found"))
73         .discardingCookies(DiscardingCookie(name = AuthTokenCookieKey))))
74     }
75     Future.successful(result)
76   }
77 }

171       }
172   }
