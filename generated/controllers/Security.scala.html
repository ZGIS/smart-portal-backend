52   /**
53     * Checks that the token is:
54     * - present in the cookie header of the request,
55     * - either in the header or in the query string,
56     * - matches a token already stored in the play cache
57     *
58     * @param p
59     * @param f
60     * @tparam A
61     * @return
62     */
63   def HasToken[A](p: BodyParser[A] = parse.anyContent)(
64     f: String => String => Request[A] => Result): Action[A] =
65     Action(p) { implicit request =>
66 
67       request.cookies.get(AuthTokenCookieKey).fold {
68         Unauthorized(Json.obj("status" -> "ERR", "message" -> "Invalid XSRF Token cookie"))
69       } { xsrfTokenCookie =>
70         logger.trace(s"cookie ${xsrfTokenCookie.value}")
71         val maybeToken = request.headers.get(AuthTokenHeader).orElse(request.getQueryString(AuthTokenUrlKey))
72 
73         maybeToken flatMap { headerToken =>
74           // ua needed to differentiate between different devices/sessions
75           val uaIdentifier: String = request.headers.get(UserAgentHeader).getOrElse(UserAgentHeaderDefault)
76           logger.trace(s"headerToken: $headerToken")
77           logger.trace(s"ua: $uaIdentifier")
78           // cache token -> maps to a String email
79           // val cacheOpt: Option[String] = cache.get[String](headerToken)
80           val cacheOpt: Option[String] = userService.checkUserSessionCacheByToken(
81             headerToken, xsrfTokenCookie.value, uaIdentifier)
82           val result = cacheOpt.fold {
83             Unauthorized(Json.obj("status" -> "ERR", "message" -> "No server-side session"))
84               .discardingCookies(DiscardingCookie(name = AuthTokenCookieKey))
85           } {
86             email =>
87               val cookieForUSerAndDevice = passwordHashing.testSessionCookie(headerToken, email, uaIdentifier)
88               logger.trace(s"testcookie: $cookieForUSerAndDevice")
89               if (xsrfTokenCookie.value == headerToken && cookieForUSerAndDevice) {
90                 logger.trace(s"request for active session: $email / $headerToken / $uaIdentifier")
91                 f(headerToken)(email)(request)
92               }
93               else {
94                 Unauthorized(Json.obj("status" -> "ERR", "message" -> "Invalid Token"))
95                   .discardingCookies(DiscardingCookie(name = AuthTokenCookieKey))
96               }
97           }
98           Some(result)
99 
100         } getOrElse Unauthorized(Json.obj("status" -> "ERR", "message" -> "No token header found"))
101           .discardingCookies(DiscardingCookie(name = AuthTokenCookieKey))
102       }
103     }
