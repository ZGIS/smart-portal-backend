107   def insert: Action[JsValue] = HasTokenAsync(parse.json) {
108     token =>
109       authUser =>
110         implicit request => {
111           logger.debug(request.toString)
112 
113           //TODO parse JSON to MDMetadataSet and convert that to XML
114           val mdMetadata = MdMetadata.fromJson(((request.body.as[JsObject]) \ "metadata").get)
115 
116           //FIXME SR I find that pretty hard to read. Is there a better way of chaining WS calls?
117           // see https://www.playframework.com/documentation/2.5.x/ScalaWS#chaining-ws-calls
118           val futureResponse: Future[(WSResponse, WSResponse)] = for {
119             getCapaResponse <- wsClient.url(CSW_OPERATIONS_METADATA_URL).get()
120             insertResponse <- {
121               //check if getCapaResponse indicates, that the CSW can perform Transactions
122               val operation = (getCapaResponse.xml \\ "OperationsMetadata" \\ "Operation").filter(node => {
123                 logger.debug(s"Attribute 'name'=${node.attribute("name").get.text.toString}")
124                 node.attribute("name").get.text.equals("Transaction")
125               })
126               logger.debug(s"operation = ${operation.toString()}")
127               if (operation.isEmpty) {
128                 throw new UnsupportedOperationException("CSW does not support Transaction.")
129               }
130 
131               //insert MDMEtadata in insert template
132               logger.debug(s"MD_MetadataXML: ${mdMetadata.get.toXml().toString()}")
133               val rule = new RuleTransformer(new AddMDMetadataToInsert(mdMetadata.get.toXml()))
134               val finalXML = rule.transform(cswtInsertXml)
135               logger.debug(s"finalXml: ${finalXML.toString()}")
136               wsClient.url(CSW_URL).post(finalXML.toString())
137             }
138             updateIngesterIndexResponse <- {
139               val response = wsClient.url(INGESTER_UPDATE_INDEX_URL).get()
140               response.onSuccess {
141                 case response if response.status == 200 => {
142                   logger.info(
143                     s"Successfully called $INGESTER_UPDATE_INDEX_URL (${response.status} - ${response.statusText})");
144                   logger.info(s"response: ${response.body}");
145                 }
146                 case response => {
147                   logger.warn(
148                     s"Call to '$INGESTER_UPDATE_INDEX_URL' returned error: ${response.status} - ${response.statusText}");
149                   logger.warn(s"response: ${response.body}");
150                 }
151               }
152               response
153             }
154           } yield (insertResponse, updateIngesterIndexResponse)
155 
156           futureResponse.recover {
157             case e: Exception =>
158               logger.error("Insert CSW threw exception", e)
159               val error = ErrorResult(s"Exception (${e.getClass.getCanonicalName}) on CSW insert: ${e.getMessage}",
160                 None)
161               InternalServerError(Json.toJson(error)).as(JSON)
162           }
163 
164           futureResponse.map(response => {
165             logger.debug(response._1.xml.toString())
166             response._1.xml match {
167               case e: Elem if e.label == "TransactionResponse" => {
168                 val fileIdentifier = (e \\ "InsertResult" \\ "BriefRecord" \\ "identifier").text
169                 logger.debug(s"Adding ${mdMetadata.get.fileIdentifier} to default collection of $authUser.")
170                 val added = collectionsService.addMdResourceToUserDefaultCollection(CSW_URL, mdMetadata.get, authUser)
171                 if (!added) {
172                   logger.warn(s"Could not add ${mdMetadata.get.fileIdentifier} to default collection of $authUser.")
173                 }
174                 Ok(Json.obj("type" -> "success", "fileIdentifier" -> fileIdentifier,
175                   "message" -> s"Inserted as ${fileIdentifier}."))
176               }
177               case e: Elem if e.label == "ExceptionReport" => {
178                 val message = (e \\ "Exception" \\ "ExceptionText").text
179                 //TODO SR make this InternalServerError
180                 val error = ErrorResult(message, None)
181                 InternalServerError(Json.toJson(error)).as(JSON)
182               }
183               case _ => {
184                 val error = ErrorResult(s"Unexpected Response from CSW: ${response._1.status} - ${response._1.statusText}",
185                   Some(response._1.body))
186                 InternalServerError(Json.toJson(error)).as(JSON)
187               }
188             }
189           })
190         }
191   }
