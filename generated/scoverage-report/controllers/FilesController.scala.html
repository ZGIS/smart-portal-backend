142             // Google Java upload stuff
143             import scala.collection.JavaConverters._
144

146 //            val authenticatedStorageOptions = StorageOptions.newBuilder().setProjectId(googleProjectId)

151             // val noCredentials = NoCredentials.getInstance()
152

154             val roAcl = Acl.of(Acl.User.ofAllUsers(), Acl.Role.READER)
155             val acls: java.util.List[Acl] = List(roAcl).asJava
156 
157             // the inputstream is closed by default, so we don't need to close it here
158             val blob: Blob = storage.create(BlobInfo.newBuilder(googleStorageBucket, filename).setAcl(acls).build(),
159               Files.asByteSource(tmpFile).openStream())
160 
161             // return the public download link
162             val owcEntry = fileMetadata(blob.getMediaLink(), contentType, authUser)
163             val insertOk = collectionsService.addPlainFileEntryToUserDefaultCollection(owcEntry, authUser)
164 
165             if (insertOk) {
166               logger.debug(s"file upload $filename to ${blob.getMediaLink}")
167               Ok(Json.obj("status" -> "OK", "message" -> s"file uploaded $filename.", "file" -> blob.getMediaLink(), "entry" -> owcEntry.toJson))
168             } else {
169               logger.error("file metadata insert failed.")
170               BadRequest(Json.obj("status" -> "ERR", "message" -> "file metadata insert failed."))
171             }
172           }.getOrElse {
173             logger.error("file upload failed")
174             BadRequest(Json.obj("status" -> "ERR", "message" -> "file upload failed."))
175           }
176   }
