29 /**
30   * This Class saves valid values for Metadata Editor Comboboxes
31   *
32   * @param standardValue index of standard value
33   * @param values        list of values
34   * @param descriptions  optional list of descriptions
35   */
36 
37 // TODO SR For more complicated things (like valid BBOXes or so) value needs to become an
38 // object See https://github.com/angular/angular/issues/4843#issuecomment-206583992
39 class ValidValues(
40                    val standardValue: Int,
41                    val values: List[String],
42                    val descriptions: Option[List[String]]
43                  ) {
44   require(values.nonEmpty, "values list must not be empty")
45   require(descriptions.isEmpty || descriptions.get.length == values.length,
46     "decriptions list must either be None or same length as values list")
47   require(standardValue >= 0 && standardValue < values.length, "standardValue must be within values list length")
48   require(values.distinct.size == values.size, "all values must be unique")
49   require(descriptions.isEmpty || (descriptions.get.distinct.size == descriptions.get.size),
50     "all descriptions must be unique")
51 
52 }
