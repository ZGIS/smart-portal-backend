106               jsuser.username,

111               ZonedDateTime.now.withZoneSameInstant(ZoneId.of(appTimeZone)))

114               logger.error("User create error.")

121             // found user with that email already

319           BadRequest(Json.obj("status" -> "ERR", "message" -> "User not found."))

322           val updateUser = User(dbuser.email,
323             dbuser.username,
324             dbuser.firstname,
325             dbuser.lastname,
326             cryptPass,
327             "ACTIVE:PASSWORDUPDATE",
328             ZonedDateTime.now.withZoneSameInstant(ZoneId.of(appTimeZone)))
329 
330           userDAO.updatePassword(updateUser).fold {
331             logger.error("Password update error.")
332             BadRequest(Json.obj("status" -> "ERR", "message" -> "Password update error."))
333           } { user =>
334             val uaIdentifier: String = request.headers.get(UserAgentHeader).getOrElse(UserAgentHeaderDefault)
335             // logger.info(s"Logging in username from $uaIdentifier")
336             cache.remove(token)
337             val newtoken = passwordHashing.createSessionCookie(user.username, uaIdentifier)
338             cache.set(newtoken, user.username)
339             Ok(Json.obj("status" -> "OK", "token" -> newtoken, "username" -> user.username))
340               .withCookies(Cookie(AuthTokenCookieKey, newtoken, None, httpOnly = false))
341           }
