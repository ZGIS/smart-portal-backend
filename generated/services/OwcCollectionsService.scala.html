219   def generateMdResource(catalogUrl: String, mdMetadata: MdMetadata, userMetaEntry: UserMetaRecord): OwcResource = {
220 
221     lazy val bboxFormat = new BboxArrayFormat
222 
223     val updatedTime = Try(OffsetDateTime.of(mdMetadata.citation.ciDate, LocalTime.of(12, 0), ZoneOffset.UTC))
224       .getOrElse(OffsetDateTime.now(ZoneId.systemDefault()))
225     // val baseLink = new URL(s"http://portal.smart-project.info/context/resource/${URLEncoder.encode(mdMetadata.fileIdentifier, "UTF-8")}_copy_${UUID.randomUUID().toString}")
226     val baseLink = new URL(s"https://portal.smart-project.info/context/resource/${userMetaEntry.uuid.toString}")
227 
228     val cswGetCapaOps = OwcOperation(
229       code = "GetCapabilities",
230       method = "GET",
231       mimeType = Some("application/xml"),
232       requestUrl = new URL("https://portal.smart-project.info/pycsw/csw?SERVICE=CSW&VERSION=2.0.2&REQUEST=GetCapabilities"))
233 
234     val cswGetRecordOps = OwcOperation(
235       code = "GetRecordById",
236       method = "GET",
237       mimeType = Some("application/xml"),
238       requestUrl = new URL(s"https://portal.smart-project.info/pycsw/csw?request=GetRecordById&service=CSW&version=2.0.2&elementSetName=full&outputSchema=http://www.isotc211.org/2005/gmd&id=${mdMetadata.fileIdentifier}"))
239 
240     val cswOffering = OwcOffering(
241       code = OwcOfferingType.CSW.code,
242       operations = List(cswGetCapaOps, cswGetRecordOps)
243     )
244 
245     val viaLink = OwcLink(
246       href = baseLink,
247       mimeType = Some("application/json"),
248       rel = "via")
249 
250     val cswLink = OwcLink(
251       href = cswGetRecordOps.requestUrl,
252       mimeType = cswGetRecordOps.mimeType,
253       rel = "via")
254 
255     val owcResource = OwcResource(
256       id = baseLink,
257       geospatialExtent = bboxFormat.parseBboxArray(mdMetadata.extent.mapExtentCoordinates).toOption,
258       title = mdMetadata.title,
259       subtitle = mdMetadata.abstrakt.toOption(),
260       updateDate = updatedTime,
261       author = List(OwcAuthor(name = mdMetadata.responsibleParty.individualName.toOption(),
262         email = parseEmailStringtoEmailAddress(mdMetadata.responsibleParty.email),
263         uri = Some(new URL(mdMetadata.responsibleParty.orgWebLinkage)))),
264       publisher = mdMetadata.responsibleParty.pointOfContact.toOption(),
265       rights = mdMetadata.distribution.useLimitation.toOption(),
266       temporalExtent = GeoDateParserUtils.parseOffsetDateString(Some(mdMetadata.extent.temporalExtent)),
267 
268       // links.alternates[] and rel=alternate
269       contentDescription = List(),
270 
271       // aka links.previews[] and rel=icon (atom)
272       preview = List(),
273 
274       // aka links.data[] and rel=enclosure (atom)
275       contentByRef = List(),
276 
277       // aka links.via[] & rel=via
278       resourceMetadata = List(viaLink, cswLink),
279       offering = List(cswOffering),
280       minScaleDenominator = Try(mdMetadata.scale.toDouble).toOption,
281       maxScaleDenominator = None,
282       active = None,
283       keyword = mdMetadata.keywords.map(w => OwcCategory(term = w, scheme = None, label = None)),
284       folder = None)
285 
286     logger.trace(Json.prettyPrint(owcResource.toJson))
287 
288     owcResource
289   }

496   /**
497     * returns an Option[EmailAddress] object if parsing is successful, for those mdMetadata email joint arrays
498     *
499     * @param emailString the string that might comprise of or contain an email address
500     * @return Option[EmailAddress]
501     */
502   def parseEmailStringtoEmailAddress(emailString: String): Option[EmailAddress] = {
503 
504     if (EmailAddress.isValid(emailString)) {
505       Some(EmailAddress(emailString))
506     } else {
507       if (emailString.contains(",")) {
508         val commaFree = emailString.split(",").head.trim
509 
510         Try {
511           EmailAddress(commaFree)
512         } match {
513           case Success(e) => Some(e)
514           case _ => None
515         }
516       } else {
517         None
518       }
519     }
520   }
