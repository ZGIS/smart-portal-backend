51   /**
52     * reading cell values from a sheet is not as straightforward apparently
53     *
54     * @param row
55     * @param cell
56     * @param sheet
57     * @return
58     */
59   def getCellValueAsStringOption(row: Int, cell: Int, sheet: Sheet): Option[String] = {
60     val theRow = sheet.getRow(row)
61     if (theRow != null) {
62 
63       val lastAccessibleCell: Int = if (theRow.getLastCellNum-1 <= theRow.getPhysicalNumberOfCells) {
64         theRow.getLastCellNum-1
65       } else {
66         theRow.getPhysicalNumberOfCells
67       }
68       logger.trace(s"getCellValueAsStringOption.lastCell: ${lastAccessibleCell+1} : lastAccessibleCell ${lastAccessibleCell}")
69       if (lastAccessibleCell >= cell) {
70         logger.trace(s"getCellValueAsStringOption => row: $row, cell: $cell, sheet: ${sheet.getSheetName}," +
71           s"getCellType: ${sheet.getRow(row).getCell(cell).getCellType}")
72         sheet.getRow(row).getCell(cell).getCellType match {
73           case Cell.CELL_TYPE_NUMERIC => val value = Try {
74             sheet.getRow(row).getCell(cell).getNumericCellValue.toString
75           }.toOption
76             logger.trace(s"CELL_TYPE_NUMERIC: $value")
77             value
78           case Cell.CELL_TYPE_STRING => val value = Try {
79             val stringValue = sheet.getRow(row).getCell(cell).getRichStringCellValue.getString
80             if (stringValue == null) "" else stringValue
81           }.toOption
82             logger.trace(s"CELL_TYPE_STRING: $value")
83             value
84           case Cell.CELL_TYPE_BLANK => logger.debug(s"getCellValueAsStringOption cell type => row: $row, cell: $cell, sheet: ${sheet.getSheetName}," +
85             "getCellType: CELL_TYPE_BLANK")
86             None
87           case _ => logger.error(s"getCellValueAsStringOption cell type => row: $row, cell: $cell, sheet: ${sheet.getSheetName}," +
88             "unexpected error get cell type")
89             None
90         }
91       } else {
92         logger.warn(s"beyond cell range of row => row: $row, cell: $cell, sheet: ${sheet.getSheetName}," +
93           s"getLastCellNum: ${sheet.getRow(row).getLastCellNum}")
94         None
95       }
96     } else {
97       logger.debug("reaching into the void")
98       None
99     }
100   }

102   /**
103     * finds synonym ffor a list of main keywords of a category, uses lookup sheet
104     *
105     * @param keywords_list
106     * @param synonymSheet
107     * @return
108     */
109   def findSynonymsForKeywords(keywords_list: List[String], synonymSheet: org.apache.poi.ss.usermodel.Sheet): List[String] = {
110     logger.trace(s"findSynonymsForKeywords => keywords_list: ${keywords_list.mkString("::")}, synonymSheet: ${synonymSheet.getSheetName}")
111     val buf = scala.collection.mutable.ArrayBuffer.empty[String]
112     keywords_list.foreach { keyword =>
113       logger.trace(s"findSynonymsForKeywords.foreach keyword: $keyword")
114       val lastRow = if (synonymSheet.getLastRowNum < synonymSheet.getPhysicalNumberOfRows) synonymSheet.getLastRowNum else synonymSheet.getPhysicalNumberOfRows
115       logger.debug(s"synonymsheet.lastRow: ${lastRow}")
116       for (i <- 1 to lastRow) {
117         if (synonymSheet.getRow(i).getLastCellNum > 1) {
118           logger.trace(s"findSynonymsForKeywords.foreach synonymSheet.getLastRowNum: ${synonymSheet.getLastRowNum}")
119           if (getCellValueAsStringOption(i, 0, synonymSheet).isDefined) {
120             logger.trace(s"findSynonymsForKeywords.foreach getCellValueAsString($i, 0, ${synonymSheet.getLastRowNum}): " +
121               s"${getCellValueAsStringOption(i, 0, synonymSheet)}")
122             if (getCellValueAsStringOption(i, 0, synonymSheet).get.equalsIgnoreCase(keyword)) {
123               getCellValueAsStringOption(i, 1, synonymSheet).get.split(",").map(kw => kw.trim.toLowerCase).foreach(lkw => buf += lkw)
124             }
125           }
126         } else {
127           logger.debug(s"beyond cell range of row => row: $i, cell: 1, sheet: ${synonymSheet.getSheetName}," +
128             s"getLastCellNum: ${synonymSheet.getRow(i).getLastCellNum}")
129         }
130       }
131     }
132     val resultBuf = buf.toList
133     logger.debug(s"findSynonymsForKeywords => results buf: ${resultBuf.mkString("::")}")
134     resultBuf
135   }

137   /**
138     * reads in an XLSX Sheet with categories definition and a keywords lookup sheet and produces a list of Categories
139     * encapsulated in [[CategoryHolder]] to be dropped as RDF/XML eventually
140     *
141     * @param domainSheet
142     * @param synonymSheet
143     * @return
144     */
145   def buildCategoriesFromSheet(domainSheet: org.apache.poi.ss.usermodel.Sheet,
146                                synonymSheet: org.apache.poi.ss.usermodel.Sheet): List[CategoryHolder] = {
147     val buf = scala.collection.mutable.ListBuffer.empty[CategoryHolder]
148     val lastParent = scala.collection.mutable.ListBuffer.empty[String]
149     val parentDefault = "main"
150     lastParent += parentDefault
151     val lastRow = if (domainSheet.getLastRowNum < domainSheet.getPhysicalNumberOfRows) domainSheet.getLastRowNum else domainSheet.getPhysicalNumberOfRows
152     logger.warn(s"domainSheet.lastRow: ${lastRow}")
153     for (i <- 0 to lastRow) {
154       val cellValue = getCellValueAsStringOption(i, 0, domainSheet)
155       logger.trace(s"domainSheet.getRow($i).getCell(0).getCell: $cellValue")
156       if (cellValue.isDefined) {
157         val hierarchy_number = getCellValueAsStringOption(i, 0, domainSheet).get
158         val item_name = getCellValueAsStringOption(i, 1, domainSheet).getOrElse("")
159         logger.trace("row(+1) " + i + " hierarchy_number " + hierarchy_number + " item_name " + item_name)
160         if (hierarchy_number.equalsIgnoreCase("hierarchy_number")) {
161           if (!item_name.equalsIgnoreCase("item_name")) {
162             if (regMatcher.findFirstIn(item_name).isDefined) {
163               // val plist = regMatcher.findAllIn(item_name)
164               // lastParent += plist.toSeq.mkString("")
165               val plist = regMatcher.findAllIn(item_name).toList
166               logger.trace(s"regMatcher.findAllIn($item_name): ${plist.mkString("::")}")
167               lastParent += plist(1)
168             }
169           }
170         } else {
171           val lastParentValue = lastParent.last.trim
172           logger.trace(s"lastParentValue: ${lastParentValue}")
173           try {
174             logger.trace(s"before cell $i 4, after lastParentValue $lastParentValue")
175             val description = getCellValueAsStringOption(i, 4, domainSheet).getOrElse("")
176             logger.trace(s"before cell $i 5, after description $description")
177             val trimmed = getCellValueAsStringOption(i, 5, domainSheet).getOrElse("").split(",").map(_.trim)
178             val keywords_low = trimmed.map(_.toLowerCase).toList
179             val synonyms_low = findSynonymsForKeywords(keywords_low, synonymSheet)
180             val keyword_content = keywords_low ++ synonyms_low
181             logger.trace(s"before cell $i 6, after keyword_content ${keyword_content.mkString("::")}")
182             val icon = getCellValueAsStringOption(i, 6, domainSheet).getOrElse("")
183             logger.trace(s"before cell $i 7, after icon $icon")
184             val bg_icon = getCellValueAsStringOption(i, 7, domainSheet).getOrElse("")
185             logger.trace(s"before cell $i 8, after bg_icon $bg_icon")
186             val query_string = getCellValueAsStringOption(i, 8, domainSheet).getOrElse("")
187             if (i <= 10 && query_string.isEmpty) {
188               logger.warn(s"before cell $i categoryClass, after query_string $query_string")
189             } else {
190               logger.trace(s"before cell $i categoryClass, after query_string $query_string")
191             }
192             val categoryClass = if (lastParentValue.equalsIgnoreCase("main")) {
193               "MainCategory"
194             } else {
195               "ChildCategory"
196             }
197             logger.trace(s"before cat number $i")
198             val cat = CategoryHolder(
199               hierarchyNumber = hierarchy_number,
200               id = i,
201               parent = lastParentValue,
202               itemName = item_name,
203               description = description,
204               keywordContent = keyword_content,
205               queryString = query_string,
206               icon = icon,
207               bgIcon = bg_icon,
208               categoryClass = categoryClass)
209             logger.trace(s"after cat number $i")
210             logger.debug(cat.toRdf)
211             buf += cat
212 
213           } catch {
214             case ex: Exception => logger.error(s"new CategoryHolder error on row $i: ${ex.getMessage} ${ex.getStackTrace.mkString("\n")}")
215           }
216         }
217       }
218     }
219     logger.info(s"domainSheet.lastRow: ${lastRow}")
220     val resultBuf = buf.toList
221     logger.info(s"buildCategoriesFromSheet => results buf(${resultBuf.length}): ${resultBuf.map(kw => kw.hierarchyNumber).mkString("::")}")
222     resultBuf
223   }
