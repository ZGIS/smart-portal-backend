292     request =>
293       request.body.validate[OwcResource] fold(
294         errors => {
295           logger.error(JsError.toJson(errors).toString())
296           val error: ErrorResult = ErrorResult("replace resource failed",
297             Some(StringEscapeUtils.escapeJson(errors.mkString("; "))))
298           BadRequest(Json.toJson(error)).as(JSON)
299         },
300         owcResource => {
301           logger.trace(Json.prettyPrint(owcResource.toJson))
302           val collectionDoc = collectionsService.getOwcContextsForUserAndId(
303             userOption = Some(request.user), owcContextIdOption = Some(owcContextId)).headOption
304           val updatedCollection = collectionDoc.map {
305             o =>
306               val newResources = o.resource.filterNot(r => r.id.equals(owcResource.id))
307               val updated = o.copy(resource = newResources ++ Seq(owcResource))
308               collectionsService.updateCollection(updated, request.user)
309           }.getOrElse(None)
310           updatedCollection.fold {
311             val error: ErrorResult = ErrorResult("database update for resource failed", None)
312             BadRequest(Json.toJson(error)).as(JSON)
313           } {
314             theDoc => {
315               val resourceTest = theDoc.resource.find(r => r.id.equals(owcResource.id)).exists(
316                 tResource => tResource.equals(owcResource))
317               if (resourceTest) {
318                 Ok(Json.obj("message" -> "owcResource replaced in owcContext",
319                   "document" -> theDoc.toJson, "entry" -> owcResource.toJson))
320               }
321               else {
322                 val error: ErrorResult = ErrorResult("owcResource could not be replaced in owcContext", None)
323                 BadRequest(Json.toJson(error)).as(JSON)
324               }
325             }
326           }
