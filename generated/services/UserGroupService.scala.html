36   /**
37     * check if user can edit this particular group
38     *
39     * @param user
40     * @param userGroup
41     * @return
42     */
43   def amIPowerUserForGroup(user: User, userGroup: UserGroup): Boolean = {
44     dbSession.viaConnection(implicit connection => {
45       UserGroup.findUserGroupsForUser(user)
46         .find(ug => ug.uuid.equals(userGroup.uuid))
47         .exists(ug => ug.hasUsersLevel.exists(ul => ul.users_accountsubject.equals(user.accountSubject) && ul.userlevel >= 1))
48     })
49   }
50 
51   /**
52     * finds all groups a user is part of
53     *
54     * @param user
55     * @return
56     */
57   def getUsersOwnUserGroups(user: User): Seq[UserGroup] = {
58     dbSession.viaTransaction(implicit connection => {
59       UserGroup.findUserGroupsForUser(user)
60     })
61   }
62 
63   /**
64     * finds a group by its if the user is part of it
65     *
66     * @param user
67     * @param uuid
68     * @return
69     */
70   def findUsersOwnUserGroupsById(user: User, uuid: UUID): Option[UserGroup] = {
71     dbSession.viaTransaction(implicit connection => {
72       UserGroup.findUserGroupsForUser(user)
73         .find(ug => ug.uuid.equals(uuid))
74     })
75   }

77   /**
78     * will create a new group, assumes that calling user is listed with rights
79     *
80     * @param user
81     * @param userGroup
82     * @return
83     */
84   def createUsersOwnUserGroup(user: User, userGroup: UserGroup): Option[UserGroup] = {
85     dbSession.viaTransaction(implicit connection => {
86       UserGroup.createUserGroup(userGroup: UserGroup)
87     })
88   }
89 
90   /**
91     * will update a group if the calling user is poweruser
92     *
93     * @param user
94     * @param userGroup
95     * @return
96     */
97   def updateUsersOwnUserGroup(user: User, userGroup: UserGroup): Option[UserGroup] = {
98     val edit = amIPowerUserForGroup(user, userGroup)
99     if (edit) {
100       dbSession.viaTransaction(implicit connection => {
101         UserGroup.updateUserGroup(userGroup: UserGroup)
102       })
103     } else {
104       logger.warn(s"user ${user.accountSubject} illegally attempted to edit group ${userGroup.uuid}")
105       None
106     }
107   }
108 
109   /**
110     * will delete a group if the calling user is poweruser
111     *
112     * @param user
113     * @param uuid
114     * @return
115     */
116   def deleteUsersOwnUserGroup(user: User, uuid: UUID): Boolean = {
117     val userGroupOption = dbSession.viaConnection(implicit connection => UserGroup.findUserGroupsById(uuid))
118     userGroupOption.exists {
119       userGroup =>
120         val edit = amIPowerUserForGroup(user, userGroup)
121         if (edit) {
122           dbSession.viaTransaction(implicit connection => {
123             UserGroup.deleteUserGroup(userGroup: UserGroup)
124           })
125         } else {
126           logger.warn(s"user ${user.accountSubject} illegally attempted to edit group ${userGroup.uuid}")
127           false
128         }
129     }
130   }
131 
132   def getOwcContextsRightsMatrixForUser(user: User): Seq[OwcContextsRightsMatrix] = {
133     val userGroupsList = getUsersOwnUserGroups(user)
134 
135     val originalOwnContextsNativeRights = dbSession.viaConnection(implicit connection => {
136       OwcContextDAO.findOwcContextsByNativeOwnerBrief(user).distinct.map(uco =>
137         OwcContextsRightsMatrix(
138           owcContextId = uco.owcContextId,
139           queryingUserAccountSubject = user.accountSubject,
140           origOwnerAccountSubject = uco.origOwnerAccountSubject,
141           viaGroups = Seq(),
142           contextIntrinsicVisibility = uco.contextOwnersVisibility,
143           queryingUserAccessLevel = 2)
144       )
145     })
146 
147     val nativeRightsPerContextInGroups = dbSession.viaConnection(implicit connection => {
148       userGroupsList.flatMap {
149         g =>
150           val viaGroup = g.name
151           val ownLevel = g.hasUsersLevel.find(_.users_accountsubject.contentEquals(user.accountSubject)).map(_.userlevel).getOrElse(-1)
152 
153           val contextsRights: Seq[OwcContextsRightsMatrix] = g.hasOwcContextsVisibility.flatMap {
154             o =>
155               val groupContextsBriefTuples: Seq[UserHasOwcRightsNative] = OwcContextDAO.findOwcContextsByContextBrief(o.owc_context_id)
156 
157               val rightsBundle: Seq[OwcContextsRightsMatrix] = groupContextsBriefTuples.map {
158                 tup =>
159                   OwcContextsRightsMatrix(
160                     owcContextId = tup.owcContextId,
161                     queryingUserAccountSubject = user.accountSubject,
162                     origOwnerAccountSubject = tup.origOwnerAccountSubject,
163                     viaGroups = Seq(viaGroup),
164                     contextIntrinsicVisibility = tup.contextOwnersVisibility,
165                     queryingUserAccessLevel = ownLevel)
166               }
167               rightsBundle
168           }
169           contextsRights
170       }
171     })
172 
173     originalOwnContextsNativeRights ++ nativeRightsPerContextInGroups
174   }
