38 /**
39   * RectangleReader
40   */
41 class RectangleReader extends Reads[Rectangle] with ClassnameLogger {
42 
43   private lazy val ctx = SpatialContext.GEO
44   private lazy val jtsCtx = JtsSpatialContext.GEO
45   private lazy val jtsGeoJsonReader = jtsCtx.getFormats().getReader(ShapeIO.GeoJSON)
46 
47   /**
48     *
49     * @param geoJsonString
50     * @return
51     */
52   def jsToRect(geoJsonString: String): Option[Rectangle] = {
53     Try(jtsGeoJsonReader.read(geoJsonString).asInstanceOf[Rectangle]) match {
54       case Success(rect: Rectangle) => Some(rect)
55       case Failure(ex) => {
56         logger.warn(s"Problem reading geoJsonString to rectangle: ${ex.getMessage}")
57         None
58       }
59     }
60   }
61 
62   /**
63     *
64     * @param geoJsonJsValue
65     * @return
66     */
67   def jsToRect(geoJsonJsValue: JsValue): Option[Rectangle] = {
68     val geoJsonString = Json.stringify(geoJsonJsValue)
69     jsToRect(geoJsonString)
70   }
71 
72   /**
73     *
74     * @param json
75     * @return
76     */
77   def reads(json: JsValue): JsResult[Rectangle] = {
78 
79     logger.warn(s"trying js to rect ${Json.stringify(json)}")
80     val rectOpt = jsToRect(json)
81 
82     rectOpt match {
83       case Some(s) => {
84         JsSuccess(s)
85       }
86       case _ => {
87         logger.error("JsError ValidationError error.expected.geometry")
88         JsError(Seq(JsPath() -> Seq(ValidationError("error.expected.rectangle"))))
89       }
90     }
91   }
92 }

107         Json.parse(geoJsonWriter.toString(ctx.getWorldBounds))

109     }

126 class JtsPolygonReader extends Reads[Option[Rectangle]] with ClassnameLogger {
127 
128   private lazy val ctx = SpatialContext.GEO
129   private lazy val jtsCtx = JtsSpatialContext.GEO
130   private lazy val geoJsonReader = ctx.getFormats().getReader(ShapeIO.GeoJSON)
131   private lazy val jtsGeoJsonReader = jtsCtx.getFormats().getReader(ShapeIO.GeoJSON)
132 
133   /**
134     *
135     * @param geoJsonString
136     * @return
137     */
138   def jsToRect(geoJsonString: String): Option[Rectangle] = {
139 
140     val parse = Try {
141       val geom = jtsGeoJsonReader.read(geoJsonString)
142       logger.trace(s"jts geom ${geom.getBoundingBox}")
143       geom
144     }
145 
146     parse match {
147       case Success(shape: Shape) => Some(shape.getBoundingBox)
148       case Failure(ex) => {
149         logger.error(s"Problem reading geoJsonString to rectangle: ${ex.getMessage}")
150         None
151       }
152     }
153   }
154 
155   /**
156     *
157     * @param geoJsonJsValue
158     * @return
159     */
160   def jsToRect(geoJsonJsValue: JsValue): Option[Rectangle] = {
161     val geoJsonString = Json.stringify(geoJsonJsValue)
162     jsToRect(geoJsonString)
163   }
164 
165   /**
166     *
167     * @param json
168     * @return
169     */
170   def reads(json: JsValue): JsResult[Option[Rectangle]] = {
171     val rectOpt = jsToRect(json)
172     rectOpt match {
173       case Some(s) => {
174         JsSuccess(Some(s))
175       }
176       case None => {
177         JsSuccess(None)
178       }
179       case _ => {
180         logger.error("JsError ValidationError error.expected.geometry")
181         JsError(Seq(JsPath() -> Seq(ValidationError("error.expected.geometry"))))
182       }
183     }
184   }
