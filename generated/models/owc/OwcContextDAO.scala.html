45   /**
46     * Parse an OwcContext from a ResultSet
47     */
48   private def owcContextParser(implicit connection: Connection): RowParser[OwcContext] = {
49     str("id") ~
50       get[Option[String]]("area_of_interest") ~
51       get[Option[String]]("spec_reference") ~
52       get[Option[String]]("context_metadata") ~
53       str("language") ~
54       str("title") ~
55       get[Option[String]]("subtitle") ~
56       get[OffsetDateTime]("update_date") ~
57       get[Option[String]]("authors") ~
58       get[Option[String]]("publisher") ~
59       get[Option[String]]("creator_application") ~
60       get[Option[String]]("creator_display") ~
61       get[Option[String]]("rights") ~
62       get[Option[String]]("time_interval_of_interest") ~
63       get[Option[String]]("keyword") map {
64       case
65         id ~
66           bboxText ~
67           specReferenceUuids ~
68           contextMetaUuids ~
69           language ~
70           title ~
71           subtitle ~
72           updateDate ~
73           authorsUuids ~
74           publisher ~
75           creatorAppUuid ~
76           creatorDispUuid ~
77           rights ~
78           temporalText ~
79           keywordUuids
80       => {
81         OwcContext(
82           id = new URL(id),
83           areaOfInterest = GeoDateParserUtils.createOptionalBbox(bboxText),
84           specReference = specReferenceUuids.map(u =>
85             OwcOfferingDAO.findByPropertiesUUID[OwcLink](Some(u))(OwcLinkEvidence, connection).toList).getOrElse(List()), // aka links.profiles[] & rel=profile
86           contextMetadata = contextMetaUuids.map(u =>
87             OwcOfferingDAO.findByPropertiesUUID[OwcLink](Some(u))(OwcLinkEvidence, connection).toList).getOrElse(List()), // aka links.via[] & rel=via
88           language = language,
89           title = title,
90           subtitle = subtitle,
91           updateDate = updateDate,
92           author = authorsUuids.map(u => OwcOfferingDAO.findByPropertiesUUID[OwcAuthor](Some(u))(OwcAuthorEvidence, connection).toList).getOrElse(List()),
93           publisher = publisher,
94           creatorApplication = creatorAppUuid.map(u => OwcCreatorApplicationDAO.findOwcCreatorApplicationByUuid(UUID.fromString(u))).getOrElse(None),
95           creatorDisplay = creatorDispUuid.map(u => OwcCreatorDisplayDAO.findOwcCreatorDisplayByUuid(UUID.fromString(u))).getOrElse(None),
96           rights = rights,
97           timeIntervalOfInterest = GeoDateParserUtils.parseOffsetDateString(temporalText),
98           keyword = keywordUuids.map(u =>
99             OwcOfferingDAO.findByPropertiesUUID[OwcCategory](Some(u))(OwcCategoryEvidence, connection).toList).getOrElse(List()),
100           resource = OwcResourceDAO.findOwcResourcesForOwcContext(id.toString).toList)
101       }
102     }
103   }

304   /**
305     * takes a list of [[OwcResource]] and tests that non of these exist,
306     * and if so creates them in the database and returns success of the overall create
307     *
308     * @param resources
309     * @param connection
310     * @return
311     */
312   def preCreateCheckOwcResources(resources: List[OwcResource])(implicit connection: Connection): Boolean = {
313     if (resources.nonEmpty) {
314       val idList = resources.map(o => o.id.toString)
315       val exists = idList.exists(id => OwcResourceDAO.findOwcResourceById(id).isDefined)
316       if (exists) {
317         logger.error(s"(createContext/resources) OwcResource with ID one of: ${idList.mkString(" ++ ")} exists already, recommend abort")
318         false
319       } else {
320         val insert = resources.map(OwcResourceDAO.createOwcResource(_)).filter(_.isDefined)
321         insert.length == resources.length
322       }
323     } else {
324       true
325     }
326   }

374   /**
375     * create/insert into db Owc Context
376     *
377     * @param owcContextUnfixed should make issue on github for owc geojson and portal backend
378     *                          and then remove the fix see [[OwcGeoJsonFixes]]
379     * @param user
380     * @param visibility
381     * @param rightsRelationType
382     * @return
383     */
384   def createOwcContext(owcContextUnfixed: OwcContext, user: User, visibility: Int, rightsRelationType: String)
385                       (implicit connection: Connection): Option[OwcContext] = {
386 
387     val owcContext = OwcGeoJsonFixes.fixRelPropertyForOwcLinks(owcContextUnfixed)
388 
389     val preCreateCheckOwcAuthorsForContext = OwcResourceDAO.preCreateCheckOwcAuthors(owcContext.author)
390     val preCreateCheckOwcLinksForContext = OwcResourceDAO.preCreateCheckOwcLinks(owcContext.specReference ++ owcContext.contextMetadata)
391     val preCreateCheckOwcResourcesForContext = preCreateCheckOwcResources(owcContext.resource)
392     val preCreateCheckOwcCategoriesForContext = OwcResourceDAO.preCreateCheckOwcCategories(owcContext.keyword)
393     val preCreateCheckOwcCreatorApplicationForContext = preCreateCheckOwcCreatorApplication(owcContext)
394     val preCreateCheckOwcCreatorDisplayForContext = preCreateCheckOwcCreatorDisplay(owcContext)
395 
396     if (user.isActive && preCreateCheckOwcAuthorsForContext && preCreateCheckOwcLinksForContext &&
397       preCreateCheckOwcResourcesForContext && preCreateCheckOwcCategoriesForContext &&
398       preCreateCheckOwcCreatorApplicationForContext && preCreateCheckOwcCreatorDisplayForContext) {
399 
400       val owcContextInsert = SQL(
401         s"""insert into $tableOwcContexts values (
402            |{id},
403            |{area_of_interest},
404            |{spec_reference},
405            |{context_metadata},
406            |{language},
407            |{title},
408            |{subtitle},
409            |{update_date},
410            |{authors},
411            |{publisher},
412            |{creator_application},
413            |{creator_display},
414            |{rights},
415            |{time_interval_of_interest},
416            |{keyword})""".stripMargin).on(
417         'id -> owcContext.id.toString,
418         'area_of_interest -> owcContext.areaOfInterest.map(r => GeoDateParserUtils.rectToWkt(r)),
419         'spec_reference -> owcContext.specReference.map(_.uuid.toString).mkString(":"), // links.profiles[] and rel=profile
420         'context_metadata -> owcContext.contextMetadata.map(_.uuid.toString).mkString(":"), // aka links.via[] & rel=via
421         'language -> owcContext.language,
422         'title -> owcContext.title,
423         'subtitle -> owcContext.subtitle,
424         'update_date -> owcContext.updateDate.toLocalDateTime,
425         'authors -> owcContext.author.map(_.uuid.toString).mkString(":"),
426         'publisher -> owcContext.publisher,
427         'creator_application -> owcContext.creatorApplication.map(_.uuid.toString),
428         'creator_display -> owcContext.creatorDisplay.map(_.uuid.toString),
429         'rights -> owcContext.rights,
430         'time_interval_of_interest -> owcContext.timeIntervalOfInterest.map(dates => GeoDateParserUtils.writeOffsetDatesAsDateString(dates)).getOrElse(None),
431         'keyword -> owcContext.keyword.map(_.uuid.toString).mkString(":")
432       ).executeUpdate()
433 
434       // this references an existing owc_context_id (the newly created one from above)
435       val resourceDepsInsertCount = owcContext.resource.map {
436         owcResource =>
437           SQL(
438             s"""insert into $tableOwcContextHasOwcResources  values (
439                |{owc_context_id}, {owc_resource_id}
440                |)
441                  """.stripMargin).on(
442             'owc_context_id -> owcContext.id.toString,
443             'owc_resource_id -> owcResource.id.toString
444           ).executeUpdate()
445       }.sum
446 
447       // this references an existing owc_context_id (the newly created one from above)
448       val rightsDepsInsert = SQL(
449         s"""insert into $tableUserHasOwcContextRights values (
450            |{accountsubject}, {owcContextId}, {rights_relation_type}, {visibility}
451            |)""".stripMargin).on(
452         'accountsubject -> user.accountSubject,
453         'owcContextId -> owcContext.id.toString,
454         'rights_relation_type -> rightsRelationType,
455         'visibility -> visibility
456       ).executeUpdate()
457 
458       val allInserts = resourceDepsInsertCount + rightsDepsInsert + owcContextInsert
459       val testValue = owcContext.resource.size + 1 + 1
460 
461       if (allInserts == testValue) {
462         Some(owcContext)
463       } else {
464         logger.error("OwcContext/create: one of " +
465           s"resourceDepsInsertCount: $resourceDepsInsertCount (should be ${owcContext.resource.size}) + " +
466           s"rightsDepsInsert: $rightsDepsInsert (1) + owcContextInsert: $owcContextInsert (1) failed")
467         logger.error("OwcContext couldn't be created")
468         // we need to think where to place rollback most appropriately
469         connection.rollback()
470         None
471       }
472     } else {
473       logger.error("OwcContext/create: one of " +
474         s"user.isActive: ${user.isActive} + preCreateCheckOwcAuthorsForContext: $preCreateCheckOwcAuthorsForContext  " +
475         s"preCreateCheckOwcResourcesForContext: $preCreateCheckOwcResourcesForContext + " +
476         s"preCreateCheckOwcCategoriesForContext: $preCreateCheckOwcCategoriesForContext + " +
477         s"preCreateCheckOwcCreatorApplicationForContext: $preCreateCheckOwcCreatorApplicationForContext + " +
478         s"preCreateCheckOwcCreatorDisplayForContext: $preCreateCheckOwcCreatorDisplayForContext was not successful")
479       logger.error("Precondition failed, won't create OwcContext")
480       // we need to think where to place rollback most appropriately
481       connection.rollback()
482       None
483     }
484   }

557   /**
558     * pre Check for resources Update For an [[OwcContext]],
559     * will delete, update and insert required [[OwcResource]] in order to make OwcContext Update consistent
560     *
561     * @param owcContext
562     * @param connection
563     * @return
564     */
565   def preUpdateCheckOwcResources(owcContext: OwcContext)(implicit connection: Connection): Boolean = {
566     // get current list,
567     val current: List[URL] = owcContext.resource.map(_.id)
568 
569     logger.trace(s"preUpdateCheckOwcResources: current: List[URL] ${current.map(_.toString).mkString}")
570 
571     // get old list,
572     val oldOwcContext = findOwcContextById(owcContext.id)
