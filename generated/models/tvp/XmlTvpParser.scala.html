51     *
52     * @param source
53     * @return
54     */
55   def parseOm2Measurements(source: Source): Seq[TimeValuePair] = {
56 
57     val eventReader = new XMLEventReader(source)
58 
59     // phentime = "time";
60     var datetime: String = null
61     // GMLID = "identifier";
62     var foiId: String = null
63     // UOM = "uom";
64     var measUnit: String = null
65     // result = "value";
66     var measValue: String = null
67     // OBSPROP = "observedProperty";
68     var geom: Option[String] = None
69     // SFPOS = "pos";
70     var obsProp: String = null
71     // PROC = "procedure";
72     var procedure: String = null
73 
74     val tvps = ArrayBuffer[TimeValuePair]()
75 
76     for (event <- eventReader) {
77       event match {
78         case EvElemStart(prefix, label, attributes, ns) => {
79           if (label == OMOBS) {
80             logger.trace(s"$prefix:$label start")
81             // new observation
82             obsProp = "" // OBSPROP = "observedProperty";
83             foiId = "" // GMLID = "identifier";
84             geom = None // SFPOS = "pos";
85             measUnit = "" // UOM = "uom";
86           }
87           if (label == GMLID) {
88             // this would be the observation id of the enclosing OM_Observation
89             val next = eventReader.next().asInstanceOf[EvText]
90             logger.trace(s"$prefix:$label ${next.text}")
91           }
92           if (label == OBSPROP) {
93             val att = extractPrefixedAtrrib(attributes, X_HREF)
94             logger.trace(s"$prefix:$label $att")
95             obsProp = att
96           }
97           if (label == PROC) {
98             val att = extractPrefixedAtrrib(attributes, X_HREF)
99             logger.trace(s"$prefix:$label $att")
100             procedure = att
101           }
102           if (label == TIMEPOS) {
103             val next = eventReader.next().asInstanceOf[EvText]
104             logger.trace(s"$prefix:$label ${next.text}")
105             datetime = next.text
106           }
107           if (label == FOI) {
108             // should check if actual geometry is in here or an xlink:href
109             val att = extractPrefixedAtrrib(attributes, X_HREF)
110             logger.trace(s"$prefix:$label $att")
111             foiId = att
112           }
113           if (label == SFPOS) {
114             val next = eventReader.next().asInstanceOf[EvText]
115             logger.trace(s"$prefix:$label ${Some(next.text)}")
116             geom = Some(next.text)
117           }
118           if (label == RESULT) {
119             val next = eventReader.next().asInstanceOf[EvText]
120             measUnit = extractAttrib(attributes, UOM)
121             logger.trace(s"$prefix:$label ${next.text} ${extractAttrib(attributes, UOM)}")
122             measValue = next.text
123           }
124         }
125         // If we reach the end of an item element, we add it to the list
126         case EvElemEnd(prefix, label) => {
127           if (label == OMOBS) {
128             logger.debug(s"$prefix:$label end")
129             // full build and add
130             val tvp = TimeValuePair(
131               datetime,
132               foiId,
133               measUnit,
134               measValue,
135               geom,
136               obsProp,
137               procedure
138             )
139             logger.trace(tvp.toJson.toString())
140             tvps.append(tvp)
141           }
142         }
143         case EvText(t) => {
144           if (t.trim.size > 0) {
145             logger.debug(s"missed text: $t ")
146           }
147         }
148         case _ => // ignore
149       }
150     }
151 
152     tvps.toSeq
153   }

155   /**
156     *
157     * @param attrs
158     * @param prefixedLabel
159     * @return
160     */
161   def extractPrefixedAtrrib(attrs: Iterable[MetaData], prefixedLabel: String): String = {
162     val attOpt = attrs.find(meta => meta.prefixedKey.equalsIgnoreCase(prefixedLabel))
163     attOpt.map(meta => meta.value.text).getOrElse("")
164   }
165 
166   /**
167     *
168     * @param attrs
169     * @param label
170     * @return
171     */
172   def extractAttrib(attrs: Iterable[MetaData], label: String): String = {
173     val attOpt = attrs.find(meta => meta.key.equalsIgnoreCase(label))
174     attOpt.map(meta => meta.value.text).getOrElse("")
175   }
