45 /**
46   * provides access to ValidValues configuration. You need to have `smart.metadata.validValues` configured in
47   * your application configuration.
48   * @param configuration
49   */
50 @Singleton
51 class MetadataService @Inject()(portalConfig: PortalConfig) extends ClassnameLogger {
52 
53   lazy val metadataValidValues: Option[util.List[Configuration]] =
54     portalConfig.metadataValidValues
55 
56   /**
57     * returns an Option of ValidValues for a given topic. If topic os not found,
58     * it returns None.
59     * @param topic
60     * @return
61     */
62   def getValidValuesFor(topic: String): Option[ValidValues] = {
63     import scala.collection.JavaConversions._
64 
65     logger.debug(s"returning valid values for '${topic}'")
66     val configurations = metadataValidValues.get
67     val validValuesConf = configurations.filter(conf => conf.getString("for").getOrElse("").equals(topic))
68     if (validValuesConf.isEmpty()) {
69       logger.warn(s"Could not find ValidValues for ${topic}")
70       None
71     }
72     else {
73       val conf = validValuesConf.head
74       logger.debug(s"Found: ${validValuesConf.head}")
75 
76       Some(ValidValues.parseConfiguration(validValuesConf.head))
77     }
78   }
