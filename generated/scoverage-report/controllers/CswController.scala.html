83     val validValuesOption = metadataService.getValidValuesFor(topic)
84     if (validValuesOption.isEmpty) {

93   /** calls CSW:Transaction -> Insert
94     * FIXME should implement HasToken(parse.json) instead of Action.async, to only allow authenticated users
95     */
96   def insert: Action[AnyContent] = Action.async { request =>
97     logger.debug(request.body.asJson.toString)
98 
99     //TODO parse JSON to MDMetadataSet and convert that to XML
100 
101     val futureResponse: Future[WSResponse] = for {
102       getCapaResponse <- wsClient.url(CSW_OPERATIONS_METADATA_URL).get()
103       insertResponse <- {
104         //check if getCapaResponse indicates, that the CSW can perform Transactions
105         val operation = (getCapaResponse.xml \\ "OperationsMetadata" \\ "Operation").filter(node => {
106           logger.debug(s"Attribute 'name'=${node.attribute("name").get.text.toString}")
107           node.attribute("name").get.text.equals("Transaction")
108         })
109         logger.debug(s"operation = ${operation.toString()}")
110         if (operation.isEmpty) {
111           throw new UnsupportedOperationException("CSW does not support Transaction.")
112         }
113 
114         //insert MDMEtadata in insert template
115         val rule = new RuleTransformer(new AddMDMetadataToInsert(mdMetadata))
116         val finalXML = rule.transform(cswtInsertXml)
117         logger.debug(s"finalXml: ${finalXML.toString()}")
118         wsClient.url(CSW_URL).post(finalXML.toString())
119       }
120     } yield insertResponse
121 
122     futureResponse.recover {
123       case e: Exception =>
124         logger.error("Insert CSW threw exception", e)
125         InternalServerError(e.getMessage)
126     }
127 
128     futureResponse.map(response => {
129       logger.error(response.xml.toString())
130       response.xml match {
131         case e: Elem if e.label == "TransactionResponse" => {
132           val fileIdentifier = (e \\ "InsertResult" \\ "BriefRecord" \\ "identifier").text
133           Ok(Json.obj("type" -> "success", "fileIdentifier" -> fileIdentifier,
134             "message" -> s"Inserted as ${fileIdentifier}."))
135         }
136         case e: Elem if e.label == "ExceptionReport" => {
137           val message = (e \\ "Exception" \\ "ExceptionText").text
138           //TODO SR make this InternalServerError
139           Ok(Json.obj("type" -> "danger", "message" -> message))
140         }
141         case _ => InternalServerError(s"Unexpected Response: ${response}")
142       }
143     })
144   }
