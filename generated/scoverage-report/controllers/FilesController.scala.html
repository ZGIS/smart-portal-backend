142             theFile.ref.moveTo(tmpFile)

144             // Google Java upload stuff
145             import scala.collection.JavaConverters._
146 
147 //            val serviceAccountCredentials = ServiceAccountCredentials.fromStream(new FileInputStream(googleClientSecret))
148 //            val authenticatedStorageOptions = StorageOptions.newBuilder().setProjectId(googleProjectId)

152             val storage: Storage = StorageOptions.getDefaultInstance().getService()
153             // val noCredentials = NoCredentials.getInstance()
154 
155             // Modify access list to allow all users with link to read file
156             val roAcl = Acl.of(Acl.User.ofAllUsers(), Acl.Role.READER)
157             val acls: java.util.List[Acl] = List(roAcl).asJava
158 
159             // the inputstream is closed by default, so we don't need to close it here
160             val blob: Blob = storage.create(BlobInfo.newBuilder(googleStorageBucket, filename).setAcl(acls).build(),
161               Files.asByteSource(tmpFile).openStream())
162 
163             // return the public download link
164             val owcEntry = fileMetadata(filename, contentType, authUser, blob.getMediaLink())
165             val insertOk = collectionsService.addPlainFileEntryToUserDefaultCollection(owcEntry, authUser)
166 
167             if (insertOk) {
168               logger.debug(s"file upload $filename to ${blob.getMediaLink}")
169               //FIXME SR do we also want to have a general "return" object? Status is always in the respone so it does not need to be in here
170               Ok(Json.obj("status" -> "OK", "message" -> s"file uploaded $filename.", "file" -> blob.getMediaLink(), "entry" -> owcEntry.toJson))
171             } else {
172               logger.error("file metadata insert failed.")
173               val error = ErrorResult("File metadata insert failed.", None)
174               BadRequest(Json.toJson(error)).as(JSON)
175             }
176           }.getOrElse {
177             logger.error("file upload failed")
178             val error = ErrorResult("File upload failed.", None)
179             BadRequest(Json.toJson(error)).as(JSON)
180           }
181   }
