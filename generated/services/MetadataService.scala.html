45 /**
46   * provides access to ValidValues configuration. You need to have `smart.metadata.validValues` configured in
47   * your application configuration.
48   * @param configuration
49   */
50 class MetadataService @Inject()(configuration: Configuration) extends ClassnameLogger {
51 
52   lazy val metadataValidValues: Option[util.List[Configuration]] =
53     configuration.getConfigList("smart.metadata.validValues")
54 
55   /**
56     * returns an Option of ValidValues for a given topic. If topic os not found,
57     * it returns None.
58     * @param topic
59     * @return
60     */
61   def getValidValuesFor(topic: String): Option[ValidValues] = {
62     import scala.collection.JavaConversions._
63 
64     logger.debug(s"returning valid values for '${topic}'")
65     val configurations = metadataValidValues.get
66     val validValuesConf = configurations.filter(conf => conf.getString("for").getOrElse("").equals(topic))
67     if (validValuesConf.isEmpty()) {
68       logger.warn(s"Could not find ValidValues for ${topic}")
69       None
70     }
71     else {
72       val conf = validValuesConf.head
73       logger.debug(s"Found: ${validValuesConf.head}")
74 
75       Some(ValidValues.parseConfiguration(validValuesConf.head))
76     }
77   }
