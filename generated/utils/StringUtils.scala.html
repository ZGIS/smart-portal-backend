33   /**
34     * Converts String to Option
35     * @param s
36     */
37   implicit class OptionUuidConverters(val s: String) {
38 
39     /**
40       * Converts String to Option[UUID]. None when trimmed String is empty or UUID fails.
41       *
42       * @return Given UUID encapsulated in Option or None if empty after trimming
43       */
44     def toUuidOption: Option[UUID] = {
45       if (s.trim.isEmpty) {
46         None
47       }
48       else {
49         Try {
50           UUID.fromString(s)
51         } match {
52           case Success(uuid) => Some(uuid)
53           case Failure(ex) => {
54             logger.error("Couldn't convert String to UUID", ex)
55             None
56           }
57         }
58       }
59     }
60   }
61 
62   /**
63     * Converts String to Option
64     * @param s
65     */
66   implicit class OptionConverters(val s: String) {
67 
68     /**
69       * Converts String to Option. None when trimmed String is empty.
70       *
71       * @return Given String encapsulated in Option or None if empty after trimming
72       */
73     def toOption(): Option[String] = {
74       if (s.trim.isEmpty) {
75         None
76       }
77       else {
78         Some(s)
79       }
80     }
81   }
