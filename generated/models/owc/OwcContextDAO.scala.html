42   /**
43     * Parse an OwcContext from a ResultSet
44     */
45   private def owcContextParser(implicit connection: Connection): RowParser[OwcContext] = {
46     str("id") ~
47       get[Option[String]]("area_of_interest") ~
48       get[Option[String]]("spec_reference") ~
49       get[Option[String]]("context_metadata") ~
50       str("language") ~
51       str("title") ~
52       get[Option[String]]("subtitle") ~
53       get[OffsetDateTime]("update_date") ~
54       get[Option[String]]("authors") ~
55       get[Option[String]]("publisher") ~
56       get[Option[String]]("creator_application") ~
57       get[Option[String]]("creator_display") ~
58       get[Option[String]]("rights") ~
59       get[Option[String]]("time_interval_of_interest") ~
60       get[Option[String]]("keyword") map {
61       case
62         id ~
63           bboxText ~
64           specReferenceUuids ~
65           contextMetaUuids ~
66           language ~
67           title ~
68           subtitle ~
69           updateDate ~
70           authorsUuids ~
71           publisher ~
72           creatorAppUuid ~
73           creatorDispUuid ~
74           rights ~
75           temporalText ~
76           keywordUuids
77       => {
78         OwcContext(
79           id = new URL(id),
80           areaOfInterest = GeoDateParserUtils.createOptionalBbox(bboxText),
81           specReference = specReferenceUuids.map(u =>
82             OwcOfferingDAO.findByPropertiesUUID[OwcLink](Some(u))(OwcLinkEvidence, connection).toList).getOrElse(List()), // aka links.profiles[] & rel=profile
83           contextMetadata = contextMetaUuids.map(u =>
84             OwcOfferingDAO.findByPropertiesUUID[OwcLink](Some(u))(OwcLinkEvidence, connection).toList).getOrElse(List()), // aka links.via[] & rel=via
85           language = language,
86           title = title,
87           subtitle = subtitle,
88           updateDate = updateDate,
89           author = authorsUuids.map(u => OwcOfferingDAO.findByPropertiesUUID[OwcAuthor](Some(u))(OwcAuthorEvidence, connection).toList).getOrElse(List()),
90           publisher = publisher,
91           creatorApplication = creatorAppUuid.map(u => OwcCreatorApplicationDAO.findOwcCreatorApplicationByUuid(UUID.fromString(u))).getOrElse(None),
92           creatorDisplay = creatorDispUuid.map(u => OwcCreatorDisplayDAO.findOwcCreatorDisplayByUuid(UUID.fromString(u))).getOrElse(None),
93           rights = rights,
94           timeIntervalOfInterest = GeoDateParserUtils.parseOffsetDateString(temporalText),
95           keyword = keywordUuids.map(u =>
96             OwcOfferingDAO.findByPropertiesUUID[OwcCategory](Some(u))(OwcCategoryEvidence, connection).toList).getOrElse(List()),
97           resource = OwcResourceDAO.findOwcResourcesForOwcContext(id.toString).toList)
98       }
99     }
100   }

253   /**
254     * takes a list of [[OwcResource]] and tests that non of these exist,
255     * and if so creates them in the database and returns success of the overall create
256     *
257     * @param resources
258     * @param connection
259     * @return
260     */
261   def preCreateCheckOwcResources(resources: List[OwcResource])(implicit connection: Connection): Boolean = {
262     if (resources.nonEmpty) {
263       val idList = resources.map(o => o.id.toString)
264       val exists = idList.exists(id => OwcResourceDAO.findOwcResourceById(id).isDefined)
265       if (exists) {
266         logger.error(s"(createContext/resources) OwcResource with ID one of: ${idList.mkString(" ++ ")} exists already, recommend abort")
267         false
268       } else {
269         val insert = resources.map(OwcResourceDAO.createOwcResource(_)).filter(_.isDefined)
270         insert.length == resources.length
271       }
272     } else {
273       true
274     }
275   }

323   /**
324     * create/insert into db Owc Context
325     *
326     * @param owcContextUnfixed should make issue on github for owc geojson and portal backend
327     *                          and then remove the fix see [[OwcGeoJsonFixes]]
328     * @param user
329     * @param visibility
330     * @param rightsRelationType
331     * @return
332     */
333   def createOwcContext(owcContextUnfixed: OwcContext, user: User, visibility: Int, rightsRelationType: String)
334                       (implicit connection: Connection): Option[OwcContext] = {
335 
336     val owcContext = OwcGeoJsonFixes.fixRelPropertyForOwcLinks(owcContextUnfixed)
337 
338     val preCreateCheckOwcAuthorsForContext = OwcResourceDAO.preCreateCheckOwcAuthors(owcContext.author)
339     val preCreateCheckOwcLinksForContext = OwcResourceDAO.preCreateCheckOwcLinks(owcContext.specReference ++ owcContext.contextMetadata)
340     val preCreateCheckOwcResourcesForContext = preCreateCheckOwcResources(owcContext.resource)
341     val preCreateCheckOwcCategoriesForContext = OwcResourceDAO.preCreateCheckOwcCategories(owcContext.keyword)
342     val preCreateCheckOwcCreatorApplicationForContext = preCreateCheckOwcCreatorApplication(owcContext)
343     val preCreateCheckOwcCreatorDisplayForContext = preCreateCheckOwcCreatorDisplay(owcContext)
344 
345     if (user.isActive && preCreateCheckOwcAuthorsForContext && preCreateCheckOwcLinksForContext &&
346       preCreateCheckOwcResourcesForContext && preCreateCheckOwcCategoriesForContext &&
347       preCreateCheckOwcCreatorApplicationForContext && preCreateCheckOwcCreatorDisplayForContext) {
348 
349       val owcContextInsert = SQL(
350         s"""insert into $tableOwcContexts values (
351            |{id},
352            |{area_of_interest},
353            |{spec_reference},
354            |{context_metadata},
355            |{language},
356            |{title},
357            |{subtitle},
358            |{update_date},
359            |{authors},
360            |{publisher},
361            |{creator_application},
362            |{creator_display},
363            |{rights},
364            |{time_interval_of_interest},
365            |{keyword})""".stripMargin).on(
366         'id -> owcContext.id.toString,
367         'area_of_interest -> owcContext.areaOfInterest.map(r => GeoDateParserUtils.rectToWkt(r)),
368         'spec_reference -> owcContext.specReference.map(_.uuid.toString).mkString(":"), // links.profiles[] and rel=profile
369         'context_metadata -> owcContext.contextMetadata.map(_.uuid.toString).mkString(":"), // aka links.via[] & rel=via
370         'language -> owcContext.language,
371         'title -> owcContext.title,
372         'subtitle -> owcContext.subtitle,
373         'update_date -> owcContext.updateDate.toLocalDateTime,
374         'authors -> owcContext.author.map(_.uuid.toString).mkString(":"),
375         'publisher -> owcContext.publisher,
376         'creator_application -> owcContext.creatorApplication.map(_.uuid.toString),
377         'creator_display -> owcContext.creatorDisplay.map(_.uuid.toString),
378         'rights -> owcContext.rights,
379         'time_interval_of_interest -> owcContext.timeIntervalOfInterest.map(dates => GeoDateParserUtils.writeOffsetDatesAsDateString(dates)).getOrElse(None),
380         'keyword -> owcContext.keyword.map(_.uuid.toString).mkString(":")
381       ).executeUpdate()
382 
383       // this references an existing owc_context_id (the newly created one from above)
384       val resourceDepsInsertCount = owcContext.resource.map {
385         owcResource =>
386           SQL(
387             s"""insert into $tableOwcContextHasOwcResources  values (
388                |{owc_context_id}, {owc_resource_id}
389                |)
390                  """.stripMargin).on(
391             'owc_context_id -> owcContext.id.toString,
392             'owc_resource_id -> owcResource.id.toString
393           ).executeUpdate()
394       }.sum
395 
396       // this references an existing owc_context_id (the newly created one from above)
397       val rightsDepsInsert = SQL(
398         s"""insert into $tableUserHasOwcContextRights values (
399            |{accountsubject}, {owcContextId}, {rights_relation_type}, {visibility}
400            |)""".stripMargin).on(
401         'accountsubject -> user.accountSubject,
402         'owcContextId -> owcContext.id.toString,
403         'rights_relation_type -> rightsRelationType,
404         'visibility -> visibility
405       ).executeUpdate()
406 
407       val allInserts = resourceDepsInsertCount + rightsDepsInsert + owcContextInsert
408       val testValue = owcContext.resource.size + 1 + 1
409 
410       if (allInserts == testValue) {
411         Some(owcContext)
412       } else {
413         logger.error("OwcContext/create: one of " +
414           s"resourceDepsInsertCount: $resourceDepsInsertCount (should be ${owcContext.resource.size}) + " +
415           s"rightsDepsInsert: $rightsDepsInsert (1) + owcContextInsert: $owcContextInsert (1) failed")
416         logger.error("OwcContext couldn't be created")
417         // we need to think where to place rollback most appropriately
418         connection.rollback()
419         None
420       }
421     } else {
422       logger.error("OwcContext/create: one of " +
423         s"user.isActive: ${user.isActive} + preCreateCheckOwcAuthorsForContext: $preCreateCheckOwcAuthorsForContext  " +
424         s"preCreateCheckOwcResourcesForContext: $preCreateCheckOwcResourcesForContext + " +
425         s"preCreateCheckOwcCategoriesForContext: $preCreateCheckOwcCategoriesForContext + " +
426         s"preCreateCheckOwcCreatorApplicationForContext: $preCreateCheckOwcCreatorApplicationForContext + " +
427         s"preCreateCheckOwcCreatorDisplayForContext: $preCreateCheckOwcCreatorDisplayForContext was not successful")
428       logger.error("Precondition failed, won't create OwcContext")
429       // we need to think where to place rollback most appropriately
430       connection.rollback()
431       None
432     }
433   }

512   /**
513     * pre Check for resources Update For an [[OwcContext]],
514     * will delete, update and insert required [[OwcResource]] in order to make OwcContext Update consistent
515     *
516     * @param owcContext
517     * @param connection
518     * @return
519     */
520   def preUpdateCheckOwcResources(owcContext: OwcContext)(implicit connection: Connection): Boolean = {
521     // get current list,
522     val current: List[URL] = owcContext.resource.map(_.id)
523 
524     logger.trace(s"preUpdateCheckOwcResources: current: List[URL] ${current.map(_.toString).mkString}")
525 
526     // get old list,
527     val oldOwcContext = findOwcContextById(owcContext.id)
528     val old: List[URL] = oldOwcContext.map(o => o.resource.map(_.id)).getOrElse(List())
529 
530     logger.trace(s"preUpdateCheckOwcResources: old: List[URL] ${current.map(_.toString).mkString}")
531 
532     // in old but not current -> delete
533     val toBeDeleted = old.diff(current)
534 
535     logger.trace(s"preUpdateCheckOwcResources: toBeDeleted: List[URL] ${current.map(_.toString).mkString}")
536 
537     val deleted: Boolean = oldOwcContext.exists { owcContext =>
538       owcContext.resource.filter(o => toBeDeleted.contains(o.id))
539         .map { owcResource =>
540           // delete relation from tableOwcContextHasOwcResources table before finally deleting referenced OwcResource
541           val deletedOwcResourceRelation = SQL(
542             s"""Delete from $tableOwcContextHasOwcResources where
543 owc_context_id = {owc_context_id} and owc_resource_id = {owc_resource_id}""".stripMargin).on(
544             'owc_context_id -> owcContext.id.toString,
545             'owc_resource_id -> owcResource.id.toString
546           ).executeUpdate().equals(1)
547 
548           val deletedOwcResource = OwcResourceDAO.deleteOwcResource(owcResource)
549 
550           deletedOwcResource && deletedOwcResourceRelation
551         }.count(_ == true) == toBeDeleted.length
552     }
553 
554     // in both lists -> update
555     val toBeUpdated = current.intersect(old)
556 
557     logger.trace(s"preUpdateCheckOwcResources: toBeUpdated: List[URL] ${current.map(_.toString).mkString}")
558 
559     val updated = owcContext.resource.filter(o => toBeUpdated.contains(o.id))
560       .map(OwcResourceDAO.updateOwcResource(_))
561       .count(_.isDefined) == toBeUpdated.length
562 
563     // in current but not in old -> insert
564     val toBeInserted = current.diff(old)
565 
566     logger.trace(s"preUpdateCheckOwcResources: toBeInserted: List[URL] ${current.map(_.toString).mkString}")
567 
568     val inserted = owcContext.resource.filter(o => toBeInserted.contains(o.id))
569       .map { owcResource =>
570         val createdOwcResource = OwcResourceDAO.createOwcResource(owcResource)
571 
572         // Now also insert new relation into tableOwcContextHasOwcResources table for the created and to be referenced OwcResource
573         val createdOwcResourceRelation = SQL(
574           s"""Insert into $tableOwcContextHasOwcResources values ({owc_context_id}, {owc_resource_id})""".stripMargin).on(
575           'owc_context_id -> owcContext.id.toString,
576           'owc_resource_id -> owcResource.id.toString
577         ).executeUpdate().equals(1)
578 
579         if (createdOwcResourceRelation) createdOwcResource else None
580       }
581       .count(_.isDefined) == toBeInserted.length
582 
583     if (deleted && updated && inserted) {
