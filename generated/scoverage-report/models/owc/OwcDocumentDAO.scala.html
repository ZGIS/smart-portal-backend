86   def instantiateOwcFeatureType(id: String, featureType: String, bboxAsWkt: Option[String]): OwcFeatureType = {
87 
88     val emptyProfile = OwcLink(UUID.randomUUID(), "profile", None, "http://www.opengis.net/spec/owc-atom/1.0/req/core", None)
89     val emptySelf = OwcLink(UUID.randomUUID(), "self", Some("application/json"), "http://example.com/empty", None)
90     val now = ZonedDateTime.now.withZoneSameInstant(ZoneId.systemDefault())
91     logger.debug(s"instatiate for $featureType - $id")
92 
93     val emptyDefaultProps = OwcProperties(
94       UUID.randomUUID(),
95       "en",
96       "empty properties",
97       Some("no property information was found"),
98       Some(now),
99       Some("automated system response"),
100       None,
101       List(),
102       List(),
103       None,
104       None,
105       List(),
106       List(emptyProfile, emptySelf)
107     )
108 
109     val rectOpt = createOptionalBbox(bboxAsWkt)
110     val props = owcPropertiesDAO.findOwcPropertiesForOwcFeatureType(id).getOrElse(emptyDefaultProps)
111 
112     featureType match {
113       case "OwcEntry" => {
114         logger.debug(s"found OwcEntry ${id}")
115         val offerings = owcOfferingDAO.findOwcOfferingsForOwcEntry(id).toList
116         OwcEntry(id, rectOpt, props, offerings)
117       }
118       case "OwcDocument" => {
119         logger.debug(s"found OwcDocument ${id}")
120         val entries = findOwcEntriesForOwcDocument(id).toList
121         OwcDocument(id, rectOpt, props, entries)
122       }
123       case _ => throw new InvalidClassException(featureType, s"Unknown feature type $featureType")
124     }
125   }

193     * @param owcEntry
194     * @return

196   def createOwcEntry(owcEntry: OwcEntry): Option[OwcEntry] = {
197 
198     db.withTransaction {
199       implicit connection => {
200 
201         val sql = SQL(
202           s"""
203           insert into $tableOwcFeatureTypes values (
204             {id}, {feature_type}, {bbox}
205           )
206         """).on(
207           'id -> owcEntry.id,
208           'feature_type -> "OwcEntry",
209           'bbox -> owcEntry.bbox.map(rect => rectToWkt(rect))
210         )
211 
212         val rowCount = sql.executeUpdate()
213 
214         logger.debug(s"inserted ${owcEntry.getClass.getSimpleName} $rowCount ${sql.toString}")
215 
216         owcEntry.offerings.foreach {
217           owcOffering => {
218             if (owcOfferingDAO.findOwcOfferingByUuid(owcOffering.uuid).isEmpty) {
219               owcOfferingDAO.createOwcOffering(owcOffering)
220             }
221 
222             SQL(
223               s"""insert into $tableOwcEntriesHasOwcOfferings  values (
224                  |{owc_feature_types_as_entry_id}, {owc_offerings_uuid}
225                  |)
226                """.stripMargin).on(
227               'owc_feature_types_as_entry_id -> owcEntry.id,
228               'owc_offerings_uuid -> owcOffering.uuid.toString
229             ).executeUpdate()
230           }
231         }
232 
233         // the single dependent owc properties object
234         if (owcPropertiesDAO.findOwcPropertiesByUuid(owcEntry.properties.uuid).isEmpty) {
235           owcPropertiesDAO.createOwcProperties(owcEntry.properties)
236         }
237 
238         SQL(
239           s"""insert into $tableOwcFeatureTypesHasOwcProperties  values (
240              |{owc_feature_types_id}, {owc_properties_uuid}
241              |)
242                """.stripMargin).on(
243           'owc_feature_types_id -> owcEntry.id,
244           'owc_properties_uuid -> owcEntry.properties.uuid.toString
245         ).executeUpdate()
246 
247         rowCount match {
248           case 1 => Some(owcEntry)
249           case _ => None
250         }
251       }
252     }

284         owcEntry.offerings.filter {
285           offering => {
286             SQL(s"""select owc_offerings_uuid from $tableOwcEntriesHasOwcOfferings where owc_offerings_uuid = {uuid}""").on(
287               'uuid -> offering.uuid.toString
288             ).as(SqlParser.str("owc_offerings_uuid") *).isEmpty
289           }
290         }.foreach(owcOfferingDAO.deleteOwcOffering(_))
291 
292         if (SQL(s"""select owc_properties_uuid from $tableOwcFeatureTypesHasOwcProperties where owc_properties_uuid = {uuid}""").on(
293           'uuid -> owcEntry.properties.uuid.toString
294         ).as(SqlParser.str("owc_properties_uuid") *).isEmpty) {
295           owcPropertiesDAO.deleteOwcProperties(owcEntry.properties)
296         }
297       }
298     )
299 
300     rowCount match {
301       case 1 => true
302       case _ => false
303     }
304   }
