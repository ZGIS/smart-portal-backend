272   def callJenaKubeReload: Future[Option[ErrorResult]] = {
273     val digestHash = MessageDigest.getInstance("SHA-512").digest(appSecret.getBytes("UTF-8"))
274     val hex = digestHash.map("%02X" format _).mkString
275     wsClient.url(ADMIN_JENA_UPDATE_URL).withHeaders("Authorization" -> s"Bearer $hex").get().map { response =>
276       if (response.status == 200) {
277         logger.info(
278           s"Successfully called $ADMIN_JENA_UPDATE_URL (${response.status} - ${response.statusText})")
279         logger.info(s"response: ${response.body}")
280         None
281       } else {
282         logger.warn(
283           s"Call to '$ADMIN_JENA_UPDATE_URL' returned error: ${response.status} - ${response.statusText}")
284         logger.warn(s"response: ${response.body}")
285         Some(ErrorResult("Update of running vocab service failed.", Some(s"Call to '$ADMIN_JENA_UPDATE_URL' returned error: ${response.status} - ${response.statusText}")))
286       }
287     }
288   }
