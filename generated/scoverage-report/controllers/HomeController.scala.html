87   /**
88     * login with JSON (from Angular / form)
89     * {"username":"akmoch","password":"testpass123"}
90     *
91     * Create the authentication token and sets the cookie [[AuthTokenCookieKey]]
92     * for AngularJS to use also in X-XSRF-TOKEN in HTTP header.
93     *
94     * @return
95     */
96   def login = Action(parse.json) { implicit request =>
97     request.body.validate[LoginCredentials].fold(
98       errors => {
99         logger.error(JsError.toJson(errors).toString())
100         BadRequest(Json.obj("status" -> "ERR", "message" -> JsError.toJson(errors)))
101       },
102       valid = credentials => {
103         // find user in db and compare password stuff
104         userDAO.authenticate(credentials.username, credentials.password).fold {
105           logger.error("User or password wrong.")
106           Unauthorized(Json.obj("status" -> "ERR", "message" -> "Username or password wrong."))
107         } { user =>
108           val uaIdentifier: String = request.headers.get(UserAgentHeader).getOrElse(UserAgentHeaderDefault)
109           // logger.debug(s"Logging in username from $uaIdentifier")
110           val token = passwordHashing.createSessionCookie(user.username, uaIdentifier)
111           cache.set(token, user.username)
112           Ok(Json.obj("status" -> "OK", "token" -> token, "username" -> user.username, "userprofile" -> user.asProfileJs()))
113             .withCookies(Cookie(AuthTokenCookieKey, token, None, httpOnly = false))
114         }
115       })
116   }
117 
118   /**
119     * Log-out a user. Invalidates the authentication token.
120     *
121     * Discard the cookie [[AuthTokenCookieKey]] to have AngularJS no longer set the
122     * X-XSRF-TOKEN in HTTP header.
123     */
124   def logout = HasToken(parse.empty) { token =>
125     username => implicit request =>
126       cache.remove(token)
127       Ok.discardingCookies(DiscardingCookie(name = AuthTokenCookieKey))
128   }
129

135   def gconnect = Action { implicit request =>
136     val data = request.body.asJson.getOrElse(Json.obj("status" -> "ERR", "message" -> "gconnect went south"))
137     logger.debug(data.toString())
138     // do lots of Google OAuth2 stuff
139     Unauthorized(data)
140   }
141 
142   /**
143     * Disconnect the Google login
144     *
145     * @return
146     */
147   def gdisconnect = HasToken(parse.empty) { token =>
148     username => implicit request =>
149       cache.remove(token)
150       // do some Google OAuth2 unregisteR
151       Ok.discardingCookies(DiscardingCookie(name = AuthTokenCookieKey))
152   }
