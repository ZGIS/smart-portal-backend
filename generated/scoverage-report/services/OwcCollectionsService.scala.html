347   /**
348     * returns an Option[EmailAddress] object if parsing is successful, for those mdMetadata email joint arrays
349     *
350     * @param emailString the string that might comprise of or contain an email address
351     * @return Option[EmailAddress]
352     */
353   def parseEmailStringtoEmailAddress(emailString: String): Option[EmailAddress] = {
354 
355     if (EmailAddress.isValid(emailString)) {
356       Some(EmailAddress(emailString))
357     } else {
358       if (emailString.contains(",")) {
359         val commaFree = emailString.split(",").head.trim
360 
361         Try {
362           EmailAddress(commaFree)
363         } match {
364           case Success(e) => Some(e)
365           case _ => None
366         }
367       } else {
368         None
369       }
370     }
