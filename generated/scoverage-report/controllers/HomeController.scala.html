86   /**
87     * login with JSON (from Angular / form)
88     * {"username":"akmoch","password":"testpass123"}
89     *
90     * Create the authentication token and sets the cookie [[AuthTokenCookieKey]]
91     * for AngularJS to use also in X-XSRF-TOKEN in HTTP header.
92     *
93     * @return
94     */
95   def login = Action(parse.json) { implicit request =>
96     request.body.validate[LoginCredentials].fold(
97       errors => {
98         logger.error(JsError.toJson(errors).toString())
99         BadRequest(Json.obj("status" -> "ERR", "message" -> JsError.toJson(errors)))
100       },
101       valid = credentials => {
102         // find user in db and compare password stuff
103         userDAO.authenticate(credentials.username, credentials.password).fold {
104           logger.error("User or password wrong.")
105           Unauthorized(Json.obj("status" -> "ERR", "message" -> "Username or password wrong."))
106         } { user =>
107           val uaIdentifier: String = request.headers.get(UserAgentHeader).getOrElse(UserAgentHeaderDefault)
108           // logger.debug(s"Logging in username from $uaIdentifier")
109           val token = passwordHashing.createSessionCookie(user.username, uaIdentifier)
110           cache.set(token, user.username)
111           Ok(Json.obj("status" -> "OK", "token" -> token, "username" -> user.username, "userprofile" -> user.asProfileJs()))
112             .withCookies(Cookie(AuthTokenCookieKey, token, None, httpOnly = false))
113         }
114       })
115   }
116 
117   /**
118     * Log-out a user. Invalidates the authentication token.
119     *
120     * Discard the cookie [[AuthTokenCookieKey]] to have AngularJS no longer set the
121     * X-XSRF-TOKEN in HTTP header.
122     */
123   def logout = HasToken(parse.empty) { token =>
124     username => implicit request =>
125       cache.remove(token)
126       Ok.discardingCookies(DiscardingCookie(name = AuthTokenCookieKey))
127   }
128

137     // do lots of Google OAuth2 stuff

139   }

145     */

147     username => implicit request =>
148       cache.remove(token)
149       // do some Google OAuth2 unregisteR
150       Ok.discardingCookies(DiscardingCookie(name = AuthTokenCookieKey))
151   }
