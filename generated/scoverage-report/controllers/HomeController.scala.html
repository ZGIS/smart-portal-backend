111   /**
112     * login with JSON (from Angular / form)
113     * {"email":"akmoch","password":"testpass123"}
114     *
115     * Create the authentication token and sets the cookie [[AuthTokenCookieKey]]
116     * for AngularJS to use also in X-XSRF-TOKEN in HTTP header.
117     *
118     * @return
119     */
120   def login: Action[JsValue] = Action(parse.json) { implicit request =>
121     request.body.validate[LoginCredentials].fold(
122       errors => {
123         logger.error(JsError.toJson(errors).toString())
124         val error = ErrorResult("Could not validate request.", Some(JsError.toJson(errors).toString()))
125         BadRequest(Json.toJson(error)).as(JSON)
126       },
127       valid = credentials => {
128         // find user in db and compare password stuff
129         userDAO.authenticate(credentials.email, credentials.password).fold {
130           logger.error("User email or password wrong.")
131           val error = ErrorResult("User email or password wrong.", None)
132           Unauthorized(Json.toJson(error)).as(JSON)
133         } { user =>
134           val uaIdentifier: String = request.headers.get(UserAgentHeader).getOrElse(UserAgentHeaderDefault)
135           // logger.debug(s"Logging in email from $uaIdentifier")
136           val token = passwordHashing.createSessionCookie(user.email, uaIdentifier)
137           cache.set(token, user.email)
138           Ok(Json.obj("status" -> "OK", "token" -> token, "email" -> user.email, "userprofile" -> user.asProfileJs()))
139             .withCookies(Cookie(AuthTokenCookieKey, token, None, httpOnly = false))
140         }
141       })
142   }

144   /**
145     * https://developers.google.com/recaptcha/docs/verify
146     *
147     * @param recaptcaChallenge
148     * @return
149     */
150   def recaptchaValidate(recaptcaChallenge: String): Action[AnyContent] = Action.async { implicit request =>
151     // TODO check from where URL referer comes from
152     ws.url(recaptcaVerifyUrl)
153       .withHeaders("Accept" -> "application/json")
154       .withRequestTimeout(10000.millis)
155       .withQueryString("secret" -> reCaptchaSecret,
156         "response" -> recaptcaChallenge).get().map {
157       response =>
158         val success = (response.json \ "success").as[Boolean]
159         if (success) {
160           Ok(Json.obj("status" -> "OK", "message" -> "granted", "success" -> JsBoolean(true)))
161         }
162         else {
163           val errors = (response.json \ "error-codes")
164           val jsErrors = errors.getOrElse(JsString("No further errors"))
165           val error = ErrorResult("User email or password wrong.", Some(jsErrors.toString()))
166           BadRequest(Json.toJson(error)).as(JSON)
167         }
168     }.recover {
169       case e: Exception =>
170         val error = ErrorResult("Exception on validating captcha.", Some(e.getMessage))
171         InternalServerError(Json.toJson(error)).as(JSON)
172     }
173   }
174 
175   /**
176     * Log-out a user. Invalidates the authentication token.
177     *
178     * Discard the cookie [[AuthTokenCookieKey]] to have AngularJS no longer set the
179     * X-XSRF-TOKEN in HTTP header.
180     */
181   def logout: Action[Unit] = HasToken(parse.empty) { token =>
182     email =>
183       implicit request =>
184         cache.remove(token)
185         Ok.discardingCookies(DiscardingCookie(name = AuthTokenCookieKey))
186   }
