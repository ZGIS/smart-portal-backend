63   /**
64     * Loads a time series from a SOS server. A Timeseries "configuration" has to be passd within the POST
65     * request.
66     *
67     * @return
68     */
69   def getTimeseries: Action[JsValue] = Action.async(parse.json) { request =>
70     request.body.validate[Timeseries].fold(
71       errors => {
72         logger.error(JsError.toJson(errors).toString())
73         val error = ErrorResult("Error while validating request.", Some(JsError.toJson(errors).toString))
74         Future {
75           BadRequest(Json.toJson(error)).as(JSON)
76         }
77       },
78 
79       timeseries => {
80         val responseFuture = wsClient.url(timeseries.sosUrl)
81           .withHeaders(CONTENT_TYPE -> "application/xml")
82           .post(getObservationRequestXml(timeseries))
83 
84         responseFuture.recover {
85           case e: Exception =>
86             logger.error("Sos get capabilities threw exception", e)
87             val error = ErrorResult(s"Exception (${e.getClass.getCanonicalName}) on SOS get capabilities.",
88               Some(e.getMessage))
89             InternalServerError(Json.toJson(error)).as(JSON)
90         }
91 
92         responseFuture.map(response => {
93           if (response.status != 200) {
94             logger.error(s"Calling SOS ${timeseries.sosUrl} HTTP result status ${response.status} on GetObservation")
95             val error = ErrorResult(s"Server ${timeseries.sosUrl} responded with ${response.status} on GetObservation",
96               Some(response.body))
97             InternalServerError(Json.toJson(error)).as(JSON)
98           }
99           else if (!response.header("Content-Type").get.contains("application/xml")) {
100             logger.error(s"SOS GetCapabilities result was not XML but ${response.header("Content-Type").get}")
101             val error = ErrorResult(
102               s"SOS GetCapabilities result was not XML but ${response.header("Content-Type").get}",
103               Some(response.body))
104             InternalServerError(Json.toJson(error)).as(JSON)
105           }
106           else {
107             val tvp = new XmlTvpParser().parseOm2Measurements(Source.fromString(response.body))
108             val tsdata =
109               if (tvp.isEmpty) {
110                 TimeseriesData(
111                   x = Seq.empty[String],
112                   y = Seq.empty[String],
113                   name = timeseries.timeseriesName
114                 )
115               }
116               else {
117                 TimeseriesData(
118                   x = tvp.map(_.datetime),
119                   y = tvp.map(_.measValue),
120                   name = timeseries.timeseriesName
121                 )
122               }
123             val result = timeseries.copy(data = Some(tsdata), uom = Some(tvp.head.measUnit))
124             Ok(result.toJson()).as(JSON)
125           }
126         })
127       }
128     )
129   }
