41 final case class User(email: EmailAddress,
42                 accountSubject: String,
43                 firstname: String,
44                 lastname: String,
45                 password: String,
46                 laststatustoken: String,
47                 laststatuschange: ZonedDateTime) extends ClassnameLogger {

49   /**
50     * toString, should be more explicit with datetime handling
51     *
52     * @return
53     */
54   override def toString: String = {
55     f"""User(
56         |${email.value},
57         |${accountSubject},
58         |${firstname},
59         |${lastname},
60         |***,
61         |${laststatustoken},
62         |${laststatuschange}

66   /**
