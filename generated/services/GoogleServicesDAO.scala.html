164     *
165     * @param vocabString the vocabString
166     * @param pathPrefix sub directory where the file should go /prefix/prefix no trailing slash
167     * @return
168     */
169   def updateVocabFileGoogleBucket(vocabString: String, fileName: String, pathPrefix: String): Either[ErrorResult, LocalBlobInfo] = {
170     // Google Java upload stuff
171     import java.nio.charset.StandardCharsets.UTF_8
172 
173     // val storage: Storage = authenticatedStorageOptions.getService()
174     val storage: Storage = StorageOptions.getDefaultInstance().getService()
175     // val noCredentials = NoCredentials.getInstance()
176 
177     val blobTry: Try[LocalBlobInfo] = Try {
178       val blobId = BlobId.of(googleStorageBucket, pathPrefix + "/" + fileName)
179       val blob = storage.get(blobId)
180       if (blob != null) {
181         val channel = blob.writer()
182         channel.write(ByteBuffer.wrap(vocabString.getBytes(UTF_8)))
183         channel.close()
184         LocalBlobInfo.newFrom(blob)
185       } else {
186         throw new IllegalArgumentException("original blob for file could not be retrieved.")
187       }
188     }
189 
190     blobTry match {
191       case Success(blobInfo) => Right(blobInfo)
192       case Failure(ex) =>
193         logger.error(s"Blob update in cloud storage failed. ${ex.getLocalizedMessage}")
194         Left(ErrorResult("Blob update in cloud storage failed.", Some(ex.getLocalizedMessage)))
