112           //TODO parse JSON to MDMetadataSet and convert that to XML

123               logger.trace(Json.prettyPrint(mdMetadata.toJson()))
124               //FIXME SR I find that pretty hard to read. Is there a better way of chaining WS calls?
125               // see https://www.playframework.com/documentation/2.5.x/ScalaWS#chaining-ws-calls
126               val futureResponse: Future[(WSResponse, WSResponse)] = for {
127                 getCapaResponse <- wsClient.url(CSW_OPERATIONS_METADATA_URL).get()
128                 insertResponse <- {
129                   //check if getCapaResponse indicates, that the CSW can perform Transactions
130                   val operation = (getCapaResponse.xml \\ "OperationsMetadata" \\ "Operation").filter(node => {
131                     logger.debug(s"Attribute 'name'=${node.attribute("name").get.text.toString}")
132                     node.attribute("name").get.text.equals("Transaction")
133                   })
134                   logger.debug(s"operation = ${operation.toString()}")
135                   if (operation.isEmpty) {
136                     throw new UnsupportedOperationException("CSW does not support Transaction.")
137                   }
138 
139                   //insert MDMEtadata in insert template
140                   logger.debug(s"MD_MetadataXML: ${mdMetadata.toXml().toString()}")
141                   val rule = new RuleTransformer(new AddMDMetadataToInsert(mdMetadata.toXml()))
142                   val finalXML = rule.transform(cswtInsertXml)
143                   logger.debug(s"finalXml: ${finalXML.toString()}")
144                   wsClient.url(CSW_URL).post(finalXML.toString())
145                 }
146                 updateIngesterIndexResponse <- {
147                   val response = wsClient.url(INGESTER_UPDATE_INDEX_URL).get()
148                   response.onSuccess {
149                     case response if response.status == 200 => {
150                       logger.info(
151                         s"Successfully called $INGESTER_UPDATE_INDEX_URL (${response.status} - ${response.statusText})");

153                     }

157                       logger.warn(s"response: ${response.body}");

161                 }
162               } yield (insertResponse, updateIngesterIndexResponse)
163 
164               futureResponse.recover {

167                   val error = ErrorResult(s"Exception (${e.getClass.getCanonicalName}) on CSW insert: ${e.getMessage}",
168                     None)
169                   InternalServerError(Json.toJson(error)).as(JSON)
170               }
171 
172               futureResponse.map(response => {
173                 logger.debug(response._1.xml.toString())
174                 response._1.xml match {
175                   case e: Elem if e.label == "TransactionResponse" => {
176                     val fileIdentifier = (e \\ "InsertResult" \\ "BriefRecord" \\ "identifier").text
177                     logger.debug(s"Adding ${mdMetadata.fileIdentifier} to default collection of $authUser.")
178                     val added = collectionsService.addMdResourceToUserDefaultCollection(CSW_URL, mdMetadata, authUser)
179                     val userMetaEntryOk = userService.insertUserMetaRecordEntry(CSW_URL, mdMetadata.fileIdentifier, authUser)
180                     if (added && userMetaEntryOk.isDefined) {
181                       Ok(Json.obj("type" -> "success", "fileIdentifier" -> fileIdentifier,
182                         "message" -> s"Inserted as ${fileIdentifier} and reference entry added to your data collection."))
183                     } else {
184                       val error = ErrorResult(s"Could not add ${mdMetadata.fileIdentifier} to your collection of $authUser.", None)
185                       logger.warn(error.message)
186                       BadRequest(Json.toJson(error)).as(JSON)
187                     }
188                   }
189                   case e: Elem if e.label == "ExceptionReport" => {
190                     val message = (e \\ "Exception" \\ "ExceptionText").text
191                     //TODO SR make this InternalServerError
192                     val error = ErrorResult(message, None)
193                     InternalServerError(Json.toJson(error)).as(JSON)
194                   }
195                   case _ => {
196                     val error = ErrorResult(s"Unexpected Response from CSW: ${response._1.status} - ${response._1.statusText}",
197                       Some(response._1.body))
198                     InternalServerError(Json.toJson(error)).as(JSON)
199                   }
200                 }
201               })
202             }
203           )
