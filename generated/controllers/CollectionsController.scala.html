291     request =>

293         errors => {
294           logger.error(JsError.toJson(errors).toString())

296             Some(StringEscapeUtils.escapeJson(errors.mkString("; "))))
297           BadRequest(Json.toJson(error)).as(JSON)
298         },
299         owcResource => {
300           logger.trace(Json.prettyPrint(owcResource.toJson))
301           val collectionDoc = collectionsService.getOwcContextsForUserAndId(
302             userOption = Some(request.user), owcContextIdOption = Some(owcContextId)).headOption
303           val updatedCollection = collectionDoc.map {
304             o =>
305               val newResources = o.resource.filterNot(r => r.id.equals(owcResource.id))
306               val updated = o.copy(resource = newResources ++ Seq(owcResource))
307               collectionsService.updateCollection(updated, request.user)
308           }.getOrElse(None)
309           updatedCollection.fold {
310             val error: ErrorResult = ErrorResult("database update for resource failed", None)
311             BadRequest(Json.toJson(error)).as(JSON)
312           } {
313             theDoc => {
314               val resourceTest = theDoc.resource.find(r => r.id.equals(owcResource.id)).exists(
315                 tResource => tResource.equals(owcResource))
316               if (resourceTest) {
317                 Ok(Json.obj("message" -> "owcResource replaced in owcContext",
318                   "document" -> theDoc.toJson, "entry" -> owcResource.toJson))
319               }
320               else {
321                 val error: ErrorResult = ErrorResult("owcResource could not be replaced in owcContext", None)
322                 BadRequest(Json.toJson(error)).as(JSON)
323               }
324             }
325           }
326         }
