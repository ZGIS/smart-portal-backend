291             // check if new email is used already by another user than yourself
292             userService.findUserByEmailAddress(incomingProfileJs.email).fold {
293 
294               // found none, good, let's update your profile in the database (but not password nor accountsubject)

322     * should require "double check" in gui to issue this, requires to accept and store new session secret
323     *

325     */
326   def updatePassword(): Action[JsValue] = defaultAuthAction(parse.json) {
327     request =>
328       request.body.validate[PasswordUpdateCredentials].fold(
329         errors => {
330           logger.error(JsError.toJson(errors).toString())
331           val error = ErrorResult("Could not validate request.", Some(JsError.toJson(errors).toString()))
332           BadRequest(Json.toJson(error)).as(JSON)
333         },
334         valid = credentials => {
335           // find user in db
336           val dbuser = request.user
337           // compare password
338           if (passwordHashing.validatePassword(credentials.oldPassword, dbuser.password)) {
339             val newCryptPass = passwordHashing.createHash(credentials.newPassword)
340             val updateUser = dbuser.copy(
341               password = newCryptPass,
342               laststatustoken = s"${StatusToken.ACTIVE}:PASSWORDUPDATE",
343               laststatuschange = ZonedDateTime.now.withZoneSameInstant(ZoneId.of(appTimeZone)))
344 
345             userService.updatePassword(updateUser).fold {
346               logger.error("Password update error.")
347               val error = ErrorResult("Password update error.", None)
348               BadRequest(Json.toJson(error)).as(JSON)
349             } { user =>
350               val uaIdentifier: String = request.headers.get(UserAgentHeader).getOrElse(UserAgentHeaderDefault)
351               // cache.remove(token)
352               userService.removeUserSessionCache(request.user.email, request.authenticatedRequest.userSession.token)
353               emailService.sendPasswordUpdateEmail(user.email, "Password Update on GW HUB", user.firstname)
354               // val newtoken = passwordHashing.createSessionCookie(user.email, uaIdentifier)
355               // cache.set(newtoken, user.email.value)
356               val newtoken = userService.upsertUserSession(user.email.value, uaIdentifier)
357               Ok(Json.obj("status" -> "OK", "token" -> newtoken, "email" -> user.email.value))
358                 .withCookies(Cookie(AuthTokenCookieKey, newtoken, None, httpOnly = false))
359             }
360           } else {
361             logger.error("Password update error, not matching.")
362             val error = ErrorResult("Password update error, not matching.", None)
363             BadRequest(Json.toJson(error)).as(JSON)
364           }
365         })
366   }

480   def gconnect: Action[JsValue] = Action(parse.json) { implicit request =>
481     request.body.validate[GAuthCredentials].fold(
482       errors => {
483         logger.error(JsError.toJson(errors).toString())
484         val error = ErrorResult("Could not validate request.", Some(JsError.toJson(errors).toString()))
485         BadRequest(Json.toJson(error)).as(JSON)
486       },
487       valid = gAuthCredentials => {
488         val accessTokenResult = googleService.getGoogleAuthorization(gAuthCredentials)
489         accessTokenResult match {
490           case Left(errorResult) => {
491             logger.error(errorResult.message + errorResult.details.mkString)
492             InternalServerError(Json.toJson(errorResult)).as(JSON)
493           }
494           case Right(googleTokenResponse) => {
495             // Get profile info from ID token
496             val gAuthPayload = googleTokenResponse.parseIdToken().getPayload
497             // Use this value as a key to identify a user.
498             val userId = gAuthPayload.getSubject()
499             val email = gAuthPayload.getEmail()
500             val emailVerified = gAuthPayload.getEmailVerified()
501             // get further payload things that is easy in java but needs to work in scala too
502             val name = Option(gAuthPayload.get("name").asInstanceOf[String])
503             val familyName = Option(gAuthPayload.get("family_name").asInstanceOf[String])
504             val givenName = Option(gAuthPayload.get("given_name").asInstanceOf[String])
505             if ((gAuthCredentials.accesstype.equalsIgnoreCase("LOGIN") || gAuthCredentials.accesstype.equalsIgnoreCase("REGISTER")) &&
506               emailVerified && EmailAddress.isValid(email)) {
507 
508               val emailAddress = EmailAddress(email)
509               // viaConnection lookup email block
510               val firstUserLookup = userService.findUserByEmailAddress(emailAddress)
511 
512               firstUserLookup.fold {
513                 // 'error, no user': well, must be a new user coming through from Google
514                 val cryptPass = passwordHashing.createHash(UUID.randomUUID().toString)
515 
516                 // Use access token to call API for refresh or check-up?
517                 // val googleAccessToken = googleTokenResponse.getAccessToken()
518                 // val googleCredential = new GoogleCredential().setAccessToken(googleAccessToken)
519                 val newUser = User(emailAddress,
520                   s"google:${userId}",
521                   givenName.getOrElse(name.getOrElse(emailAddress.mailbox.value)),
522                   familyName.getOrElse(""),
523                   cryptPass,
524                   s"${StatusToken.ACTIVE}:REGCONFIRMED",
525                   ZonedDateTime.now.withZoneSameInstant(ZoneId.of(appTimeZone)))
526 
527                 val createUserCall = userService.createUser(newUser)
528

530                   logger.error("User create error.")
531                   val error = ErrorResult("Error while creating user", None)

541                     InternalServerError(Json.toJson(error)).as(JSON)
542                   } { owcDoc =>
543                     logger.info(s"New user registered ${createdUser.accountSubject}. Email went out $emailWentOut. " +
544                       s"User Collection ${owcDoc.id.toString}")
545                     val uaIdentifier: String = request.headers.get(UserAgentHeader).getOrElse(UserAgentHeaderDefault)
546                     // logger.debug(s"Logging in email from $uaIdentifier")
547                     // val token = passwordHashing.createSessionCookie(createdUser.email, uaIdentifier)

557                 // val token = passwordHashing.createSessionCookie(existingUser.email.value, uaIdentifier)
