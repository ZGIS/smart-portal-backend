58   /**
59     * create Session Cookie
60     *
61     * @param username
62     * @param useragent
63     * @return
64     */
65   def createSessionCookie(username: String, useragent: String): String = {
66     val in = appSecret + username + useragent.toUpperCase
67     val digest = MessageDigest.getInstance("SHA-512").digest(in.getBytes("UTF-8"))
68     // Logger.trace(s"createSessionCookie: ${toHex(digest)} app secret: $appSecret , email: $email, useragent: $useragent >>> $in")
69     toHex(digest)
70   }

107   /**
108     *
109     * @param password
110     * @return
111     */
112   def createHash(password: Array[Char]): String = {
113     val salt = nextRandomSalt
114     val hash = pbkdf2(password, salt, PBKDF2_ITERATIONS, HASH_BYTE_SIZE)
115     PBKDF2_ITERATIONS + ":" + toHex(salt) + ":" + toHex(hash)
116   }

118   /**
119     *
120     * @return
121     */
122   private def nextRandomSalt(): Array[Byte] = {
123     val random = new SecureRandom()
124     val salt = Array.ofDim[Byte](SALT_BYTE_SIZE)
125     random.nextBytes(salt)
126     salt
127   }

139   /**
140     * the work horse...
141     *
142     * @param password
143     * @param correctHash
144     * @return
145     */
146   def validatePassword(password: Array[Char], correctHash: String): Boolean = {
147     val params = correctHash.split(":")
148     val iterations = Integer.parseInt(params(ITERATION_INDEX))
149     val salt = fromHex(params(SALT_INDEX))
150     val hash = fromHex(params(PBKDF2_INDEX))
151     val testHash = pbkdf2(password, salt, iterations, hash.length)
152     slowEquals(hash, testHash)
153   }
154 
155   /**
156     * the other work horse
157     *
158     * @param a
159     * @param b
160     * @return
161     */
162   private def slowEquals(a: Array[Byte], b: Array[Byte]): Boolean = {
163     val range = 0 until scala.math.min(a.length, b.length)
164     val diff = range.foldLeft(a.length ^ b.length) {
165       case (acc, i) => acc | a(i) ^ b(i)
166     }
167     diff == 0
168   }
169 
170   /**
171     * crypto magic
172     *
173     * @param password
174     * @param salt
175     * @param iterations
176     * @param bytes
177     * @return
178     */
179   private def pbkdf2(password: Array[Char], salt: Array[Byte], iterations: Int, bytes: Int): Array[Byte] = {
180     val spec = new PBEKeySpec(password, salt, iterations, bytes * 8)
181     val skf = SecretKeyFactory.getInstance(PBKDF2_ALGORITHM)
182     skf.generateSecret(spec).getEncoded()
183   }
184 
185   /**
186     *
187     * @param hex
188     * @return
189     */
190   private def fromHex(hex: String): Array[Byte] = {
191     hex.sliding(2, 2).toArray.map(Integer.parseInt(_, 16).toByte)
192   }
