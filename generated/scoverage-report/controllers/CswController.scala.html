44 private class AddMDMetadataToInsert(xml: Node) extends RewriteRule {
45   override def transform(n: Node): Seq[Node] = n match {
46     case e: Elem if e.label == "Insert" => {
47       new Elem(e.prefix, "Insert", e.attributes, e.scope, e.minimizeEmpty, (xml).toSeq: _*)
48     }
49     case x => x
50   }
51 }

84     val validValuesOption = metadataService.getValidValuesFor(topic)
85     if (validValuesOption.isEmpty) {
86       //TODO SR should we have a generic error return object that we return as JSON?

94   /** calls CSW:Transaction -> Insert
95     * FIXME should implement HasToken(parse.json) instead of Action.async, to only allow authenticated users
96     */
97   def insert: Action[AnyContent] = Action.async { request =>
98     logger.debug(request.body.asJson.toString)
99 
100     //TODO parse JSON to MDMetadataSet and convert that to XML
101     if (request.body.asJson.isEmpty) {
102       BadRequest("Posted data was no valid JSON document")
103     }
104 
105     val mdMetadata = MdMetadata.fromJson(((request.body.asJson.get) \ "metadata").get)
106 
107     val futureResponse: Future[WSResponse] = for {
108       getCapaResponse <- wsClient.url(CSW_OPERATIONS_METADATA_URL).get()
109       insertResponse <- {
110         //check if getCapaResponse indicates, that the CSW can perform Transactions
111         val operation = (getCapaResponse.xml \\ "OperationsMetadata" \\ "Operation").filter(node => {
112           logger.debug(s"Attribute 'name'=${node.attribute("name").get.text.toString}")
113           node.attribute("name").get.text.equals("Transaction")
114         })
115         logger.debug(s"operation = ${operation.toString()}")
116         if (operation.isEmpty) {
117           throw new UnsupportedOperationException("CSW does not support Transaction.")
118         }
119 
120         //insert MDMEtadata in insert template
121         logger.debug(s"MD_MetadataXML: ${mdMetadata.get.toXml().toString()}")
122         val rule = new RuleTransformer(new AddMDMetadataToInsert(mdMetadata.get.toXml()))
123         val finalXML = rule.transform(cswtInsertXml)
124         logger.debug(s"finalXml: ${finalXML.toString()}")
125         wsClient.url(CSW_URL).post(finalXML.toString())
126       }
127     } yield insertResponse
128 
129     futureResponse.recover {
130       case e: Exception =>
131         logger.error("Insert CSW threw exception", e)
132         InternalServerError(e.getMessage)
133     }
134 
135     futureResponse.map(response => {
136       logger.debug(response.xml.toString())
137       response.xml match {
138         case e: Elem if e.label == "TransactionResponse" => {
139           val fileIdentifier = (e \\ "InsertResult" \\ "BriefRecord" \\ "identifier").text
140           //TODO collectionsService.addEntryToUserDefaultCollection(mdMetadata, cachedSecUser)
141           Ok(Json.obj("type" -> "success", "fileIdentifier" -> fileIdentifier,
142             "message" -> s"Inserted as ${fileIdentifier}."))
143         }
144         case e: Elem if e.label == "ExceptionReport" => {
145           val message = (e \\ "Exception" \\ "ExceptionText").text
146           //TODO SR make this InternalServerError
147           Ok(Json.obj("type" -> "danger", "message" -> message))
148         }
149         case _ => InternalServerError(s"Unexpected Response: ${response}")
150       }
151     })
152   }
