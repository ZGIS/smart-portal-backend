27 /**
28   * Model of OWS Context Documents and provide GeoJson encoding thereof (and maybe AtomXML)
29   * An OWC document is an extended FeatureCollection, where the features (aka entries) hold a variety of metadata
30   * about the things they provide the context for (i.e. other data sets, services, metadata records)
31   * OWC documents do not duplicate a CSW MD_Metadata record, but a collection of referenced resources;
32   *
33   * http://www.opengeospatial.org/standards/owc
34   *
35   * Classically, the WMC documents (Web Map Context documents) were a list of WMS layers for a web map viewer
36   * with a certain context, i.e. title, Bounding Box and a few visualisation properties like scale/zoom,
37   * OWC has superseded that concept into a generic collection of resources:
38   *
39   * We use OWC primarily in the form of collections of case studies, preferably with at least two offerings per entry:
40   * 1) a web visualisable form, e.g.WMS, WFS, SOS ...
41   * 2) a CSW addressable MD_Metadata record according to the resource
42   *
43   * The OWC JSON Encoding is a profile of GeoJSON FeatureCollection, the XML encoding is a profile of Atom/GeoRSS Feed
44   */
45

100 /**

102   * which should correspond to the offering type code (e.g. WMS, WFS ..)
103   *
104   * @param code        operation code, e.g. GetCapabilities
105   * @param method      GET, POST ...
106   * @param contentType e.g. "application/xml", for expected return type (accept header?)

108   * @param request     only need to hold data when method is POST
109   * @param result      could hold inline result of the request, not sure if we need
110   */

126 /**
127   * companion object for [[OwcOperation]]
128   */
129 object OwcOperation extends ClassnameLogger {
130   /**
131     *
132     * @param jsonString
133     * @return
134     */
135   def parseJson(jsonString: String): Option[OwcOperation] = parseJson(Json.parse(jsonString))
136 
137   /**
138     *
139     * @param json
140     * @return
141     */
142   def parseJson(json: JsValue): Option[OwcOperation] = {
143     val resultFromJson: JsResult[OwcOperation] = Json.fromJson[OwcOperation](json)
144     resultFromJson match {
145       case JsSuccess(r: OwcOperation, path: JsPath) => Some(r)
146       case e: JsError => None
147     }
148   }
149 }
