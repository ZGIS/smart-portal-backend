79     * @return
80     */
81   def queryActiveSessions(token: Option[String], max: Option[Int], email: Option[String]): Seq[UserSession] = {
82     dbSession.viaConnection{ implicit connection =>
83 
84       val count = max.getOrElse(100)
85       (token, email) match {
86         case (Some(to), Some(em)) =>
87           val viaToken = UserSession.findUserSessionByToken(to, count).toSeq
88           val viaEmail = UserSession.findUserSessionByEmail(em, count)
89           (viaToken ++ viaEmail).filter(ul => ul.token.contains(to) && ul.email.contains(em))
90         case (Some(to), None) =>
91           UserSession.findUserSessionByToken(to, count).toSeq
92         case (None, Some(em)) =>
93           UserSession.findUserSessionByEmail(em, count)
94         case (None, None) =>
95           UserSession.getAllUserSessions(count)
96       }
97     }
98   }

100   def blockUnblockUsers(command: String, email: String): Option[User] = {
101     val userOpt = dbSession.viaConnection( implicit connection => {
102       UserDAO.findUserByEmailAsString(email)
103     })
104     userOpt.fold[Option[User]]{
105       None
106     } {
107       user =>
108         val statusToken = if (command.equalsIgnoreCase("BLOCK")) StatusToken.BLOCKED else StatusToken.ACTIVE
109         val updateUser = User(
110           user.email,
111           user.accountSubject,
112           user.firstname,
113           user.lastname,
114           "***",
115           s"${statusToken.value}:BY-ADMIN",
116           ZonedDateTime.now.withZoneSameInstant(ZoneId.of(appTimeZone)))
117 
118         // result of Option can be evaluated upstream in controller
119         dbSession.viaTransaction { implicit connection =>
120           UserDAO.updateNoPass(updateUser)
121         }
122 
123     }
124   }
