84     * Loads a time series from a SOS server. A Timeseries "configuration" has to be passd within the POST

89   def getTimeseries: Action[JsValue] = optionalAuthenticationAction.async(parse.json) {
90     request =>
91       request.body.validate[Timeseries].fold(
92         errors => {
93           logger.error(JsError.toJson(errors).toString())
94           val error = ErrorResult("Error while validating request.", Some(JsError.toJson(errors).toString))
95           Future {
96             BadRequest(Json.toJson(error)).as(JSON)
97           }
98         },
99 
100         timeseries => {
101           val sosXmlRequest = getObservationRequestXml(timeseries)
102           val responseFuture = wsClient.url(timeseries.sosUrl)
103             .withHeaders(CONTENT_TYPE -> "application/xml")
104             .post(sosXmlRequest)
105 
106           responseFuture.recover {
107             case e: Exception =>
108               logger.error("Sos GetObservation threw exception", e)
109               val error = ErrorResult(s"Exception (${e.getClass.getCanonicalName}) on SOS get capabilities.",
110                 Some(e.getMessage))
111               InternalServerError(Json.toJson(error)).as(JSON)
112           }
113 
114           responseFuture.map { response =>
115             if (response.status != 200) {
116               logger.error(s"Calling SOS ${timeseries.sosUrl} HTTP result status ${response.status} on GetObservation")
117               val error = ErrorResult(s"Server ${timeseries.sosUrl} responded with ${response.status} on GetObservation",
118                 Some(response.body))
119               InternalServerError(Json.toJson(error)).as(JSON)
120             }
121             else if (!response.header("Content-Type").get.contains("application/xml")) {
122               logger.error(s"SOS GetObservation result was not XML but ${response.header("Content-Type").get}")
123               val error = ErrorResult(
124                 s"SOS GetObservation result was not XML but ${response.header("Content-Type").get}",
125                 Some(response.body))
126               InternalServerError(Json.toJson(error)).as(JSON)
127             }
128             else if (timeseries.responseFormat.isDefined && !timeseries.responseFormat.get.equalsIgnoreCase("http://www.opengis.net/om/2.0")) {
129               logger.error(s"SOS GetObservation responseFormat unsupported for this request; ${timeseries.responseFormat.getOrElse("none")}")
130               val error = ErrorResult(
131                 s"SOS GetObservation responseFormat unsupported for this request; ${timeseries.responseFormat.getOrElse("none")}",
132                 Some(response.body))
133               InternalServerError(Json.toJson(error)).as(JSON)
134             }
135             else {
136               val tvp = new XmlTvpParser().parseOm2Measurements(Source.fromString(response.body))
137               val tsdata =
138                 if (tvp.isEmpty) {
139                   TimeseriesData(
140                     x = Seq.empty[String],
141                     y = Seq.empty[String],
142                     name = timeseries.timeseriesName
143                   )
144                 }
145                 else {
146                   TimeseriesData(
147                     x = tvp.map(_.datetime),
148                     y = tvp.map(_.measValue),
149                     name = timeseries.timeseriesName
150                   )
151                 }
152               val uom = if (tvp.isEmpty) {
153                 if (timeseries.uom.isEmpty) {
154                   timeseries.observedProperty
155                 } else {
156                   timeseries.uom.getOrElse("empty")
157                 }
158               } else tvp.head.measUnit
159               val result = timeseries.copy(data = Some(tsdata), uom = Some(uom))
160 
161               val logRequest = UserLinkLogging(id = None,
162                 timestamp = ZonedDateTime.now.withZoneSameInstant(ZoneId.of(appTimeZone)),
163                 ipaddress = Some(request.remoteAddress),
164                 useragent = request.headers.get(UserAgentHeader),
165                 email = request.optionalSession.map(_.email),
166                 link = sosXmlRequest,
167                 referer = request.headers.get(RefererHeader))
168 
169               val updated = userService.logLinkInfo(logRequest)
170 
171               logger.trace(logRequest.toString)
172               Ok(result.toJson()).as(JSON)
173             }
174           }
175         }
176       )
177   }

297                       .withHeaders("x-filename" -> fileFile.getName))
298                 }
299 
300                 case Left(errorResult) => InternalServerError(Json.toJson(errorResult)).as(JSON)
301               }
302             }
303           })
304         }

309     * a bit of a crippling nested fold but the write-out as file for xls/csv only does Unit return,
310     * thought it would be nice to add some IO security around it

312     * @param responseFormat
313     * @param sosResponseAsText
314     * @param sosCapabilities
315     * @param sosXmlRequest
316     * @return
317     */
318   def exportToFile(responseFormat: Option[String], sosResponseAsText: String,
319                    sosCapabilities: SosCapabilities, sosXmlRequest: String): Either[ErrorResult, File] = {
320 
321     val updatedTime = OffsetDateTime.now(ZoneId.of(appTimeZone))
322     val fileNameTmpl = "export-" + Try(URLEncoder.encode(sosCapabilities.title.replace(" ", "_"), "UTF-8")
323       + "-" + updatedTime.format(formatter)).getOrElse("-sosdata")
324     val pathOfUploadTmp = Paths.get(uploadDataPath)
325     val intermTempDir = Files.createTempDirectory(pathOfUploadTmp, "sos-export-")
326 
327     responseFormat match {
328       case Some(WML2.value) => {
329         val wml2 = wml2Exporter.getWml2ExportFromSosGetObs(sosResponseAsText, sosCapabilities, sosXmlRequest)
330 
331         if (wml2.isEmpty) {
332           logger.error("Couldn't extract OM2 into SpreadSheet from this SOS GetObservation response")
333           Left(ErrorResult(message = "Couldn't extract OM2 into SpreadSheet from this SOS GetObservation response",
334             details = Some("Sheet is empty, no data?")))
335         } else {
336           val tmpFile = new File(intermTempDir.resolve(fileNameTmpl + WML2.fileSuffix).toAbsolutePath.toString)
337           scala.xml.XML.save(tmpFile.getAbsolutePath, wml2.get.head, "UTF-8", true, null)
338           // return ref handle
339           Right(tmpFile)
340         }
341       }
342       case Some(spreadSheet) if spreadSheet.equals(XLS.value) || spreadSheet.equals(CSV.value) => {
343         val format = SosDataFormat.apply(spreadSheet)
344         val tmpFile = new File(intermTempDir.resolve(fileNameTmpl + format.fileSuffix).toAbsolutePath.toString)
345         val sheets = xlsExporter.getSpreadsheetExportFromSosGetObs(sosResponseAsText, sosCapabilities, sosXmlRequest)
346         val saved = saveSpreadSheet(sheets, format, tmpFile.getAbsolutePath)
347         saved.map(err => Left(err)).getOrElse(Right(tmpFile))
348       }
349       case Some(otherText) =>
350         logger.error(s"Couldn't extract OM2 into SpreadSheet from this SOS GetObservation response, unsupported fomat:? $otherText")
351         Left(ErrorResult(message = s"Couldn't extract OM2 into SpreadSheet from this SOS GetObservation response, unsupported fomat:? $otherText", details = None))
352     }
353   }

367       } {

371             // TODO here add spreadsheet with metadata export info
372             wb.saveAsXlsx(filename)
373             None
374           } else if (format.equals(CSV)) {

380           } else {
381             logger.error("Couldn't extract OM2 into SpreadSheet from this SOS GetObservation response")
382             Some(ErrorResult(
383               message = "Couldn't extract OM2 into SpreadSheet from this SOS GetObservation response",
384               details = Some(s"Unsupported format $format?")))
385           }
386       }
387     } catch {
388       case ex: Exception =>
389         logger.error(s"Couldn't extract OM2 into SpreadSheet from this SOS GetObservation response, ${ex.getLocalizedMessage}")
390         Some(ErrorResult(message = "Couldn't extract OM2 into SpreadSheet from this SOS GetObservation response", details = Some(ex.getLocalizedMessage)))
