51     *
52     * @param source
53     * @return
54     */
55   def parseOm2Measurements(source: Source): Seq[TimeValuePair] = {
56 
57     val eventReader = new XMLEventReader(source)
58 
59     // phentime = "time";
60     var datetime: String = null
61     // GMLID = "identifier";
62     var foiId: String = null
63     // UOM = "uom";
64     var measUnit: String = null
65     // result = "value";
66     var measValue: String = null
67     // OBSPROP = "observedProperty";
68     var geom: Option[String] = None
69     // SFPOS = "pos";
70     var obsProp: String = null
71     // PROC = "procedure";
72     var procedure: String = null
73 
74     val tvps = ArrayBuffer[TimeValuePair]()
75 
76     for (event <- eventReader) {
77       event match {
78         case EvElemStart(prefix, label, attributes, ns) => {
79           if (label == OMOBS) {
80             logger.trace(s"$prefix:$label start")
81             // new obs
82             obsProp = "" // OBSPROP = "observedProperty";
83             foiId = "" // GMLID = "identifier";
84             geom = None // SFPOS = "pos";
85             measUnit = "" // UOM = "uom";
86           }
87           if (label == GMLID) {
88             // this would be the observation id of the enclosing OM_Observation
89             val next = eventReader.next().asInstanceOf[EvText]
90             logger.trace(s"$prefix:$label ${next.text}")
91           }
92 
93           if (label == OBSPROP) {
94             val att = extractPrefixedAtrrib(attributes, X_HREF)
95             logger.trace(s"$prefix:$label $att")
96             obsProp = att
97           }
98           if (label == PROC) {
99             val att = extractPrefixedAtrrib(attributes, X_HREF)
100             logger.trace(s"$prefix:$label $att")
101             procedure = att
102           }
103           if (label == TIMEPOS) {
104             val next = eventReader.next().asInstanceOf[EvText]
105             logger.trace(s"$prefix:$label ${next.text}")
106             datetime = next.text
107           }
108           if (label == FOI) {
109             // should check if actual geometry is in here or an xlink:href
110             val att = extractPrefixedAtrrib(attributes, X_HREF)
111             logger.trace(s"$prefix:$label $att")
112             foiId = att
113           }
114           if (label == SFPOS) {
115             val next = eventReader.next().asInstanceOf[EvText]
116             logger.trace(s"$prefix:$label ${Some(next.text)}")
117             geom = Some(next.text)
118           }
119           if (label == RESULT) {
120             val next = eventReader.next().asInstanceOf[EvText]
121             measUnit = extractAttrib(attributes, UOM)
122             logger.trace(s"$prefix:$label ${next.text} ${extractAttrib(attributes, UOM)}")
123             measValue = next.text
124           }
125         }
126         // If we reach the end of an item element, we add it to the list
127         case EvElemEnd(prefix, label) => {
128           if (label == OMOBS) {
129             logger.debug(s"$prefix:$label end")
130             // full build and add
131             val tvp = TimeValuePair(
132               datetime,
133               foiId,
134               measUnit,
135               measValue,
136               geom,
137               obsProp,
138               procedure
139             )
140             logger.trace(tvp.toJson.toString())
141             tvps.append(tvp)
142           }
143         }
144         case EvText(t) => {
145           if (t.trim.size > 0) {
146             logger.debug(s"missed text: $t ")
147           }
148         }
149         case _ => // ignore
150       }
151     }
152 
153     tvps.toSeq
154   }

156   /**
157     *
158     * @param attrs
159     * @param prefixedLabel
160     * @return
161     */
162   def extractPrefixedAtrrib(attrs: Iterable[MetaData], prefixedLabel: String): String = {
163     val attOpt = attrs.find(meta => meta.prefixedKey.equalsIgnoreCase(prefixedLabel))
164     attOpt.map(meta => meta.value.text).getOrElse("")
165   }
166 
167   /**
168     *
169     * @param attrs
170     * @param label
171     * @return
172     */
173   def extractAttrib(attrs: Iterable[MetaData], label: String): String = {
174     val attOpt = attrs.find(meta => meta.key.equalsIgnoreCase(label))
175     attOpt.map(meta => meta.value.text).getOrElse("")
176   }
