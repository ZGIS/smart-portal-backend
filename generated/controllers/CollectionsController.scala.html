317               collectionsService.updateCollection(updated, request.user)

320             val error: ErrorResult = ErrorResult("database update for resource failed", None)
321             BadRequest(Json.toJson(error)).as(JSON)
322           } {
323             theDoc => {
324               val resourceTest = theDoc.resource.find(r => r.id.equals(owcResource.id)).exists(
325                 tResource => tResource.equals(owcResource))
326               if (resourceTest) {
327                 Ok(Json.obj("message" -> "owcResource replaced in owcContext",
328                   "document" -> theDoc.toJson, "entry" -> owcResource.toJson))
329               }
330               else {
331                 val error: ErrorResult = ErrorResult("owcResource could not be replaced in owcContext", None)
332                 BadRequest(Json.toJson(error)).as(JSON)
