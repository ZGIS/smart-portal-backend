28 /**
29   * Security actions that should be used by all controllers that need to protect their actions.
30   * Can be composed to fine-tune access control.
31   *
32   * from https://github.com/mariussoutier/play-angular-require-seed/blob/master/app/controllers/Security.scala
33   */
34 trait Security { self: Controller =>
35 
36   val cache: CacheApi
37   val configuration: Configuration
38 
39   val AuthTokenCookieKey = "XSRF-TOKEN"
40   val AuthTokenHeader = "X-XSRF-TOKEN"
41   val AuthTokenUrlKey = "auth"
42 
43   val UserAgentHeader = "User-Agent"
44   val UserAgentHeaderDefault = "Default-UA/1.0"
45 
46   /**
47     * Checks that the token is:
48     * - present in the cookie header of the request,
49     * - either in the header or in the query string,
50     * - matches a token already stored in the play cache
51     *
52     * @param p
53     * @param f
54     * @tparam A
55     * @return
56     */
57   def HasToken[A](p: BodyParser[A] = parse.anyContent)(
58     f: String => String => Request[A] => Result): Action[A] =
59   Action(p) { implicit request =>
60 
61     request.cookies.get(AuthTokenCookieKey).fold {
62       Unauthorized(Json.obj("status" -> "ERR", "message" -> "Invalid XSRF Token cookie"))
63     } { xsrfTokenCookie =>
64       // Logger.debug(s"cookie ${xsrfTokenCookie.value}")
65       val maybeToken = request.headers.get(AuthTokenHeader).orElse(request.getQueryString(AuthTokenUrlKey))
66 
67       maybeToken flatMap { token =>
68         // ua needed to differentiate between different devices/sessions
69         val uaIdentifier: String = request.headers.get(UserAgentHeader).getOrElse(UserAgentHeaderDefault)
70 
71         // cache token -> maps to a String username
72         cache.get[String](token) map { username =>
73           lazy val passwordHashing = new PasswordHashing(configuration)
74           val cookieForUSerAndDevice = passwordHashing.testSessionCookie(token, username, uaIdentifier)
75           if (xsrfTokenCookie.value == token && cookieForUSerAndDevice) {
76             Logger.debug(s"request for active session: $username / $token / $uaIdentifier")
77             f(token)(username)(request)
78           } else {
79             Unauthorized(Json.obj("status" -> "ERR", "message" -> "Invalid Token"))
80           }
81         }
82       } getOrElse Unauthorized(Json.obj("status" -> "ERR", "message" -> "No Token"))
83     }
84   }
