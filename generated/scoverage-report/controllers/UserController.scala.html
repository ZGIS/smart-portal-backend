108               jsuser.username,

113               ZonedDateTime.now.withZoneSameInstant(ZoneId.of(appTimeZone)))

115             userDAO.createWithNps(newUser).fold {
116               logger.error("User create error.")
117               BadRequest(Json.obj("status" -> "ERR", "message" -> "User create error."))
118             } { user =>
119               logger.info("New user registered.")

123             // found user with that email already

321           BadRequest(Json.obj("status" -> "ERR", "message" -> "User not found."))

324           val updateUser = User(dbuser.email,
325             dbuser.username,
326             dbuser.firstname,
327             dbuser.lastname,
328             cryptPass,
329             "ACTIVE:PASSWORDUPDATE",
330             ZonedDateTime.now.withZoneSameInstant(ZoneId.of(appTimeZone)))
331 
332           userDAO.updatePassword(updateUser).fold {
333             logger.error("Password update error.")
334             BadRequest(Json.obj("status" -> "ERR", "message" -> "Password update error."))
335           } { user =>
336             val uaIdentifier: String = request.headers.get(UserAgentHeader).getOrElse(UserAgentHeaderDefault)
337             // logger.info(s"Logging in username from $uaIdentifier")
338             cache.remove(token)
339             val newtoken = passwordHashing.createSessionCookie(user.username, uaIdentifier)
340             cache.set(newtoken, user.username)
341             Ok(Json.obj("status" -> "OK", "token" -> newtoken, "username" -> user.username))
342               .withCookies(Cookie(AuthTokenCookieKey, newtoken, None, httpOnly = false))
343           }
344
