53   def sendRegistrationEmail(emailTo: String, subject: String, usernameTo: String, linkId: String): Boolean = {
54 
55     val emailText =
56       """Hello %s,
57         |thank you for registering on the GW HUB.
58         |Please click on the following link to confirm your registration:
59         |
60         |%s/api/v1/users/register/%s
61         |
62         |If you have any question please email us to %s.
63         |
64         |Your GW HUB Team
65       """.format(usernameTo, portalApiHost, linkId, emailFrom).stripMargin
66 
67     logger.debug(emailText)
68 
69     val from = new Email(emailFrom)
70     val to = new Email(emailTo)
71     val content = new Content("text/plain", emailText)
72     val mail = new Mail(from, subject, to, content)
73 
74     try {
75 
76       val request = new Request()
77       request.method = Method.POST
78       request.endpoint = "mail/send"
79       request.body = mail.build()
80       val response = sg.api(request)
81       logger.trace(s"mail api response status: ${response.statusCode}")
82       logger.trace(s"mail api response.body: ${response.body}")
83       logger.trace(s"response.headers: ${response.headers}")
84       logger.trace(s"mail api response status: ${response.statusCode}")
85 
86       true
87     } catch {
88       case ioex: IOException => {
89         logger.error("IO Messaging exception: " + ioex.getLocalizedMessage)
90         false
91       }
92       case e: Exception => {
93         logger.error("Other email problem: " + e.getLocalizedMessage)

95       }
96     }
