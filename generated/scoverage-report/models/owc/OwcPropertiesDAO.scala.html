49   /**
50     * Parse a OwcAuthor from a ResultSet
51     */
52   val owcAuthorParser = {
53     str("uuid") ~
54       str("name") ~
55       get[Option[String]]("email") ~
56       get[Option[String]]("uri") map {
57       case uuid ~ name ~ email ~ uri =>
58         OwcAuthor(UUID.fromString(uuid), name, email, uri)
59     }
60   }

62     * Parse a OwcCategory from a ResultSet
63     */
64   val owcCategoryParser = {
65     str("uuid") ~
66       str("scheme") ~
67       str("term") ~
68       get[Option[String]]("label") map {
69       case uuid ~ scheme ~ term ~ label =>
70         OwcCategory(UUID.fromString(uuid), scheme, term, label)
71     }

87   /**
88     * Parse a OwcProperties from a ResultSet
89     */
90   val owcPropertiesParser: RowParser[OwcProperties] = {
91     str("uuid") ~
92       str("language") ~
93       str("title") ~
94       get[Option[String]]("subtitle") ~
95       get[Option[ZonedDateTime]]("updated") ~
96       get[Option[String]]("generator") ~
97       get[Option[String]]("rights") ~
98       get[Option[String]]("creator") ~
99       get[Option[String]]("publisher") map {
100 
101       case uuid ~ language ~ title ~ subtitle ~ updated ~ generator ~ rights ~ creator ~ publisher =>
102         OwcProperties(
103           UUID.fromString(uuid),
104           language,
105           title,
106           subtitle,
107           updated,
108           generator,
109           rights,
110           findOwcAuthorsByPropertiesUUID(UUID.fromString(uuid)).toList,
111           findOwcAuthorsAsContributorsByPropertiesUUID(UUID.fromString(uuid)).toList,
112           creator,
113           publisher,
114           findOwcCategoriesByPropertiesUUID(UUID.fromString(uuid)).toList,
115           findOwcLinksByPropertiesUUID(UUID.fromString(uuid)).toList
116         )
117     }

473   /**
474     * create owc properties in the database, title has additionally unique constraint, uuid will be unique and primary key
475     *
476     * @param owcProperties
477     * @return
478     */
479   def createOwcProperties(owcProperties: OwcProperties): Option[OwcProperties] = {
480 
481     db.withTransaction {
482       implicit connection => {
483 
484         val rowCount = SQL(
485           s"""
486           insert into $tableOwcProperties values (
487             {uuid}, {language}, {title}, {subtitle}, {updated}, {generator}, {rights}, {creator}, {publisher}
488           )
489         """).on(
490           'uuid -> owcProperties.uuid.toString,
491           'language -> owcProperties.language,
492           'title -> owcProperties.title,
493           'subtitle -> owcProperties.subtitle,
494           'updated -> owcProperties.updated,
495           'generator -> owcProperties.generator,
496           'rights -> owcProperties.rights,
497           'creator -> owcProperties.creator,
498           'publisher -> owcProperties.publisher
499         ).executeUpdate()
500 
501         owcProperties.authors.foreach {
502           author => {
503             if (findOwcAuthorByUuid(author.uuid).isEmpty) {
504               createOwcAuthor(author)
505             }
506 
507             SQL(
508               s"""insert into $tableOwcPropertiesHasOwcAuthors values (
509                  |{owc_properties_uuid}, {owc_authors_uuid}
510                  |)
511                """.stripMargin).on(
512               'owc_properties_uuid -> owcProperties.uuid.toString,
513               'owc_authors_uuid -> author.uuid.toString
514             ).executeUpdate()
515           }
516         }
517 
518         owcProperties.contributors.foreach {
519           authorAsContributor => {
520             if (findOwcAuthorByUuid(authorAsContributor.uuid).isEmpty) {
521               createOwcAuthor(authorAsContributor)
522             }
523 
524             SQL(
525               s"""insert into $tableOwcPropertiesHasOwcAuthorsAsContributors values (
526                  |{owc_properties_uuid}, {owc_authors_as_contributors_uuid}
527                  |)
528                """.stripMargin).on(
529               'owc_properties_uuid -> owcProperties.uuid.toString,
530               'owc_authors_as_contributors_uuid -> authorAsContributor.uuid.toString
531             ).executeUpdate()
532           }
533         }
534 
535         owcProperties.categories.foreach {
536           category => {
537             if (findOwcCategoriesByUuid(category.uuid).isEmpty) {
538               createOwcCategory(category)
539             }
540 
541             SQL(
542               s"""insert into $tableOwcPropertiesHasOwcCategories  values (
543                  |{owc_properties_uuid}, {owc_categories_uuid}
544                  |)
545                """.stripMargin).on(
546               'owc_properties_uuid -> owcProperties.uuid.toString,
547               'owc_categories_uuid -> category.uuid.toString
548             ).executeUpdate()
549           }
550         }
551 
552         owcProperties.links.foreach {
553           owcLink => {
554             if (findOwcLinksByUuid(owcLink.uuid).isEmpty) {
555               createOwcLink(owcLink)
556             }
557 
558             SQL(
559               s"""insert into $tableOwcPropertiesHasOwcLinks  values (
560                  |{owc_properties_uuid}, {owc_links_uuid}
561                  |)
562                """.stripMargin).on(
563               'owc_properties_uuid -> owcProperties.uuid.toString,
564               'owc_links_uuid -> owcLink.uuid.toString
565             ).executeUpdate()
566           }
567         }
568 
569         rowCount match {
570           case 1 => Some(owcProperties)
571           case _ => None
572         }
573       }
574     }
575   }

739       }
740     }

743       implicit connection => {
744         (owcProperties.authors ++ owcProperties.contributors).distinct.filter {
