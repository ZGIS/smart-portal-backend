59                       (f: String => String => Request[A] => Future[Result]): Action[A] =
60     Action.async(p) { implicit request =>

63 
64         request.cookies.get(AuthTokenCookieKey).fold {
65           logger.error(request.cookies.toString);
66           Future[Result] {Unauthorized(Json.obj("status" -> "ERR", "message" -> "Invalid XSRF Token cookie"))}
67         } { xsrfTokenCookie =>
68           logger.trace(s"cookie ${xsrfTokenCookie.value}")
69           val maybeToken = request.headers.get(AuthTokenHeader).orElse(request.getQueryString(AuthTokenUrlKey))
70 
71           maybeToken flatMap { token =>
72             // ua needed to differentiate between different devices/sessions
73             val uaIdentifier: String = request.headers.get(UserAgentHeader).getOrElse(UserAgentHeaderDefault)
74             logger.trace(s"token: $token")
75             logger.trace(s"ua: $uaIdentifier")
76             // cache token -> maps to a String email
77             cache.get[String](token) map { email =>
78               // lazy val passwordHashing = new PasswordHashing(configuration)
79               val cookieForUSerAndDevice = passwordHashing.testSessionCookie(token, email, uaIdentifier)
80               logger.trace(s"testcookie: $cookieForUSerAndDevice")
81               if (xsrfTokenCookie.value == token && cookieForUSerAndDevice) {
82                 logger.trace(s"request for active session: $email / $token / $uaIdentifier")
83                 f(token)(email)(request)
84               }
85               else {
86                 Future[Result] {Unauthorized(Json.obj("status" -> "ERR", "message" -> "Invalid Token"))}
87               }
88             }
89           } getOrElse {
