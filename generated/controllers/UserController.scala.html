327   def updatePassword(): Action[JsValue] = defaultAuthAction(parse.json) {
328     request =>
329       request.body.validate[PasswordUpdateCredentials].fold(
330         errors => {
331           logger.error(JsError.toJson(errors).toString())
332           val error = ErrorResult("Could not validate request.", Some(JsError.toJson(errors).toString()))
333           BadRequest(Json.toJson(error)).as(JSON)
334         },
335         valid = credentials => {
336           // find user in db
337           val dbuser = request.user
338           // compare password
339           if (passwordHashing.validatePassword(credentials.oldPassword, dbuser.password)) {
340             val newCryptPass = passwordHashing.createHash(credentials.newPassword)
341             val updateUser = dbuser.copy(
342               password = newCryptPass,
343               laststatustoken = s"${StatusToken.ACTIVE}:PASSWORDUPDATE",
344               laststatuschange = ZonedDateTime.now.withZoneSameInstant(ZoneId.of(portalConfig.appTimeZone)))
345 
346             userService.updatePassword(updateUser).fold {
347               logger.error("Password update error.")
348               val error = ErrorResult("Password update error.", None)
349               BadRequest(Json.toJson(error)).as(JSON)
350             } { user =>
351               val uaIdentifier: String = request.headers.get(UserAgentHeader).getOrElse(UserAgentHeaderDefault)
352               // cache.remove(token)
353               userService.removeUserSessionCache(request.user.email, request.authenticatedRequest.userSession.token)
354               emailService.sendPasswordUpdateEmail(user.email, "Password Update on GW HUB", user.firstname)
355               // val newtoken = passwordHashing.createSessionCookie(user.email, uaIdentifier)
356               // cache.set(newtoken, user.email.value)
357               val newtoken = userService.upsertUserSession(user.email.value, uaIdentifier)
358               Ok(Json.obj("status" -> "OK", "token" -> newtoken, "email" -> user.email.value))
359                 .withCookies(Cookie(AuthTokenCookieKey, newtoken, None, httpOnly = false))
360             }
361           } else {
362             logger.error("Password update error, not matching.")
363             val error = ErrorResult("Password update error, not matching.", None)
364             BadRequest(Json.toJson(error)).as(JSON)
365           }
366         })
367   }

483     request.body.validate[GAuthCredentials].fold(
484       errors => {
485         logger.error(JsError.toJson(errors).toString())
486         val error = ErrorResult("Could not validate request.", Some(JsError.toJson(errors).toString()))
487         BadRequest(Json.toJson(error)).as(JSON)
488       },
489       valid = gAuthCredentials => {
490         val accessTokenResult = googleService.getGoogleAuthorization(gAuthCredentials)
491         accessTokenResult match {
492           case Left(errorResult) => {
493             logger.error(errorResult.message + errorResult.details.mkString)
494             InternalServerError(Json.toJson(errorResult)).as(JSON)
495           }
496           case Right(googleTokenResponse) => {
497             // Get profile info from ID token
498             val gAuthPayload = googleTokenResponse.parseIdToken().getPayload
499             // Use this value as a key to identify a user.
500             val userId = gAuthPayload.getSubject()
501             val email = gAuthPayload.getEmail()
502             val emailVerified = gAuthPayload.getEmailVerified()
503             // get further payload things that is easy in java but needs to work in scala too
504             val name = Option(gAuthPayload.get("name").asInstanceOf[String])
505             val familyName = Option(gAuthPayload.get("family_name").asInstanceOf[String])
506             val givenName = Option(gAuthPayload.get("given_name").asInstanceOf[String])
507             if ((gAuthCredentials.accesstype.equalsIgnoreCase("LOGIN") || gAuthCredentials.accesstype.equalsIgnoreCase("REGISTER")) &&
508               emailVerified && EmailAddress.isValid(email)) {
509 
510               val emailAddress = EmailAddress(email)
511               // viaConnection lookup email block
512               val firstUserLookup = userService.findUserByEmailAddress(emailAddress)
513 
514               firstUserLookup.fold {
515                 // 'error, no user': well, must be a new user coming through from Google
516                 val cryptPass = passwordHashing.createHash(UUID.randomUUID().toString)
517 
518                 // Use access token to call API for refresh or check-up?
519                 // val googleAccessToken = googleTokenResponse.getAccessToken()
520                 // val googleCredential = new GoogleCredential().setAccessToken(googleAccessToken)
521                 val newUser = User(emailAddress,
522                   s"google:${userId}",
523                   givenName.getOrElse(name.getOrElse(emailAddress.mailbox.value)),
524                   familyName.getOrElse(""),
525                   cryptPass,
526                   s"${StatusToken.ACTIVE}:REGCONFIRMED",
527                   ZonedDateTime.now.withZoneSameInstant(ZoneId.of(portalConfig.appTimeZone)))
