29 /**
30  * This controller creates an `Action` that demonstrates how to write
31  * simple asynchronous code in a controller. It uses a timer to
32  * asynchronously delay sending a response for 1 second.
33  *
34  * @param actorSystem We need the `ActorSystem`'s `Scheduler` to
35  * run code after a delay.
36  * @param exec We need an `ExecutionContext` to execute our
37  * asynchronous code.
38  */
39 @Singleton
40 class AsyncController @Inject() (actorSystem: ActorSystem)(implicit exec: ExecutionContext) extends Controller {
41 
42   /**
43    * Create an Action that returns a plain text message after a delay
44    * of 1 second.
45    *
46    * The configuration in the `routes` file means that this method
47    * will be called when the application receives a `GET` request with
48    * a path of `/message`.
49    */
50   def message = Action.async {
51     getFutureMessage(1.second).map { msg => Ok(msg) }
52   }
53 
54   private def getFutureMessage(delayTime: FiniteDuration): Future[String] = {
55     val promise: Promise[String] = Promise[String]()
56     actorSystem.scheduler.scheduleOnce(delayTime) { promise.success("Hi!") }
57     promise.future
58   }
59
