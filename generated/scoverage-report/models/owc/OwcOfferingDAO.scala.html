182   /**
183     * pre Check for OwcOperations Update For an OwcOffering, will delete, update and insert
184     * required operations in order to make OwcOffering Update consistent
185     *
186     * @param owcOffering
187     * @return
188     */
189   private def preUpdateCheckOperationsForOffering(owcOffering: OwcOffering)(implicit connection: Connection): Boolean = {
190 
191     // get current list,
192     val current: List[UUID] = owcOffering.operations.map(_.uuid)
193 
194     // get old list,
195     val old: List[UUID] = findOwcOfferingByUuid(owcOffering.uuid)
196       .map(o => o.operations.map(_.uuid)).getOrElse(List())
197 
198     // in old but not current -> delete
199     val toBeDeleted = old.diff(current)
200     val deleted = owcOffering.operations.filter(o => toBeDeleted.contains(o.uuid))
201       .map(OwcOperationDAO.deleteOwcOperation(_))
202       .count(_ == true) == toBeDeleted.length
203 
204     // in both lists -> update
205     val toBeUpdated = current.intersect(old)
206     val updated = owcOffering.operations.filter(o => toBeUpdated.contains(o.uuid))
207       .map(OwcOperationDAO.updateOwcOperation(_))
208       .count(_.isDefined) == toBeUpdated.length
209 
210     // in current but not in old -> insert
211     val toBeInserted = current.diff(old)
212     val inserted = owcOffering.operations.filter(o => toBeInserted.contains(o.uuid))
213       .map(OwcOperationDAO.createOwcOperation(_))
214       .count(_.isDefined) == toBeInserted.length
215 
216     if (deleted && updated && inserted) {
217       true
218     } else {
219       logger.error(s"(updateOffering/operation) one of deleted: $deleted , updated: $updated , inserted: $inserted not complete, recommend abort")
220       false
221     }
222   }

398   /**
399     * finds owc props like author, category, ink, content, stylesets, operation relation via its uuids
400     * obviously this could be further generalised
401     *
402     * @param propertiesUuid A String, which may contain 0, 1 or more UUIDs, multiple UUIDs must be separated by ':' colons
403     * @param a              implicit type evidence
404     * @tparam A Type of OwcAuthor, OwcCategory, OwcLink, OwcContent, OwcStyleSet, OwcOperation
405     * @return Sequence of objects of type A if found, otherwise empty Seq
406     */
407   def findByPropertiesUUID[A](propertiesUuid: Option[String])(implicit a: A, connection: Connection): Seq[A] = {
408 
409     import utils.StringUtils._
410 
411     val values = propertiesUuid.map(_.split(":").toSeq).map(
412       potUuids => {
413         val uuids = potUuids.map(_.toUuidOption).filter(_.isDefined).map(_.get)
414         uuids.map { uid =>
415           a match {
416             case a: OwcAuthor => OwcAuthorDAO.findOwcAuthorByUuid(uid).asInstanceOf[Option[A]]
417             case a: OwcCategory => OwcCategoryDAO.findOwcCategoryByUuid(uid).asInstanceOf[Option[A]]
418             case a: OwcLink => OwcLinkDAO.findOwcLinkByUuid(uid).asInstanceOf[Option[A]]
419             case a: OwcContent => OwcContentDAO.findOwcContentByUuid(uid).asInstanceOf[Option[A]]
420             case a: OwcCreatorApplication => OwcCreatorApplicationDAO.findOwcCreatorApplicationByUuid(uid).asInstanceOf[Option[A]]
421             case a: OwcCreatorDisplay => OwcCreatorDisplayDAO.findOwcCreatorDisplayByUuid(uid).asInstanceOf[Option[A]]
422             case a: OwcStyleSet => OwcStyleSetDAO.findOwcStyleSetByUuid(uid).asInstanceOf[Option[A]]
423             case a: OwcOperation => OwcOperationDAO.findOwcOperationByUuid(uid).asInstanceOf[Option[A]]
424             case a: OwcOffering => findOwcOfferingByUuid(uid).asInstanceOf[Option[A]]
425             case _ => throw new InvalidClassException(s"The evidence parameter type ${a.getClass.getCanonicalName} not supported here")
426           }
427 
428         }.filter(_.isDefined).map(_.get)
429       }
430     )
431     values.getOrElse(Seq())
432   }
