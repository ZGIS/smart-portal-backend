29 /**
30   * This Class saves valid values for Metadata Editor Comboboxes
31   *
32   * @param standardValue index of standard value
33   * @param values        list of values
34   * @param descriptions  optional list of descriptions
35   */
36 
37 // TODO SR For more complicated things (like valid BBOXes or so) value needs to become an object See https://github.com/angular/angular/issues/4843#issuecomment-206583992
38 class ValidValues(
39                    val standardValue: Int,
40                    val values: List[String],
41                    val descriptions: Option[List[String]]
42                  ) {
43   require(values.nonEmpty, "values list must not be empty")
44   require(descriptions.isEmpty || descriptions.get.length == values.length,
45     "decriptions list must either be None or same length as values list")
46   require(standardValue >= 0 && standardValue < values.length, "standardValue must be within values list length")
47   require(values.distinct.size == values.size, "all values must be unique")
48   require(descriptions.isEmpty || (descriptions.get.distinct.size == descriptions.get.size),
49     "all descriptions must be unique")
50 
51 }
