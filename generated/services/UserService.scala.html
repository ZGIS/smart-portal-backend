41     * basic auth compare
42     *
43     * @param loginCredentials
44     * @return
45     */
46   def authenticateLocal(loginCredentials: LoginCredentials): Either[ErrorResult, User] = {
47 
48     val userOpt = dbSession.viaConnection { implicit connection =>
49       UserDAO.findUserByEmailAsString(loginCredentials.email).filter(user =>
50         passwordHashing.validatePassword(loginCredentials.password, user.password)
51       )
52     }
53     if (userOpt.isDefined) {
54       // here is correct auth branch
55       val user = userOpt.get
56       if (user.isBlocked) {
57         Left(ErrorResult("Your account was temporarily disabled. Please contact the administrator.", None))
58       } else {
59         Right(user)
60       }
61 
62     } else {
63       // should never indicate if user not found or just password wrong?
64       Left(ErrorResult("User email or password wrong.", None))
65     }
66   }

201     * check for an active server-side session
202     *
203     * @param xsrfToken
204     * @param xsrfTokenCookie
205     * @param userAgentHeader
206     * @return
207     */
208   def checkUserSessionCacheByToken(xsrfToken: String, xsrfTokenCookie: String, userAgentHeader: String): Option[String] = {
209 
210     // TODO here replace cache access with DB
211     // val userEmailOpt = cache.get[String](xsrfToken)
212     val userEmailOpt = dbSession.viaConnection(implicit connection => {
213       UserSession.findUserSessionByToken(xsrfToken)
214     })
215 
216     userEmailOpt.fold[Option[String]] {
217       logger.debug(s"no session for xsrfToken: $xsrfToken")
218       None
219     } {
220       session =>
221       val cookieForUSerAndUserAgent = passwordHashing.testSessionCookie(xsrfToken, session.email, userAgentHeader)
222       logger.trace(s"testcookie: $cookieForUSerAndUserAgent")
223       if (xsrfToken == xsrfTokenCookie && cookieForUSerAndUserAgent) {
224         logger.trace(s"request for active session: ${session.email} / $xsrfToken / $userAgentHeader")
225         Some(session.email)
226       } else {
227         logger.error("Invalid Token")
228         None
229       }
230     }
231   }
