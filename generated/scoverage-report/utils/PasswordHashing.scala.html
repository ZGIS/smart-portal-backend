30 /**
31   * provides reasonably current  and secure measures of password hashing as of 2015/2016
32   * This Scala implementation of password hashing was inspired by:
33   * https://github.com/dholbrook/scala-password-hash

42   val PBKDF2_ALGORITHM = "PBKDF2WithHmacSHA1"

58   /**
59     * create Session Cookie
60     *
61     * @param username
62     * @param useragent
63     * @return
64     */
65   def createSessionCookie(username: String, useragent: String): String = {
66     val in = appSecret + username + useragent.toUpperCase
67     val digest = MessageDigest.getInstance("SHA-512").digest(in.getBytes("UTF-8"))
68     // Logger.trace(s"createSessionCookie: ${toHex(digest)} app secret: $appSecret , username: $username, useragent: $useragent >>> $in")
69     toHex(digest)
70   }

72   /**
73     * test Session Cookie with token, username, uaIdentifier, based on validatePassword
74     *
75     * @param cookieHash
76     * @param username
77     * @param useragent
78     * @return
79     */
80   def testSessionCookie(cookieHash: String, username: String, useragent: String): Boolean = {
81     val in = appSecret + username + useragent.toUpperCase
82     val digestHash = MessageDigest.getInstance("SHA-512").digest(in.getBytes("UTF-8"))
83     val testHash = fromHex(cookieHash)
84     // Logger.trace(s"testSessionCookie: testhash ${toHex(testHash)} vs newdigest ${toHex(digestHash)} app secret: $appSecret cookieHash: $cookieHash, username: $username, $useragent: useragent  >>> $in")
85     slowEquals(digestHash, testHash)
86   }

106   /**
107     *
108     * @param password
109     * @return
110     */
111   def createHash(password: Array[Char]): String = {
112     val salt = nextRandomSalt
113     val hash = pbkdf2(password, salt, PBKDF2_ITERATIONS, HASH_BYTE_SIZE)
114     PBKDF2_ITERATIONS + ":" + toHex(salt) + ":" + toHex(hash)
115   }
116 
117   /**
118     *
119     * @return
120     */
121   private def nextRandomSalt(): Array[Byte] = {
122     val random = new SecureRandom()
123     val salt = Array.ofDim[Byte](SALT_BYTE_SIZE)
124     random.nextBytes(salt)
125     salt
126   }

134   def validatePassword(password: String, correctHash: String): Boolean = {

138   /**
139     * the work horse...
140     *
141     * @param password
142     * @param correctHash
143     * @return
144     */
145   def validatePassword(password: Array[Char], correctHash: String): Boolean = {
146     val params = correctHash.split(":")
147     val iterations = Integer.parseInt(params(ITERATION_INDEX))
148     val salt = fromHex(params(SALT_INDEX))
149     val hash = fromHex(params(PBKDF2_INDEX))
150     val testHash = pbkdf2(password, salt, iterations, hash.length)
151     slowEquals(hash, testHash)
152   }
153 
154   /**
155     * the other work horse
156     *
157     * @param a
158     * @param b
159     * @return
160     */
161   private def slowEquals(a: Array[Byte], b: Array[Byte]): Boolean = {
162     val range = 0 until scala.math.min(a.length, b.length)
163     val diff = range.foldLeft(a.length ^ b.length) {
164       case (acc, i) => acc | a(i) ^ b(i)
165     }
166     diff == 0
167   }
168 
169   /**
170     * crypto magic
171     *
172     * @param password
173     * @param salt
174     * @param iterations
175     * @param bytes
176     * @return
177     */
178   private def pbkdf2(password: Array[Char], salt: Array[Byte], iterations: Int, bytes: Int): Array[Byte] = {
179     val spec = new PBEKeySpec(password, salt, iterations, bytes * 8)
180     val skf = SecretKeyFactory.getInstance(PBKDF2_ALGORITHM)
181     skf.generateSecret(spec).getEncoded()
182   }
183 
184   /**
185     *
186     * @param hex
187     * @return
188     */
189   private def fromHex(hex: String): Array[Byte] = {
190     hex.sliding(2, 2).toArray.map(Integer.parseInt(_, 16).toByte)
191   }
