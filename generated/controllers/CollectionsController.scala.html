456   private def calculateKeywordScore(owcContext: OwcContext, keywords: Seq[String]): Double = {
457     val numQueryKeywords = keywords.length
458     val numContextKeywords = owcContext.keyword.length
459     val numResourcesKeywords = owcContext.resource.map(_.keyword.length).sum
460     val directContextHits = keywords.map(kw => owcContext.keyword.count(p => p.term.contentEquals(kw))).sum
461     val directResourceHits = keywords.map(
462       kw => owcContext.resource.map(res => res.keyword.count(p => p.term.contentEquals(kw))).sum).sum
463 
464     val closeContextHits = keywords.map(kw => owcContext.keyword.count(p => p.term.contains(kw))).sum
465     val closeResourceHits = keywords.map(
466       kw => owcContext.resource.map(res => res.keyword.count(p => p.term.contains(kw))).sum).sum
467 
468     ( (directContextHits / numContextKeywords) +
469       (closeContextHits * 0.3 / numContextKeywords) +
470       (directResourceHits / numResourcesKeywords) +
471       ((closeResourceHits / numResourcesKeywords) * 0.3)) / numQueryKeywords
472 
473   }
