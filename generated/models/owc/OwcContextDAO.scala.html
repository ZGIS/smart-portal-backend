42   /**
43     * Parse an OwcContext from a ResultSet
44     */
45   private def owcContextParser(implicit connection: Connection): RowParser[OwcContext] = {
46     str("id") ~
47       get[Option[String]]("area_of_interest") ~
48       get[Option[String]]("spec_reference") ~
49       get[Option[String]]("context_metadata") ~
50       str("language") ~
51       str("title") ~
52       get[Option[String]]("subtitle") ~
53       get[OffsetDateTime]("update_date") ~
54       get[Option[String]]("authors") ~
55       get[Option[String]]("publisher") ~
56       get[Option[String]]("creator_application") ~
57       get[Option[String]]("creator_display") ~
58       get[Option[String]]("rights") ~
59       get[Option[String]]("time_interval_of_interest") ~
60       get[Option[String]]("keyword") map {
61       case
62         id ~
63           bboxText ~
64           specReferenceUuids ~
65           contextMetaUuids ~
66           language ~
67           title ~
68           subtitle ~
69           updateDate ~
70           authorsUuids ~
71           publisher ~
72           creatorAppUuid ~
73           creatorDispUuid ~
74           rights ~
75           temporalText ~
76           keywordUuids
77       => {
78         OwcContext(
79           id = new URL(id),
80           areaOfInterest = GeoDateParserUtils.createOptionalBbox(bboxText),
81           specReference = specReferenceUuids.map(u =>
82             OwcOfferingDAO.findByPropertiesUUID[OwcLink](Some(u))(OwcLinkEvidence, connection).toList).getOrElse(List()), // aka links.profiles[] & rel=profile
83           contextMetadata = contextMetaUuids.map(u =>
84             OwcOfferingDAO.findByPropertiesUUID[OwcLink](Some(u))(OwcLinkEvidence, connection).toList).getOrElse(List()), // aka links.via[] & rel=via
85           language = language,
86           title = title,
87           subtitle = subtitle,
88           updateDate = updateDate,
89           author = authorsUuids.map(u => OwcOfferingDAO.findByPropertiesUUID[OwcAuthor](Some(u))(OwcAuthorEvidence, connection).toList).getOrElse(List()),
90           publisher = publisher,
91           creatorApplication = creatorAppUuid.map(u => OwcCreatorApplicationDAO.findOwcCreatorApplicationByUuid(UUID.fromString(u))).getOrElse(None),
92           creatorDisplay = creatorDispUuid.map(u => OwcCreatorDisplayDAO.findOwcCreatorDisplayByUuid(UUID.fromString(u))).getOrElse(None),
93           rights = rights,
94           timeIntervalOfInterest = GeoDateParserUtils.parseOffsetDateString(temporalText),
95           keyword = keywordUuids.map(u =>
96             OwcOfferingDAO.findByPropertiesUUID[OwcCategory](Some(u))(OwcCategoryEvidence, connection).toList).getOrElse(List()),
97           resource = OwcResourceDAO.findOwcResourcesForOwcContext(id.toString).toList)
98       }
99     }
100   }

271   /**
272     * takes a list of [[OwcResource]] and tests that non of these exist,
273     * and if so creates them in the database and returns success of the overall create
274     *
275     * @param resources
276     * @param connection
277     * @return
278     */
279   def preCreateCheckOwcResources(resources: List[OwcResource])(implicit connection: Connection): Boolean = {
280     if (resources.nonEmpty) {
281       val idList = resources.map(o => o.id.toString)
282       val exists = idList.exists(id => OwcResourceDAO.findOwcResourceById(id).isDefined)
283       if (exists) {
284         logger.error(s"(createContext/resources) OwcResource with ID one of: ${idList.mkString(" ++ ")} exists already, recommend abort")
285         false
286       } else {
287         val insert = resources.map(OwcResourceDAO.createOwcResource(_)).filter(_.isDefined)
288         insert.length == resources.length
289       }
290     } else {
291       true
292     }
293   }

341   /**
342     * create/insert into db Owc Context
343     *
344     * @param owcContextUnfixed should make issue on github for owc geojson and portal backend
345     *                          and then remove the fix see [[OwcGeoJsonFixes]]
346     * @param user
347     * @param visibility
348     * @param rightsRelationType
349     * @return
350     */
351   def createOwcContext(owcContextUnfixed: OwcContext, user: User, visibility: Int, rightsRelationType: String)
352                       (implicit connection: Connection): Option[OwcContext] = {
353 
354     val owcContext = OwcGeoJsonFixes.fixRelPropertyForOwcLinks(owcContextUnfixed)
355 
356     val preCreateCheckOwcAuthorsForContext = OwcResourceDAO.preCreateCheckOwcAuthors(owcContext.author)
357     val preCreateCheckOwcLinksForContext = OwcResourceDAO.preCreateCheckOwcLinks(owcContext.specReference ++ owcContext.contextMetadata)
358     val preCreateCheckOwcResourcesForContext = preCreateCheckOwcResources(owcContext.resource)
359     val preCreateCheckOwcCategoriesForContext = OwcResourceDAO.preCreateCheckOwcCategories(owcContext.keyword)
360     val preCreateCheckOwcCreatorApplicationForContext = preCreateCheckOwcCreatorApplication(owcContext)
361     val preCreateCheckOwcCreatorDisplayForContext = preCreateCheckOwcCreatorDisplay(owcContext)
362 
363     if (user.isActive && preCreateCheckOwcAuthorsForContext && preCreateCheckOwcLinksForContext &&
364       preCreateCheckOwcResourcesForContext && preCreateCheckOwcCategoriesForContext &&
365       preCreateCheckOwcCreatorApplicationForContext && preCreateCheckOwcCreatorDisplayForContext) {
366 
367       val owcContextInsert = SQL(
368         s"""insert into $tableOwcContexts values (
369            |{id},
370            |{area_of_interest},
371            |{spec_reference},
372            |{context_metadata},
373            |{language},
374            |{title},
375            |{subtitle},
376            |{update_date},
377            |{authors},
378            |{publisher},
379            |{creator_application},
380            |{creator_display},
381            |{rights},
382            |{time_interval_of_interest},
383            |{keyword})""".stripMargin).on(
384         'id -> owcContext.id.toString,
385         'area_of_interest -> owcContext.areaOfInterest.map(r => GeoDateParserUtils.rectToWkt(r)),
386         'spec_reference -> owcContext.specReference.map(_.uuid.toString).mkString(":"), // links.profiles[] and rel=profile
387         'context_metadata -> owcContext.contextMetadata.map(_.uuid.toString).mkString(":"), // aka links.via[] & rel=via
388         'language -> owcContext.language,
389         'title -> owcContext.title,
390         'subtitle -> owcContext.subtitle,
391         'update_date -> owcContext.updateDate.toLocalDateTime,
392         'authors -> owcContext.author.map(_.uuid.toString).mkString(":"),
393         'publisher -> owcContext.publisher,
394         'creator_application -> owcContext.creatorApplication.map(_.uuid.toString),
395         'creator_display -> owcContext.creatorDisplay.map(_.uuid.toString),
396         'rights -> owcContext.rights,
397         'time_interval_of_interest -> owcContext.timeIntervalOfInterest.map(dates => GeoDateParserUtils.writeOffsetDatesAsDateString(dates)).getOrElse(None),
398         'keyword -> owcContext.keyword.map(_.uuid.toString).mkString(":")
399       ).executeUpdate()
400 
401       // this references an existing owc_context_id (the newly created one from above)
402       val resourceDepsInsertCount = owcContext.resource.map {
403         owcResource =>
404           SQL(
405             s"""insert into $tableOwcContextHasOwcResources  values (
406                |{owc_context_id}, {owc_resource_id}
407                |)
408                  """.stripMargin).on(
409             'owc_context_id -> owcContext.id.toString,
410             'owc_resource_id -> owcResource.id.toString
411           ).executeUpdate()
412       }.sum
413 
414       // this references an existing owc_context_id (the newly created one from above)
415       val rightsDepsInsert = SQL(
416         s"""insert into $tableUserHasOwcContextRights values (
417            |{accountsubject}, {owcContextId}, {rights_relation_type}, {visibility}
418            |)""".stripMargin).on(
419         'accountsubject -> user.accountSubject,
420         'owcContextId -> owcContext.id.toString,
421         'rights_relation_type -> rightsRelationType,
422         'visibility -> visibility
423       ).executeUpdate()
424 
425       val allInserts = resourceDepsInsertCount + rightsDepsInsert + owcContextInsert
426       val testValue = owcContext.resource.size + 1 + 1
427 
428       if (allInserts == testValue) {
429         Some(owcContext)
430       } else {
431         logger.error("OwcContext/create: one of " +
432           s"resourceDepsInsertCount: $resourceDepsInsertCount (should be ${owcContext.resource.size}) + " +
433           s"rightsDepsInsert: $rightsDepsInsert (1) + owcContextInsert: $owcContextInsert (1) failed")
434         logger.error("OwcContext couldn't be created")
435         // we need to think where to place rollback most appropriately
436         connection.rollback()
437         None
438       }
439     } else {
440       logger.error("OwcContext/create: one of " +
441         s"user.isActive: ${user.isActive} + preCreateCheckOwcAuthorsForContext: $preCreateCheckOwcAuthorsForContext  " +
442         s"preCreateCheckOwcResourcesForContext: $preCreateCheckOwcResourcesForContext + " +
443         s"preCreateCheckOwcCategoriesForContext: $preCreateCheckOwcCategoriesForContext + " +
444         s"preCreateCheckOwcCreatorApplicationForContext: $preCreateCheckOwcCreatorApplicationForContext + " +
445         s"preCreateCheckOwcCreatorDisplayForContext: $preCreateCheckOwcCreatorDisplayForContext was not successful")
446       logger.error("Precondition failed, won't create OwcContext")
447       // we need to think where to place rollback most appropriately
448       connection.rollback()
449       None
450     }
451   }

530   /**
531     * pre Check for resources Update For an [[OwcContext]],
532     * will delete, update and insert required [[OwcResource]] in order to make OwcContext Update consistent
533     *
534     * @param owcContext
535     * @param connection
536     * @return
537     */
538   def preUpdateCheckOwcResources(owcContext: OwcContext)(implicit connection: Connection): Boolean = {
539     // get current list,
540     val current: List[URL] = owcContext.resource.map(_.id)
541 
542     logger.trace(s"preUpdateCheckOwcResources: current: List[URL] ${current.map(_.toString).mkString}")
543 
544     // get old list,
545     val oldOwcContext = findOwcContextById(owcContext.id)
546     val old: List[URL] = oldOwcContext.map(o => o.resource.map(_.id)).getOrElse(List())
547 
548     logger.trace(s"preUpdateCheckOwcResources: old: List[URL] ${current.map(_.toString).mkString}")
549 
550     // in old but not current -> delete
551     val toBeDeleted = old.diff(current)
552 
553     logger.trace(s"preUpdateCheckOwcResources: toBeDeleted: List[URL] ${current.map(_.toString).mkString}")
554 
555     val deleted: Boolean = oldOwcContext.exists { owcContext =>
556       owcContext.resource.filter(o => toBeDeleted.contains(o.id))
557         .map { owcResource =>
558           // delete relation from tableOwcContextHasOwcResources table before finally deleting referenced OwcResource
559           val deletedOwcResourceRelation = SQL(
560             s"""Delete from $tableOwcContextHasOwcResources where
561 owc_context_id = {owc_context_id} and owc_resource_id = {owc_resource_id}""".stripMargin).on(
562             'owc_context_id -> owcContext.id.toString,
563             'owc_resource_id -> owcResource.id.toString
564           ).executeUpdate().equals(1)
565 
566           val deletedOwcResource = OwcResourceDAO.deleteOwcResource(owcResource)
567 
568           deletedOwcResource && deletedOwcResourceRelation
569         }.count(_ == true) == toBeDeleted.length
570     }
571 
572     // in both lists -> update
573     val toBeUpdated = current.intersect(old)
574 
575     logger.trace(s"preUpdateCheckOwcResources: toBeUpdated: List[URL] ${current.map(_.toString).mkString}")
576 
577     val updated = owcContext.resource.filter(o => toBeUpdated.contains(o.id))
578       .map(OwcResourceDAO.updateOwcResource(_))
579       .count(_.isDefined) == toBeUpdated.length
580 
581     // in current but not in old -> insert
