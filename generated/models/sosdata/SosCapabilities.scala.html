70     Json.fromJson[SosCapabilities](json) match {

85   def fromXml(nodeSeq: NodeSeq, sosUrl: String): Option[SosCapabilities] = {

91             sosUrl = sosUrl,

103               "procedure"),
104             responseFormats = Some(

106               (nodeSeq \\ "OperationsMetadata" \ "Operation") filter(n => (n \ "@name" toString) == "GetObservation"),

119     }

122   private def parseValuesFromOperationsMetadata(nodeseq: NodeSeq, parameter: String): Seq[String] = {
123     val param = (nodeseq \ "Parameter") filter(node => (node \ "@name" toString) == parameter)

125   }
