41     * basic auth compare
42     *
43     * @param loginCredentials
44     * @return
45     */
46   def authenticateLocal(loginCredentials: LoginCredentials): Either[ErrorResult, User] = {
47 
48     val userOpt = dbSession.viaConnection { implicit connection =>
49       UserDAO.findUserByEmailAsString(loginCredentials.email).filter(user =>
50         passwordHashing.validatePassword(loginCredentials.password, user.password)
51       )
52     }
53     if (userOpt.isDefined) {
54       // here is correct auth branch
55       val user = userOpt.get
56       if (user.isBlocked) {
57         val status = user.getToken
58         Left(ErrorResult(s"Your account was ${status.value}. Please contact the administrator.", None))
59       } else {
60         Right(user)
61       }
62 
63     } else {
64       // should never indicate if user not found or just password wrong?
65       Left(ErrorResult("User email or password wrong.", None))
66     }
67   }

237   /**
238     * check for an active server-side session
239     *
240     * @param xsrfToken
241     * @param xsrfTokenCookie
242     * @param userAgentHeader
243     * @return
244     */
245   def getUserSessionByToken(xsrfToken: String, xsrfTokenCookie: String, userAgentHeader: String): Option[UserSession] = {
246     dbSession.viaConnection(implicit connection => {
247       UserSession.findUserSessionByToken(xsrfToken, 1)
248     })
249       .fold[Option[UserSession]] {
250         logger.debug(s"no session for xsrfToken: $xsrfToken")
251         None
252       } {
253         session =>
254           val cookieForUSerAndUserAgent = passwordHashing.testSessionCookie(xsrfToken, session.email, userAgentHeader)
255           logger.trace(s"testcookie: $cookieForUSerAndUserAgent")
256           if (xsrfToken == xsrfTokenCookie && cookieForUSerAndUserAgent) {
257             logger.trace(s"request for active session: ${session.email} / $xsrfToken / $userAgentHeader")
258             Some(session)
259           } else {
260             logger.error("Invalid Token")
261             None
262           }
263     }
264   }
