182   /**
183     * pre Check for OwcOperations Update For an OwcOffering, will delete, update and insert
184     * required operations in order to make OwcOffering Update consistent
185     *
186     * @param owcOffering
187     * @return
188     */
189   private def preUpdateCheckOperationsForOffering(owcOffering: OwcOffering)(implicit connection: Connection): Boolean = {
190 
191     // get current list,
192     val current: List[UUID] = owcOffering.operations.map(_.uuid)
193 
194     // get old list,
195     val oldOwcOffering = findOwcOfferingByUuid(owcOffering.uuid)
196     val old: List[UUID] = oldOwcOffering.map(o => o.operations.map(_.uuid)).getOrElse(List())
197 
198     // in old but not current -> delete
199     val toBeDeleted = old.diff(current)
200     val deleted = oldOwcOffering.map { owcOffering =>
201       owcOffering.operations.filter(o => toBeDeleted.contains(o.uuid))
202         .map(OwcOperationDAO.deleteOwcOperation(_))
203     }.count(_ == true) == toBeDeleted.length
204 
205     // in both lists -> update
206     val toBeUpdated = current.intersect(old)
207     val updated = owcOffering.operations.filter(o => toBeUpdated.contains(o.uuid))
208       .map(OwcOperationDAO.updateOwcOperation(_))
209       .count(_.isDefined) == toBeUpdated.length
210 
211     // in current but not in old -> insert
212     val toBeInserted = current.diff(old)
213     val inserted = owcOffering.operations.filter(o => toBeInserted.contains(o.uuid))
214       .map(OwcOperationDAO.createOwcOperation(_))
215       .count(_.isDefined) == toBeInserted.length
216 
217     if (deleted && updated && inserted) {
218       true
219     } else {
220       logger.error(s"(updateOffering/operation) one of deleted: $deleted , updated: $updated , inserted: $inserted not complete, recommend abort")
221       false
222     }
223   }

401   /**
402     * finds owc props like author, category, ink, content, stylesets, operation relation via its uuids
403     * obviously this could be further generalised
404     *
405     * @param propertiesUuid A String, which may contain 0, 1 or more UUIDs, multiple UUIDs must be separated by ':' colons
406     * @param a              implicit type evidence
407     * @tparam A Type of OwcAuthor, OwcCategory, OwcLink, OwcContent, OwcStyleSet, OwcOperation
408     * @return Sequence of objects of type A if found, otherwise empty Seq
409     */
410   def findByPropertiesUUID[A](propertiesUuid: Option[String])(implicit a: A, connection: Connection): Seq[A] = {
411 
412     import utils.StringUtils._
413 
414     val values = propertiesUuid.map(_.split(":").toSeq).map(
415       potUuids => {
416         val uuids = potUuids.map(_.toUuidOption).filter(_.isDefined).map(_.get)
417         uuids.map { uid =>
418           a match {
419             case a: OwcAuthor => OwcAuthorDAO.findOwcAuthorByUuid(uid).asInstanceOf[Option[A]]
420             case a: OwcCategory => OwcCategoryDAO.findOwcCategoryByUuid(uid).asInstanceOf[Option[A]]
421             case a: OwcLink => OwcLinkDAO.findOwcLinkByUuid(uid).asInstanceOf[Option[A]]
422             case a: OwcContent => OwcContentDAO.findOwcContentByUuid(uid).asInstanceOf[Option[A]]
423             case a: OwcCreatorApplication => OwcCreatorApplicationDAO.findOwcCreatorApplicationByUuid(uid).asInstanceOf[Option[A]]
424             case a: OwcCreatorDisplay => OwcCreatorDisplayDAO.findOwcCreatorDisplayByUuid(uid).asInstanceOf[Option[A]]
425             case a: OwcStyleSet => OwcStyleSetDAO.findOwcStyleSetByUuid(uid).asInstanceOf[Option[A]]
426             case a: OwcOperation => OwcOperationDAO.findOwcOperationByUuid(uid).asInstanceOf[Option[A]]
427             case a: OwcOffering => findOwcOfferingByUuid(uid).asInstanceOf[Option[A]]
428             case _ => throw new InvalidClassException(s"The evidence parameter type ${a.getClass.getCanonicalName} not supported here")
429           }
430 
431         }.filter(_.isDefined).map(_.get)
432       }
433     )
434     values.getOrElse(Seq())
