44 final case class CategoryHolder(hierarchyNumber: String = "",

46                           parent: String = "main",
47                           itemName: String = "",
48                           description: String = "",
49                           keywordContent: List[String] = List(),

53                           categoryClass: String = "MainCategory") extends ClassnameLogger {
54   def toRdf(vocabUrl: String): String = {
55     val keywordString = if (keywordContent.nonEmpty) {
56       keywordContent.mkString(", ")
57     } else {
58       ""
59     }
60 
61     s"""<rdf:Description rdf:about="${vocabUrl}/categories.rdfs#$id"
62        |    categories:id="$id"
63        |    categories:hierarchy_number="$hierarchyNumber"
64        |    categories:parent="$parent"
65        |    categories:item_name="$itemName"
66        |    categories:description="$description"
67        |    categories:keyword_content="$keywordString"
68        |    categories:query_string="$queryString"
69        |    categories:icon="$icon"
70        |    categories:bg_icon="$bgIcon">
71        |  <rdf:type rdf:resource="${vocabUrl}/categories.rdfs#$categoryClass"/>
72        |</rdf:Description>""".stripMargin
73   }
74 }
