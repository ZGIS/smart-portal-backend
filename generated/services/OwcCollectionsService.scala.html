271   def generateMdResource(catalogUrl: String, mdMetadata: MdMetadata, userMetaEntry: UserMetaRecord): OwcResource = {
272 
273     lazy val bboxFormat = new BboxArrayFormat
274 
275     val updatedTime = Try(OffsetDateTime.of(mdMetadata.citation.ciDate, LocalTime.of(12, 0), ZoneOffset.UTC))
276       .getOrElse(OffsetDateTime.now(ZoneId.systemDefault()))
277     // val baseLink = new URL(s"http://portal.smart-project.info/context/resource/${URLEncoder.encode(mdMetadata.fileIdentifier, "UTF-8")}_copy_${UUID.randomUUID().toString}")
278     val baseLink = new URL(s"https://portal.smart-project.info/context/resource/${userMetaEntry.uuid.toString}")
279 
280     val cswGetCapaOps = OwcOperation(
281       code = "GetCapabilities",
282       method = "GET",
283       mimeType = Some("application/xml"),
284       requestUrl = new URL("https://portal.smart-project.info/pycsw/csw?SERVICE=CSW&VERSION=2.0.2&REQUEST=GetCapabilities"))
285 
286     val cswGetRecordOps = OwcOperation(
287       code = "GetRecordById",
288       method = "GET",
289       mimeType = Some("application/xml"),
290       requestUrl = new URL(s"https://portal.smart-project.info/pycsw/csw?request=GetRecordById&service=CSW&version=2.0.2&elementSetName=full&outputSchema=http://www.isotc211.org/2005/gmd&id=${mdMetadata.fileIdentifier}"))
291 
292     val cswOffering = OwcOffering(
293       code = OwcOfferingType.CSW.code,
294       operations = List(cswGetCapaOps, cswGetRecordOps)
295     )
296 
297     val viaLink = OwcLink(
298       href = baseLink,
299       mimeType = Some("application/json"),
300       rel = "via")
301 
302     val cswLink = OwcLink(
303       href = cswGetRecordOps.requestUrl,
304       mimeType = cswGetRecordOps.mimeType,
305       rel = "via")
306 
307     val owcResource = OwcResource(
308       id = baseLink,
309       geospatialExtent = bboxFormat.parseBboxArray(mdMetadata.extent.mapExtentCoordinates).toOption,
310       title = mdMetadata.title,
311       subtitle = mdMetadata.abstrakt.toOption(),
312       updateDate = updatedTime,
313       author = List(OwcAuthor(name = mdMetadata.responsibleParty.individualName.toOption(),
314         email = parseEmailStringtoEmailAddress(mdMetadata.responsibleParty.email),
315         uri = Some(new URL(mdMetadata.responsibleParty.orgWebLinkage)))),
316       publisher = mdMetadata.responsibleParty.pointOfContact.toOption(),
317       rights = mdMetadata.distribution.useLimitation.toOption(),
318       temporalExtent = GeoDateParserUtils.parseOffsetDateString(Some(mdMetadata.extent.temporalExtent)),
319 
320       // links.alternates[] and rel=alternate
321       contentDescription = List(),
322 
323       // aka links.previews[] and rel=icon (atom)
324       preview = List(),
325 
326       // aka links.data[] and rel=enclosure (atom)
327       contentByRef = List(),
328 
329       // aka links.via[] & rel=via
330       resourceMetadata = List(viaLink, cswLink),
331       offering = List(cswOffering),
332       minScaleDenominator = Try(mdMetadata.scale.toDouble).toOption,
333       maxScaleDenominator = None,
334       active = None,
335       keyword = mdMetadata.keywords.map(w => OwcCategory(term = w, scheme = None, label = None)),
336       folder = None)
337 
338     logger.trace(Json.prettyPrint(owcResource.toJson))
339 
340     owcResource
341   }

552     * @return Option[EmailAddress]

554   def parseEmailStringtoEmailAddress(emailString: String): Option[EmailAddress] = {
555 
556     if (EmailAddress.isValid(emailString)) {
557       Some(EmailAddress(emailString))
558     } else {
559       if (emailString.contains(",")) {
560         val commaFree = emailString.split(",").head.trim
561 
562         Try {
563           EmailAddress(commaFree)
564         } match {
565           case Success(e) => Some(e)
566           case _ => None
567         }
568       } else {
569         None
570       }
