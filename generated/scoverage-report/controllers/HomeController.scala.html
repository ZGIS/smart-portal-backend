166     // TODO check from where URL referer comes from
167     ws.url(recaptcaVerifyUrl)
168       .withHeaders("Accept" -> "application/json")
169       .withRequestTimeout(10000.millis)
170       .withQueryString("secret" -> reCaptchaSecret,
171         "response" -> recaptcaChallenge).get().map {
172       response =>
173         val success = (response.json \ "success").as[Boolean]
174         if (success) {
175           Ok(Json.obj("status" -> "OK", "message" -> "granted", "success" -> JsBoolean(true)))
176         }
177         else {
178           val errors = (response.json \ "error-codes")
179           val jsErrors = errors.getOrElse(JsString("No further errors"))
180           val error = ErrorResult("User email or password wrong.", Some(jsErrors.toString()))
181           BadRequest(Json.toJson(error)).as(JSON)
182         }
183     }.recover {
184       case e: Exception =>
185         val error = ErrorResult("Exception on validating captcha.", Some(e.getMessage))
186         InternalServerError(Json.toJson(error)).as(JSON)
187     }
188   }
