81     */
82   def queryActiveSessions(token: Option[String], max: Option[Int], email: Option[String]): Seq[UserSession] = {
83     dbSession.viaConnection{ implicit connection =>
84 
85       val count = max.getOrElse(100)
86       (token, email) match {
87         case (Some(to), Some(em)) =>
88           val viaToken = UserSession.findUserSessionByToken(to, count).toSeq
89           val viaEmail = UserSession.findUserSessionByEmail(em, count)
90           (viaToken ++ viaEmail).filter(ul => ul.token.contains(to) && ul.email.contains(em))
91         case (Some(to), None) =>
92           UserSession.findUserSessionByToken(to, count).toSeq
93         case (None, Some(em)) =>
94           UserSession.findUserSessionByEmail(em, count)
95         case (None, None) =>
96           UserSession.getAllUserSessions(count)
97       }
98     }
99   }

101   def blockUnblockUsers(command: String, email: String): Option[User] = {
102     val userOpt = dbSession.viaConnection( implicit connection => {
103       UserDAO.findUserByEmailAsString(email)
104     })
105     userOpt.fold[Option[User]]{
106       None
107     } {
108       user =>
109         val statusToken = if (command.equalsIgnoreCase("BLOCK")) StatusToken.BLOCKED else StatusToken.ACTIVE
110         val updateUser = User(
111           user.email,
112           user.accountSubject,
113           user.firstname,
114           user.lastname,
115           "***",
116           s"${statusToken.value}:BY-ADMIN",
117           ZonedDateTime.now.withZoneSameInstant(ZoneId.of(appTimeZone)))
118 
119         // result of Option can be evaluated upstream in controller
120         dbSession.viaTransaction { implicit connection =>
121           UserDAO.updateNoPass(updateUser)
122         }
123 
124     }
125   }
