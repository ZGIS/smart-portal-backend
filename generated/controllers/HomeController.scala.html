106   /**
107     * login with JSON (from Angular / form)
108     * {"email":"akmoch","password":"testpass123"}
109     *
110     * Create the authentication token and sets the cookie [[AuthTokenCookieKey]]
111     * for AngularJS to use also in X-XSRF-TOKEN in HTTP header.
112     *
113     * @return
114     */
115   def login: Action[JsValue] = Action(parse.json) { implicit request =>
116     request.body.validate[LoginCredentials].fold(
117       errors => {
118         logger.error(JsError.toJson(errors).toString())
119         val error = ErrorResult("Could not validate request.", Some(JsError.toJson(errors).toString()))
120         BadRequest(Json.toJson(error)).as(JSON)
121       },
122       valid = credentials => {
123         // authenticate user in db
124         userService.authenticateLocal(credentials) match {
125           case Left(error) =>
126             logger.error(error.message)
127             Unauthorized(Json.toJson(error)).as(JSON)
128 
129           case Right(user) =>
130             val uaIdentifier: String = request.headers.get(UserAgentHeader).getOrElse(UserAgentHeaderDefault)
131             logger.trace(s"Logging in email from $uaIdentifier")
132             val token = userService.upsertUserSession(user.email.value, uaIdentifier)
133             logger.trace(s"Logging in setting cache $token")
134             Ok(Json.obj("status" -> "OK", "token" -> token, "email" -> user.email.value, "userprofile" -> user.asProfileJs))
135               .withCookies(Cookie(AuthTokenCookieKey, token, None, httpOnly = false))
136 
137           case _ =>
138             val error = ErrorResult("Could not validate request.", None)
139             logger.error(error.message)
140             InternalServerError(Json.toJson(error)).as(JSON)
141         }
142       })
143   }
144 
145   /**
146     * https://developers.google.com/recaptcha/docs/verify
147     *
148     * @param recaptcaChallenge
149     * @return
150     */
151   def recaptchaValidate(recaptcaChallenge: String): Action[AnyContent] = Action.async { implicit request =>
152     // TODO check from where URL referer comes from
153     ws.url(recaptcaVerifyUrl)
154       .withHeaders("Accept" -> "application/json")
155       .withRequestTimeout(10000.millis)
156       .withQueryString("secret" -> reCaptchaSecret,
157         "response" -> recaptcaChallenge).get().map {
158       response =>
159         val success = (response.json \ "success").as[Boolean]
160         if (success) {
161           Ok(Json.obj("status" -> "OK", "message" -> "granted", "success" -> JsBoolean(true)))
162         }
163         else {
164           val errors = (response.json \ "error-codes")
165           val jsErrors = errors.getOrElse(JsString("No further errors"))
166           val error = ErrorResult("User email or password wrong.", Some(jsErrors.toString()))
167           BadRequest(Json.toJson(error)).as(JSON)
168         }
169     }.recover {
170       case e: Exception =>
171         val error = ErrorResult("Exception on validating captcha.", Some(e.getMessage))
172         InternalServerError(Json.toJson(error)).as(JSON)
173     }
174   }
