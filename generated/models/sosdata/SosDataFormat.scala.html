24 trait SosDataFormat {

26   def contentType: String
27   def fileSuffix: String

30 object SosDataFormat {

32   case object OM20 extends SosDataFormat {

34     val contentType = "application/xml"

36   }
37 
38   case object WML2 extends SosDataFormat {
39     val value = "http://www.opengis.net/waterml/2.0"
40     val contentType = "application/xml"
41     val fileSuffix = ".wml"
42   }
43 
44   case object XLS extends SosDataFormat {
45     val value = "application/vnd.ms-excel"
46     val contentType = "application/vnd.ms-excel"
47     val fileSuffix = ".xlsx"
48   }
49 
50   case object CSV extends SosDataFormat {
51     val value = "text/csv"
52     val contentType = "text/csv"
53     val fileSuffix = ".csv"
54   }
55 
56   def apply(v: String) : SosDataFormat = {
57     v match {
58       case "http://www.opengis.net/om/2.0" => OM20
59       case "http://www.opengis.net/waterml/2.0" => WML2
60       case "application/vnd.ms-excel" => XLS
61       case "text/csv" => CSV
62       case _ => throw new IllegalArgumentException(s"Value $v is not a supported data format value")
63     }
64   }
65 
66   val supportedParserFormats: List[String] = List(OM20.value, WML2.value)
67 
68   val supportedExportFormats: List[String] = List(WML2.value, XLS.value, CSV.value)
69 
70   def isValid(v: String) : Boolean = {
71     Try (apply(v)).isSuccess
72   }
73 
74   def isParser(f: SosDataFormat) : Boolean = {
75     supportedParserFormats.contains(f.value)
76   }
77 
78   def isExport(f: SosDataFormat) : Boolean = {
79     supportedExportFormats.contains(f.value)
80   }
