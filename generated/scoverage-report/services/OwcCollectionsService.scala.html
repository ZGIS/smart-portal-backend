201       code = "GetRecordById",

205 
206     val cswOffering = OwcOffering(
207       code = OwcOfferingType.CSW.code,
208       operations = List(cswGetCapaOps, cswGetRecordOps)
209     )
210 
211     val viaLink = OwcLink(
212       href = baseLink,
213       mimeType = Some("application/json"),
214       rel = "via")
215 
216     val cswLink = OwcLink(
217       href = cswGetRecordOps.requestUrl,
218       mimeType = cswGetRecordOps.mimeType,
219       rel = "via")
220 
221     val userLookup = sessionHolder.viaConnection { implicit connection =>
222       UserDAO.findUserByEmailAsString(authUser)
223     }
224 
225     userLookup.fold {
226       // user not found, then can't insert, shouldn't happen though
227       logger.error("User not found, can't access users collection")

230       user =>
231         val collectionLookup = sessionHolder.viaConnection { implicit connection =>
232           OwcContextDAO.findUserDefaultOwcContext(user)
233         }
234         collectionLookup.fold {
235           // collection not found, then can't insert, should probably also not happen usually
236           logger.error("Collection not found, can't access users collection")

239           owcDoc =>
240 
241             val owcResource = OwcResource(
242               id = baseLink,
243               geospatialExtent = bboxFormat.parseBboxArray(mdMetadata.extent.mapExtentCoordinates).toOption,

248                 email = parseEmailStringtoEmailAddress(mdMetadata.responsibleParty.email),
249                 uri = Some(new URL(mdMetadata.responsibleParty.orgWebLinkage)))),

251               rights = mdMetadata.distribution.useLimitation.toOption(),

254               // links.alternates[] and rel=alternate
255               contentDescription = List(),
256 
257               // aka links.previews[] and rel=icon (atom)
258               preview = List(),
259 
260               // aka links.data[] and rel=enclosure (atom)
261               contentByRef = List(),
262 
263               // aka links.via[] & rel=via
264               resourceMetadata = List(viaLink, cswLink),
265               offering = List(cswOffering),
266               minScaleDenominator = Try(mdMetadata.scale.toDouble).toOption,
267               maxScaleDenominator = None,
268               active = None,
269               keyword = mdMetadata.keywords.map(w => OwcCategory(term = w, scheme = None, label = None)),
270               folder = None)
271 
272             val entries = owcDoc.resource ++ Seq(owcResource)
273             val newDoc = owcDoc.copy(resource = entries)
274 
275             sessionHolder.viaTransaction { implicit connection =>

277             }

365   /**
366     * returns an Option[EmailAddress] object if parsing is successful, for those mdMetadata email joint arrays
367     *
368     * @param emailString the string that might comprise of or contain an email address
369     * @return Option[EmailAddress]
370     */
371   def parseEmailStringtoEmailAddress(emailString: String): Option[EmailAddress] = {
372 
373     if (EmailAddress.isValid(emailString)) {
374       Some(EmailAddress(emailString))
375     } else {
376       if (emailString.contains(",")) {
377         val commaFree = emailString.split(",").head.trim
378 
379         Try {
380           EmailAddress(commaFree)
381         } match {
382           case Success(e) => Some(e)
383           case _ => None
384         }
385       } else {
386         None
387       }
388     }
