337                   logger.error("Password update error.")

344                   cache.set(newtoken, user.username)
345                   Ok(Json.obj("status" -> "OK", "token" -> newtoken, "username" -> user.username))
346                     .withCookies(Cookie(AuthTokenCookieKey, newtoken, None, httpOnly = false))
