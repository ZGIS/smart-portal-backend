162     * @param pathPrefix sub directory where the file should go /prefix/prefix no trailing slash

165   def updateVocabFileGoogleBucket(vocabString: String, fileName: String, pathPrefix: String): Either[ErrorResult, LocalBlobInfo] = {
166     // Google Java upload stuff
167     import java.nio.charset.StandardCharsets.UTF_8
168 
169     // val storage: Storage = authenticatedStorageOptions.getService()
170     val storage: Storage = StorageOptions.getDefaultInstance().getService()
171     // val noCredentials = NoCredentials.getInstance()
172 
173     val blobTry: Try[LocalBlobInfo] = Try {
174       val blobId = BlobId.of(googleStorageBucket, pathPrefix + "/" + fileName)
175       val blob = storage.get(blobId)
176       if (blob != null) {
177         val channel = blob.writer()
178         channel.write(ByteBuffer.wrap(vocabString.getBytes(UTF_8)))
179         channel.close()
180         LocalBlobInfo.newFrom(blob)
181       } else {
182         throw new IllegalArgumentException("original blob for file could not be retrieved.")
183       }
184     }
185 
186     blobTry match {
187       case Success(blobInfo) => Right(blobInfo)
188       case Failure(ex) =>
189         logger.error(s"Blob update in cloud storage failed. ${ex.getLocalizedMessage}")
