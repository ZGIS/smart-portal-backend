100   /**
101     * login with JSON (from Angular / form)
102     * {"email":"akmoch","password":"testpass123"}
103     *
104     * Create the authentication token and sets the cookie [[AuthTokenCookieKey]]
105     * for AngularJS to use also in X-XSRF-TOKEN in HTTP header.
106     *
107     * @return
108     */
109   def login: Action[JsValue] = Action(parse.json) { implicit request =>
110     request.body.validate[LoginCredentials].fold(
111       errors => {
112         logger.error(JsError.toJson(errors).toString())
113         val error = ErrorResult("Could not validate request.", Some(JsError.toJson(errors).toString()))
114         BadRequest(Json.toJson(error)).as(JSON)
115       },
116       valid = credentials => {
117         // authenticate user in db
118         userService.authenticateLocal(credentials) match {
119           case Left(error) =>
120             logger.error(error.message)
121             Unauthorized(Json.toJson(error)).as(JSON)
122 
123           case Right(user) =>
124             val uaIdentifier: String = request.headers.get(UserAgentHeader).getOrElse(UserAgentHeaderDefault)
125             logger.trace(s"Logging in email from $uaIdentifier")
126             val token = userService.upsertUserSession(user.email.value, uaIdentifier)
127             logger.trace(s"Logging in setting cache $token")
128             Ok(Json.obj("status" -> "OK", "token" -> token, "email" -> user.email.value, "userprofile" -> user.asProfileJs))
129               .withCookies(Cookie(AuthTokenCookieKey, token, None, httpOnly = false))
130 
131           case _ =>
132             val error = ErrorResult("Could not validate request.", None)
133             logger.error(error.message)
134             InternalServerError(Json.toJson(error)).as(JSON)
135         }
136       })
137   }
138 
139   /**
140     * https://developers.google.com/recaptcha/docs/verify
141     *
142     * @param recaptcaChallenge
143     * @return
144     */
145   def recaptchaValidate(recaptcaChallenge: String): Action[AnyContent] = Action.async { implicit request =>
146     // TODO check from where URL referer comes from
147     ws.url(portalConfig.recaptcaVerifyUrl)
148       .withHeaders("Accept" -> "application/json")
149       .withRequestTimeout(10000.millis)
150       .withQueryString("secret" -> portalConfig.reCaptchaSecret,
151         "response" -> recaptcaChallenge).get().map {
152       response =>
153         val success = (response.json \ "success").as[Boolean]
154         if (success) {
155           Ok(Json.obj("status" -> "OK", "message" -> "granted", "success" -> JsBoolean(true)))
156         }
157         else {
158           val errors = (response.json \ "error-codes")
159           val jsErrors = errors.getOrElse(JsString("No further errors"))
160           val error = ErrorResult("User email or password wrong.", Some(jsErrors.toString()))
161           BadRequest(Json.toJson(error)).as(JSON)
162         }
163     }.recover {
164       case e: Exception =>
165         val error = ErrorResult("Exception on validating captcha.", Some(e.getMessage))
166         InternalServerError(Json.toJson(error)).as(JSON)
167     }
168   }
