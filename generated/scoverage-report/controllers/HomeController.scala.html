108   /**
109     * login with JSON (from Angular / form)
110     * {"email":"akmoch","password":"testpass123"}
111     *
112     * Create the authentication token and sets the cookie [[AuthTokenCookieKey]]
113     * for AngularJS to use also in X-XSRF-TOKEN in HTTP header.
114     *
115     * @return
116     */
117   def login: Action[JsValue] = Action(parse.json) { implicit request =>
118     request.body.validate[LoginCredentials].fold(
119       errors => {
120         logger.error(JsError.toJson(errors).toString())
121         val error = ErrorResult("Could not validate request.", Some(JsError.toJson(errors).toString()))
122         BadRequest(Json.toJson(error)).as(JSON)
123       },
124       valid = credentials => {
125         // find user in db and compare password stuff
126         userDAO.authenticate(credentials.email, credentials.password).fold {
127           logger.error("User email or password wrong.")
128           val error = ErrorResult("User email or password wrong.", None)
129           Unauthorized(Json.toJson(error)).as(JSON)
130         } { user =>
131           val uaIdentifier: String = request.headers.get(UserAgentHeader).getOrElse(UserAgentHeaderDefault)
132           // logger.debug(s"Logging in email from $uaIdentifier")
133           val token = passwordHashing.createSessionCookie(user.email, uaIdentifier)
134           cache.set(token, user.email)
135           Ok(Json.obj("status" -> "OK", "token" -> token, "email" -> user.email, "userprofile" -> user.asProfileJs()))
136             .withCookies(Cookie(AuthTokenCookieKey, token, None, httpOnly = false))
137         }
138       })
139   }

141   /**
142     * https://developers.google.com/recaptcha/docs/verify
143     *
144     * @param recaptcaChallenge
145     * @return
146     */
147   def recaptchaValidate(recaptcaChallenge: String): Action[AnyContent] = Action.async { implicit request =>
148     // TODO check from where URL referer comes from
149     ws.url(recaptcaVerifyUrl)
150       .withHeaders("Accept" -> "application/json")
151       .withRequestTimeout(10000.millis)
152       .withQueryString("secret" -> reCaptchaSecret,
153         "response" -> recaptcaChallenge).get().map {
154       response =>
155         val success = (response.json \ "success").as[Boolean]
156         if (success) {
157           Ok(Json.obj("status" -> "OK", "message" -> "granted", "success" -> JsBoolean(true)))
158         }
159         else {
160           val errors = (response.json \ "error-codes")
161           val jsErrors = errors.getOrElse(JsString("No further errors"))
162           val error = ErrorResult("User email or password wrong.", Some(jsErrors.toString()))
163           BadRequest(Json.toJson(error)).as(JSON)
164         }
165     }.recover {
166       case e: Exception =>
167         val error = ErrorResult("Exception on validating captcha.", Some(e.getMessage))
168         InternalServerError(Json.toJson(error)).as(JSON)
169     }
170   }
171 
172   /**
173     * Log-out a user. Invalidates the authentication token.
174     *
175     * Discard the cookie [[AuthTokenCookieKey]] to have AngularJS no longer set the
176     * X-XSRF-TOKEN in HTTP header.
177     */
178   def logout: Action[Unit] = HasToken(parse.empty) { token =>
179     email =>
180       implicit request =>
181         cache.remove(token)
182         Ok.discardingCookies(DiscardingCookie(name = AuthTokenCookieKey))
183   }
