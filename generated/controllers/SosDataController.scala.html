85   def getTimeseries: Action[JsValue] = optionalAuthenticationAction.async(parse.json) {
86     request =>
87         request.body.validate[Timeseries].fold(
88           errors => {
89             logger.error(JsError.toJson(errors).toString())
90             val error = ErrorResult("Error while validating request.", Some(JsError.toJson(errors).toString))
91             Future {
92               BadRequest(Json.toJson(error)).as(JSON)
93             }
94           },
95 
96           timeseries => {
97             val sosXmlRequest = getObservationRequestXml(timeseries)
98             val responseFuture = wsClient.url(timeseries.sosUrl)
99               .withHeaders(CONTENT_TYPE -> "application/xml")
100               .post(sosXmlRequest)
101 
102             responseFuture.recover {
103               case e: Exception =>
104                 logger.error("Sos GetObservation threw exception", e)
105                 val error = ErrorResult(s"Exception (${e.getClass.getCanonicalName}) on SOS get capabilities.",
106                   Some(e.getMessage))
107                 InternalServerError(Json.toJson(error)).as(JSON)
108             }
109 
110             responseFuture.map { response =>
111               if (response.status != 200) {
112                 logger.error(s"Calling SOS ${timeseries.sosUrl} HTTP result status ${response.status} on GetObservation")
113                 val error = ErrorResult(s"Server ${timeseries.sosUrl} responded with ${response.status} on GetObservation",
114                   Some(response.body))
115                 InternalServerError(Json.toJson(error)).as(JSON)
116               }
117               else if (!response.header("Content-Type").get.contains("application/xml")) {
118                 logger.error(s"SOS GetObservation result was not XML but ${response.header("Content-Type").get}")
119                 val error = ErrorResult(
120                   s"SOS GetObservation result was not XML but ${response.header("Content-Type").get}",
121                   Some(response.body))
122                 InternalServerError(Json.toJson(error)).as(JSON)
123               }
124               else if (timeseries.responseFormat.isDefined && !timeseries.responseFormat.get.equalsIgnoreCase("http://www.opengis.net/om/2.0")) {
125                 logger.error(s"SOS GetObservation responseFormat unsupported for this request; ${timeseries.responseFormat.getOrElse("none")}")
126                 val error = ErrorResult(
127                   s"SOS GetObservation responseFormat unsupported for this request; ${timeseries.responseFormat.getOrElse("none")}",
128                   Some(response.body))
129                 InternalServerError(Json.toJson(error)).as(JSON)
130               }
131               else {
132                 val tvp = new XmlTvpParser().parseOm2Measurements(Source.fromString(response.body))
133                 val tsdata =
134                   if (tvp.isEmpty) {
135                     TimeseriesData(
136                       x = Seq.empty[String],
137                       y = Seq.empty[String],
138                       name = timeseries.timeseriesName
139                     )
140                   }
141                   else {
142                     TimeseriesData(
143                       x = tvp.map(_.datetime),
144                       y = tvp.map(_.measValue),
145                       name = timeseries.timeseriesName
146                     )
147                   }
148                 val uom = if (tvp.isEmpty) {
149                   if (timeseries.uom.isEmpty) {
150                     timeseries.observedProperty
151                   } else {
152                     timeseries.uom.getOrElse("empty")
153                   }
154                 } else tvp.head.measUnit
155                 val result = timeseries.copy(data = Some(tsdata), uom = Some(uom))
156 
157                 val logRequest = UserLinkLogging(id = None,
158                   timestamp = ZonedDateTime.now.withZoneSameInstant(ZoneId.of(appTimeZone)),
159                   ipaddress = Some(request.remoteAddress),
160                   useragent = request.headers.get(UserAgentHeader),
161                   email = request.optionalSession.map(_.email),
162                   link = sosXmlRequest,
163                   referer = request.headers.get(RefererHeader))
164 
165                 val updated = userService.logLinkInfo(logRequest)
166 
167                 logger.trace(logRequest.toString)
168                 Ok(result.toJson()).as(JSON)
169               }
170             }
171           }
172         )
