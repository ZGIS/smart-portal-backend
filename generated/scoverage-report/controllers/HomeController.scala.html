98   /**
99     * login with JSON (from Angular / form)
100     * {"username":"akmoch","password":"testpass123"}
101     *
102     * Create the authentication token and sets the cookie [[AuthTokenCookieKey]]
103     * for AngularJS to use also in X-XSRF-TOKEN in HTTP header.
104     *
105     * @return
106     */
107   def login = Action(parse.json) { implicit request =>
108     request.body.validate[LoginCredentials].fold(
109       errors => {
110         logger.error(JsError.toJson(errors).toString())
111         BadRequest(Json.obj("status" -> "ERR", "message" -> JsError.toJson(errors)))
112       },
113       valid = credentials => {
114         // find user in db and compare password stuff
115         userDAO.authenticate(credentials.username, credentials.password).fold {
116           logger.error("User or password wrong.")
117           BadRequest(Json.obj("status" -> "ERR", "message" -> "Username or password wrong."))
118         } { user =>
119           val uaIdentifier: String = request.headers.get(UserAgentHeader).getOrElse(UserAgentHeaderDefault)
120           logger.info(s"Logging in username from $uaIdentifier")
121           val token = passwordHashing.createSessionCookie("username", uaIdentifier)
122           cache.set(token, user.username)
123           Ok(Json.obj("status" -> "OK", "token" -> token, "username" -> "username"))
124             .withCookies(Cookie(AuthTokenCookieKey, token, None, httpOnly = false))
125         }
126       })
127   }
128 
129   /**
130     * Log-out a user. Invalidates the authentication token.
131     *
132     * Discard the cookie [[AuthTokenCookieKey]] to have AngularJS no longer set the
133     * X-XSRF-TOKEN in HTTP header.
134     */
135   def logout = HasToken(parse.empty) { token =>
136     username => implicit request =>
137       cache.remove(token)
138       Ok.discardingCookies(DiscardingCookie(name = AuthTokenCookieKey))
139   }

162       Ok.discardingCookies(DiscardingCookie(name = AuthTokenCookieKey))
163   }
