125             UserDAO.createUser(newUser).fold {
126               logger.error("User create error.")
127               val error = ErrorResult("User create error.", None)

130               val emailWentOut = emailService.sendRegistrationEmail(jsuser.email, "Please confirm your GW HUB account", jsuser.firstname, regLinkId)

132               // creating default collection only after registration, account is only barely usable here

134             }
135           } { user =>
136             // found user with that email already
137             logger.error("Email already in use.")

358   def updatePassword(): Action[JsValue] = HasToken(parse.json) {
359     token =>
360       cachedSecUserEmail =>
361         implicit request =>
362           request.body.validate[PasswordUpdateCredentials].fold(
363             errors => {
364               logger.error(JsError.toJson(errors).toString())
365               val error = ErrorResult("Could not validate request.", Some(JsError.toJson(errors).toString()))
366               BadRequest(Json.toJson(error)).as(JSON)
367             },
368             valid = credentials => {
369               // find user in db
370               sessionHolder.viaTransaction { implicit connection =>
371                 UserDAO.findUserByEmailAddress(credentials.email).fold {
372                   logger.error("User not found.")
373                   val error = ErrorResult("User not found", None)
374                   BadRequest(Json.toJson(error)).as(JSON)
375                 } { dbuser =>
376                   // compare password
377                   if (passwordHashing.validatePassword(credentials.oldPassword, dbuser.password)) {
378                     val newCryptPass = passwordHashing.createHash(credentials.newPassword)
379                     val updateUser = dbuser.copy(
380                       password = newCryptPass,
381                       laststatustoken = s"${StatusToken.ACTIVE}:PASSWORDUPDATE",
382                       laststatuschange = ZonedDateTime.now.withZoneSameInstant(ZoneId.of(appTimeZone)))
383 
384                     UserDAO.updatePassword(updateUser).fold {
385                       logger.error("Password update error.")
386                       val error = ErrorResult("Password update error.", None)
387                       BadRequest(Json.toJson(error)).as(JSON)
388                     } { user =>
389                       val uaIdentifier: String = request.headers.get(UserAgentHeader).getOrElse(UserAgentHeaderDefault)
390                       cache.remove(token)
391                       emailService.sendPasswordUpdateEmail(user.email, "Password Update on GW HUB", user.firstname)
392                       val newtoken = passwordHashing.createSessionCookie(user.email, uaIdentifier)
393                       cache.set(newtoken, user.email.value)
394                       Ok(Json.obj("status" -> "OK", "token" -> newtoken, "email" -> user.email.value))
395                         .withCookies(Cookie(AuthTokenCookieKey, newtoken, None, httpOnly = false))
396                     }
397                   } else {
398                     logger.error("Password update error, not matching.")
399                     val error = ErrorResult("Password update error, not matching.", None)
400                     BadRequest(Json.toJson(error)).as(JSON)
401                   }
402                 }
403               }
404             })
405   }

509   def gconnect: Action[JsValue] = Action(parse.json) { implicit request =>
510     request.body.validate[GAuthCredentials].fold(
511       errors => {
512         logger.error(JsError.toJson(errors).toString())
513         val error = ErrorResult("Could not validate request.", Some(JsError.toJson(errors).toString()))
514         BadRequest(Json.toJson(error)).as(JSON)
515       },
516       valid = credentials => {
517         if (!new java.io.File(googleClientSecret).exists) {
518           val error = ErrorResult("Service JSON file not available", None)
519           InternalServerError(Json.toJson(error)).as(JSON)
520         } else {
521           // do lots of Google OAuth2 stuff
522           val clientSecrets = GoogleClientSecrets.load(
523             JacksonFactory.getDefaultInstance(), new FileReader(googleClientSecret))
524 
525           // 988846878323-bkja0j1tgep5ojthfr2e92ao8n7iksab.apps.googleusercontent.com
526           val clientId = clientSecrets.getDetails().getClientId()
527           logger.debug(s"clientId $clientId")
528           // Specify the same redirect URI that you use with your web
529           // app. If you don't have a web version of your app, you can
530           // specify an empty string.
531           val tokenResponse: GoogleTokenResponse = new GoogleAuthorizationCodeTokenRequest(
532             new NetHttpTransport(),
533             JacksonFactory.getDefaultInstance(),
534             "https://www.googleapis.com/oauth2/v4/token",
535             clientId,
536             clientSecrets.getDetails().getClientSecret(),
537             credentials.authcode,
538             gauthRedirectUrl).execute()
539 
540           val accessToken = tokenResponse.getAccessToken()
541           // Use access token to call API
542           val credential = new GoogleCredential().setAccessToken(accessToken)
543           // Get profile info from ID token
544           val idToken: GoogleIdToken = tokenResponse.parseIdToken()
545           val payload = idToken.getPayload()
546           // Use this value as a key to identify a user.
547           val userId = payload.getSubject()
548           val email = payload.getEmail()
549           val emailVerified = payload.getEmailVerified()
550           // get further payload things that is easy in java but needs to work in scala too
551           val name = Option(payload.get("name").asInstanceOf[String])
552           val pictureUrl = Option(payload.get("picture").asInstanceOf[String])
553           val locale = Option(payload.get("locale").asInstanceOf[String])
554           val familyName = Option(payload.get("family_name").asInstanceOf[String])
555           val givenName = Option(payload.get("given_name").asInstanceOf[String])
556           if ((credentials.accesstype.equalsIgnoreCase("LOGIN") || credentials.accesstype.equalsIgnoreCase("REGISTER")) &&
557             emailVerified && EmailAddress.isValid(email)) {
558 
559             val emailAddress = EmailAddress(email)
560             sessionHolder.viaTransaction { implicit connection =>
561               UserDAO.findUserByEmailAddress(emailAddress).fold {
562                 // well, must be a new user coming through from Google
563                 val regLinkId = java.util.UUID.randomUUID().toString()
564                 val cryptPass = passwordHashing.createHash(regLinkId)
565 
566                 val newUser = User(emailAddress,
567                   s"google:${userId}",
568                   givenName.getOrElse(name.getOrElse(emailAddress.mailbox.value)),
569                   familyName.getOrElse(""),
570                   cryptPass,
571                   s"${StatusToken.ACTIVE}:REGCONFIRMED",
572                   ZonedDateTime.now.withZoneSameInstant(ZoneId.of(appTimeZone)))
573 
574                 UserDAO.createUser(newUser).fold {
575                   logger.error("User create error.")
576                   val error = ErrorResult("Error while creating user", None)
577                   InternalServerError(Json.toJson(error)).as(JSON)
578                 } { user =>
579                   val emailWentOut = emailService.sendConfirmationEmail(newUser.email, "Your GW HUB account is now active", newUser.firstname)
580                   // all good here, creating default collection (Unit)
581                   collectionsService.createUserDefaultCollection(newUser)
582 
583                   logger.info(s"New user registered. Email went out $emailWentOut")
584                   val uaIdentifier: String = request.headers.get(UserAgentHeader).getOrElse(UserAgentHeaderDefault)
585                   // logger.debug(s"Logging in email from $uaIdentifier")
