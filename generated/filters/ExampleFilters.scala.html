27 /**
28  * This is a simple filter that adds a header to all requests. It's
29  * added to the application's list of filters by the
30  * [[Filters]] class.
31  *
32  * @param mat This object is needed to handle streaming of requests
33  * and responses.
34  * @param exec This class is needed to execute code asynchronously.
35  * It is used below by the `map` method.
36  */
37 @Singleton
38 class ExampleFilters @Inject()(
39     implicit override val mat: Materializer,
40     exec: ExecutionContext) extends Filter {
41 
42   override def apply(nextFilter: RequestHeader => Future[Result])
43            (requestHeader: RequestHeader): Future[Result] = {
44     // Run the next filter in the chain. This will call other filters
45     // and eventually call the action. Take the result and modify it
46     // by adding a new header.
47     nextFilter(requestHeader).map { result =>
48       result.withHeaders("X-ExampleFilter" -> "foo")
49     }
50   }
51 
52 }
