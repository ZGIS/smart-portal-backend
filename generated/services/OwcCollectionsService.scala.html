485   /**
486     * returns an Option[EmailAddress] object if parsing is successful, for those mdMetadata email joint arrays
487     *
488     * @param emailString the string that might comprise of or contain an email address
489     * @return Option[EmailAddress]
490     */
491   def parseEmailStringtoEmailAddress(emailString: String): Option[EmailAddress] = {
492 
493     if (EmailAddress.isValid(emailString)) {
494       Some(EmailAddress(emailString))
495     } else {
496       if (emailString.contains(",")) {
497         val commaFree = emailString.split(",").head.trim
498 
499         Try {
500           EmailAddress(commaFree)
501         } match {
502           case Success(e) => Some(e)
503           case _ => None
504         }
505       } else {
506         None
507       }
508     }
509   }
