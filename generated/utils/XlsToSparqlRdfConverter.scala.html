49     * reading cell values from a sheet is not as straightforward apparently
50     *
51     * @param row
52     * @param cell
53     * @param sheet
54     * @return
55     */
56   def getCellValueAsStringOption(row: Int, cell: Int, sheet: Sheet): Option[String] = {
57     val theRow = sheet.getRow(row)
58     if (theRow != null) {
59 
60       val lastCell = if (theRow.getLastCellNum < theRow.getPhysicalNumberOfCells) {
61         theRow.getLastCellNum
62       } else {
63         theRow.getPhysicalNumberOfCells
64       }
65       logger.trace(s"getCellValueAsStringOption.lastCell: ${lastCell}")
66       if (lastCell > cell) {
67         logger.trace(s"getCellValueAsStringOption => row: $row, cell: $cell, sheet: ${sheet.getSheetName}," +
68           s"getCellType: ${sheet.getRow(row).getCell(cell).getCellType}")
69         sheet.getRow(row).getCell(cell).getCellType match {
70           case Cell.CELL_TYPE_NUMERIC => val value = Try {
71             sheet.getRow(row).getCell(cell).getNumericCellValue.toString
72           }.toOption
73             logger.trace(s"CELL_TYPE_NUMERIC: $value")
74             value
75           case Cell.CELL_TYPE_STRING => val value = Try {
76             val stringValue = sheet.getRow(row).getCell(cell).getRichStringCellValue.getString
77             if (stringValue == null) "" else stringValue
78           }.toOption
79             logger.trace(s"CELL_TYPE_STRING: $value")
80             value
81           case Cell.CELL_TYPE_BLANK => logger.debug(s"getCellValueAsStringOption cell type => row: $row, cell: $cell, sheet: ${sheet.getSheetName}," +
82             "getCellType: CELL_TYPE_BLANK")
83             None
84           case _ => logger.error(s"getCellValueAsStringOption cell type => row: $row, cell: $cell, sheet: ${sheet.getSheetName}," +
85             "unexpected error get cell type")
86             None
87         }
88       } else {
89         logger.debug(s"beyond cell range of row => row: $row, cell: $cell, sheet: ${sheet.getSheetName}," +
90           s"getLastCellNum: ${sheet.getRow(row).getLastCellNum}")
91         None
92       }
93     } else {
94       logger.debug("reaching into the void")
95       None
96     }
97   }

99   /**
100     * finds synonym ffor a list of main keywords of a category, uses lookup sheet
101     *
102     * @param keywords_list
103     * @param synonymSheet
104     * @return
105     */
106   def findSynonymsForKeywords(keywords_list: List[String], synonymSheet: org.apache.poi.ss.usermodel.Sheet): List[String] = {
107     logger.trace(s"findSynonymsForKeywords => keywords_list: ${keywords_list.mkString("::")}, synonymSheet: ${synonymSheet.getSheetName}")
108     val buf = scala.collection.mutable.ArrayBuffer.empty[String]
109     keywords_list.foreach { keyword =>
110       logger.trace(s"findSynonymsForKeywords.foreach keyword: $keyword")
111       val lastRow = if (synonymSheet.getLastRowNum < synonymSheet.getPhysicalNumberOfRows) synonymSheet.getLastRowNum else synonymSheet.getPhysicalNumberOfRows
112       logger.warn(s"synonymsheet.lastRow: ${lastRow}")
113       for (i <- 1 to lastRow) {
114         if (synonymSheet.getRow(i).getLastCellNum > 1) {
115           logger.trace(s"findSynonymsForKeywords.foreach synonymSheet.getLastRowNum: ${synonymSheet.getLastRowNum}")
116           if (getCellValueAsStringOption(i, 0, synonymSheet).isDefined) {
117             logger.trace(s"findSynonymsForKeywords.foreach getCellValueAsString($i, 0, ${synonymSheet.getLastRowNum}): " +
118               s"${getCellValueAsStringOption(i, 0, synonymSheet)}")
119             if (getCellValueAsStringOption(i, 0, synonymSheet).get.equalsIgnoreCase(keyword)) {
120               getCellValueAsStringOption(i, 1, synonymSheet).get.split(",").map(kw => kw.trim.toLowerCase).foreach(lkw => buf += lkw)
121             }
122           }
123         } else {
124           logger.debug(s"beyond cell range of row => row: $i, cell: 1, sheet: ${synonymSheet.getSheetName}," +
125             s"getLastCellNum: ${synonymSheet.getRow(i).getLastCellNum}")
126         }
127       }
128     }
129     val resultBuf = buf.toList
130     logger.debug(s"findSynonymsForKeywords => results buf: ${resultBuf.mkString("::")}")
131     resultBuf
132   }

134   /**
135     * reads in an XLSX Sheet with categories definition and a keywords lookup sheet and produces a list of Categories
136     * encapsulated in [[CategoryHolder]] to be dropped as RDF/XML eventually
137     *
138     * @param domainSheet
139     * @param synonymSheet
140     * @return
141     */
142   def buildCategoriesFromSheet(domainSheet: org.apache.poi.ss.usermodel.Sheet,
143                                synonymSheet: org.apache.poi.ss.usermodel.Sheet): List[CategoryHolder] = {
144     val buf = scala.collection.mutable.ListBuffer.empty[CategoryHolder]
145     val lastParent = scala.collection.mutable.ListBuffer.empty[String]
146     val parentDefault = "main"
147     lastParent += parentDefault
148     val lastRow = if (domainSheet.getLastRowNum < domainSheet.getPhysicalNumberOfRows) domainSheet.getLastRowNum else domainSheet.getPhysicalNumberOfRows
149     logger.warn(s"domainSheet.lastRow: ${lastRow}")
150     for (i <- 0 to lastRow) {
151       val cellValue = getCellValueAsStringOption(i, 0, domainSheet)
152       logger.trace(s"domainSheet.getRow($i).getCell(0).getCell: $cellValue")
153       if (cellValue.isDefined) {
154         val hierarchy_number = getCellValueAsStringOption(i, 0, domainSheet).get
155         val item_name = getCellValueAsStringOption(i, 1, domainSheet).getOrElse("")
156         logger.trace("row(+1) " + i + " hierarchy_number " + hierarchy_number + " item_name " + item_name)
157         if (hierarchy_number.equalsIgnoreCase("hierarchy_number")) {
158           if (!item_name.equalsIgnoreCase("item_name")) {
159             if (regMatcher.findFirstIn(item_name).isDefined) {
160               // val plist = regMatcher.findAllIn(item_name)
161               // lastParent += plist.toSeq.mkString("")
162               val plist = regMatcher.findAllIn(item_name).toList
163               logger.trace(s"regMatcher.findAllIn($item_name): ${plist.mkString("::")}")
164               lastParent += plist(1)
165             }
166           }
167         } else {
168           val lastParentValue = lastParent.last.trim
169           logger.trace(s"lastParentValue: ${lastParentValue}")
170           try {
171             logger.trace(s"before cell $i 4, after lastParentValue $lastParentValue")
172             val description = getCellValueAsStringOption(i, 4, domainSheet).getOrElse("")
173             logger.trace(s"before cell $i 5, after description $description")
174             val trimmed = getCellValueAsStringOption(i, 5, domainSheet).getOrElse("").split(",").map(_.trim)
175             val keywords_low = trimmed.map(_.toLowerCase).toList
176             val synonyms_low = findSynonymsForKeywords(keywords_low, synonymSheet)
177             val keyword_content = keywords_low ++ synonyms_low
178             logger.trace(s"before cell $i 6, after keyword_content ${keyword_content.mkString("::")}")
179             val icon = getCellValueAsStringOption(i, 6, domainSheet).getOrElse("")
180             logger.trace(s"before cell $i 7, after icon $icon")
181             val bg_icon = getCellValueAsStringOption(i, 7, domainSheet).getOrElse("")
182             logger.trace(s"before cell $i 8, after bg_icon $bg_icon")
183             val query_string = getCellValueAsStringOption(i, 8, domainSheet).getOrElse("")
184             logger.trace(s"before cell $i categoryClass, after query_string $query_string")
185             val categoryClass = if (lastParentValue.equalsIgnoreCase("main")) {
186               "MainCategory"
187             } else {
188               "ChildCategory"
189             }
190             logger.trace(s"before cat number $i")
191             val cat = CategoryHolder(
192               hierarchyNumber = hierarchy_number,
193               id = i,
194               parent = lastParentValue,
195               itemName = item_name,
196               description = description,
197               keywordContent = keyword_content,
198               queryString = query_string,
199               icon = icon,
200               bgIcon = bg_icon,
201               categoryClass = categoryClass)
202             logger.trace(s"after cat number $i")
203             logger.debug(cat.toRdf)
204             buf += cat
205 
206           } catch {
207             case ex: Exception => logger.error(s"new CategoryHolder error on row $i: ${ex.getMessage} ${ex.getStackTrace.mkString("\n")}")
208           }
209         }
210       }
211     }
212     logger.info(s"domainSheet.lastRow: ${lastRow}")
213     val resultBuf = buf.toList
214     logger.info(s"buildCategoriesFromSheet => results buf(${resultBuf.length}): ${resultBuf.map(kw => kw.hierarchyNumber).mkString("::")}")
215     resultBuf
216   }
