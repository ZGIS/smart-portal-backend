39 /**
40   * inserts the MDMetadata into the TransactionInsert XML
41   *
42   * @param xml
43   */
44 private class AddMDMetadataToInsert(xml: Node) extends RewriteRule {
45   override def transform(n: Node): Seq[Node] = n match {
46     case e: Elem if e.label == "Insert" => {
47       new Elem(e.prefix, "Insert", e.attributes, e.scope, e.minimizeEmpty, (xml).toSeq: _*)
48     }
49     case x => x
50   }
51 }

97   /** calls CSW:Transaction -> Insert
98     * FIXME should implement HasToken(parse.json) instead of Action.async, to only allow authenticated users
99     */
100   def insert: Action[JsValue] = HasTokenAsync(parse.json) {
101     token =>
102       authUser =>
103         implicit request => {
104           logger.debug(request.toString)
105 
106           //TODO parse JSON to MDMetadataSet and convert that to XML
107           /*
108                     if (request.body) {
109                       BadRequest("Posted data was no valid JSON document")
110                     }
111           */
112           val mdMetadata = MdMetadata.fromJson(((request.body.as[JsObject]) \ "metadata").get)
113 
114           val futureResponse: Future[WSResponse] = for {
115             getCapaResponse <- wsClient.url(CSW_OPERATIONS_METADATA_URL).get()
116             insertResponse <- {
117               //check if getCapaResponse indicates, that the CSW can perform Transactions
118               val operation = (getCapaResponse.xml \\ "OperationsMetadata" \\ "Operation").filter(node => {
119                 logger.debug(s"Attribute 'name'=${node.attribute("name").get.text.toString}")
120                 node.attribute("name").get.text.equals("Transaction")
121               })
122               logger.debug(s"operation = ${operation.toString()}")
123               if (operation.isEmpty) {
124                 throw new UnsupportedOperationException("CSW does not support Transaction.")
125               }
126 
127               //insert MDMEtadata in insert template
128               logger.debug(s"MD_MetadataXML: ${mdMetadata.get.toXml().toString()}")
129               val rule = new RuleTransformer(new AddMDMetadataToInsert(mdMetadata.get.toXml()))
130               val finalXML = rule.transform(cswtInsertXml)
131               logger.debug(s"finalXml: ${finalXML.toString()}")
132               wsClient.url(CSW_URL).post(finalXML.toString())
133             }
134             updateIngesterIndexResponse <- {
135               wsClient.url(INGESTER_UPDATE_INDEX_URL).get()
136             }
137           } yield insertResponse
138 
139           futureResponse.recover {
140             case e: Exception =>
141               logger.error("Insert CSW threw exception", e)
142               InternalServerError(e.getMessage)
143           }
144 
145           futureResponse.map(response => {
146             logger.debug(response.xml.toString())
147             response.xml match {
148               case e: Elem if e.label == "TransactionResponse" => {
149                 val fileIdentifier = (e \\ "InsertResult" \\ "BriefRecord" \\ "identifier").text
150                 logger.debug(s"Adding ${mdMetadata.get.fileIdentifier} to default collection of $authUser.")
151                 val added = collectionsService.addMdEntryToUserDefaultCollection(mdMetadata.get, authUser)
152                 if (!added) {
153                   logger.warn(s"Could not add ${mdMetadata.get.fileIdentifier} to default collection of $authUser.")
154                 }
155                 Ok(Json.obj("type" -> "success", "fileIdentifier" -> fileIdentifier,
156                   "message" -> s"Inserted as ${fileIdentifier}."))
157               }
158               case e: Elem if e.label == "ExceptionReport" => {
159                 val message = (e \\ "Exception" \\ "ExceptionText").text
160                 //TODO SR make this InternalServerError
161                 Ok(Json.obj("type" -> "danger", "message" -> message))
162               }
163               case _ => InternalServerError(s"Unexpected Response: ${response}")
164             }
165           })
166         }
167   }
