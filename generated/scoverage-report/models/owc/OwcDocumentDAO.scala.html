78   /**
79     * instantiate an OwcFeatureType either OwcEntry, or OwcDocument
80     *
81     * @param id
82     * @param featureType
83     * @param bboxAsWkt
84     * @return
85     */
86   def instantiateOwcFeatureType(id: String, featureType: String, bboxAsWkt: Option[String]): OwcFeatureType = {
87 
88     val emptyProfile = OwcLink(UUID.randomUUID(), "profile", None, "http://www.opengis.net/spec/owc-atom/1.0/req/core", None)
89     val emptySelf = OwcLink(UUID.randomUUID(), "self", Some("application/json"), "http://example.com/empty", None)
90     val now = ZonedDateTime.now.withZoneSameInstant(ZoneId.systemDefault())
91     logger.debug(s"instatiate for $featureType - $id")
92 
93     val emptyDefaultProps = OwcProperties(
94       UUID.randomUUID(),
95       "en",
96       "empty properties",
97       Some("no property information was found"),
98       Some(now),
99       Some("automated system response"),
100       None,
101       List(),
102       List(),
103       None,
104       None,
105       List(),
106       List(emptyProfile, emptySelf)
107     )
108 
109     val rectOpt = createOptionalBbox(bboxAsWkt)
110     val props = owcPropertiesDAO.findOwcPropertiesForOwcFeatureType(id).getOrElse(emptyDefaultProps)
111 
112     featureType match {
113       case "OwcEntry" => {
114         logger.debug(s"found OwcEntry ${id}")
115         val offerings = owcOfferingDAO.findOwcOfferingsForOwcEntry(id).toList
116         OwcEntry(id, rectOpt, props, offerings)
117       }
118       case "OwcDocument" => {
119         logger.debug(s"found OwcDocument ${id}")
120         val entries = findOwcEntriesForOwcDocument(id).toList
121         OwcDocument(id, rectOpt, props, entries)
122       }
123       case _ => throw new InvalidClassException(featureType, s"Unknown feature type $featureType")
124     }
125   }

207   /**
208     * create an owc entry with dependent offerings and properties
209     *
210     * @param owcEntry
211     * @return
212     */
213   def createOwcEntry(owcEntry: OwcEntry): Option[OwcEntry] = {
214 
215     db.withTransaction {
216       implicit connection => {
217 
218         val sql = SQL(
219           s"""
220           insert into $tableOwcFeatureTypes values (
221             {id}, {feature_type}, {bbox}
222           )
223         """).on(
224           'id -> owcEntry.id,
225           'feature_type -> "OwcEntry",
226           'bbox -> owcEntry.bbox.map(rect => rectToWkt(rect))
227         )
228 
229         val rowCount = sql.executeUpdate()
230 
231         logger.debug(s"inserted ${owcEntry.getClass.getSimpleName} $rowCount ${sql.toString}")
232 
233         owcEntry.offerings.foreach {
234           owcOffering => {
235             if (owcOfferingDAO.findOwcOfferingByUuid(owcOffering.uuid).isEmpty) {
236               owcOfferingDAO.createOwcOffering(owcOffering)
237             }
238 
239             SQL(
240               s"""insert into $tableOwcEntriesHasOwcOfferings  values (
241                  |{owc_feature_types_as_entry_id}, {owc_offerings_uuid}
242                  |)
243                """.stripMargin).on(
244               'owc_feature_types_as_entry_id -> owcEntry.id,
245               'owc_offerings_uuid -> owcOffering.uuid.toString
246             ).executeUpdate()
247           }
248         }
249 
250         // the single dependent owc properties object
251         if (owcPropertiesDAO.findOwcPropertiesByUuid(owcEntry.properties.uuid).isEmpty) {
252           owcPropertiesDAO.createOwcProperties(owcEntry.properties)
253         }
254 
255         SQL(
256           s"""insert into $tableOwcFeatureTypesHasOwcProperties  values (
257              |{owc_feature_types_id}, {owc_properties_uuid}
258              |)
259                """.stripMargin).on(
260           'owc_feature_types_id -> owcEntry.id,
261           'owc_properties_uuid -> owcEntry.properties.uuid.toString
262         ).executeUpdate()
263 
264         rowCount match {
265           case 1 => Some(owcEntry)
266           case _ => None
267         }
268       }
269     }
270   }

301       }
302     }
303 
304     db.withConnection(
305       implicit connection => {
306 
307         owcEntry.offerings.filter {
308           offering => {
309             SQL(s"""select owc_offerings_uuid from $tableOwcEntriesHasOwcOfferings where owc_offerings_uuid = {uuid}""").on(
310               'uuid -> offering.uuid.toString
311             ).as(SqlParser.str("owc_offerings_uuid") *).isEmpty
312           }
313         }.foreach(owcOfferingDAO.deleteOwcOffering(_))
314 
315         if (SQL(s"""select owc_properties_uuid from $tableOwcFeatureTypesHasOwcProperties where owc_properties_uuid = {uuid}""").on(
316           'uuid -> owcEntry.properties.uuid.toString
317         ).as(SqlParser.str("owc_properties_uuid") *).isEmpty) {
318           owcPropertiesDAO.deleteOwcProperties(owcEntry.properties)
319         }
320       }
321     )
322 
323     rowCount match {
324       case 1 => true
325       case _ => false
326     }
327   }
