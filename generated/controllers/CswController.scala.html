113   /**
114     * trying to generalise for all transaction types
115     *
116     * @param transaction
117     * @return
118     */
119   def initiateTransactionRequest(transaction: CswTransactionWithIndexUpdate): Future[(WSResponse, WSResponse)] = {
120     //FIXME SR I find that pretty hard to read. Is there a better way of chaining WS calls?
121     // see https://www.playframework.com/documentation/2.5.x/ScalaWS#chaining-ws-calls
122     for {
123       getCapaResponse <- wsClient.url(CSW_OPERATIONS_METADATA_URL).get()
124       transactionResponse <- {
125         //check if getCapaResponse indicates, that the CSW can perform Transactions
126         val operation = (getCapaResponse.xml \\ "OperationsMetadata" \\ "Operation").filter(node => {
127           logger.debug(s"Attribute 'name'=${node.attribute("name").get.text.toString}")
128           node.attribute("name").get.text.equals("Transaction")
129         })
130         logger.debug(s"operation = ${operation.toString()}")
131         if (operation.isEmpty) {
132           throw new UnsupportedOperationException("CSW does not support Transaction.")
133         }
134 
135         // prepare transaction template
136         val finalXML = transaction.transform
137 
138         logger.debug(s"finalXml: ${finalXML.toString()}")
139         wsClient.url(cswInternalApiUrl).post(finalXML.toString())
140       }
141 
142       updateIngesterIndexResponse <- {
143         // if (transactionResponse.status == 200) {
144         val responseTuple = if (transaction.transactionType.equals(CswTransctionType.DELETE)) {
145           (wsClient.url(INGESTER_DELETE_FROM_INDEX_URL).withQueryString("fileIdentifier" -> transaction.fileIdentifier)get(), INGESTER_DELETE_FROM_INDEX_URL)
146         } else {
147           (wsClient.url(INGESTER_UPDATE_INDEX_URL).get(), INGESTER_UPDATE_INDEX_URL)
148         }
149         responseTuple._1.onSuccess {
150           case response if response.status == 200 => {
151             logger.info(
152               s"Successfully called '${responseTuple._2}' (${response.status} - ${response.statusText})")
153             logger.info(s"response: ${response.body}")
154           }
155           case response => {
156             logger.warn(
157               s"Call to '${responseTuple._2}' returned error: ${response.status} - ${response.statusText}")
158             logger.warn(s"response: ${response.body}")
159           }
160         }
161         responseTuple._1
162 
163       }
164     } yield (transactionResponse, updateIngesterIndexResponse)
165   }

167   /**
168     * check the response of the tansaction
169     *
170     * @param response
171     * @param transaction
172     * @return
173     */
174   def evaluateTransactionResponse(response: WSResponse, transaction: CswTransactionWithIndexUpdate): Either[ErrorResult, UUID] = {
175     logger.debug(response.xml.toString())
176     response.xml match {
177       case e: Elem if e.label == "TransactionResponse" =>
178 
179         if (transaction.responsePositive(e)) {
180           Right(UUID.fromString(transaction.fileIdentifier))
181         } else {
182           Left(ErrorResult("Identifier not clear in TransactionResponse", None))
183         }
184 
185       case e: Elem if e.label == "ExceptionReport" =>
186         val message = (e \\ "Exception" \\ "ExceptionText").text
187         val error = ErrorResult(message, None)
188         Left(error)
189 
190       case _ =>
191         val error = ErrorResult(s"Unexpected Response from CSW: ${response.status} - ${response.statusText}",
192           Some(response.body))
193         Left(error)
194 
195     }
196   }

203   def insertMetadataRecord: Action[JsValue] = (authenticationAction andThen userAction).async(parse.json) {
204     request =>
205 
206       logger.debug(request.toString)
207       (request.body.as[JsObject] \ "metadata").validate[MdMetadata].fold(
208         errors => {
209           logger.error(JsError.toJson(errors).toString())
210           val error = ErrorResult("Could not validate metadata from your request.", Some(JsError.toJson(errors).toString()))
211           Future {
212             BadRequest(Json.toJson(error)).as(JSON)
213           }
214         },
215         mdMetadata => {
216           logger.trace(Json.prettyPrint(mdMetadata.toJson()))
217           logger.debug(s"MD_Metadata XML: ${mdMetadata.toXml()}")
218 
219           val insertTransaction = CswInsertRequest(cswtInsertXml, mdMetadata)
220           val futureResponse: Future[(WSResponse, WSResponse)] = initiateTransactionRequest(insertTransaction)
221 
222           futureResponse.recover {
223             case e: Exception =>
224               logger.error("Insert CSW threw exception", e)
225               val error = ErrorResult(s"Exception (${e.getClass.getCanonicalName}) on CSW insert: ${e.getMessage}",
226                 None)
227               InternalServerError(Json.toJson(error)).as(JSON)
228           }
229 
230           futureResponse.map { response =>
231             logger.debug("Ingester Update: " + response._2.body)
232             val responseEval = evaluateTransactionResponse(response._1, insertTransaction)
233             responseEval.fold[Result](
234               error => {
235                 logger.warn(error.message)
236                 InternalServerError(Json.toJson(error)).as(JSON)
237               },
238               uuid => {
239                 logger.debug(s"Adding ${mdMetadata.fileIdentifier} to default collection of ${request.user.email}.")
240                 val userMetaEntryOk = userService.insertUserMetaRecordEntry(portalExternalBaseUrl, mdMetadata.fileIdentifier, request.user.email)
241                 userMetaEntryOk.fold {
242                   val error = ErrorResult(s"Could not add ${mdMetadata.fileIdentifier} to your collection of ${request.user.email}.", None)
243                   logger.warn(error.message)
244                   BadRequest(Json.toJson(error)).as(JSON)
245                 } {
246                   userMetaEntry =>
247                     val owcresource = collectionsService.generateMdResource(portalExternalBaseUrl, mdMetadata, userMetaEntry)
248                     val added = collectionsService.addMdResourceToUserDefaultCollection(portalExternalBaseUrl, owcresource, userMetaEntry)
249                     if (added) {
250                       Ok(Json.obj("type" -> "success", "fileIdentifier" -> userMetaEntry.originaluuid.toString, "userMetaRecord" -> Json.toJson(userMetaEntry),
251                         "message" -> s"Inserted as ${uuid.toString} and reference entry with with refUuid ${userMetaEntry.uuid} added to your data collection."))
252                     } else {
253                       val error = ErrorResult(s"Could not add ${mdMetadata.fileIdentifier} to your collection of ${request.user.email}.", None)
254                       logger.warn(error.message)
255                       BadRequest(Json.toJson(error)).as(JSON)
256                     }
257                 }
258               })
259           }
260         }
261       )
262   }
