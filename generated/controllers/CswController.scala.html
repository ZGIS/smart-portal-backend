108   def initiateTransactionRequest(transaction: CswTransactionWithIndexUpdate): Future[(WSResponse, WSResponse)] = {
109     //FIXME SR I find that pretty hard to read. Is there a better way of chaining WS calls?

111     for {
112       getCapaResponse <- wsClient.url(CSW_OPERATIONS_METADATA_URL).get()
113       transactionResponse <- {
114         //check if getCapaResponse indicates, that the CSW can perform Transactions
115         val operation = (getCapaResponse.xml \\ "OperationsMetadata" \\ "Operation").filter(node => {
116           logger.debug(s"Attribute 'name'=${node.attribute("name").get.text.toString}")
117           node.attribute("name").get.text.equals("Transaction")
118         })
119         logger.debug(s"operation = ${operation.toString()}")
120         if (operation.isEmpty) {
121           throw new UnsupportedOperationException("CSW does not support Transaction.")
122         }
123 
124         // prepare transaction template
125         val finalXML = transaction.transform
126 
127         logger.debug(s"finalXml: ${finalXML.toString()}")
128         wsClient.url(CSW_URL).post(finalXML.toString())
129       }
130 
131       updateIngesterIndexResponse <- {
132         // if (transactionResponse.status == 200) {
133         val response = wsClient.url(INGESTER_UPDATE_INDEX_URL).get()
134         response.onSuccess {
135           case response if response.status == 200 => {
136             logger.info(

138             logger.info(s"response: ${response.body}")

159   def evaluateTransactionResponse(response: WSResponse, transaction: CswTransactionWithIndexUpdate): Either[ErrorResult, UUID] = {
160     logger.debug(response.xml.toString())
161     response.xml match {
162       case e: Elem if e.label == "TransactionResponse" =>
163 
164         if (transaction.responsePositive(e)) {
165           Right(UUID.fromString(transaction.fileIdentifier))
166         } else {
167           Left(ErrorResult("Identifier not clear in TransactionResponse", None))
168         }
169 
170       case e: Elem if e.label == "ExceptionReport" =>
171         val message = (e \\ "Exception" \\ "ExceptionText").text
172         val error = ErrorResult(message, None)
173         Left(error)
174 
175       case _ =>
176         val error = ErrorResult(s"Unexpected Response from CSW: ${response.status} - ${response.statusText}",
177           Some(response.body))
178         Left(error)
179 
180     }

236                         "message" -> s"Inserted as ${uuid.toString} and reference entry added to your data collection."))

239                       logger.warn(error.message)
