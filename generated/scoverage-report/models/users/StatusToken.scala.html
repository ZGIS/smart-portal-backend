46   def apply(v: String) : StatusToken = {
47     v match {
48       case "REGISTERED" => REGISTERED
49       case "ACTIVE" => ACTIVE
50       case "PASSWORDRESET" => PASSWORDRESET
51       case "EMAILVALIDATION" => EMAILVALIDATION
52       case _ => throw new IllegalArgumentException(s"Value $v is not a valid StatusToken value")
53     }
54   }
55 
56   val activatedTokens: List[String] = List(ACTIVE.value, PASSWORDRESET.value, EMAILVALIDATION.value)
57 
58   def isValid(v: String) : Boolean = {
59     Try (apply(v)).isSuccess
60   }
61 
62   def isActive(token: StatusToken): Boolean = {
63     activatedTokens.contains(token.value)
64   }
