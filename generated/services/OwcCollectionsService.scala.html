261   /**
262     * generate the plain MdMetadataa resource for a CSW entry reference
263     *
264     * @param externalCatalogUrl
265     * @param mdMetadata
266     * @param userMetaEntry
267     * @return
268     */
269   def generateMdResource(externalCatalogUrl: String, mdMetadata: MdMetadata, userMetaEntry: UserMetaRecord): OwcResource = {
270 
271     lazy val bboxFormat = new BboxArrayFormat
272 
273     val updatedTime = Try(OffsetDateTime.of(mdMetadata.citation.ciDate, LocalTime.of(12, 0), ZoneOffset.UTC))
274       .getOrElse(OffsetDateTime.now(ZoneId.systemDefault()))
275 
276     val baseLink = new URL(s"${portalConfig.portalExternalBaseLink}/context/resource/${userMetaEntry.uuid.toString}")
277 
278     val cswGetCapaOps = OwcOperation(
279       code = "GetCapabilities",
280       method = "GET",
281       mimeType = Some("application/xml"),
282       requestUrl = new URL(s"${portalConfig.portalExternalBaseLink}/pycsw/csw?SERVICE=CSW&VERSION=2.0.2&REQUEST=GetCapabilities"))
283 
284     val cswGetRecordOps = OwcOperation(
285       code = "GetRecordById",
286       method = "GET",
287       mimeType = Some("application/xml"),
288       requestUrl = new URL(s"${portalConfig.portalExternalBaseLink}/pycsw/csw?request=GetRecordById&service=CSW&version=2.0.2&elementSetName=full&outputSchema=http://www.isotc211.org/2005/gmd&id=${mdMetadata.fileIdentifier}"))
289 
290     val cswOffering = OwcOffering(
291       code = OwcOfferingType.CSW.code,
292       operations = List(cswGetCapaOps, cswGetRecordOps)
293     )
294 
295     val viaLink = OwcLink(
296       href = baseLink,
297       mimeType = Some("application/json"),
298       rel = "via")
299 
300     val cswLink = OwcLink(
301       href = cswGetRecordOps.requestUrl,
302       mimeType = cswGetRecordOps.mimeType,
303       rel = "via")
304 
305     val owcResource = OwcResource(
306       id = baseLink,
307       geospatialExtent = bboxFormat.parseBboxArray(mdMetadata.extent.mapExtentCoordinates).toOption,
308       title = mdMetadata.title,
309       subtitle = mdMetadata.abstrakt.toOption(),
310       updateDate = updatedTime,
311       author = List(OwcAuthor(name = mdMetadata.responsibleParty.individualName.toOption(),
312         email = parseEmailStringtoEmailAddress(mdMetadata.responsibleParty.email),
313         uri = Some(new URL(mdMetadata.responsibleParty.orgWebLinkage)))),
314       publisher = mdMetadata.responsibleParty.pointOfContact.toOption(),
315       rights = mdMetadata.distribution.useLimitation.toOption(),
316       temporalExtent = GeoDateParserUtils.parseOffsetDateString(Some(mdMetadata.extent.temporalExtent)),
317 
318       // links.alternates[] and rel=alternate
319       contentDescription = List(),
320 
321       // aka links.previews[] and rel=icon (atom)
322       preview = List(),
323 
324       // aka links.data[] and rel=enclosure (atom)
325       contentByRef = List(),
326 
327       // aka links.via[] & rel=via
328       resourceMetadata = List(viaLink, cswLink),
329       offering = List(cswOffering),
330       minScaleDenominator = Try(mdMetadata.scale.toDouble).toOption,
331       maxScaleDenominator = None,
332       active = None,
333       keyword = mdMetadata.keywords.map(w => OwcCategory(term = w, scheme = None, label = None)),
334       folder = None)
335 
336     logger.trace(Json.prettyPrint(owcResource.toJson))
337 
338     owcResource
339   }

593   def parseEmailStringtoEmailAddress(emailString: String): Option[EmailAddress] = {
594 
595     if (EmailAddress.isValid(emailString)) {
596       Some(EmailAddress(emailString))
597     } else {
598       if (emailString.contains(",")) {
599         val commaFree = emailString.split(",").head.trim
600 
601         Try {
602           EmailAddress(commaFree)
603         } match {
604           case Success(e) => Some(e)
605           case _ => None
606         }
607       } else {
608         None
609       }
610     }
