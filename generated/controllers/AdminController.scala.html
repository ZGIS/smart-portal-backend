273   def callJenaKubeReload: Future[Option[ErrorResult]] = {
274     val digestHash = MessageDigest.getInstance("SHA-512").digest(appSecret.getBytes("UTF-8"))
275     val hex = digestHash.map("%02X" format _).mkString
276     val baseHost = new URL(portalConfig.portalExternalBaseLink).getHost
277     wsClient.url(ADMIN_JENA_UPDATE_URL)
278       .withHeaders("Authorization" -> s"Bearer $hex")
279       .withQueryString("env" -> baseHost)
280       .get().map { response =>
281       if (response.status == 200) {
282         logger.info(
283           s"Successfully called $ADMIN_JENA_UPDATE_URL (${response.status} - ${response.statusText})")
284         logger.info(s"response: ${response.body}")
285         None
286       } else {
287         logger.warn(
288           s"Call to '$ADMIN_JENA_UPDATE_URL' returned error: ${response.status} - ${response.statusText}")
289         logger.warn(s"response: ${response.body}")
290         Some(ErrorResult("Update of running vocab service failed.", Some(s"Call to '$ADMIN_JENA_UPDATE_URL' returned error: ${response.status} - ${response.statusText}")))
291       }
292     }
293   }
