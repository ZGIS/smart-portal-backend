40 /**
41   * inserts the MDMetadata into the TransactionInsert XML
42   *
43   * @param xml
44   */
45 private class AddMDMetadataToInsert(xml: Node) extends RewriteRule {
46   override def transform(n: Node): Seq[Node] = n match {
47     case e: Elem if e.label == "Insert" => {
48       new Elem(e.prefix, "Insert", e.attributes, e.scope, e.minimizeEmpty, (xml).toSeq: _*)
49     }
50     case x => x
51   }
52 }

54 case class CswInsertRequest(requestTemplateXml: Node, mdMetadata: MdMetadata, transaction: String = "INSERT") extends CswTransactionWithIndexUpdate with ClassnameLogger {
55   override def fileIdentifier: String = mdMetadata.fileIdentifier
56 
57   override def transform: NodeSeq = {
58     val rule = new RuleTransformer(new AddMDMetadataToInsert(mdMetadata.toXml()))
59     rule.transform(requestTemplateXml)
60   }
61 
62   override def responsePositive(e: Elem) : Boolean = {
63     logger.debug(e.toString())
64     val resultingFileIdentifier = (e \\ "InsertResult" \\ "BriefRecord" \\ "identifier").text
65     resultingFileIdentifier.equals(fileIdentifier)
