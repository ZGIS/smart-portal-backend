156             val acls: java.util.List[Acl] = List(roAcl).asJava
157 
158             // the inputstream is closed by default, so we don't need to close it here
159             val blob: Blob = storage.create(BlobInfo.newBuilder(googleStorageBucket, filename).setAcl(acls).build(),
160               Files.asByteSource(tmpFile).openStream())
161 
162             // return the public download link
163             val owcEntry = fileMetadata(filename, contentType, authUser, blob.getMediaLink())
164             val insertOk = collectionsService.addPlainFileEntryToUserDefaultCollection(owcEntry, authUser)
165 
166             if (insertOk) {
167               logger.debug(s"file upload $filename to ${blob.getMediaLink}")
168               Ok(Json.obj("status" -> "OK", "message" -> s"file uploaded $filename.", "file" -> blob.getMediaLink(), "entry" -> owcEntry.toJson))
169             } else {
170               logger.error("file metadata insert failed.")
171               BadRequest(Json.obj("status" -> "ERR", "message" -> "file metadata insert failed."))
172             }
173           }.getOrElse {
174             logger.error("file upload failed")
175             BadRequest(Json.obj("status" -> "ERR", "message" -> "file upload failed."))
176           }
