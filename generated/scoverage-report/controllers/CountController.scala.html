28 /**
29  * This controller demonstrates how to use dependency injection to
30  * bind a component into a controller class. The class creates an
31  * `Action` that shows an incrementing count to users. The [[Counter]]
32  * object is injected by the Guice dependency injection system.
33  */
34 @Singleton
35 class CountController @Inject() (counter: Counter) extends Controller {
36 
37   /**
38    * Create an action that responds with the [[Counter]]'s current
39    * count. The result is plain text. This `Action` is mapped to
40    * `GET /count` requests by an entry in the `routes` config file.
41    */
42   def count = Action { Ok(counter.nextCount().toString) }
