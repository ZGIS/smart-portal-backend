621   /**
622     *
623     * @param owcProperties
624     * @return
625     */
626   def createOwcProperties(owcProperties: OwcProperties): Option[OwcProperties] = {
627 
628     db.withTransaction {
629       implicit connection => {
630 
631         val rowCount = SQL(
632           s"""
633           insert into $tableOwcProperties values (
634             {uuid}, {language}, {title}, {subtitle}, {updated}, {generator}, {rights}, {creator}, {publisher}
635           )
636         """).on(
637           'uuid -> owcProperties.uuid.toString,
638           'language -> owcProperties.language,
639           'title -> owcProperties.title,
640           'subtitle -> owcProperties.subtitle,
641           'updated -> owcProperties.updated,
642           'generator -> owcProperties.generator,
643           'rights -> owcProperties.rights,
644           'creator -> owcProperties.creator,
645           'publisher -> owcProperties.publisher
646         ).executeUpdate()
647 
648         owcProperties.authors.foreach {
649           author => {
650             if (findOwcAuthorByUuid(author.uuid).isEmpty) {
651               createOwcAuthor(author)
652             }
653 
654             SQL(
655               s"""insert into $tableOwcPropertiesHasOwcAuthors values (
656                  |{owc_properties_uuid}, {owc_authors_uuid}
657                  |)
658                """.stripMargin).on(
659               'owc_properties_uuid -> owcProperties.uuid.toString,
660               'owc_authors_uuid -> author.uuid.toString
661             ).executeUpdate()
662           }
663         }
664 
665         owcProperties.contributors.foreach {
666           authorAsContributor => {
667             if (findOwcAuthorByUuid(authorAsContributor.uuid).isEmpty) {
668               createOwcAuthor(authorAsContributor)
669             }
670 
671             SQL(
672               s"""insert into $tableOwcPropertiesHasOwcAuthorsAsContributors values (
673                  |{owc_properties_uuid}, {owc_authors_as_contributors_uuid}
674                  |)
675                """.stripMargin).on(
676               'owc_properties_uuid -> owcProperties.uuid.toString,
677               'owc_authors_as_contributors_uuid -> authorAsContributor.uuid.toString
678             ).executeUpdate()
679           }
680         }
681 
682         owcProperties.categories.foreach {
683           category => {
684             if (findOwcCategoriesByUuid(category.uuid).isEmpty) {
685               createOwcCategory(category)
686             }
687 
688             SQL(
689               s"""insert into $tableOwcPropertiesHasOwcCategories  values (
690                  |{owc_properties_uuid}, {owc_categories_uuid}
691                  |)
692                """.stripMargin).on(
693               'owc_properties_uuid -> owcProperties.uuid.toString,
694               'owc_categories_uuid -> category.uuid.toString
695             ).executeUpdate()
696           }
697         }
698 
699         owcProperties.links.foreach {
700           owcLink => {
701             if (findOwcLinksByUuid(owcLink.uuid).isEmpty) {
702               createOwcLink(owcLink)
703             }
704 
705             SQL(
706               s"""insert into $tableOwcPropertiesHasOwcLinks  values (
707                  |{owc_properties_uuid}, {owc_links_uuid}
708                  |)
709                """.stripMargin).on(
710               'owc_properties_uuid -> owcProperties.uuid.toString,
711               'owc_links_uuid -> owcLink.uuid.toString
712             ).executeUpdate()
713           }
714         }
715 
716         rowCount match {
717           case 1 => Some(owcProperties)
718           case _ => None
719         }
720       }
721     }
722   }
