35 
36   val cache: CacheApi
37   val configuration: Configuration
38   val passwordHashing: PasswordHashing
39 
40   val AuthTokenCookieKey = "XSRF-TOKEN"
41   val AuthTokenHeader = "X-XSRF-TOKEN"
42   val AuthTokenUrlKey = "auth"
43 
44   val UserAgentHeader = "User-Agent"
45   val UserAgentHeaderDefault = "Default-UA/1.0"
46 
47   /**
48     * Checks that the token is:
49     * - present in the cookie header of the request,
50     * - either in the header or in the query string,
51     * - matches a token already stored in the play cache
52     *
53     * @param p
54     * @param f
55     * @tparam A
56     * @return
57     */
58   def HasToken[A](p: BodyParser[A] = parse.anyContent)(
59     f: String => String => Request[A] => Result): Action[A] =
60   Action(p) { implicit request =>
61 
62     request.cookies.get(AuthTokenCookieKey).fold {
63       Unauthorized(Json.obj("status" -> "ERR", "message" -> "Invalid XSRF Token cookie"))
64     } { xsrfTokenCookie =>
65       Logger.trace(s"cookie ${xsrfTokenCookie.value}")
66       val maybeToken = request.headers.get(AuthTokenHeader).orElse(request.getQueryString(AuthTokenUrlKey))
67 
68       maybeToken flatMap { token =>
69         // ua needed to differentiate between different devices/sessions
70         val uaIdentifier: String = request.headers.get(UserAgentHeader).getOrElse(UserAgentHeaderDefault)
71         Logger.trace(s"token: $token")
72         Logger.trace(s"ua: $uaIdentifier")
73         // cache token -> maps to a String username
74         cache.get[String](token) map { username =>
75           // lazy val passwordHashing = new PasswordHashing(configuration)
76           val cookieForUSerAndDevice = passwordHashing.testSessionCookie(token, username, uaIdentifier)
77           Logger.trace(s"testcookie: $cookieForUSerAndDevice")
78           if (xsrfTokenCookie.value == token && cookieForUSerAndDevice) {
79             Logger.trace(s"request for active session: $username / $token / $uaIdentifier")
80             f(token)(username)(request)
81           } else {
82             Unauthorized(Json.obj("status" -> "ERR", "message" -> "Invalid Token"))
83           }
84         }
85       } getOrElse Unauthorized(Json.obj("status" -> "ERR", "message" -> "No Token"))
86     }
87   }
