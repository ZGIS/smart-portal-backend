60 /**
61   * inserts the MDMetadata into the TransactionInsert XML
62   *
63   * @param xml
64   */
65 private class AddMDMetadataToInsert(xml: Node) extends RewriteRule {
66   override def transform(n: Node): Seq[Node] = n match {
67     case e: Elem if e.label == "Insert" => {
68       new Elem(e.prefix, "Insert", e.attributes, e.scope, e.minimizeEmpty, (xml).toSeq: _*)
69     }
70     case x => x
71   }
72 }

74 final case class CswInsertRequest(requestTemplateXml: Node, mdMetadata: MdMetadata, transactionType: CswTransctionType = INSERT) extends CswTransactionWithIndexUpdate with ClassnameLogger {
75   override def fileIdentifier: String = mdMetadata.fileIdentifier
76 
77   override def transform: NodeSeq = {
78     val rule = new RuleTransformer(new AddMDMetadataToInsert(mdMetadata.toXml()))
79     rule.transform(requestTemplateXml)
80   }
81 
82   override def responsePositive(e: Elem) : Boolean = {
83     logger.debug(e.toString())
84     val resultingFileIdentifier = (e \\ "InsertResult" \\ "BriefRecord" \\ "identifier").text
85     resultingFileIdentifier.equals(fileIdentifier)
86   }
87 }
