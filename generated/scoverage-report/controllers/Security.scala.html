47   /**
48     * Checks that the token is:
49     * - present in the cookie header of the request,
50     * - either in the header or in the query string,
51     * - matches a token already stored in the play cache
52     *
53     * @param p
54     * @param f
55     * @tparam A
56     * @return
57     */
58   def HasToken[A](p: BodyParser[A] = parse.anyContent)(
59     f: String => String => Request[A] => Result): Action[A] =
60   Action(p) { implicit request =>
61 
62     request.cookies.get(AuthTokenCookieKey).fold {
63       Unauthorized(Json.obj("status" -> "ERR", "message" -> "Invalid XSRF Token cookie"))
64     } { xsrfTokenCookie =>
65       Logger.trace(s"cookie ${xsrfTokenCookie.value}")
66       val maybeToken = request.headers.get(AuthTokenHeader).orElse(request.getQueryString(AuthTokenUrlKey))
67 
68       maybeToken flatMap { token =>
69         // ua needed to differentiate between different devices/sessions
70         val uaIdentifier: String = request.headers.get(UserAgentHeader).getOrElse(UserAgentHeaderDefault)
71         Logger.trace(s"token: $token")
72         Logger.trace(s"ua: $uaIdentifier")
73         // cache token -> maps to a String email
74         cache.get[String](token) map { email =>
75           // lazy val passwordHashing = new PasswordHashing(configuration)
76           val cookieForUSerAndDevice = passwordHashing.testSessionCookie(token, email, uaIdentifier)
77           Logger.trace(s"testcookie: $cookieForUSerAndDevice")
78           if (xsrfTokenCookie.value == token && cookieForUSerAndDevice) {
79             Logger.trace(s"request for active session: $email / $token / $uaIdentifier")
80             f(token)(email)(request)
81           } else {
82             Unauthorized(Json.obj("status" -> "ERR", "message" -> "Invalid Token"))
83           }
84         }
85       } getOrElse Unauthorized(Json.obj("status" -> "ERR", "message" -> "No Token"))
86     }
87   }

89   /**
90     * Checks IF the token is:
91     * - present in the cookie header of the request,
92     * - either in the header or in the query string,
93     * - matches a token already stored in the play cache
94     * - WILL ALLOW also not auth authenticated ANONYMOUS user too
95     *
96     * @param p
97     * @param f
98     * @tparam A
99     * @return
100     */
101   def HasOptionalToken[A](p: BodyParser[A] = parse.anyContent)(
102     f: Option[String] => Request[A] => Result): Action[A] =
103     Action(p) { implicit request =>
104 
105       request.cookies.get(AuthTokenCookieKey).fold {
106         // Unauthorized(Json.obj("status" -> "ERR", "message" -> "Invalid XSRF Token cookie"))
107         Logger.trace("optional cookie: Invalid XSRF-Token cookie")
108         f(None)(request)
109       } { xsrfTokenCookie =>
110         Logger.trace(s"cookie ${xsrfTokenCookie.value}")
111         val maybeToken = request.headers.get(AuthTokenHeader).orElse(request.getQueryString(AuthTokenUrlKey))
112 
113         maybeToken flatMap { token =>
114           // ua needed to differentiate between different devices/sessions
115           val uaIdentifier: String = request.headers.get(UserAgentHeader).getOrElse(UserAgentHeaderDefault)
116           Logger.trace(s"token: $token")
117           Logger.trace(s"ua: $uaIdentifier")
118           // cache token -> maps to a String email
119           cache.get[String](token) map { email =>
120             // lazy val passwordHashing = new PasswordHashing(configuration)
121             val cookieForUSerAndDevice = passwordHashing.testSessionCookie(token, email, uaIdentifier)
122             Logger.trace(s"testcookie: $cookieForUSerAndDevice")
123             if (xsrfTokenCookie.value == token && cookieForUSerAndDevice) {
124               Logger.trace(s"request for active session: $email / $token / $uaIdentifier")
125               f(Some(email))(request)
126             } else {
127               // Unauthorized(Json.obj("status" -> "ERR", "message" -> "Invalid Token"))
128               Logger.trace("optional cookie: Invalid Token")
129               f(None)(request)
130             }
131           }
132         } getOrElse {
133           // Unauthorized(Json.obj("status" -> "ERR", "message" -> "No Token"))
134           Logger.trace("optional cookie: No Token")
135           f(None)(request)
136         }
137 
138       }
139     }
