59   def uploadMultipartForm: Action[MultipartFormData[TemporaryFile]] = (authenticationAction andThen userAction) (parse.multipartFormData) {
60     request =>
61       request.body.file("file").map { theFile =>
62 
63         val filename = theFile.filename
64         val contentType = theFile.contentType
65         val pathOfUploadTmp = Paths.get(uploadDataPath)
66         val intermTempDir = Files.createTempDirectory(pathOfUploadTmp, "sac-upload-")
67         val tmpFile = new File(intermTempDir.resolve(filename).toAbsolutePath.toString)
68         val handle = theFile.ref.moveTo(tmpFile)
69         val kilobytes = handle.length / 1024
70 
71         googleService.uploadFileGoogleBucket(handle) match {
72           case Left(errorResult) =>
73             logger.error(errorResult.message + errorResult.details.mkString)

75           case Right(blobInfo) =>
76             // return the public download link
77             val userFileEntryOk = userService.insertUserFileEntry(filename, request.user.email, blobInfo.mediaLink)
78             userFileEntryOk.fold {

80               val error = ErrorResult("File metadata insert failed.", None)
81               BadRequest(Json.toJson(error)).as(JSON)
82             } { userFile =>
83               val owcResource = collectionsService.fileMetadata(userFile, contentType, Try(kilobytes.toInt).toOption)
84               logger.trace(Json.prettyPrint(owcResource.toJson))
85               logger.debug(s"file upload $filename to ${blobInfo.mediaLink} with reference ${userFile.linkreference}; bob details: ${blobInfo.name} ${blobInfo.bucket} ${blobInfo.contentType} ${blobInfo.contentEncoding} ${blobInfo.size} ${blobInfo.md5}")
86               Try(handle.delete()).failed.map(ex => logger.error(ex.getLocalizedMessage))
87               Try(Files.delete(intermTempDir)).failed.map(ex => logger.error(ex.getLocalizedMessage))
88 
89               //FIXME SR do we also want to have a general "return" object? Status is always in the respone so it does not need to be in here
90               val added = collectionsService.addPlainFileResourceToUserDefaultCollection(owcResource, request.user)
91               if (added) {
92                 Ok(Json.obj("status" -> "OK", "message" -> s"file uploaded $filename.", "file" -> userFile.linkreference, "entry" -> owcResource.toJson))
93               } else {
94                 logger.error("Insert into own collection failed.")
95                 val error = ErrorResult("Insert into own collection failed.", None)
96                 BadRequest(Json.toJson(error)).as(JSON)
97               }
98             }
99           case _ =>
100             logger.error("file upload to cloud storage failed")
101             val error = ErrorResult("File upload to cloud storage failed.", None)
102             BadRequest(Json.toJson(error)).as(JSON)
103         }
104       }.getOrElse {
105         logger.error("file upload failed, no file found for upload")
106         val error = ErrorResult("file upload failed, no file found for upload", None)
107         BadRequest(Json.toJson(error)).as(JSON)
108       }
109   }
