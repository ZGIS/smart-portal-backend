28 /**
29   * Model of OWS Context Documents and provide GeoJson encoding thereof (and maybe AtomXML)
30   * An OWC document is an extended FeatureCollection, where the features (aka entries) hold a variety of metadata
31   * about the things they provide the context for (i.e. other data sets, services, metadata records)
32   * OWC documents do not duplicate a CSW MD_Metadata record, but a collection of referenced resources;
33   *
34   * http://www.opengeospatial.org/standards/owc
35   *
36   * Classically, the WMC documents (Web Map Context documents) were a list of WMS layers for a web map viewer
37   * with a certain context, i.e. title, Bounding Box and a few visualisation properties like scale/zoom,
38   * OWC has superseded that concept into a generic collection of resources:
39   *
40   * We use OWC primarily in the form of collections of case studies, preferably with at least two offerings per entry:
41   * 1) a web visualisable form, e.g.WMS, WFS, SOS ...
42   * 2) a CSW addressable MD_Metadata record according to the resource
43   *
44   * The OWC JSON Encoding is a profile of GeoJSON FeatureCollection, the XML encoding is a profile of Atom/GeoRSS Feed
45   */

190     resultFromJson match {
191       case JsSuccess(r: OwcOperation, path: JsPath) => Some(r)
192       case e: JsError => {
193         val lines = e.errors.map { tupleAction =>
194           val jsPath = tupleAction._1
195           val valErrors = tupleAction._2.map( valErr => valErr.message ).toList.mkString(" ; ")
196           jsPath.toJsonString + " >> " +  valErrors
197         }
198 
199         logger.error(s"JsError info  ${lines.mkString(" | ")}")

201       }

290   private lazy val OwcReqPattern = "^http://www.opengis.net/spec/owc-(geojson|atom)/1.0/req/(wms|wmts|wfs|wcs|csw|wps|gml|kml|geotiff|sos|netcdf|http-link)$".r
291 
292   def buildOfferingFrom(offeringType: String, operations: List[OwcOperation], content: List[String]): OwcOffering = {
293     // Todo, we might find a way to find an existing UUID from DB if entry exists
294     val uuid = UUID.randomUUID()
295 
296     val geoJsonCode = "http://www.opengis.net/spec/owc-geojson/1.0/req/" + offeringType
297 
298     offeringType match {
299       case "wms" => WmsOffering(uuid, geoJsonCode, operations, content)
300       case "wmts" => WmtsOffering(uuid, geoJsonCode, operations, content)
301       case "wfs" => WfsOffering(uuid, geoJsonCode, operations, content)
302       case "wcs" => WcsOffering(uuid, geoJsonCode, operations, content)
303       case "csw" => CswOffering(uuid, geoJsonCode, operations, content)
304       case "wps" => WpsOffering(uuid, geoJsonCode, operations, content)
305       case "gml" => GmlOffering(uuid, geoJsonCode, operations, content)
306       case "kml" => KmlOffering(uuid, geoJsonCode, operations, content)
307       case "geotiff" => GeoTiffOffering(uuid, geoJsonCode, operations, content)
308       case "sos" => SosOffering(uuid, geoJsonCode, operations, content)
309       case "netcdf" => NetCdfOffering(uuid, geoJsonCode, operations, content)
310       case "http-link" => HttpLinkOffering(uuid, geoJsonCode, operations, content)
311     }
312   }
313 
314   /**
315     *
316     * @param code
317     * @param operations
318     * @param content
319     * @return
320     */
321   def apply(code: String, operations: List[OwcOperation], content: List[String]): OwcOffering = {
322 
323     code match {
324       case OwcReqPattern(owcEncoding, offeringType) => {
325         logger.debug(s"$owcEncoding, $offeringType")
326         buildOfferingFrom(offeringType, operations, content)
327       }
328       case _ => throw new IllegalArgumentException(s"offering code cannot be used to build offering type: $code")
329     }
330   }
331 
332   /**
333     *
334     * @param arg
335     * @return
336     */
337   def unapply(arg: OwcOffering): Option[(String, List[OwcOperation], List[String])] = {
338     Some(arg.code, arg.operations, arg.content)
339   }
