64   implicit def columnToBoolean: Column[Boolean] = {
65     Column.nonNull { (value, meta) =>
66       val MetaDataItem(qualified, nullable, clazz) = meta
67       value match {
68         case bool: Boolean => Right(bool) // Provided-default case
69         case bit: Int => Right(bit == 1) // Custom conversion
70         case _ => {
71           Left(TypeDoesNotMatch(s"Cannot convert $value: ${value.asInstanceOf[AnyRef].getClass} to Boolean for column $qualified"))
72         }
73       }
74     }

81   implicit def columnToOffset: Column[OffsetDateTime] = {
82     Column.nonNull { (value, meta) =>
83       val MetaDataItem(qualified, nullable, clazz) = meta
84       value match {
85         case offsetDateTime: OffsetDateTime => Right(offsetDateTime) // default case
86         case zonedDateTime: ZonedDateTime => Right(OffsetDateTime.of(zonedDateTime.toLocalDateTime, zonedDateTime.getOffset)) // Custom conversion
87         case h2DateTime: org.h2.api.TimestampWithTimeZone => {
88           // val offsetMinutes = h2DateTime.getTimeZoneOffsetMins
89           val localDateTime = h2DateTime.toLocalDateTime
90           Right(OffsetDateTime.of(localDateTime, ZoneOffset.ofHoursMinutesSeconds(0,0,0)))
91         } // Custom conversion, H2 time stamp with timezone is experimental
92         case sqlTimeStamp: java.sql.Timestamp => {
93           val localDateTime = sqlTimeStamp.toLocalDateTime
94           Right(OffsetDateTime.of(localDateTime, ZoneOffset.ofHoursMinutesSeconds(0,0,0)))
95         } // Custom conversion to omit H2 TimeStamp with TimeZone, for Testing don't use ZoneOffsets
96         case _ => {
97           Left(TypeDoesNotMatch(s"Cannot convert $value: ${value.asInstanceOf[AnyRef].getClass} to OffsetDateTime for column $qualified"))
98         }
99       }
100     }
