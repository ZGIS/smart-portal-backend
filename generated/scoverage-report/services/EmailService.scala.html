48     * send a Registration Email with a pre-generated unique link to confirm your account

56   def sendRegistrationEmail(emailTo: String, subject: String, usernameTo: String, linkId: String): Boolean = {
57 
58     mailServerProperties.put("mail.smtp.host", emailServAddress)
59     mailServerProperties.put("mail.smtp.port", emailServPort)
60     mailServerProperties.put("mail.smtp.auth", "true")
61     mailServerProperties.put("mail.smtp.starttls.enable", "true")
62 
63     getMailSession = Session.getDefaultInstance(mailServerProperties, null)
64 
65     val emailText =
66       """Hello %s,
67         |thank you for registering on the GW HUB.
68         |Please click on the following link to confirm your registration:
69         |
70         |http://dev.smart-project.info/api/v1/users/register/%s
71         |
72         |If you have any question please email us to %s.
73         |
74         |Your GW HUB Team
75       """.format(usernameTo, linkId, emailFrom).stripMargin
76 
77     try {
78       // Create a default MimeMessage object.
79       val message = new MimeMessage(getMailSession)
80       // Set From: header field of the header.
81       message.setFrom(new InternetAddress(emailFrom))
82       // Set To: header field of the header.
83       message.addRecipient(Message.RecipientType.TO, new InternetAddress(emailTo))
84       // Set Subject: header field. eg "Please confirm your GW HUB account"
85       message.setSubject(subject)
86 
87       message.setText(emailText)
88       logger.trace(emailText)
89 
90       // Send message
91       val transport: Transport = getMailSession.getTransport("smtp")
92       transport.connect(emailServAddress, emailUsername, emailPassword)
93       transport.sendMessage(message, message.getAllRecipients)
94       transport.close()
95       true
96     } catch {
97       case mex: MessagingException => {
98         logger.error("Messaging exception: " + mex.getLocalizedMessage)
99         false
100       }
101       case e: Exception => {
102         logger.error("Other email problem: " + e.getLocalizedMessage)
103         false
104       }
105     }
106   }
