265     request =>

290               Ok(Json.toJson(user.asProfileJs))
291             }
292           } else {
293             // check if new email is used already by another user than yourself
294             userService.findUserByEmailAddress(incomingProfileJs.email).fold {
295 
296               // found none, good, let's update your profile in the database (but not password nor accountsubject)

324     * should require "double check" in gui to issue this, requires to accept and store new session secret

328   def updatePassword(): Action[JsValue] = defaultAuthAction(parse.json) {
329     request =>
330       request.body.validate[PasswordUpdateCredentials].fold(
331         errors => {
332           logger.error(JsError.toJson(errors).toString())
333           val error = ErrorResult("Could not validate request.", Some(JsError.toJson(errors).toString()))
334           BadRequest(Json.toJson(error)).as(JSON)
335         },
336         valid = credentials => {
337           // find user in db
338           val dbuser = request.user
339           // compare password
340           if (passwordHashing.validatePassword(credentials.oldPassword, dbuser.password)) {
341             val newCryptPass = passwordHashing.createHash(credentials.newPassword)
342             val updateUser = dbuser.copy(
343               password = newCryptPass,
344               laststatustoken = s"${StatusToken.ACTIVE}:PASSWORDUPDATE",
345               laststatuschange = ZonedDateTime.now.withZoneSameInstant(ZoneId.of(appTimeZone)))
346 
347             userService.updatePassword(updateUser).fold {
348               logger.error("Password update error.")
349               val error = ErrorResult("Password update error.", None)
350               BadRequest(Json.toJson(error)).as(JSON)
351             } { user =>
352               val uaIdentifier: String = request.headers.get(UserAgentHeader).getOrElse(UserAgentHeaderDefault)
353               // cache.remove(token)
354               userService.removeUserSessionCache(request.user.email, request.authenticatedRequest.userSession.token)
355               emailService.sendPasswordUpdateEmail(user.email, "Password Update on GW HUB", user.firstname)
356               // val newtoken = passwordHashing.createSessionCookie(user.email, uaIdentifier)
357               // cache.set(newtoken, user.email.value)
358               val newtoken = userService.upsertUserSession(user.email.value, uaIdentifier)
359               Ok(Json.obj("status" -> "OK", "token" -> newtoken, "email" -> user.email.value))
360                 .withCookies(Cookie(AuthTokenCookieKey, newtoken, None, httpOnly = false))
361             }
362           } else {
363             logger.error("Password update error, not matching.")
364             val error = ErrorResult("Password update error, not matching.", None)
365             BadRequest(Json.toJson(error)).as(JSON)
366           }
367         })
368   }

489       valid = gAuthCredentials => {
490         val accessTokenResult = googleService.getGoogleAuthorization(gAuthCredentials)
491         accessTokenResult match {
492           case Left(errorResult) => {
493             logger.error(errorResult.message + errorResult.details.mkString)
494             InternalServerError(Json.toJson(errorResult)).as(JSON)
495           }
496           case Right(googleTokenResponse) => {
497             // Get profile info from ID token
498             val gAuthPayload = googleTokenResponse.parseIdToken().getPayload
499             // Use this value as a key to identify a user.
500             val userId = gAuthPayload.getSubject()
501             val email = gAuthPayload.getEmail()
502             val emailVerified = gAuthPayload.getEmailVerified()
503             // get further payload things that is easy in java but needs to work in scala too
504             val name = Option(gAuthPayload.get("name").asInstanceOf[String])
505             val familyName = Option(gAuthPayload.get("family_name").asInstanceOf[String])
506             val givenName = Option(gAuthPayload.get("given_name").asInstanceOf[String])
507             if ((gAuthCredentials.accesstype.equalsIgnoreCase("LOGIN") || gAuthCredentials.accesstype.equalsIgnoreCase("REGISTER")) &&
508               emailVerified && EmailAddress.isValid(email)) {
509 
510               val emailAddress = EmailAddress(email)
511               // viaConnection lookup email block
512               val firstUserLookup = userService.findUserByEmailAddress(emailAddress)
513 
514               firstUserLookup.fold {
515                 // 'error, no user': well, must be a new user coming through from Google
516                 val cryptPass = passwordHashing.createHash(UUID.randomUUID().toString)
517
