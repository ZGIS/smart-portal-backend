82     * Loads a time series from a SOS server. A Timeseries "configuration" has to be passd within the POST
83     * request.

87   def getTimeseries: Action[JsValue] = optionalAuthenticationAction.async(parse.json) {
88     request =>
89       request.body.validate[Timeseries].fold(
90         errors => {
91           logger.error(JsError.toJson(errors).toString())
92           val error = ErrorResult("Error while validating request.", Some(JsError.toJson(errors).toString))
93           Future {
94             BadRequest(Json.toJson(error)).as(JSON)
95           }
96         },
97 
98         timeseries => {
99           val sosXmlRequest = getObservationRequestXml(timeseries)
100           val responseFuture = wsClient.url(timeseries.sosUrl)
101             .withHeaders(CONTENT_TYPE -> "application/xml")
102             .post(sosXmlRequest)
103 
104           responseFuture.recover {
105             case e: Exception =>
106               logger.error("Sos GetObservation threw exception", e)
107               val error = ErrorResult(s"Exception (${e.getClass.getCanonicalName}) on SOS get capabilities.",
108                 Some(e.getMessage))
109               InternalServerError(Json.toJson(error)).as(JSON)
110           }
111 
112           responseFuture.map { response =>
113             if (response.status != 200) {
114               logger.error(s"Calling SOS ${timeseries.sosUrl} HTTP result status ${response.status} on GetObservation")
115               val error = ErrorResult(s"Server ${timeseries.sosUrl} responded with ${response.status} on GetObservation",
116                 Some(response.body))
117               InternalServerError(Json.toJson(error)).as(JSON)
118             }
119             else if (!response.header("Content-Type").get.contains("application/xml")) {
120               logger.error(s"SOS GetObservation result was not XML but ${response.header("Content-Type").get}")
121               val error = ErrorResult(
122                 s"SOS GetObservation result was not XML but ${response.header("Content-Type").get}",
123                 Some(response.body))
124               InternalServerError(Json.toJson(error)).as(JSON)
125             }
126             else if (timeseries.responseFormat.isDefined && !timeseries.responseFormat.get.equalsIgnoreCase("http://www.opengis.net/om/2.0")) {
127               logger.error(s"SOS GetObservation responseFormat unsupported for this request; ${timeseries.responseFormat.getOrElse("none")}")
128               val error = ErrorResult(
129                 s"SOS GetObservation responseFormat unsupported for this request; ${timeseries.responseFormat.getOrElse("none")}",
130                 Some(response.body))
131               InternalServerError(Json.toJson(error)).as(JSON)
132             }
133             else {
134               val tvp = new XmlTvpParser().parseOm2Measurements(Source.fromString(response.body))
135               val tsdata =
136                 if (tvp.isEmpty) {
137                   TimeseriesData(
138                     x = Seq.empty[String],
139                     y = Seq.empty[String],
140                     name = timeseries.timeseriesName
141                   )
142                 }
143                 else {
144                   TimeseriesData(
145                     x = tvp.map(_.datetime),
146                     y = tvp.map(_.measValue),
147                     name = timeseries.timeseriesName
148                   )
149                 }
150               val uom = if (tvp.isEmpty) {
151                 if (timeseries.uom.isEmpty) {
152                   timeseries.observedProperty
153                 } else {
154                   timeseries.uom.getOrElse("empty")
155                 }
156               } else tvp.head.measUnit
157               val result = timeseries.copy(data = Some(tsdata), uom = Some(uom))
158 
159               val logRequest = UserLinkLogging(id = None,
160                 timestamp = ZonedDateTime.now.withZoneSameInstant(ZoneId.of(portalConfig.appTimeZone)),
161                 ipaddress = Some(request.remoteAddress),
162                 useragent = request.headers.get(UserAgentHeader),
163                 email = request.optionalSession.map(_.email),
164                 link = sosXmlRequest,
165                 referer = request.headers.get(RefererHeader))
166 
167               val updated = userService.logLinkInfo(logRequest)
168 
169               logger.trace(logRequest.toString)
170               Ok(result.toJson()).as(JSON)
171             }
172           }
173         }
174       )
175   }

316   def exportToFile(responseFormat: Option[String], sosResponseAsText: String,
317                    sosCapabilities: SosCapabilities, sosXmlRequest: String): Either[ErrorResult, File] = {
318 
319     val updatedTime = OffsetDateTime.now(ZoneId.of(portalConfig.appTimeZone))
320     val fileNameTmpl = "export-" + Try(URLEncoder.encode(sosCapabilities.title.replace(" ", "_"), "UTF-8")
321       + "-" + updatedTime.format(formatter)).getOrElse("-sosdata")
322     val pathOfUploadTmp = Paths.get(portalConfig.uploadDataPath)
323     val intermTempDir = Files.createTempDirectory(pathOfUploadTmp, "sos-export-")
324 
325     responseFormat match {
326       case Some(WML2.value) => {
327         val wml2 = wml2Exporter.getWml2ExportFromSosGetObs(sosResponseAsText, sosCapabilities, sosXmlRequest)
328 
329         if (wml2.isEmpty) {
330           logger.error("Couldn't extract OM2 into SpreadSheet from this SOS GetObservation response")
331           Left(ErrorResult(message = "Couldn't extract OM2 into SpreadSheet from this SOS GetObservation response",
332             details = Some("Sheet is empty, no data?")))
333         } else {
334           val tmpFile = new File(intermTempDir.resolve(fileNameTmpl + WML2.fileSuffix).toAbsolutePath.toString)
335           scala.xml.XML.save(tmpFile.getAbsolutePath, wml2.get.head, "UTF-8", true, null)
336           // return ref handle
337           Right(tmpFile)
338         }
339       }
340       case Some(spreadSheet) if spreadSheet.equals(XLS.value) || spreadSheet.equals(CSV.value) => {
341         val format = SosDataFormat.apply(spreadSheet)
342         val tmpFile = new File(intermTempDir.resolve(fileNameTmpl + format.fileSuffix).toAbsolutePath.toString)
343         val sheets = xlsExporter.getSpreadsheetExportFromSosGetObs(sosResponseAsText, sosCapabilities, sosXmlRequest)
344         val saved = saveSpreadSheet(sheets, format, tmpFile.getAbsolutePath)
345         saved.map(err => Left(err)).getOrElse(Right(tmpFile))
346       }
347       case Some(otherText) =>
348         logger.error(s"Couldn't extract OM2 into SpreadSheet from this SOS GetObservation response, unsupported fomat:? $otherText")
349         Left(ErrorResult(message = s"Couldn't extract OM2 into SpreadSheet from this SOS GetObservation response, unsupported fomat:? $otherText", details = None))
350     }
351   }
