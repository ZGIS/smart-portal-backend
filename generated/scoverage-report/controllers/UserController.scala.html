107             val newUser = User(jsuser.email,

113               ZonedDateTime.now.withZoneSameInstant(ZoneId.of(appTimeZone)))

115             userDAO.createWithNps(newUser).fold {
116               logger.error("User create error.")
117               BadRequest(Json.obj("status" -> "ERR", "message" -> "User create error."))
118             } { user =>
119               val emailWentOut = emailService.sendRegistrationEmail(jsuser.email, "Please confirm your GW HUB account", jsuser.firstname, regLinkId)

121               // creating default collection only after registration, account is only barely usable here

124           } { user =>
125             // found user with that email already
126             logger.error("Email already in use.")

129         } { user =>
