46   /**
47     * Parse an OwcContext from a ResultSet
48     */
49   private def owcContextParser(implicit connection: Connection): RowParser[OwcContext] = {
50     str("id") ~
51       get[Option[String]]("area_of_interest") ~
52       get[Option[String]]("spec_reference") ~
53       get[Option[String]]("context_metadata") ~
54       str("language") ~
55       str("title") ~
56       get[Option[String]]("subtitle") ~
57       get[OffsetDateTime]("update_date") ~
58       get[Option[String]]("authors") ~
59       get[Option[String]]("publisher") ~
60       get[Option[String]]("creator_application") ~
61       get[Option[String]]("creator_display") ~
62       get[Option[String]]("rights") ~
63       get[Option[String]]("time_interval_of_interest") ~
64       get[Option[String]]("keyword") map {
65       case
66         id ~
67           bboxText ~
68           specReferenceUuids ~
69           contextMetaUuids ~
70           language ~
71           title ~
72           subtitle ~
73           updateDate ~
74           authorsUuids ~
75           publisher ~
76           creatorAppUuid ~
77           creatorDispUuid ~
78           rights ~
79           temporalText ~
80           keywordUuids
81       => {
82         OwcContext(
83           id = new URL(id),
84           areaOfInterest = GeoDateParserUtils.createOptionalBbox(bboxText),
85           specReference = specReferenceUuids.map(u =>
86             OwcOfferingDAO.findByPropertiesUUID[OwcLink](Some(u))(OwcLinkEvidence, connection).toList).getOrElse(List()), // aka links.profiles[] & rel=profile
87           contextMetadata = contextMetaUuids.map(u =>
88             OwcOfferingDAO.findByPropertiesUUID[OwcLink](Some(u))(OwcLinkEvidence, connection).toList).getOrElse(List()), // aka links.via[] & rel=via
89           language = language,
90           title = title,
91           subtitle = subtitle,
92           updateDate = updateDate,
93           author = authorsUuids.map(u => OwcOfferingDAO.findByPropertiesUUID[OwcAuthor](Some(u))(OwcAuthorEvidence, connection).toList).getOrElse(List()),
94           publisher = publisher,
95           creatorApplication = creatorAppUuid.map(u => OwcCreatorApplicationDAO.findOwcCreatorApplicationByUuid(UUID.fromString(u))).getOrElse(None),
96           creatorDisplay = creatorDispUuid.map(u => OwcCreatorDisplayDAO.findOwcCreatorDisplayByUuid(UUID.fromString(u))).getOrElse(None),
97           rights = rights,
98           timeIntervalOfInterest = GeoDateParserUtils.parseOffsetDateString(temporalText),
99           keyword = keywordUuids.map(u =>
100             OwcOfferingDAO.findByPropertiesUUID[OwcCategory](Some(u))(OwcCategoryEvidence, connection).toList).getOrElse(List()),
101           resource = OwcResourceDAO.findOwcResourcesForOwcContext(id.toString).toList)
102       }
103     }
104   }

305   /**
306     * takes a list of [[OwcResource]] and tests that non of these exist,
307     * and if so creates them in the database and returns success of the overall create
308     *
309     * @param resources
310     * @param connection
311     * @return
312     */
313   def preCreateCheckOwcResources(resources: List[OwcResource])(implicit connection: Connection): Boolean = {
314     if (resources.nonEmpty) {
315       val idList = resources.map(o => o.id.toString)
316       val exists = idList.exists(id => OwcResourceDAO.findOwcResourceById(id).isDefined)
317       if (exists) {
318         logger.error(s"(createContext/resources) OwcResource with ID one of: ${idList.mkString(" ++ ")} exists already, recommend abort")
319         false
320       } else {
321         val insert = resources.map(OwcResourceDAO.createOwcResource(_)).filter(_.isDefined)
322         insert.length == resources.length
323       }
324     } else {
325       true
326     }
327   }

375   /**
376     * create/insert into db Owc Context
377     *
378     * @param owcContextUnfixed should make issue on github for owc geojson and portal backend
379     *                          and then remove the fix see [[OwcGeoJsonFixes]]
380     * @param user
381     * @param visibility
382     * @param rightsRelationType
383     * @return
384     */
385   def createOwcContext(owcContextUnfixed: OwcContext, user: User, visibility: Int, rightsRelationType: String)
386                       (implicit connection: Connection): Option[OwcContext] = {
387 
388     val owcContext = OwcGeoJsonFixes.fixRelPropertyForOwcLinks(owcContextUnfixed)
389 
390     val preCreateCheckOwcAuthorsForContext = OwcResourceDAO.preCreateCheckOwcAuthors(owcContext.author)
391     val preCreateCheckOwcLinksForContext = OwcResourceDAO.preCreateCheckOwcLinks(owcContext.specReference ++ owcContext.contextMetadata)
392     val preCreateCheckOwcResourcesForContext = preCreateCheckOwcResources(owcContext.resource)
393     val preCreateCheckOwcCategoriesForContext = OwcResourceDAO.preCreateCheckOwcCategories(owcContext.keyword)
394     val preCreateCheckOwcCreatorApplicationForContext = preCreateCheckOwcCreatorApplication(owcContext)
395     val preCreateCheckOwcCreatorDisplayForContext = preCreateCheckOwcCreatorDisplay(owcContext)
396 
397     if (user.isActive && preCreateCheckOwcAuthorsForContext && preCreateCheckOwcLinksForContext &&
398       preCreateCheckOwcResourcesForContext && preCreateCheckOwcCategoriesForContext &&
399       preCreateCheckOwcCreatorApplicationForContext && preCreateCheckOwcCreatorDisplayForContext) {
400 
401       val owcContextInsert = SQL(
402         s"""insert into $tableOwcContexts values (
403            |{id},
404            |{area_of_interest},
405            |{spec_reference},
406            |{context_metadata},
407            |{language},
408            |{title},
409            |{subtitle},
410            |{update_date},
411            |{authors},
412            |{publisher},
413            |{creator_application},
414            |{creator_display},
415            |{rights},
416            |{time_interval_of_interest},
417            |{keyword})""".stripMargin).on(
418         'id -> owcContext.id.toString,
419         'area_of_interest -> owcContext.areaOfInterest.map(r => GeoDateParserUtils.rectToWkt(r)),
420         'spec_reference -> owcContext.specReference.map(_.uuid.toString).mkString(":"), // links.profiles[] and rel=profile
421         'context_metadata -> owcContext.contextMetadata.map(_.uuid.toString).mkString(":"), // aka links.via[] & rel=via
422         'language -> owcContext.language,
423         'title -> owcContext.title,
424         'subtitle -> owcContext.subtitle,
425         'update_date -> owcContext.updateDate.toLocalDateTime,
426         'authors -> owcContext.author.map(_.uuid.toString).mkString(":"),
427         'publisher -> owcContext.publisher,
428         'creator_application -> owcContext.creatorApplication.map(_.uuid.toString),
429         'creator_display -> owcContext.creatorDisplay.map(_.uuid.toString),
430         'rights -> owcContext.rights,
431         'time_interval_of_interest -> owcContext.timeIntervalOfInterest.map(dates => GeoDateParserUtils.writeOffsetDatesAsDateString(dates)).getOrElse(None),
432         'keyword -> owcContext.keyword.map(_.uuid.toString).mkString(":")
433       ).executeUpdate()
434 
435       // this references an existing owc_context_id (the newly created one from above)
436       val resourceDepsInsertCount = owcContext.resource.map {
437         owcResource =>
438           SQL(
439             s"""insert into $tableOwcContextHasOwcResources  values (
440                |{owc_context_id}, {owc_resource_id}
441                |)
442                  """.stripMargin).on(
443             'owc_context_id -> owcContext.id.toString,
444             'owc_resource_id -> owcResource.id.toString
445           ).executeUpdate()
446       }.sum
447 
448       // this references an existing owc_context_id (the newly created one from above)
449       val rightsDepsInsert = SQL(
450         s"""insert into $tableUserHasOwcContextRights values (
451            |{accountsubject}, {owcContextId}, {rights_relation_type}, {visibility}
452            |)""".stripMargin).on(
453         'accountsubject -> user.accountSubject,
454         'owcContextId -> owcContext.id.toString,
455         'rights_relation_type -> rightsRelationType,
456         'visibility -> visibility
457       ).executeUpdate()
458 
459       val allInserts = resourceDepsInsertCount + rightsDepsInsert + owcContextInsert
460       val testValue = owcContext.resource.size + 1 + 1
461 
462       if (allInserts == testValue) {
463         Some(owcContext)
464       } else {
465         logger.error("OwcContext/create: one of " +
466           s"resourceDepsInsertCount: $resourceDepsInsertCount (should be ${owcContext.resource.size}) + " +
467           s"rightsDepsInsert: $rightsDepsInsert (1) + owcContextInsert: $owcContextInsert (1) failed")
468         logger.error("OwcContext couldn't be created")
469         // we need to think where to place rollback most appropriately
470         connection.rollback()
471         None
472       }
473     } else {
474       logger.error("OwcContext/create: one of " +
475         s"user.isActive: ${user.isActive} + preCreateCheckOwcAuthorsForContext: $preCreateCheckOwcAuthorsForContext  " +
476         s"preCreateCheckOwcResourcesForContext: $preCreateCheckOwcResourcesForContext + " +
477         s"preCreateCheckOwcCategoriesForContext: $preCreateCheckOwcCategoriesForContext + " +
478         s"preCreateCheckOwcCreatorApplicationForContext: $preCreateCheckOwcCreatorApplicationForContext + " +
479         s"preCreateCheckOwcCreatorDisplayForContext: $preCreateCheckOwcCreatorDisplayForContext was not successful")
480       logger.error("Precondition failed, won't create OwcContext")
481       // we need to think where to place rollback most appropriately
482       connection.rollback()
483       None
484     }
485   }

564   /**
