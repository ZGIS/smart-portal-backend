82   /**
83     * instantiate an OwcFeatureType either OwcEntry, or OwcDocument
84     * @param id
85     * @param featureType
86     * @param bboxAsWkt
87     * @return
88     */
89   def instantiateOwcFeatureType(id: String, featureType: String, bboxAsWkt: Option[String]): OwcFeatureType = {
90 
91     val emptyProfile = OwcLink(UUID.randomUUID(), "profile", None, "http://www.opengis.net/spec/owc-atom/1.0/req/core", None)
92     val emptySelf = OwcLink(UUID.randomUUID(), "self", Some("application/json"), "http://example.com/empty", None)
93     val now = ZonedDateTime.now.withZoneSameInstant(ZoneId.systemDefault())
94 
95     val emptyDefaultProps = OwcProperties(
96       UUID.randomUUID(),
97       "en",
98       "empty properties",
99       Some("no property information was found"),
100       Some(now),
101       Some("automated system response"),
102       None,
103       List(),
104       List(),
105       None,
106       None,
107       List(),
108       List(emptyProfile, emptySelf)
109     )
110 
111     val rectOpt = createOptionalBbox(bboxAsWkt)
112     val props = owcPropertiesDAO.findOwcPropertiesForOwcFeatureType(id).getOrElse(emptyDefaultProps)
113 
114     featureType match {
115       case "OwcEntry" => {
116         val offerings = owcOfferingDAO.findOwcOfferingsForOwcEntry(id).toList
117         OwcEntry(id, rectOpt, props, offerings)
118       }
119       case "OwcDocument" => {
120         val entries = findOwcEntriesForOwcDocument(id).toList
121         OwcDocument(id, rectOpt, props, entries)
122       }
123       case _ => throw new InvalidClassException(featureType, s"Unknown feature type $featureType")
124     }
125   }

186   /**
187     * create an owc entry with dependent offerings and properties
188     *
189     * @param owcEntry
190     * @return
191     */
192   def createOwcEntry(owcEntry: OwcEntry): Option[OwcEntry] = {
193 
194     db.withTransaction {
195       implicit connection => {
196 
197         val rowCount = SQL(
198           s"""
199           insert into $tableOwcFeatureTypes values (
200             {id}, {feature_type}, {bbox}
201           )
202         """).on(
203           'id -> owcEntry.id,
204           'feature_type -> owcEntry.getClass.getSimpleName,
205           'bbox -> owcEntry.bbox.map( rect => rectToWkt(rect) )
206         ).executeUpdate()
207 
208         owcEntry.offerings.foreach {
209           owcOffering => {
210             if (owcOfferingDAO.findOwcOfferingByUuid(owcOffering.uuid).isEmpty) {
211               owcOfferingDAO.createOwcOffering(owcOffering)
212             }
213 
214             SQL(
215               s"""insert into $tableOwcEntriesHasOwcOfferings  values (
216                  |{owc_feature_types_as_entry_id}, {owc_offerings_uuid}
217                  |)
218                """.stripMargin).on(
219               'owc_feature_types_as_entry_id -> owcEntry.id,
220               'owc_offerings_uuid -> owcOffering.uuid.toString
221             ).executeUpdate()
222           }
223         }
224 
225         // the single dependent owc properties object
226         if (owcPropertiesDAO.findOwcPropertiesByUuid(owcEntry.properties.uuid).isEmpty) {
227           owcPropertiesDAO.createOwcProperties(owcEntry.properties)
228         }
229 
230         SQL(
231           s"""insert into $tableOwcFeatureTypesHasOwcProperties  values (
232              |{owc_feature_types_id}, {owc_properties_uuid}
233              |)
234                """.stripMargin).on(
235           'owc_feature_types_id -> owcEntry.id,
236           'owc_properties_uuid -> owcEntry.properties.uuid.toString
237         ).executeUpdate()
238 
239         rowCount match {
240           case 1 => Some(owcEntry)
241           case _ => None
242         }
243       }
244     }
245   }
