63   def fromJson(json: JsValue): Option[SosCapabilities] = {
64     Json.fromJson[SosCapabilities](json) match {
65       case JsSuccess(r: SosCapabilities, path: JsPath) => Some(r)
66       case e: JsError => {
67         val lines = e.errors.map { tupleAction =>
68           val jsPath = tupleAction._1
69           val valErrors = tupleAction._2.map(valErr => valErr.message).toList.mkString(" ; ")
70           jsPath.toJsonString + " >> " + valErrors
71         }
72 
73         logger.error(s"JsError info  ${lines.mkString(" | ")}")
74         None
75       }
76     }

79   def fromXml(nodeSeq: NodeSeq, sosUrl: String): Option[SosCapabilities] = {
80     try {
81       nodeSeq.head.label match {
82         case "Capabilities" =>
83           Some(SosCapabilities(
84             (nodeSeq \ "ServiceIdentification" \ "Title").text,
85             sosUrl,
86             parseValuesFromOperationsMetadata(
87               (nodeSeq \\ "OperationsMetadata" \ "Operation") filter(n => (n \ "@name" toString) == "GetObservation"),
88               "featureOfInterest"),
89             parseValuesFromOperationsMetadata(
90               (nodeSeq \\ "OperationsMetadata" \ "Operation") filter(n => (n \ "@name" toString) == "GetObservation"),
91               "offering"),
92             parseValuesFromOperationsMetadata(
93               (nodeSeq \\ "OperationsMetadata" \ "Operation") filter(n => (n \ "@name" toString) == "GetObservation"),
94               "observedProperty"),
95             parseValuesFromOperationsMetadata(
96               (nodeSeq \\ "OperationsMetadata" \ "Operation") filter(n => (n \ "@name" toString) == "GetObservation"),
97               "procedure")
98           ))
99         case _ =>
100           throw new IllegalArgumentException(f"Expected MDMetadataNode but found  ${nodeSeq.head.label}")
101       }
102     }
103     catch {
104       //FIXME SR replace by specific exceptions
105       case e: Exception => logger.warn(f"Exception on parsing MD_Metadata: ${e.getMessage}", e)
106         None
107     }  }

109   private def parseValuesFromOperationsMetadata(nodeseq: NodeSeq, parameter: String): Seq[String] = {
110     val param = (nodeseq \ "Parameter") filter(node => (node \ "@name" toString) == parameter)
111     (param \ "AllowedValues" \\ "Value").map(_.text)
112   }
