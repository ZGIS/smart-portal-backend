39   /**
40     * Parse an OwcResource from a ResultSet
41     */
42   private def owcResourceParser(implicit connection: Connection): RowParser[OwcResource] = {
43     str("id") ~
44       str("title") ~
45       get[Option[String]]("subtitle") ~
46       get[OffsetDateTime]("update_date") ~
47       get[Option[String]]("authors") ~
48       get[Option[String]]("publisher") ~
49       get[Option[String]]("rights") ~
50       get[Option[String]]("geospatial_extent") ~
51       get[Option[String]]("temporal_extent") ~
52       get[Option[String]]("content_description") ~
53       get[Option[String]]("preview") ~
54       get[Option[String]]("content_by_ref") ~
55       get[Option[String]]("offering") ~
56       get[Option[Boolean]]("active") ~
57       get[Option[String]]("resource_metadata") ~
58       get[Option[String]]("keyword") ~
59       get[Option[Double]]("min_scale_denominator") ~
60       get[Option[Double]]("max_scale_denominator") ~
61       get[Option[String]]("folder") map {
62       case
63         id ~
64           title ~
65           subtitle ~
66           updateDate ~
67           authorsUuids ~
68           publisher ~
69           rights ~
70           bboxText ~
71           temporalText ~
72           contentDescUuids ~
73           previewUuids ~
74           contentByRefUuids ~
75           offeringUuids ~
76           active ~
77           resMetaUuids ~
78           keywordUuids ~
79           minScale ~
80           maxScale ~
81           folder
82       => {
83         OwcResource(
84           id = new URL(id),
85           title = title,
86           subtitle = subtitle,
87           updateDate = updateDate,
88           author = authorsUuids.map(u => OwcOfferingDAO.findByPropertiesUUID[OwcAuthor](Some(u))(OwcAuthorEvidence, connection).toList).getOrElse(List()),
89           publisher = publisher,
90           rights = rights,
91           geospatialExtent = GeoDateParserUtils.createOptionalBbox(bboxText),
92           temporalExtent = GeoDateParserUtils.parseOffsetDateString(temporalText),
93           contentDescription = contentDescUuids.map(u =>
94             OwcOfferingDAO.findByPropertiesUUID[OwcLink](Some(u))(OwcLinkEvidence, connection).toList).getOrElse(List()), // links.alternates[] and rel=alternate
95           preview = previewUuids.map(u =>
96             OwcOfferingDAO.findByPropertiesUUID[OwcLink](Some(u))(OwcLinkEvidence, connection).toList).getOrElse(List()), // aka links.previews[] and rel=icon (atom)
97           contentByRef = contentByRefUuids.map(u =>
98             OwcOfferingDAO.findByPropertiesUUID[OwcLink](Some(u))(OwcLinkEvidence, connection).toList).getOrElse(List()), // aka links.data[] and rel=enclosure (atom)
99           offering = offeringUuids.map(u =>
100             OwcOfferingDAO.findByPropertiesUUID[OwcOffering](Some(u))(OwcOfferingEvidence, connection).toList).getOrElse(List()),
101           active = active,
102           resourceMetadata = resMetaUuids.map(u =>
103             OwcOfferingDAO.findByPropertiesUUID[OwcLink](Some(u))(OwcLinkEvidence, connection).toList).getOrElse(List()), // aka links.via[] & rel=via
104           keyword = keywordUuids.map(u =>
105             OwcOfferingDAO.findByPropertiesUUID[OwcCategory](Some(u))(OwcCategoryEvidence, connection).toList).getOrElse(List()),
106           minScaleDenominator = minScale,
107           maxScaleDenominator = maxScale,
108           folder = folder)
109       }
110     }
111   }

166     * takes a list of OwcAuthor (e.g. for createOwcResource) and tests that non of these exist,
167     * and if so creates them in the database and returns success of the overall create
168     *
169     * @param authors
170     * @param connection
171     * @return
172     */
173   def preCreateCheckOwcAuthors(authors: List[OwcAuthor])(implicit connection: Connection): Boolean = {
174     if (authors.nonEmpty) {
175       val uuidString = authors.map(_.uuid.toString).mkString(":")
176       val exists = findByPropertiesUUID[OwcAuthor](Some(uuidString))(OwcAuthorEvidence, connection).toList.nonEmpty
177       if (exists) {
178         logger.error(s"(createResourceOrContext/authors) OwcAuthor with UUID one of: ${uuidString} exists already, recommend abort")
179         false
180       } else {
181         val insert = authors.map(OwcAuthorDAO.createOwcAuthor(_)).filter(_.isDefined)
182         insert.length == authors.length
183       }
184     } else {
185       true
186     }
187   }

261   /**
262     * create an owc resource with dependent offerings and properties
263     *
264     * @param owcResourceUnfixed should make issue on github for owc geojson and portal backend
265     *                           and then remove the fix see [[OwcGeoJsonFixes]]
266     * @return
267     */
268   def createOwcResource(owcResourceUnfixed: OwcResource)(implicit connection: Connection): Option[OwcResource] = {
269 
270     val owcResource = OwcGeoJsonFixes.fixRelPropertyForOwcLinks(owcResourceUnfixed)
271 
272     val preCreateCheckOwcAuthorsForResource = preCreateCheckOwcAuthors(owcResource.author)
273     val preCreateCheckOwcLinksForResource = preCreateCheckOwcLinks(owcResource.contentDescription ++ owcResource.preview ++
274       owcResource.contentByRef ++ owcResource.resourceMetadata)
275     val preCreateCheckOwcOfferingsForResource = preCreateCheckOwcOfferings(owcResource.offering)
276     val preCreateCheckOwcCategoriesForResource = preCreateCheckOwcCategories(owcResource.keyword)
277 
278     if (preCreateCheckOwcAuthorsForResource && preCreateCheckOwcLinksForResource &&
279       preCreateCheckOwcOfferingsForResource && preCreateCheckOwcCategoriesForResource) {
280 
281       val rowCount = SQL(
282         s"""insert into $tableOwcResources values (
283            |{id}, {title},
284            |{subtitle}, {update_date},
285            |{authors}, {publisher},
286            |{rights}, {geospatial_extent},
287            |{temporal_extent}, {content_description},
288            |{preview}, {content_by_ref},
289            |{offering}, {active},
290            |{resource_metadata}, {keyword},
291            |{min_scale_denominator}, {max_scale_denominator},
292            |{folder} )""".stripMargin).on(
293         'id -> owcResource.id.toString,
294         'title -> owcResource.title,
295         'subtitle -> owcResource.subtitle,
296         'update_date -> owcResource.updateDate.toLocalDateTime,
297         'authors -> owcResource.author.map(_.uuid.toString).mkString(":"),
298         'publisher -> owcResource.publisher,
299         'rights -> owcResource.rights,
300         'geospatial_extent -> owcResource.geospatialExtent.map(r => GeoDateParserUtils.rectToWkt(r)),
301         'temporal_extent -> owcResource.temporalExtent.map(dates => GeoDateParserUtils.writeOffsetDatesAsDateString(dates)).getOrElse(None),
302         'content_description -> owcResource.contentDescription.map(_.uuid.toString).mkString(":"), // links.alternates[] and rel=alternate
303         'preview -> owcResource.preview.map(_.uuid.toString).mkString(":"), // aka links.previews[] and rel=icon (atom)
304         'content_by_ref -> owcResource.contentByRef.map(_.uuid.toString).mkString(":"), // aka links.data[] and rel=enclosure (atom)
305         'offering -> owcResource.offering.map(_.uuid.toString).mkString(":"),
306         'active -> owcResource.active,
307         'resource_metadata -> owcResource.resourceMetadata.map(_.uuid.toString).mkString(":"), // aka links.via[] & rel=via
308         'keyword -> owcResource.keyword.map(_.uuid.toString).mkString(":"),
309         'min_scale_denominator -> owcResource.minScaleDenominator,
310         'max_scale_denominator -> owcResource.maxScaleDenominator,
311         'folder -> owcResource.folder
312       ).executeUpdate()
313 
314       rowCount match {
315         case 1 => Some(owcResource)
316         case _ =>
317           logger.error("OwcResource couldn't be created")
318           // we need to think where to place rollback most appropriately
319           connection.rollback()
320           None
321       }
322     } else {
323       logger.error("OwcResource/create: one of " +
324         s"preCreateCheckOwcAuthorsForResource: $preCreateCheckOwcAuthorsForResource  " +
325         s"preCreateCheckOwcLinksForResource: $preCreateCheckOwcLinksForResource + " +
326         s"preCreateCheckOwcOfferingsForResource: $preCreateCheckOwcOfferingsForResource + " +
327         s"preCreateCheckOwcCategoriesForResource: $preCreateCheckOwcCategoriesForResource was not successful")
328       logger.error("Precondition failed, won't create OwcResource")
329       // we need to think where to place rollback most appropriately
330       connection.rollback()
331       None
332     }
333   }

335   /**
336     * pre Check for authors Update For an owcAggregator of type [[OwcContext]] or [[OwcResource]],
337     * will delete, update and insert required [[OwcAuthor]] in order to make owcAggregator type Update consistent
338     *
339     * @param owcAggregator of type [[OwcContext]] or [[OwcResource]]
340     * @param connection
341     * @tparam A of type [[OwcContext]] or [[OwcResource]]
342     * @return
343     */
344   def preUpdateCheckOwcAuthors[A](owcAggregator: A)(implicit connection: Connection): Boolean = {
345 
346     // get current list of full OwcAuthor objects,
347     val currentOwcAuthors = owcAggregator match {
348       case owcAggregator: OwcResource => owcAggregator.author
349       case owcAggregator: OwcContext => owcAggregator.author
350       case _ => throw new InvalidClassException(s"The evidence parameter type ${owcAggregator.getClass.getCanonicalName} not supported here")
351     }
352 
353     // get current list of UUIDs,
354     val current: List[UUID] = currentOwcAuthors.map(_.uuid)
355 
356     // get old list of OwcAuthor,
357     val oldOwcAuthors: List[OwcAuthor] = owcAggregator match {
358       case owcAggregator: OwcResource => findOwcResourceById(owcAggregator.id).map(o => o.author).getOrElse(List())
359       case owcAggregator: OwcContext => OwcContextDAO.findOwcContextById(owcAggregator.id).map(o => o.author).getOrElse(List())
360       case _ => throw new InvalidClassException(s"The evidence parameter type ${owcAggregator.getClass.getCanonicalName} not supported here")
361     }
362 
363     // get old list of UUIDs,
364     val old: List[UUID] = oldOwcAuthors.map(_.uuid)
365 
366     // in old but not current -> delete
367     val toBeDeleted = old.diff(current)
368     val deleted = oldOwcAuthors.filter(o => toBeDeleted.contains(o.uuid))
369       .map(OwcAuthorDAO.deleteOwcAuthor(_))
370       .count(_ == true) == toBeDeleted.length
371 
372     // in both lists -> update
373     val toBeUpdated = current.intersect(old)
374     val updated = currentOwcAuthors.filter(o => toBeUpdated.contains(o.uuid))
375       .map(OwcAuthorDAO.updateOwcAuthor(_))
376       .count(_.isDefined) == toBeUpdated.length
377 
378     // in current but not in old -> insert
379     val toBeInserted = current.diff(old)
380     val inserted = currentOwcAuthors.filter(o => toBeInserted.contains(o.uuid))
381       .map(OwcAuthorDAO.createOwcAuthor(_))
382       .count(_.isDefined) == toBeInserted.length
383 
384     if (deleted && updated && inserted) {
385       true
386     } else {
387       logger.error(s"(updateAggregator/authors) one of deleted: $deleted , updated: $updated , inserted: $inserted not complete, recommend abort")
388       false
389     }
390   }
