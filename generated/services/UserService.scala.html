42     * basic auth compare
43     *
44     * @param loginCredentials
45     * @return
46     */
47   def authenticateLocal(loginCredentials: LoginCredentials): Either[ErrorResult, User] = {
48 
49     val userOpt = dbSession.viaConnection { implicit connection =>
50       UserDAO.findUserByEmailAsString(loginCredentials.email).filter(user =>
51         passwordHashing.validatePassword(loginCredentials.password, user.password)
52       )
53     }
54     if (userOpt.isDefined) {
55       // here is correct auth branch
56       val user = userOpt.get
57       if (user.isBlocked) {
58         val status = user.getToken
59         Left(ErrorResult(s"Your account was ${status.value}. Please contact the administrator.", None))
60       } else {
61         Right(user)
62       }
63 
64     } else {
65       // should never indicate if user not found or just password wrong?
66       Left(ErrorResult("User email or password wrong.", None))
67     }
68   }

238   /**
239     * check for an active server-side session
240     *
241     * @param xsrfToken
242     * @param xsrfTokenCookie
243     * @param userAgentHeader
244     * @return
245     */
246   def getUserSessionByToken(xsrfToken: String, xsrfTokenCookie: String, userAgentHeader: String): Option[UserSession] = {
247     dbSession.viaConnection(implicit connection => {
248       UserSession.findUserSessionByToken(xsrfToken, 1)
249     })
250       .fold[Option[UserSession]] {
251         logger.debug(s"no session for xsrfToken: $xsrfToken")
252         None
253       } {
254         session =>
255           val cookieForUSerAndUserAgent = passwordHashing.testSessionCookie(xsrfToken, session.email, userAgentHeader)
256           logger.trace(s"testcookie: $cookieForUSerAndUserAgent")
257           if (xsrfToken == xsrfTokenCookie && cookieForUSerAndUserAgent) {
258             logger.trace(s"request for active session: ${session.email} / $xsrfToken / $userAgentHeader")
259             Some(session)
260           } else {
261             logger.error("Invalid Token")
262             None
263           }
264     }
265   }

300     val userLookup = dbSession.viaConnection { implicit connection =>
