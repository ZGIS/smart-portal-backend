274   def callJenaKubeReload: Future[Option[ErrorResult]] = {
275     val digestHash = MessageDigest.getInstance("SHA-512").digest(appSecret.getBytes("UTF-8"))
276     val hex = digestHash.map("%02X" format _).mkString
277     wsClient.url(ADMIN_JENA_UPDATE_URL).withHeaders("Authorization" -> s"Bearer $hex").get().map { response =>
278       if (response.status == 200) {
279         logger.info(
280           s"Successfully called $ADMIN_JENA_UPDATE_URL (${response.status} - ${response.statusText})")
281         logger.info(s"response: ${response.body}")
282         None
283       } else {
284         logger.warn(
285           s"Call to '$ADMIN_JENA_UPDATE_URL' returned error: ${response.status} - ${response.statusText}")
286         logger.warn(s"response: ${response.body}")
287         Some(ErrorResult("Update of running vocab service failed.", Some(s"Call to '$ADMIN_JENA_UPDATE_URL' returned error: ${response.status} - ${response.statusText}")))
288       }
289     }
290   }
