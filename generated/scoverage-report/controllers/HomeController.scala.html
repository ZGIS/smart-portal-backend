102   /**
103     * login with JSON (from Angular / form)
104     * {"email":"akmoch","password":"testpass123"}
105     *
106     * Create the authentication token and sets the cookie [[AuthTokenCookieKey]]
107     * for AngularJS to use also in X-XSRF-TOKEN in HTTP header.
108     *
109     * @return
110     */
111   def login: Action[JsValue] = Action(parse.json) { implicit request =>
112     request.body.validate[LoginCredentials].fold(
113       errors => {
114         logger.error(JsError.toJson(errors).toString())
115         val error = ErrorResult("Could not validate request.", Some(JsError.toJson(errors).toString()))
116         BadRequest(Json.toJson(error)).as(JSON)
117       },
118       valid = credentials => {
119         // find user in db and compare password stuff
120         userDAO.authenticate(credentials.email, credentials.password).fold {
121           logger.error("User email or password wrong.")
122           val error = ErrorResult("User email or password wrong.", None)
123           Unauthorized(Json.toJson(error)).as(JSON)
124         } { user =>
125           val uaIdentifier: String = request.headers.get(UserAgentHeader).getOrElse(UserAgentHeaderDefault)
126           // logger.debug(s"Logging in email from $uaIdentifier")
127           val token = passwordHashing.createSessionCookie(user.email, uaIdentifier)
128           cache.set(token, user.email)
129           Ok(Json.obj("status" -> "OK", "token" -> token, "email" -> user.email, "userprofile" -> user.asProfileJs()))
130             .withCookies(Cookie(AuthTokenCookieKey, token, None, httpOnly = false))
131         }
132       })
133   }

135   /**
136     * https://developers.google.com/recaptcha/docs/verify
137     *
138     * @param recaptcaChallenge
139     * @return
140     */
141   def recaptchaValidate(recaptcaChallenge: String): Action[AnyContent] = Action.async { implicit request =>
142     // TODO check from where URL referer comes from
143     ws.url(recaptcaVerifyUrl)
144       .withHeaders("Accept" -> "application/json")
145       .withRequestTimeout(10000.millis)
146       .withQueryString("secret" -> reCaptchaSecret,
147         "response" -> recaptcaChallenge).get().map {
148       response =>
149         val success = (response.json \ "success").as[Boolean]
150         if (success) {
151           Ok(Json.obj("status" -> "OK", "message" -> "granted", "success" -> JsBoolean(true)))
152         }
153         else {
154           val errors = (response.json \ "error-codes")
155           val jsErrors = errors.getOrElse(JsString("No further errors"))
156           val error = ErrorResult("User email or password wrong.", Some(jsErrors.toString()))
157           BadRequest(Json.toJson(error)).as(JSON)
158         }
159     }.recover {
160       case e: Exception =>
161         val error = ErrorResult("Exception on validating captcha.", Some(e.getMessage))
162         InternalServerError(Json.toJson(error)).as(JSON)
163     }
164   }
165 
166   /**
167     * Log-out a user. Invalidates the authentication token.
168     *
169     * Discard the cookie [[AuthTokenCookieKey]] to have AngularJS no longer set the
170     * X-XSRF-TOKEN in HTTP header.
171     */
172   def logout: Action[Unit] = HasToken(parse.empty) { token =>
173     email =>
174       implicit request =>
175         cache.remove(token)
176         Ok.discardingCookies(DiscardingCookie(name = AuthTokenCookieKey))
177   }
