28 /**
29   * Security actions that should be used by all controllers that need to protect their actions.
30   * Can be composed to fine-tune access control.
31   *
32   * from https://github.com/mariussoutier/play-angular-require-seed/blob/master/app/controllers/Security.scala
33   */
34 trait Security { self: Controller =>
35 
36   val cache: CacheApi
37   val configuration: Configuration
38 
39   val AuthTokenCookieKey = "XSRF-TOKEN"
40   val AuthTokenHeader = "X-XSRF-TOKEN"
41   val AuthTokenUrlKey = "auth"
42 
43   val UserAgentHeader = "User-Agent"
44   val UserAgentHeaderDefault = "Default-UA/1.0"
45 
46   /**
47     * Checks that the token is:
48     * - present in the cookie header of the request,
49     * - either in the header or in the query string,
50     * - matches a token already stored in the play cache
51     */
52   def HasToken[A](p: BodyParser[A] = parse.anyContent)(
53     f: String => String => Request[A] => Result): Action[A] =
54   Action(p) { implicit request =>
55 
56     request.cookies.get(AuthTokenCookieKey).fold {
57       Unauthorized(Json.obj("status" -> "ERR", "message" -> "Invalid XSRF Token cookie"))
58     } { xsrfTokenCookie =>
59       // Logger.debug(s"cookie ${xsrfTokenCookie.value}")
60       val maybeToken = request.headers.get(AuthTokenHeader).orElse(request.getQueryString(AuthTokenUrlKey))
61 
62       maybeToken flatMap { token =>
63         // ua needed to differentiate between different devices/sessions
64         val uaIdentifier: String = request.headers.get(UserAgentHeader).getOrElse(UserAgentHeaderDefault)
65 
66         // cache token -> maps to a String username
67         cache.get[String](token) map { username =>
68           lazy val passwordHashing = new PasswordHashing(configuration)
69           val cookieForUSerAndDevice = passwordHashing.testSessionCookie(token, username, uaIdentifier)
70           if (xsrfTokenCookie.value == token && cookieForUSerAndDevice) {
71             Logger.debug(s"request for active session: $username / $token / $uaIdentifier")
72             f(token)(username)(request)
73           } else {
74             Unauthorized(Json.obj("status" -> "ERR", "message" -> "Invalid Token"))
75           }
76         }
77       } getOrElse Unauthorized(Json.obj("status" -> "ERR", "message" -> "No Token"))
78     }
79   }
80 }
