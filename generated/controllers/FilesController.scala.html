63   def uploadMultipartForm: Action[MultipartFormData[TemporaryFile]] = (authenticationAction andThen userAction) (parse.multipartFormData) {
64     request =>
65       request.body.file("file").map { theFile =>
66 
67         val filename = theFile.filename
68         val contentType = theFile.contentType
69         val pathOfUploadTmp = Paths.get(uploadDataPath)
70         val intermTempDir = Files.createTempDirectory(pathOfUploadTmp, "sac-upload-")
71         val tmpFile = new File(intermTempDir.resolve(filename).toAbsolutePath.toString)
72         val handle = theFile.ref.moveTo(tmpFile)
73         val kilobytes = handle.length / 1024
74 
75         googleService.uploadFileGoogleBucket(handle) match {
76           case Left(errorResult) =>
77             logger.error(errorResult.message + errorResult.details.mkString)
78             InternalServerError(Json.toJson(errorResult)).as(JSON)
79           case Right(blobInfo) =>
80             // return the public download link
81             val userFileEntryOk = userService.insertUserFileEntry(filename, request.user.email, blobInfo.mediaLink)
82             userFileEntryOk.fold {
83               logger.error("file metadata insert failed.")
84               val error = ErrorResult("File metadata insert failed.", None)
85               BadRequest(Json.toJson(error)).as(JSON)
86             } { userFile =>
87               val owcResource = collectionsService.fileMetadata(userFile, contentType, Try(kilobytes.toInt).toOption)
88               logger.trace(Json.prettyPrint(owcResource.toJson))
89               logger.debug(s"file upload $filename to ${blobInfo.mediaLink} with reference ${userFile.linkreference}; bob details: ${blobInfo.name} ${blobInfo.bucket} ${blobInfo.contentType} ${blobInfo.contentEncoding} ${blobInfo.size} ${blobInfo.md5}")
90               Try(handle.delete()).failed.map(ex => logger.error(ex.getLocalizedMessage))
91               Try(Files.delete(intermTempDir)).failed.map(ex => logger.error(ex.getLocalizedMessage))
92 
93               //FIXME SR do we also want to have a general "return" object? Status is always in the respone so it does not need to be in here
94               val added = collectionsService.addPlainFileResourceToUserDefaultCollection(owcResource, request.user)
95               if (added) {
96                 Ok(Json.obj("status" -> "OK", "message" -> s"file uploaded $filename.", "file" -> userFile.linkreference, "entry" -> owcResource.toJson))
97               } else {
98                 logger.error("Insert into own collection failed.")
99                 val error = ErrorResult("Insert into own collection failed.", None)
100                 BadRequest(Json.toJson(error)).as(JSON)
101               }
102             }
103           case _ =>
104             logger.error("file upload to cloud storage failed")
105             val error = ErrorResult("File upload to cloud storage failed.", None)
106             BadRequest(Json.toJson(error)).as(JSON)
107         }
108       }.getOrElse {
109         logger.error("file upload failed, no file found for upload")
110         val error = ErrorResult("file upload failed, no file found for upload", None)
111         BadRequest(Json.toJson(error)).as(JSON)
112       }
113   }
