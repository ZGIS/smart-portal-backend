46   /**
47     * get Owc Documents For optional email And owc doc Id
48     *
49     * @param authUserOption
50     * @param idOption
51     * @return
52     */
53   def getOwcDocumentsForUserAndId(authUserOption: Option[String], idOption: Option[String]): Seq[OwcDocument] = {
54 
55     authUserOption.fold {
56       // no email provided
57       idOption.fold {
58         // TODO docs for anonymous, no id provided => all public docs (implies docs must be public)
59         owcDocumentDAO.getAllPublicOwcDocuments
60       } {
61         // TODO find doc by id for anonymous, only one doc if available (implies doc must be public)
62         id => {
63           owcDocumentDAO.findPublicOwcDocumentsById(id).toSeq
64         }
65       }
66     } { authUser => {
67       // trying to find a user from provided authuser option
68       userDAO.findUserByEmail(authUser).fold {
69         logger.warn("Provided user not found.")
70         idOption.fold {
71           // TODO docs for anonymous, no id provided => all public docs (later maybe check if public)
72           owcDocumentDAO.getAllPublicOwcDocuments
73         } {
74           // docs for anonymous, but id provided only one doc if available (and only if public)
75           id => {
76             owcDocumentDAO.findPublicOwcDocumentsById(id).toSeq
77           }
78         }
79       } { user =>
80         // we have a distinct ok user here
81         idOption.fold {
82           // docs for user, no id provided => all user visible docs
83           // TODO technically would be more than "only" publicly visible at some point
84           val publicDocs = owcDocumentDAO.getAllPublicOwcDocuments
85           val userDocs = owcDocumentDAO.findOwcDocumentByUser(user.email)
86 
87           publicDocs ++ userDocs
88         } {
89           // TODO find doc by id for provided user if visible/available (later maybe check constraint)
90           id => {
91             owcDocumentDAO.findOwcDocumentByIdAndUser(id, user.email).toSeq
92           }
93         }
94       }
95     }
96     }
97   }
