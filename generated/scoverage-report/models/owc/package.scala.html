77     Column.nonNull { (value, meta) =>
78       val MetaDataItem(qualified, nullable, clazz) = meta
79       value match {
80         case bool: Boolean => Right(bool) // Provided-default case

85       }

93   implicit def columnToOffset: Column[OffsetDateTime] = {
94     Column.nonNull { (value, meta) =>
95       val MetaDataItem(qualified, nullable, clazz) = meta
96       value match {
97         case offsetDateTime: OffsetDateTime => Right(offsetDateTime) // default case
98         case zonedDateTime: ZonedDateTime => Right(OffsetDateTime.of(zonedDateTime.toLocalDateTime, zonedDateTime.getOffset)) // Custom conversion
99         case h2DateTime: org.h2.api.TimestampWithTimeZone => {
100           // val offsetMinutes = h2DateTime.getTimeZoneOffsetMins
101           val localDateTime = h2DateTime.toLocalDateTime
102           Right(OffsetDateTime.of(localDateTime, ZoneOffset.ofHoursMinutesSeconds(0,0,0)))
103         } // Custom conversion, H2 time stamp with timezone is experimental
104         case sqlTimeStamp: java.sql.Timestamp => {
105           val localDateTime = sqlTimeStamp.toLocalDateTime
106           Right(OffsetDateTime.of(localDateTime, ZoneOffset.ofHoursMinutesSeconds(0,0,0)))
107         } // Custom conversion to omit H2 TimeStamp with TimeZone, for Testing don't use ZoneOffsets
108         case _ => {
109           Left(TypeDoesNotMatch(s"Cannot convert $value: ${value.asInstanceOf[AnyRef].getClass} to OffsetDateTime for column $qualified"))
110         }
111       }
112     }
