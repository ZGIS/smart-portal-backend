273   def callJenaKubeReload: Future[Option[ErrorResult]] = {
274     val digestHash = MessageDigest.getInstance("SHA-512").digest(appSecret.getBytes("UTF-8"))
275     val hex = digestHash.map("%02X" format _).mkString
276     wsClient.url(ADMIN_JENA_UPDATE_URL).withHeaders("Authorization" -> s"Bearer $hex").get().map { response =>
277       if (response.status == 200) {
278         logger.info(
279           s"Successfully called $ADMIN_JENA_UPDATE_URL (${response.status} - ${response.statusText})")
280         logger.info(s"response: ${response.body}")
281         None
282       } else {
283         logger.warn(
284           s"Call to '$ADMIN_JENA_UPDATE_URL' returned error: ${response.status} - ${response.statusText}")
285         logger.warn(s"response: ${response.body}")
286         Some(ErrorResult("Update of running vocab service failed.", Some(s"Call to '$ADMIN_JENA_UPDATE_URL' returned error: ${response.status} - ${response.statusText}")))
287       }
288     }
289   }
