41   /**
42     * Parse a User from a ResultSet
43     *
44     */
45   val userParser = {
46     get[String]("email") ~
47       get[String]("username") ~
48       get[String]("firstname") ~
49       get[String]("lastname") ~
50       get[String]("password") ~
51       get[String]("laststatustoken") ~
52       get[ZonedDateTime]("laststatuschange") map {
53       case email~username~firstname~lastname~password~laststatustoken~laststatuschange =>
54         User(email, username, firstname, lastname, password, laststatustoken, laststatuschange)
55     }
56   }
