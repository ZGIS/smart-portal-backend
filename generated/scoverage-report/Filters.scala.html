33 /**
34  * This class configures filters that run on every request. This
35  * class is queried by Play to get a list of filters.
36  *
37  * Play will automatically use filters from any class called
38  * `Filters` that is placed the root package. You can load filters
39  * from a different class by adding a `play.http.filters` setting to
40  * the `application.conf` configuration file.
41  *
42  * @param env Basic environment settings for the current application.
43  * @param exampleFilter A demonstration filter that adds a header to
44  * each response.
45  * @param corsFilter A demonstration filter that adds a header to
46  * each response.
47  */
48 @Singleton
49 class Filters @Inject() (
50   env: Environment,
51   exampleFilter: ExampleFilter,
52   corsFilter: CORSFilter,
53   csrfFilter: CSRFFilter,
54   securityHeadersFilter: SecurityHeadersFilter) extends HttpFilters {
55 
56   override val filters = {
57     // Use the example filter if we're running development mode. If
58     // we're running in production or test mode then don't use any
59     // filters at all.
60     if (env.mode == Mode.Dev) Seq(exampleFilter, csrfFilter, securityHeadersFilter)
61     else Seq(corsFilter, csrfFilter, securityHeadersFilter)
62   }
63 
64 }
