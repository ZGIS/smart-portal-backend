53   def sendRegistrationEmail(emailTo: String, subject: String, usernameTo: String, linkId: String): Boolean = {
54 
55     val emailText =
56       """Hello %s,
57         |thank you for registering on the GW HUB. Your account will be fully active once you confirmed your email address.
58         |Please click on the following link to confirm your registration:
59         |
60         |%s/api/v1/users/register/%s
61         |
62         |If you have any question please email us to %s.
63         |
64         |Your GW HUB Team
65       """.format(usernameTo, portalApiHost, linkId, emailFrom).stripMargin
66 
67     val from = new Email(emailFrom)
68     val to = new Email(emailTo)
69 
70     val content = new Content("text/plain", emailText)
71     val mail = new Mail(from, subject, to, content)
72 
73     val replyTo = new Email(emailReplyTo)
74     mail.setReplyTo(replyTo)
75 
76     try {
77 
78       val request = new Request()
79       request.method = Method.POST
80       request.endpoint = "mail/send"
81       request.body = mail.build()
82       val response = sg.api(request)
83       logger.trace(s"mail api response status: ${response.statusCode}")
84       logger.trace(s"mail api response.body: ${response.body}")
85       logger.trace(s"response.headers: ${response.headers}")
86       logger.trace(s"mail api response status: ${response.statusCode}")
87 
88       true
89     } catch {
90       case ioex: IOException => {
91         logger.error("IO Messaging exception: " + ioex.getLocalizedMessage)
92         false
93       }
94       case e: Exception => {
95         logger.error("Other email problem: " + e.getLocalizedMessage)
96         false
97       }
