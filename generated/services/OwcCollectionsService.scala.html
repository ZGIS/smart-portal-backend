447   /**
448     * returns an Option[EmailAddress] object if parsing is successful, for those mdMetadata email joint arrays
449     *
450     * @param emailString the string that might comprise of or contain an email address
451     * @return Option[EmailAddress]
452     */
453   def parseEmailStringtoEmailAddress(emailString: String): Option[EmailAddress] = {
454 
455     if (EmailAddress.isValid(emailString)) {
456       Some(EmailAddress(emailString))
457     } else {
458       if (emailString.contains(",")) {
459         val commaFree = emailString.split(",").head.trim
460 
461         Try {
462           EmailAddress(commaFree)
463         } match {
464           case Success(e) => Some(e)
465           case _ => None
466         }
467       } else {
468         None
469       }
470     }
471   }
