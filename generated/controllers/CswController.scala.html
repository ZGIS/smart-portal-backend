107   def insert: Action[JsValue] = (authenticationAction andThen userAction).async(parse.json) {
108     request =>
109 
110       logger.debug(request.toString)
111       //TODO parse JSON to MDMetadataSet and convert that to XML
112       // val mdMetadata = MdMetadata.fromJson(((request.body.as[JsObject]) \ "metadata").get)
113       (request.body.as[JsObject] \ "metadata").validate[MdMetadata].fold(
114         errors => {
115           logger.error(JsError.toJson(errors).toString())

118             BadRequest(Json.toJson(error)).as(JSON)
119           }

121         mdMetadata => {
122           logger.trace(Json.prettyPrint(mdMetadata.toJson()))
123           //FIXME SR I find that pretty hard to read. Is there a better way of chaining WS calls?
124           // see https://www.playframework.com/documentation/2.5.x/ScalaWS#chaining-ws-calls
125           val futureResponse: Future[(WSResponse, WSResponse)] = for {
126             getCapaResponse <- wsClient.url(CSW_OPERATIONS_METADATA_URL).get()
127             insertResponse <- {
128               //check if getCapaResponse indicates, that the CSW can perform Transactions
129               val operation = (getCapaResponse.xml \\ "OperationsMetadata" \\ "Operation").filter(node => {
130                 logger.debug(s"Attribute 'name'=${node.attribute("name").get.text.toString}")
131                 node.attribute("name").get.text.equals("Transaction")
132               })
133               logger.debug(s"operation = ${operation.toString()}")
134               if (operation.isEmpty) {
135                 throw new UnsupportedOperationException("CSW does not support Transaction.")
136               }
137 
138               //insert MDMEtadata in insert template
139               logger.debug(s"MD_MetadataXML: ${mdMetadata.toXml().toString()}")
140               val rule = new RuleTransformer(new AddMDMetadataToInsert(mdMetadata.toXml()))
141               val finalXML = rule.transform(cswtInsertXml)
142               logger.debug(s"finalXml: ${finalXML.toString()}")
143               wsClient.url(CSW_URL).post(finalXML.toString())
144             }
145             updateIngesterIndexResponse <- {
146               val response = wsClient.url(INGESTER_UPDATE_INDEX_URL).get()
147               response.onSuccess {
148                 case response if response.status == 200 => {
149                   logger.info(
150                     s"Successfully called $INGESTER_UPDATE_INDEX_URL (${response.status} - ${response.statusText})");
151                   logger.info(s"response: ${response.body}");
152                 }
153                 case response => {
154                   logger.warn(
155                     s"Call to '$INGESTER_UPDATE_INDEX_URL' returned error: ${response.status} - ${response.statusText}");
156                   logger.warn(s"response: ${response.body}");
157                 }
158               }
159               response
160             }
161           } yield (insertResponse, updateIngesterIndexResponse)
162 
163           futureResponse.recover {
164             case e: Exception =>
165               logger.error("Insert CSW threw exception", e)
166               val error = ErrorResult(s"Exception (${e.getClass.getCanonicalName}) on CSW insert: ${e.getMessage}",
167                 None)
168               InternalServerError(Json.toJson(error)).as(JSON)
169           }
170 
171           futureResponse.map(response => {
172             logger.debug(response._1.xml.toString())
173             response._1.xml match {
174               case e: Elem if e.label == "TransactionResponse" => {
175                 val fileIdentifier = (e \\ "InsertResult" \\ "BriefRecord" \\ "identifier").text
176                 logger.debug(s"Adding ${mdMetadata.fileIdentifier} to default collection of ${request.user.email}.")
177                 val userMetaEntryOk = userService.insertUserMetaRecordEntry(CSW_URL, mdMetadata.fileIdentifier, request.user.email)
178                 userMetaEntryOk.fold{
179                   val error = ErrorResult(s"Could not add ${mdMetadata.fileIdentifier} to your collection of ${request.user.email}.", None)
180                   logger.warn(error.message)
181                   BadRequest(Json.toJson(error)).as(JSON)
182                 }{
183                   userMetaEntry =>
184                     val added = collectionsService.addMdResourceToUserDefaultCollection(CSW_URL, mdMetadata, userMetaEntry)
185                     if (added) {
186                       Ok(Json.obj("type" -> "success", "fileIdentifier" -> fileIdentifier,
187                         "message" -> s"Inserted as ${fileIdentifier} and reference entry added to your data collection."))
188                     } else {
189                       val error = ErrorResult(s"Could not add ${mdMetadata.fileIdentifier} to your collection of ${request.user.email}.", None)
190                       logger.warn(error.message)
191                       BadRequest(Json.toJson(error)).as(JSON)
192                     }
193                 }
194               }
195               case e: Elem if e.label == "ExceptionReport" => {
196                 val message = (e \\ "Exception" \\ "ExceptionText").text
197                 //TODO SR make this InternalServerError
198                 val error = ErrorResult(message, None)
199                 InternalServerError(Json.toJson(error)).as(JSON)
200               }
201               case _ => {
202                 val error = ErrorResult(s"Unexpected Response from CSW: ${response._1.status} - ${response._1.statusText}",
203                   Some(response._1.body))
204                 InternalServerError(Json.toJson(error)).as(JSON)
205               }
206             }
207           })
208         }
209       )
210   }
