31 /**
32   * provides reasonably current  and secure measures of password hashing as of 2015/2016

43   val PBKDF2_ALGORITHM = "PBKDF2WithHmacSHA1"

59   /**
60     * create Session Cookie
61     *
62     * @param username
63     * @param useragent
64     * @return
65     */
66   def createSessionCookie(username: String, useragent: String): String = {
67     val in = appSecret + username + useragent.toUpperCase
68     val digest = MessageDigest.getInstance("SHA-512").digest(in.getBytes("UTF-8"))
69     // Logger.trace(s"createSessionCookie: ${toHex(digest)} app secret: $appSecret , email: $email, useragent: $useragent >>> $in")
70     toHex(digest)
71   }

76     * @param cookieHash
77     * @param username

108   /**
109     *
110     * @param password
111     * @return
112     */
113   def createHash(password: Array[Char]): String = {
114     val salt = nextRandomSalt
115     val hash = pbkdf2(password, salt, PBKDF2_ITERATIONS, HASH_BYTE_SIZE)
116     PBKDF2_ITERATIONS + ":" + toHex(salt) + ":" + toHex(hash)
117   }
118 
119   /**
120     *
121     * @return
122     */
123   private def nextRandomSalt(): Array[Byte] = {
124     val random = new SecureRandom()
125     val salt = Array.ofDim[Byte](SALT_BYTE_SIZE)
126     random.nextBytes(salt)
127     salt
128   }

136   def validatePassword(password: String, correctHash: String): Boolean = {

139 
140   /**
141     * the work horse...
142     *
143     * @param password
144     * @param correctHash
145     * @return
146     */
147   def validatePassword(password: Array[Char], correctHash: String): Boolean = {
148     val params = correctHash.split(":")
149     val iterations = Integer.parseInt(params(ITERATION_INDEX))
150     val salt = fromHex(params(SALT_INDEX))
151     val hash = fromHex(params(PBKDF2_INDEX))
152     val testHash = pbkdf2(password, salt, iterations, hash.length)
153     slowEquals(hash, testHash)
154   }
155 
156   /**
157     * the other work horse
158     *
159     * @param a
160     * @param b
161     * @return
162     */
163   private def slowEquals(a: Array[Byte], b: Array[Byte]): Boolean = {
164     val range = 0 until scala.math.min(a.length, b.length)
165     val diff = range.foldLeft(a.length ^ b.length) {
166       case (acc, i) => acc | a(i) ^ b(i)
167     }
168     diff == 0
169   }
170 
171   /**
172     * crypto magic
173     *
174     * @param password
175     * @param salt
176     * @param iterations
177     * @param bytes
178     * @return
179     */
180   private def pbkdf2(password: Array[Char], salt: Array[Byte], iterations: Int, bytes: Int): Array[Byte] = {
181     val spec = new PBEKeySpec(password, salt, iterations, bytes * 8)
182     val skf = SecretKeyFactory.getInstance(PBKDF2_ALGORITHM)
183     skf.generateSecret(spec).getEncoded()
184   }
185 
186   /**
187     *
188     * @param hex
189     * @return
190     */
191   private def fromHex(hex: String): Array[Byte] = {
192     hex.sliding(2, 2).toArray.map(Integer.parseInt(_, 16).toByte)
193   }
