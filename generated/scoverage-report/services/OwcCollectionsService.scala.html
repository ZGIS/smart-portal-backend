341   /**
342     * returns an Option[EmailAddress] object if parsing is successful, for those mdMetadata email joint arrays
343     *
344     * @param emailString the string that might comprise of or contain an email address
345     * @return Option[EmailAddress]
346     */
347   def parseEmailStringtoEmailAddress(emailString: String): Option[EmailAddress] = {
348 
349     if (EmailAddress.isValid(emailString)) {
350       Some(EmailAddress(emailString))
351     } else {
352       if (emailString.contains(",")) {
353         val commaFree = emailString.split(",").head.trim
354 
355         Try {
356           EmailAddress(commaFree)
357         } match {
358           case Success(e) => Some(e)
359           case _ => None
360         }
361       } else {
362         None
363       }
364     }
