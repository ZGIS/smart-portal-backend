93   /**
94     * create a new user group. user must already be in the sent object rights list as power user
95     *
96     * @return
97     */
98   def createUsersOwnUserGroup: Action[JsValue] = defaultAuthAction(parse.json) {
99     request =>
100       request.body.validate[UserGroup].fold(
101         errors => {
102           logger.error(JsError.toJson(errors).toString())
103           val error: ErrorResult = ErrorResult("Usergroup format could not be read.",
104             Some(StringEscapeUtils.escapeJson(errors.mkString("; "))))
105           BadRequest(Json.toJson(error)).as(JSON)
106         },
107         userGroup => {
108           // check if creating user is in the member list and has power user right
109           val isPresent = userGroup.hasUsersLevel.exists(ul => ul.users_accountsubject.equals(request.user.accountSubject) && ul.userlevel >= 2)
110           if (!isPresent) {
111             logger.error("User not listed with adequate rights. We would create a group that you can't edit, aborting.")
112             val error = ErrorResult("User not listed with adequate rights. We would create a group that you can't edit, aborting.", None)
113             BadRequest(Json.toJson(error)).as(JSON)
114           } else {
115             userGroupService.createUsersOwnUserGroup(request.user, userGroup).fold {
116               logger.error("Error creating the user group.")
117               val error = ErrorResult("Error creating the user group.", None)
118               BadRequest(Json.toJson(error)).as(JSON)
119             } {
120               ugroup =>
121                 Ok(Json.obj("status" -> "OK", "usergroup" -> Json.toJson(ugroup)))
122             }
123           }
124         })
125   }

197     request =>

199       val ugList = userGroupService.getOwcContextsRightsMatrixForUser(request.user)
200 
201       val areYouPowerUserForThisContext = ugList.filter(_.owcContextId.contentEquals(owcContextId)).exists(_.queryingUserAccessLevel >= 2)

203       if (areYouPowerUserForThisContext) {
204         val visOk = collectionsService.updateOwcContextVisibility(owcContextId, request.user, visibility)
205         if (visOk) {
206           Ok(Json.obj("status" -> "OK", "owcContext" -> owcContextId, "user" -> request.user.accountSubject, "visibility" -> visibility))
207         } else {
208           logger.error("Error changing the visibility of the context, database error.")
209           val error = ErrorResult("Error changing the visibility of the context, database error.", None)

211         }
