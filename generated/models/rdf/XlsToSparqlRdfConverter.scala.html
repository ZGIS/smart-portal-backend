59     * @param row
60     * @param cell
61     * @param sheet
62     * @return
63     */
64   def getCellValueAsStringOption(row: Int, cell: Int, sheet: Sheet): Option[String] = {
65     val theRow = sheet.getRow(row)
66     if (theRow != null) {
67 
68       val lastAccessibleCell: Int = if (theRow.getLastCellNum - 1 <= theRow.getPhysicalNumberOfCells) {
69         theRow.getLastCellNum - 1
70       } else {
71         theRow.getPhysicalNumberOfCells
72       }
73       logger.trace(s"getCellValueAsStringOption.lastCell: ${lastAccessibleCell + 1} : lastAccessibleCell ${lastAccessibleCell}")
74       //      if (lastAccessibleCell >= cell) {
75       try {
76         theRow.getCell(cell, MissingCellPolicy.RETURN_NULL_AND_BLANK)
77       } catch {
78         case ex: IllegalArgumentException =>
79           logger.warn(s"IllegalArgumentException row $row / $cell: ${ex.getMessage}")
80       }
81       if (theRow.getCell(cell, MissingCellPolicy.RETURN_NULL_AND_BLANK) != null) {
82         logger.trace(s"getCellValueAsStringOption => row: $row, cell: $cell, sheet: ${sheet.getSheetName}," +
83           s"getCellType: ${sheet.getRow(row).getCell(cell).getCellType}")
84         sheet.getRow(row).getCell(cell).getCellType match {
85           case Cell.CELL_TYPE_NUMERIC => val value = Try {
86             sheet.getRow(row).getCell(cell).getNumericCellValue.toString
87           }.toOption
88             logger.trace(s"CELL_TYPE_NUMERIC: $value")
89             value
90           case Cell.CELL_TYPE_STRING => val value = Try {
91             val stringValue = sheet.getRow(row).getCell(cell).getStringCellValue
92             if (stringValue == null) "" else new String(stringValue.getBytes("Windows-1251"), "UTF-8")
93           }.toOption
94             logger.trace(s"CELL_TYPE_STRING: $value")
95             value
96           case Cell.CELL_TYPE_BLANK => logger.warn(s"getCellValueAsStringOption cell type => row: $row, cell: $cell, sheet: ${sheet.getSheetName}," +
97             "getCellType: CELL_TYPE_BLANK")
98             None
99           case Cell.CELL_TYPE_FORMULA => logger.warn(s"getCellValueAsStringOption cell type => row: $row, cell: $cell, sheet: ${sheet.getSheetName}," +
100             "getCellType: CELL_TYPE_FORMULA")
101             None
102           case Cell.CELL_TYPE_ERROR => logger.warn(s"getCellValueAsStringOption cell type => row: $row, cell: $cell, sheet: ${sheet.getSheetName}," +
103             "getCellType: CELL_TYPE_ERROR")
104             None
105           case Cell.CELL_TYPE_BOOLEAN => logger.warn(s"getCellValueAsStringOption cell type => row: $row, cell: $cell, sheet: ${sheet.getSheetName}," +
106             "getCellType: CELL_TYPE_BOOLEAN")
107             None
108           case _ => logger.error(s"getCellValueAsStringOption cell type => row: $row, cell: $cell, sheet: ${sheet.getSheetName}," +
109             "unexpected error get cell type")
110             None
111         }
112       } else {
113         logger.debug(s"void getCell(row: $row, cell: $cell) is null")
114         None
115       }
116       //      } else {
117       //        logger.warn(s"beyond cell range of row => row: $row, cell: $cell, sheet: ${sheet.getSheetName}," +
118       //          s"getLastCellNum: ${sheet.getRow(row).getLastCellNum}")
119       //        None
120       //      }
121     } else {
122       logger.debug("reaching into the void, theRow is null")
123       None
124     }

148               getCellValueAsStringOption(i, 1, synonymSheet).get.split(",").map(kw => kw.trim.toLowerCase).foreach(lkw => buf += lkw)

154         }
155       }
156     }
157     val resultBuf = buf.toList
158     logger.debug(s"findSynonymsForKeywords => results buf: ${resultBuf.mkString("::")}")

162   /**
163     * reads in an XLSX Sheet with categories definition and a keywords lookup sheet and produces a list of Categories
164     * encapsulated in [[CategoryHolder]] to be dropped as RDF/XML eventually
165     *
166     * @param domainSheet
167     * @param synonymSheet
168     * @return
169     */
170   def buildCategoriesFromSheet(domainSheet: org.apache.poi.ss.usermodel.Sheet,
171                                synonymSheet: org.apache.poi.ss.usermodel.Sheet): List[CategoryHolder] = {
172     val buf = scala.collection.mutable.ListBuffer.empty[CategoryHolder]
173     val lastParent = scala.collection.mutable.ListBuffer.empty[String]
174     val parentDefault = "main"
175     lastParent += parentDefault
176     val lastRow = if (domainSheet.getLastRowNum < domainSheet.getPhysicalNumberOfRows) domainSheet.getLastRowNum else domainSheet.getPhysicalNumberOfRows
177     logger.info(s"domainSheet.lastRow: ${lastRow}")
178     for (i <- 0 to lastRow) {
179       val cellValue = getCellValueAsStringOption(i, 0, domainSheet)
180       logger.trace(s"domainSheet.getRow($i).getCell(0).getCell: $cellValue")
181       if (cellValue.isDefined) {
182         val hierarchy_number = getCellValueAsStringOption(i, 0, domainSheet).get
183         val item_name = getCellValueAsStringOption(i, 1, domainSheet).getOrElse("")
184         logger.trace("row(+1) " + i + " hierarchy_number " + hierarchy_number + " item_name " + item_name)
185         if (hierarchy_number.equalsIgnoreCase("hierarchy_number")) {
186           if (!item_name.equalsIgnoreCase("item_name")) {
187             if (regMatcher.findFirstIn(item_name).isDefined) {
188               // val plist = regMatcher.findAllIn(item_name)
189               // lastParent += plist.toSeq.mkString("")
190               val plist = regMatcher.findAllIn(item_name).toList
191               logger.trace(s"regMatcher.findAllIn($item_name): ${plist.mkString("::")}")
192               lastParent += plist(1)
193             }
194           }
195         } else {
196           val lastParentValue = lastParent.last.trim
197           logger.trace(s"lastParentValue: ${lastParentValue}")
198           try {
199             logger.trace(s"before cell $i 4, after lastParentValue $lastParentValue")
200             val description = getCellValueAsStringOption(i, 4, domainSheet).getOrElse("")
201             logger.trace(s"before cell $i 5, after description $description")
202             val trimmed = getCellValueAsStringOption(i, 5, domainSheet).getOrElse("").split(",").map(_.trim)
203             val keywords_low = trimmed.map(_.toLowerCase).toList
204             val synonyms_low = findSynonymsForKeywords(keywords_low, synonymSheet)
205             val keyword_content = keywords_low ++ synonyms_low
206             logger.trace(s"before cell $i 6, after keyword_content ${keyword_content.mkString("::")}")
207             val icon = getCellValueAsStringOption(i, 6, domainSheet).getOrElse("")
208             logger.trace(s"before cell $i 7, after icon $icon")
209             val bg_icon = getCellValueAsStringOption(i, 7, domainSheet).getOrElse("")
210             logger.trace(s"before cell $i 8, after bg_icon $bg_icon")
211             val query_string = getCellValueAsStringOption(i, 8, domainSheet).getOrElse("")
212             if (i <= 10 && query_string.isEmpty) {
213               logger.warn(s"before cell $i categoryClass, after query_string $query_string")
214             } else {
215               logger.trace(s"before cell $i categoryClass, after query_string $query_string")
216             }
217             val categoryClass = if (lastParentValue.equalsIgnoreCase("main")) {
218               "MainCategory"
219             } else {
220               "ChildCategory"
221             }
222             logger.trace(s"before cat number $i")
223             val cat = CategoryHolder(
224               hierarchyNumber = hierarchy_number,
225               id = i,
226               parent = lastParentValue,
227               itemName = item_name,
228               description = description,
229               keywordContent = keyword_content,
230               queryString = query_string,
231               icon = icon,
232               bgIcon = bg_icon,
233               categoryClass = categoryClass)
234             logger.trace(s"after cat number $i")
235             logger.debug(cat.toRdf(fixedVocabHost))
236             buf += cat
237 
238           } catch {
239             case ex: Exception => logger.error(s"new CategoryHolder error on row $i: ${ex.getMessage} ${ex.getStackTrace.mkString("\n")}")
240           }
241         }
242       }
243     }
244     logger.info(s"domainSheet.lastRow: ${lastRow}")
245     val resultBuf = buf.toList
246     logger.info(s"buildCategoriesFromSheet => results buf(${resultBuf.length}): ${resultBuf.map(kw => kw.hierarchyNumber).mkString("::")}")
247     resultBuf
248   }

318

320       val term_id_opt = getCellValueAsStringOption(i, 0, termsListSheet)

322         logger.warn(s"term_id_opt row $i / cell 0: opt empty ?!?!")
323       }
324       term_id_opt.foreach {

326           val corrected_id = if (term_id.endsWith(".0")) term_id.replace(".0", "") else term_id
327           // val tup_buf = scala.collection.mutable.ListBuffer.empty[(String, String)]
328 
329           val tup_list: List[(String, String)] = (1 to lastAccessibleCell).flatMap { j =>
330             val headerTermElement_opt = getCellValueAsStringOption(0, j, termsListSheet)
331             if (headerTermElement_opt.isEmpty) {
332               logger.warn(s"headerTermElement_opt row $i / cell  $j: opt empty ?!?!")

334             }
335             headerTermElement_opt.flatMap { hdr =>
336               logger.debug(s"hdr row 0 / cell $j : $hdr")
337               val termValue_opt = getCellValueAsStringOption(i, j, termsListSheet)
338               if (termValue_opt.isEmpty) {
339                 logger.debug(s"tval row $i / cell $j: opt empty")
340                 None
341               }
342               termValue_opt.map { tval =>
343                 logger.debug(s"tval row $i / cell $j: $tval")
344                 Tuple2(hdr, tval)
345               }
346             }
347           }.toList
348           val ssrh = SimplifiedSkosRdfHolder(id = corrected_id, values = tup_list)
349           logger.debug(tup_list.mkString("::"))
350           logger.debug(ssrh.toRdf("gen", "term", "terms", fixedVocabHost))

358     resultBuf
