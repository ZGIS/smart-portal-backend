61   def getCellValueAsStringOption(row: Int, cell: Int, sheet: Sheet): Option[String] = {
62     val theRow = sheet.getRow(row)
63     if (theRow != null) {
64 
65       val lastAccessibleCell: Int = if (theRow.getLastCellNum - 1 <= theRow.getPhysicalNumberOfCells) {
66         theRow.getLastCellNum - 1
67       } else {
68         theRow.getPhysicalNumberOfCells
69       }
70       logger.trace(s"getCellValueAsStringOption.lastCell: ${lastAccessibleCell + 1} : lastAccessibleCell ${lastAccessibleCell}")
71       //      if (lastAccessibleCell >= cell) {
72       try {
73         theRow.getCell(cell, MissingCellPolicy.RETURN_NULL_AND_BLANK)
74       } catch {
75         case ex: IllegalArgumentException =>
76           logger.warn(s"IllegalArgumentException row $row / $cell: ${ex.getMessage}")
77       }
78       if (theRow.getCell(cell, MissingCellPolicy.RETURN_NULL_AND_BLANK) != null) {
79         logger.trace(s"getCellValueAsStringOption => row: $row, cell: $cell, sheet: ${sheet.getSheetName}," +
80           s"getCellType: ${sheet.getRow(row).getCell(cell).getCellType}")
81         sheet.getRow(row).getCell(cell).getCellType match {
82           case Cell.CELL_TYPE_NUMERIC => val value = Try {
83             sheet.getRow(row).getCell(cell).getNumericCellValue.toString
84           }.toOption
85             logger.trace(s"CELL_TYPE_NUMERIC: $value")
86             value
87           case Cell.CELL_TYPE_STRING => val value = Try {
88             val stringValue = sheet.getRow(row).getCell(cell).getStringCellValue
89             if (stringValue == null) "" else new String(stringValue.getBytes("Windows-1251"), "UTF-8")
90           }.toOption
91             logger.trace(s"CELL_TYPE_STRING: $value")
92             value
93           case Cell.CELL_TYPE_BLANK => logger.warn(s"getCellValueAsStringOption cell type => row: $row, cell: $cell, sheet: ${sheet.getSheetName}," +
94             "getCellType: CELL_TYPE_BLANK")
95             None
96           case Cell.CELL_TYPE_FORMULA => logger.warn(s"getCellValueAsStringOption cell type => row: $row, cell: $cell, sheet: ${sheet.getSheetName}," +
97             "getCellType: CELL_TYPE_FORMULA")
98             None
99           case Cell.CELL_TYPE_ERROR => logger.warn(s"getCellValueAsStringOption cell type => row: $row, cell: $cell, sheet: ${sheet.getSheetName}," +
100             "getCellType: CELL_TYPE_ERROR")
101             None
102           case Cell.CELL_TYPE_BOOLEAN => logger.warn(s"getCellValueAsStringOption cell type => row: $row, cell: $cell, sheet: ${sheet.getSheetName}," +
103             "getCellType: CELL_TYPE_BOOLEAN")
104             None
105           case _ => logger.error(s"getCellValueAsStringOption cell type => row: $row, cell: $cell, sheet: ${sheet.getSheetName}," +
106             "unexpected error get cell type")
107             None
108         }
109       } else {
110         logger.debug(s"void getCell(row: $row, cell: $cell) is null")
111         None
112       }
113       //      } else {
114       //        logger.warn(s"beyond cell range of row => row: $row, cell: $cell, sheet: ${sheet.getSheetName}," +
115       //          s"getLastCellNum: ${sheet.getRow(row).getLastCellNum}")
116       //        None
117       //      }
118     } else {
119       logger.debug("reaching into the void, theRow is null")

126     *
127     * @param keywords_list

133     val buf = scala.collection.mutable.ArrayBuffer.empty[String]
134     keywords_list.foreach { keyword =>
135       logger.trace(s"findSynonymsForKeywords.foreach keyword: $keyword")
136       val lastRow = if (synonymSheet.getLastRowNum < synonymSheet.getPhysicalNumberOfRows) synonymSheet.getLastRowNum else synonymSheet.getPhysicalNumberOfRows
137       logger.debug(s"synonymsheet.lastRow: ${lastRow}")
138       for (i <- 1 to lastRow) {
139         if (synonymSheet.getRow(i).getLastCellNum > 1) {
140           logger.trace(s"findSynonymsForKeywords.foreach synonymSheet.getLastRowNum: ${synonymSheet.getLastRowNum}")
141           if (getCellValueAsStringOption(i, 0, synonymSheet).isDefined) {
142             logger.trace(s"findSynonymsForKeywords.foreach getCellValueAsString($i, 0, ${synonymSheet.getLastRowNum}): " +
143               s"${getCellValueAsStringOption(i, 0, synonymSheet)}")
144             if (getCellValueAsStringOption(i, 0, synonymSheet).get.equalsIgnoreCase(keyword)) {
145               getCellValueAsStringOption(i, 1, synonymSheet).get.split(",").map(kw => kw.trim.toLowerCase).foreach(lkw => buf += lkw)
146             }
147           }
148         } else {
149           logger.debug(s"beyond cell range of row => row: $i, cell: 1, sheet: ${synonymSheet.getSheetName}," +
150             s"getLastCellNum: ${synonymSheet.getRow(i).getLastCellNum}")
151         }
152       }
153     }
154     val resultBuf = buf.toList
155     logger.debug(s"findSynonymsForKeywords => results buf: ${resultBuf.mkString("::")}")
156     resultBuf
157   }

159   /**
160     * reads in an XLSX Sheet with categories definition and a keywords lookup sheet and produces a list of Categories
161     * encapsulated in [[CategoryHolder]] to be dropped as RDF/XML eventually
162     *
163     * @param domainSheet
164     * @param synonymSheet
165     * @return
166     */
167   def buildCategoriesFromSheet(domainSheet: org.apache.poi.ss.usermodel.Sheet,
168                                synonymSheet: org.apache.poi.ss.usermodel.Sheet): List[CategoryHolder] = {
169     val buf = scala.collection.mutable.ListBuffer.empty[CategoryHolder]
170     val lastParent = scala.collection.mutable.ListBuffer.empty[String]
171     val parentDefault = "main"
172     lastParent += parentDefault
173     val lastRow = if (domainSheet.getLastRowNum < domainSheet.getPhysicalNumberOfRows) domainSheet.getLastRowNum else domainSheet.getPhysicalNumberOfRows
174     logger.info(s"domainSheet.lastRow: ${lastRow}")
175     for (i <- 0 to lastRow) {
176       val cellValue = getCellValueAsStringOption(i, 0, domainSheet)
177       logger.trace(s"domainSheet.getRow($i).getCell(0).getCell: $cellValue")
178       if (cellValue.isDefined) {
179         val hierarchy_number = getCellValueAsStringOption(i, 0, domainSheet).get
180         val item_name = getCellValueAsStringOption(i, 1, domainSheet).getOrElse("")
181         logger.trace("row(+1) " + i + " hierarchy_number " + hierarchy_number + " item_name " + item_name)
182         if (hierarchy_number.equalsIgnoreCase("hierarchy_number")) {
183           if (!item_name.equalsIgnoreCase("item_name")) {
184             if (regMatcher.findFirstIn(item_name).isDefined) {
185               // val plist = regMatcher.findAllIn(item_name)
186               // lastParent += plist.toSeq.mkString("")
187               val plist = regMatcher.findAllIn(item_name).toList
188               logger.trace(s"regMatcher.findAllIn($item_name): ${plist.mkString("::")}")
189               lastParent += plist(1)
190             }
191           }
192         } else {
193           val lastParentValue = lastParent.last.trim
194           logger.trace(s"lastParentValue: ${lastParentValue}")
195           try {
196             logger.trace(s"before cell $i 4, after lastParentValue $lastParentValue")
197             val description = getCellValueAsStringOption(i, 4, domainSheet).getOrElse("")
198             logger.trace(s"before cell $i 5, after description $description")
199             val trimmed = getCellValueAsStringOption(i, 5, domainSheet).getOrElse("").split(",").map(_.trim)
200             val keywords_low = trimmed.map(_.toLowerCase).toList
201             val synonyms_low = findSynonymsForKeywords(keywords_low, synonymSheet)
202             val keyword_content = keywords_low ++ synonyms_low
203             logger.trace(s"before cell $i 6, after keyword_content ${keyword_content.mkString("::")}")
204             val icon = getCellValueAsStringOption(i, 6, domainSheet).getOrElse("")
205             logger.trace(s"before cell $i 7, after icon $icon")
206             val bg_icon = getCellValueAsStringOption(i, 7, domainSheet).getOrElse("")
207             logger.trace(s"before cell $i 8, after bg_icon $bg_icon")
208             val query_string = getCellValueAsStringOption(i, 8, domainSheet).getOrElse("")
209             if (i <= 10 && query_string.isEmpty) {
210               logger.warn(s"before cell $i categoryClass, after query_string $query_string")
211             } else {
212               logger.trace(s"before cell $i categoryClass, after query_string $query_string")
213             }
214             val categoryClass = if (lastParentValue.equalsIgnoreCase("main")) {
215               "MainCategory"
216             } else {
217               "ChildCategory"
218             }
219             logger.trace(s"before cat number $i")
220             val cat = CategoryHolder(
221               hierarchyNumber = hierarchy_number,
222               id = i,
223               parent = lastParentValue,
224               itemName = item_name,
225               description = description,
226               keywordContent = keyword_content,
227               queryString = query_string,
228               icon = icon,
229               bgIcon = bg_icon,
230               categoryClass = categoryClass)
231             logger.trace(s"after cat number $i")
232             logger.debug(cat.toRdf)
233             buf += cat
234 
235           } catch {
236             case ex: Exception => logger.error(s"new CategoryHolder error on row $i: ${ex.getMessage} ${ex.getStackTrace.mkString("\n")}")
237           }
238         }
239       }
240     }
241     logger.info(s"domainSheet.lastRow: ${lastRow}")
242     val resultBuf = buf.toList
243     logger.info(s"buildCategoriesFromSheet => results buf(${resultBuf.length}): ${resultBuf.map(kw => kw.hierarchyNumber).mkString("::")}")
244     resultBuf
245   }

315 
316     for (i <- 1 to lastRow) {
317       val term_id_opt = getCellValueAsStringOption(i, 0, termsListSheet)
318       if (term_id_opt.isEmpty) {
319         logger.warn(s"term_id_opt row $i / cell 0: opt empty ?!?!")
320       }
321       term_id_opt.foreach {
322         term_id =>
323           val corrected_id = if (term_id.endsWith(".0")) term_id.replace(".0", "") else term_id
324           // val tup_buf = scala.collection.mutable.ListBuffer.empty[(String, String)]
325 
326           val tup_list: List[(String, String)] = (1 to lastAccessibleCell).flatMap { j =>
327             val headerTermElement_opt = getCellValueAsStringOption(0, j, termsListSheet)
328             if (headerTermElement_opt.isEmpty) {
329               logger.warn(s"headerTermElement_opt row $i / cell  $j: opt empty ?!?!")

331             }

333               logger.debug(s"hdr row 0 / cell $j : $hdr")

336                 logger.debug(s"tval row $i / cell $j: opt empty")
337                 None
338               }
339               termValue_opt.map { tval =>
340                 logger.debug(s"tval row $i / cell $j: $tval")
341                 Tuple2(hdr, tval)
342               }
343             }
344           }.toList
345           val ssrh = SimplifiedSkosRdfHolder(id = corrected_id, values = tup_list)
346           logger.debug(tup_list.mkString("::"))
347           logger.debug(ssrh.toRdf("gen", "term", "terms"))

352     logger.info(s"theRow.getLastCellNum: ${theRow.getLastCellNum}")
353     val resultBuf = buf.toList
354     logger.info(s"parseSkosDcTermFromSheet => results buf(${resultBuf.length}): ${resultBuf.map(kw => kw.id).mkString("::")}")
