140   def uploadMultipartForm: Action[MultipartFormData[TemporaryFile]] = HasToken(parse.multipartFormData) {
141     token =>
142       authUser =>
143         implicit request =>
144 
145           request.body.file("file").map { theFile =>
146             import java.io.File
147             val filename = theFile.filename
148             val contentType = theFile.contentType
149             val tmpFile = new File(s"$uploadDataPath/$filename")
150             val handle = theFile.ref.moveTo(tmpFile)
151             val kilobytes = handle.length / 1024
152 
153             // Google Java upload stuff
154             import scala.collection.JavaConverters._
155 
156             // val storage: Storage = authenticatedStorageOptions.getService()
157             val storage: Storage = StorageOptions.getDefaultInstance().getService()
158             // val noCredentials = NoCredentials.getInstance()
159 
160             // Modify access list to allow all users with link to read file
161             val roAcl = Acl.of(Acl.User.ofAllUsers(), Acl.Role.READER)
162             val acls: java.util.List[Acl] = List(roAcl).asJava
163 
164             // the inputstream is closed by default, so we don't need to close it here
165             val blob: Blob = storage.create(BlobInfo.newBuilder(googleStorageBucket, filename).setAcl(acls).build(),
166               Files.asByteSource(tmpFile).openStream())
167 
168             // return the public download link
169             val owcResource = fileMetadata(filename, contentType, authUser, blob.getMediaLink(), Try(kilobytes.toInt).toOption)
170             val insertOk = collectionsService.addPlainFileResourceToUserDefaultCollection(owcResource, authUser)
171 
172             if (insertOk) {
173               logger.debug(s"file upload $filename to ${blob.getMediaLink}")
174               //FIXME SR do we also want to have a general "return" object? Status is always in the respone so it does not need to be in here
175               Ok(Json.obj("status" -> "OK", "message" -> s"file uploaded $filename.", "file" -> blob.getMediaLink(), "entry" -> owcResource.toJson))
176             } else {
177               logger.error("file metadata insert failed.")
178               val error = ErrorResult("File metadata insert failed.", None)
179               BadRequest(Json.toJson(error)).as(JSON)
180             }
181           }.getOrElse {
182             logger.error("file upload failed")
183             val error = ErrorResult("File upload failed.", None)
184             BadRequest(Json.toJson(error)).as(JSON)
185           }
186   }
