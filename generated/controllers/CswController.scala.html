114     * trying to generalise for all transaction types
115     *

119   def initiateTransactionRequest(transaction: CswTransactionWithIndexUpdate): Future[(WSResponse, WSResponse)] = {
120     //FIXME SR I find that pretty hard to read. Is there a better way of chaining WS calls?
121     // see https://www.playframework.com/documentation/2.5.x/ScalaWS#chaining-ws-calls
122     for {
123       getCapaResponse <- wsClient.url(CSW_OPERATIONS_METADATA_URL).get()
124       transactionResponse <- {
125         //check if getCapaResponse indicates, that the CSW can perform Transactions
126         val operation = (getCapaResponse.xml \\ "OperationsMetadata" \\ "Operation").filter(node => {
127           logger.debug(s"Attribute 'name'=${node.attribute("name").get.text.toString}")
128           node.attribute("name").get.text.equals("Transaction")
129         })
130         logger.debug(s"operation = ${operation.toString()}")
131         if (operation.isEmpty) {
132           throw new UnsupportedOperationException("CSW does not support Transaction.")
133         }
134 
135         // prepare transaction template
136         val finalXML = transaction.transform
137 
138         logger.debug(s"finalXml: ${finalXML.toString()}")
139         wsClient.url(CSW_URL).post(finalXML.toString())
140       }
141 
142       updateIngesterIndexResponse <- {
143         // if (transactionResponse.status == 200) {
144         val response = wsClient.url(INGESTER_UPDATE_INDEX_URL).get()
145         response.onSuccess {
146           case response if response.status == 200 => {
147             logger.info(
148               s"Successfully called $INGESTER_UPDATE_INDEX_URL (${response.status} - ${response.statusText})")
149             logger.info(s"response: ${response.body}")
150           }
151           case response => {
152             logger.warn(
153               s"Call to '$INGESTER_UPDATE_INDEX_URL' returned error: ${response.status} - ${response.statusText}")
154             logger.warn(s"response: ${response.body}")
155           }
156         }
157         response
158 
159       }
160     } yield (transactionResponse, updateIngesterIndexResponse)
161   }

163   /**
164     * check the response of the tansaction
165     *
166     * @param response
167     * @param transaction
168     * @return
169     */
170   def evaluateTransactionResponse(response: WSResponse, transaction: CswTransactionWithIndexUpdate): Either[ErrorResult, UUID] = {
171     logger.debug(response.xml.toString())
172     response.xml match {
173       case e: Elem if e.label == "TransactionResponse" =>
174 
175         if (transaction.responsePositive(e)) {
176           Right(UUID.fromString(transaction.fileIdentifier))
177         } else {
178           Left(ErrorResult("Identifier not clear in TransactionResponse", None))
179         }
180 
181       case e: Elem if e.label == "ExceptionReport" =>
182         val message = (e \\ "Exception" \\ "ExceptionText").text
183         val error = ErrorResult(message, None)
184         Left(error)
185 
186       case _ =>
187         val error = ErrorResult(s"Unexpected Response from CSW: ${response.status} - ${response.statusText}",
188           Some(response.body))
189         Left(error)
190 
191     }
192   }

199   def insertMetadataRecord: Action[JsValue] = (authenticationAction andThen userAction).async(parse.json) {
200     request =>
201 
202       logger.debug(request.toString)
203       (request.body.as[JsObject] \ "metadata").validate[MdMetadata].fold(
204         errors => {
205           logger.error(JsError.toJson(errors).toString())
206           val error = ErrorResult("Could not validate metadata from your request.", Some(JsError.toJson(errors).toString()))
207           Future {
208             BadRequest(Json.toJson(error)).as(JSON)
209           }
210         },
211         mdMetadata => {
212           logger.trace(Json.prettyPrint(mdMetadata.toJson()))
213           logger.debug(s"MD_Metadata XML: ${mdMetadata.toXml()}")
214 
215           val insertTransaction = CswInsertRequest(cswtInsertXml, mdMetadata)
216           val futureResponse: Future[(WSResponse, WSResponse)] = initiateTransactionRequest(insertTransaction)
217 
218           futureResponse.recover {
219             case e: Exception =>
220               logger.error("Insert CSW threw exception", e)
221               val error = ErrorResult(s"Exception (${e.getClass.getCanonicalName}) on CSW insert: ${e.getMessage}",
222                 None)
223               InternalServerError(Json.toJson(error)).as(JSON)
224           }
225 
226           futureResponse.map { response =>
227             logger.debug("Ingester Update: " + response._2.body)
228             val responseEval = evaluateTransactionResponse(response._1, insertTransaction)
229             responseEval.fold[Result](
230               error => {
231                 logger.warn(error.message)
232                 InternalServerError(Json.toJson(error)).as(JSON)
233               },
234               uuid => {
235                 logger.debug(s"Adding ${mdMetadata.fileIdentifier} to default collection of ${request.user.email}.")
236                 val userMetaEntryOk = userService.insertUserMetaRecordEntry(CSW_URL, mdMetadata.fileIdentifier, request.user.email)
237                 userMetaEntryOk.fold {
238                   val error = ErrorResult(s"Could not add ${mdMetadata.fileIdentifier} to your collection of ${request.user.email}.", None)
239                   logger.warn(error.message)
240                   BadRequest(Json.toJson(error)).as(JSON)
241                 } {
242                   userMetaEntry =>
243                     val owcresource = collectionsService.generateMdResource(CSW_URL, mdMetadata, userMetaEntry)
244                     val added = collectionsService.addMdResourceToUserDefaultCollection(CSW_URL, owcresource, userMetaEntry)
245                     if (added) {
246                       Ok(Json.obj("type" -> "success", "fileIdentifier" -> userMetaEntry.originaluuid.toString, "userMetaRecord" -> Json.toJson(userMetaEntry),
247                         "message" -> s"Inserted as ${uuid.toString} and reference entry with with refUuid ${userMetaEntry.uuid} added to your data collection."))
248                     } else {
249                       val error = ErrorResult(s"Could not add ${mdMetadata.fileIdentifier} to your collection of ${request.user.email}.", None)
250                       logger.warn(error.message)
251                       BadRequest(Json.toJson(error)).as(JSON)
252                     }
253                 }
254               })
255           }
256         }
257       )
