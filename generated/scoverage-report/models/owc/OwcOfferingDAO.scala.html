236   /**
237     * instantiates the concrete type of the offering,
238     * I believe there is possibly also something from Ratiopharm, like Class.forName, reflection etc,
239     * I am not even sure right now if we need the type
240     *
241     * @param uuid
242     * @param offeringType
243     * @param code
244     * @param content
245     * @return
246     */
247   def instantiateOwcOffering(uuid: String, offeringType: String, code: String, content: Option[String]): OwcOffering = {
248     val uuidObj = UUID.fromString(uuid)
249     val ops = findOwcOperationsByOfferingUUID(UUID.fromString(uuid)).toList
250     val contentList = content.map(text => List(text)).getOrElse(List())
251 
252     offeringType match {
253       case "WmsOffering" => WmsOffering(uuidObj, code, ops, contentList)
254       case "WmtsOffering" => WmtsOffering(uuidObj, code, ops, contentList)
255       case "WfsOffering" => WfsOffering(uuidObj, code, ops, contentList)
256       case "WcsOffering" => WcsOffering(uuidObj, code, ops, contentList)
257       case "CswOffering" => CswOffering(uuidObj, code, ops, contentList)
258       case "WpsOffering" => WpsOffering(uuidObj, code, ops, contentList)
259       case "GmlOffering" => GmlOffering(uuidObj, code, ops, contentList)
260       case "KmlOffering" => KmlOffering(uuidObj, code, ops, contentList)
261       case "GeoTiffOffering" => GeoTiffOffering(uuidObj, code, ops, contentList)
262       case "SosOffering" => SosOffering(uuidObj, code, ops, contentList)
263       case "NetCdfOffering" => NetCdfOffering(uuidObj, code, ops, contentList)
264       case "HttpLinkOffering" => HttpLinkOffering(uuidObj, code, ops, contentList)
265       case _ => throw new InvalidClassException(offeringType, s"Unknown Offering type $offeringType")
266     }
267   }

331   def createOwcOffering(owcOffering: OwcOffering): Option[OwcOffering] = {
332     db.withTransaction {
333       implicit connection => {
334 
335         val rowCount = SQL(
336           s"""
337           insert into $tableOwcOfferings values (

342           'offering_type -> owcOffering.getClass.getSimpleName,
343           'code -> owcOffering.code,
344           'content -> owcOffering.content.headOption
345         ).executeUpdate()
346 
347         owcOffering.operations.foreach {
348           owcOperation => {
349             if (findOwcOperationByUuid(owcOperation.uuid).isEmpty) {

355                  |{owc_offerings_uuid}, {owc_operations_uuid}
356                  |)
357                """.stripMargin).on(
358               'owc_offerings_uuid -> owcOffering.uuid.toString,
359               'owc_operations_uuid -> owcOperation.uuid.toString
360             ).executeUpdate()
361           }
362         }
363
