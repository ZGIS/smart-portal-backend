645   /**
646     * create owc properties in the database, title has additionally unique constraint, uuid will be unique and primary key
647     *
648     * @param owcProperties
649     * @return
650     */
651   def createOwcProperties(owcProperties: OwcProperties): Option[OwcProperties] = {
652 
653     db.withTransaction {
654       implicit connection => {
655 
656         val rowCount = SQL(
657           s"""
658           insert into $tableOwcProperties values (
659             {uuid}, {language}, {title}, {subtitle}, {updated}, {generator}, {rights}, {creator}, {publisher}
660           )
661         """).on(
662           'uuid -> owcProperties.uuid.toString,
663           'language -> owcProperties.language,
664           'title -> owcProperties.title,
665           'subtitle -> owcProperties.subtitle,
666           'updated -> owcProperties.updated,
667           'generator -> owcProperties.generator,
668           'rights -> owcProperties.rights,
669           'creator -> owcProperties.creator,
670           'publisher -> owcProperties.publisher
671         ).executeUpdate()
672 
673         owcProperties.authors.foreach {
674           author => {
675             if (findOwcAuthorByUuid(author.uuid).isEmpty) {
676               createOwcAuthor(author)
677             }
678 
679             SQL(
680               s"""insert into $tableOwcPropertiesHasOwcAuthors values (
681                  |{owc_properties_uuid}, {owc_authors_uuid}
682                  |)
683                """.stripMargin).on(
684               'owc_properties_uuid -> owcProperties.uuid.toString,
685               'owc_authors_uuid -> author.uuid.toString
686             ).executeUpdate()
687           }
688         }
689 
690         owcProperties.contributors.foreach {
691           authorAsContributor => {
692             if (findOwcAuthorByUuid(authorAsContributor.uuid).isEmpty) {
693               createOwcAuthor(authorAsContributor)
694             }
695 
696             SQL(
697               s"""insert into $tableOwcPropertiesHasOwcAuthorsAsContributors values (
698                  |{owc_properties_uuid}, {owc_authors_as_contributors_uuid}
699                  |)
700                """.stripMargin).on(
701               'owc_properties_uuid -> owcProperties.uuid.toString,
702               'owc_authors_as_contributors_uuid -> authorAsContributor.uuid.toString
703             ).executeUpdate()
704           }
705         }
706 
707         owcProperties.categories.foreach {
708           category => {
709             if (findOwcCategoriesByUuid(category.uuid).isEmpty) {
710               createOwcCategory(category)
711             }
712 
713             SQL(
714               s"""insert into $tableOwcPropertiesHasOwcCategories  values (
715                  |{owc_properties_uuid}, {owc_categories_uuid}
716                  |)
717                """.stripMargin).on(
718               'owc_properties_uuid -> owcProperties.uuid.toString,
719               'owc_categories_uuid -> category.uuid.toString
720             ).executeUpdate()
721           }
722         }
723 
724         owcProperties.links.foreach {
725           owcLink => {
726             if (findOwcLinksByUuid(owcLink.uuid).isEmpty) {
727               createOwcLink(owcLink)
728             }
729 
730             SQL(
731               s"""insert into $tableOwcPropertiesHasOwcLinks  values (
732                  |{owc_properties_uuid}, {owc_links_uuid}
733                  |)
734                """.stripMargin).on(
735               'owc_properties_uuid -> owcProperties.uuid.toString,
736               'owc_links_uuid -> owcLink.uuid.toString
737             ).executeUpdate()
738           }
739         }
740 
741         rowCount match {
742           case 1 => Some(owcProperties)
743           case _ => None
744         }
745       }
746     }
747   }
