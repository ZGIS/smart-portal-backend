111   /**
112     * login with JSON (from Angular / form)
113     * {"email":"akmoch","password":"testpass123"}
114     *
115     * Create the authentication token and sets the cookie [[AuthTokenCookieKey]]
116     * for AngularJS to use also in X-XSRF-TOKEN in HTTP header.
117     *
118     * @return
119     */
120   def login: Action[JsValue] = Action(parse.json) { implicit request =>
121     request.body.validate[LoginCredentials].fold(
122       errors => {
123         logger.error(JsError.toJson(errors).toString())
124         val error = ErrorResult("Could not validate request.", Some(JsError.toJson(errors).toString()))
125         BadRequest(Json.toJson(error)).as(JSON)
126       },
127       valid = credentials => {
128         // authenticate user in db
129         userService.authenticateLocal(credentials) match {
130           case Left(error) =>
131             logger.error(error.message)
132             Unauthorized(Json.toJson(error)).as(JSON)
133 
134           case Right(user) =>
135             val uaIdentifier: String = request.headers.get(UserAgentHeader).getOrElse(UserAgentHeaderDefault)
136             logger.trace(s"Logging in email from $uaIdentifier")
137             val token = userService.upsertUserSessionCache(user.email.value, uaIdentifier)
138             logger.trace(s"Logging in setting cache $token")
139             Ok(Json.obj("status" -> "OK", "token" -> token, "email" -> user.email.value, "userprofile" -> user.asProfileJs))
140               .withCookies(Cookie(AuthTokenCookieKey, token, None, httpOnly = false))
141 
142           case _ =>
143             val error = ErrorResult("Could not validate request.", None)
144             logger.error(error.message)
145             InternalServerError(Json.toJson(error)).as(JSON)
146         }
147       })
148   }

150   /**
151     * https://developers.google.com/recaptcha/docs/verify
152     *
153     * @param recaptcaChallenge
154     * @return
155     */
156   def recaptchaValidate(recaptcaChallenge: String): Action[AnyContent] = Action.async { implicit request =>
157     // TODO check from where URL referer comes from
158     ws.url(recaptcaVerifyUrl)
159       .withHeaders("Accept" -> "application/json")
160       .withRequestTimeout(10000.millis)
161       .withQueryString("secret" -> reCaptchaSecret,
162         "response" -> recaptcaChallenge).get().map {
163       response =>
164         val success = (response.json \ "success").as[Boolean]
165         if (success) {
166           Ok(Json.obj("status" -> "OK", "message" -> "granted", "success" -> JsBoolean(true)))
167         }
168         else {
169           val errors = (response.json \ "error-codes")
170           val jsErrors = errors.getOrElse(JsString("No further errors"))
171           val error = ErrorResult("User email or password wrong.", Some(jsErrors.toString()))
172           BadRequest(Json.toJson(error)).as(JSON)
173         }
174     }.recover {
175       case e: Exception =>
176         val error = ErrorResult("Exception on validating captcha.", Some(e.getMessage))
177         InternalServerError(Json.toJson(error)).as(JSON)
178     }
179   }
180 
181   /**
182     * Log-out a user. Invalidates the authentication token.
183     *
184     * Discard the cookie [[AuthTokenCookieKey]] to have AngularJS no longer set the
185     * X-XSRF-TOKEN in HTTP header.
186     */
187   def logout: Action[Unit] = HasToken(parse.empty) {
188     token =>
189       email =>
190         implicit request =>
191           userService.removeUserSessionCache(email, token)
192           Ok.discardingCookies(DiscardingCookie(name = AuthTokenCookieKey))
193   }
