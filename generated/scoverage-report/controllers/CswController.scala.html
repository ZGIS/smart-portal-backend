44   override def transform(n: Node): Seq[Node] = n match {
45     case e: Elem if e.label == "Insert" => {
46       new Elem(e.prefix, "Insert", e.attributes, e.scope, e.minimizeEmpty, (xml).toSeq: _*)

48     case x => x
49   }

83   def getValidValuesFor(topic: String): Action[AnyContent] = Action { request =>
84     import scala.collection.JavaConversions._
85 
86     logger.debug(s"returning valid values for '${topic}'")
87     val configurations = metadataValidValues.get
88     val validValuesConf = configurations.filter(conf => conf.getString("for").getOrElse("").equals(topic))
89     if (validValuesConf.isEmpty()) {
90       //TODO SR should we have a generic error return object that we return as JSON?
91       BadRequest(s"There are not valid values for '${topic}'")
92     }
93     else {
94       val conf = validValuesConf.head
95       val result = ValidValues.parseConfiguration(validValuesConf.head)
96       Ok(Json.toJson(result))
97     }

100   /** calls CSW:Transaction -> Insert
101     * FIXME should implement HasToken(parse.json) instead of Action.async, to only allow authenticated users
102     */
103   def insert: Action[AnyContent] = Action.async { request =>
104     logger.debug(request.body.asJson.toString)
105 
106     //TODO parse JSON to MDMetadataSet and convert that to XML
107 
108     val futureResponse: Future[WSResponse] = for {
109       getCapaResponse <- wsClient.url(CSW_OPERATIONS_METADATA_URL).get()
110       insertResponse <- {
111         //check if getCapaResponse indicates, that the CSW can perform Transactions
112         val operation = (getCapaResponse.xml \\ "OperationsMetadata" \\ "Operation").filter(node => {
113           logger.debug(s"Attribute 'name'=${node.attribute("name").get.text.toString}")
114           node.attribute("name").get.text.equals("Transaction")
115         })
116         logger.debug(s"operation = ${operation.toString()}")
117         if (operation.isEmpty) {
118           throw new UnsupportedOperationException("CSW does not support Transaction.")
119         }
120 
121         //insert MDMEtadata in insert template
122         val rule = new RuleTransformer(new AddMDMetadataToInsert(mdMetadata))
123         val finalXML = rule.transform(cswtInsertXml)
124         logger.error(finalXML.toString())
125         wsClient.url(CSW_URL).post(finalXML.toString())
126       }
127     } yield insertResponse
128 
129     futureResponse.recover {
130       case e: Exception =>
131         logger.error("Insert CSW threw exception", e)
132         InternalServerError(e.getMessage)
133     }
134 
135     futureResponse.map(response => {
136       logger.error(response.xml.toString())
137       response.xml match {
138         case e: Elem if e.label == "TransactionResponse" => {
139           val fileIdentifier = (e \\ "InsertResult" \\ "BriefRecord" \\ "identifier").text
140           Ok(Json.obj("type" -> "success", "fileIdentifier" -> fileIdentifier,
141             "message" -> s"Inserted as ${fileIdentifier}."))
142         }
143         case e: Elem if e.label == "ExceptionReport" => {
144           val message = (e \\ "Exception" \\ "ExceptionText").text
145           //TODO SR make this InternalServerError
146           Ok(Json.obj("type" -> "danger", "message" -> message))
147         }
148         case _ => InternalServerError(s"Unexpected Response: ${response}")
149       }
150     })
151   }
