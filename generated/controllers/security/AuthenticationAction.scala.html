44 /**
45   * the action checks if the incoming requets is for a valid server-side session (our original HasToken),
46   * right for an incoming UserRequest, for direct securing of a controller function (compose per function as required)
47   *
48   * @param userService
49   */
50 @Singleton
51 class AuthenticationAction @Inject()(val userService: UserService) extends ActionBuilder[AuthenticatedRequest]
52   with ActionRefiner[Request, AuthenticatedRequest] with ClassnameLogger {
53 
54   def refine[A](request: Request[A]): Future[Either[Result, AuthenticatedRequest[A]]] = {
55 
56     val result = request.cookies.get(AuthTokenCookieKey).fold[Either[Result, AuthenticatedRequest[A]]] {
57       Left(Results.Unauthorized(Json.obj("status" -> "ERR", "message" -> "Invalid XSRF Token cookie")))
58     } { xsrfTokenCookie =>
59       logger.trace(s"cookie ${xsrfTokenCookie.value}")
60       request.headers.get(AuthTokenHeader).orElse(request.getQueryString(AuthTokenUrlKey)).map {
61         headerToken =>
62           // ua needed to differentiate between different devices/sessions
63           val uaIdentifier: String = request.headers.get(UserAgentHeader).getOrElse(UserAgentHeaderDefault)
64           logger.trace(s"headerToken: $headerToken")
65           logger.trace(s"ua: $uaIdentifier")
66           userService.getUserSessionByToken(headerToken, xsrfTokenCookie.value, uaIdentifier).fold[Either[Result, AuthenticatedRequest[A]]] {
67             Left(Results.Unauthorized(Json.obj("status" -> "ERR", "message" -> "No server-side session"))
68               .discardingCookies(DiscardingCookie(name = AuthTokenCookieKey)))
69           } {
70             userSession =>
71               Right(new AuthenticatedRequest[A](userSession, request))
72           }
73 
74       }.getOrElse(Left(Results.Unauthorized(Json.obj("status" -> "ERR", "message" -> "No token header found"))
75         .discardingCookies(DiscardingCookie(name = AuthTokenCookieKey))))
76     }
77     Future.successful(result)
78   }
79 }
