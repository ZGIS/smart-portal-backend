39   /**
40     * basic auth compare
41     *
42     * @param loginCredentials
43     * @return
44     */
45   def authenticateLocal(loginCredentials: LoginCredentials): Either[ErrorResult, User] = {
46 
47     val userOpt = dbSession.viaConnection { implicit connection =>
48       UserDAO.findUserByEmailAsString(loginCredentials.email).filter(user =>
49         passwordHashing.validatePassword(loginCredentials.password, user.password)
50       )
51     }
52     if (userOpt.isDefined) {
53       // here is correct auth branch
54       val user = userOpt.get
55       if (user.isBlocked) {
56         val status = user.getToken
57         Left(ErrorResult(s"Your account was ${status.value}. Please contact the administrator.", None))
58       } else {
59         Right(user)
60       }
61 
62     } else {
63       // should never indicate if user not found or just password wrong?
64       Left(ErrorResult("User email or password wrong.", None))
65     }
66   }

236   /**
237     * check for an active server-side session
238     *
239     * @param xsrfToken
240     * @param xsrfTokenCookie
241     * @param userAgentHeader
242     * @return
243     */
244   def getUserSessionByToken(xsrfToken: String, xsrfTokenCookie: String, userAgentHeader: String): Option[UserSession] = {
245     dbSession.viaConnection(implicit connection => {
246       UserSession.findUserSessionByToken(xsrfToken, 1)
247     })
248       .fold[Option[UserSession]] {
249         logger.debug(s"no session for xsrfToken: $xsrfToken")
250         None
251       } {
252         session =>
253           val cookieForUSerAndUserAgent = passwordHashing.testSessionCookie(xsrfToken, session.email, userAgentHeader)
254           logger.trace(s"testcookie: $cookieForUSerAndUserAgent")
255           if (xsrfToken == xsrfTokenCookie && cookieForUSerAndUserAgent) {
256             logger.trace(s"request for active session: ${session.email} / $xsrfToken / $userAgentHeader")
257             Some(session)
258           } else {
259             logger.error("Invalid Token")
260             None
261           }
262     }
263   }

298   def insertUserFileEntry(filename: String, authUser: String, filelink: String): Option[UserFile] = {
299     val userLookup = dbSession.viaConnection { implicit connection =>
300       UserDAO.findUserByEmailAsString(authUser)

302 
303     userLookup.fold[Option[UserFile]] {

308       user =>

310           uuid = UUID.randomUUID(),
311           users_accountsubject = user.accountSubject,
312           originalfilename = filename,
313           linkreference = filelink,

315           laststatuschange = ZonedDateTime.now.withZoneSameInstant(ZoneId.of(appTimeZone)))

317         dbSession.viaTransaction(implicit connection => {
