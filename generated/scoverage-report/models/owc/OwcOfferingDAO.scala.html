264   def instantiateOwcOffering(uuid: String, offeringType: String, code: String, content: Option[String]): OwcOffering = {
265     val uuidObj = UUID.fromString(uuid)
266     val ops = findOwcOperationsByOfferingUUID(UUID.fromString(uuid)).toList
267     val contentList = content.map(text => List(text)).getOrElse(List())
268 
269     offeringType match {
270       case "WmsOffering" => WmsOffering(uuidObj, code, ops, contentList)
271       case "WmtsOffering" => WmtsOffering(uuidObj, code, ops, contentList)
272       case "WfsOffering" => WfsOffering(uuidObj, code, ops, contentList)
273       case "WcsOffering" => WcsOffering(uuidObj, code, ops, contentList)
274       case "CswOffering" => CswOffering(uuidObj, code, ops, contentList)
275       case "WpsOffering" => WpsOffering(uuidObj, code, ops, contentList)
276       case "GmlOffering" => GmlOffering(uuidObj, code, ops, contentList)
277       case "KmlOffering" => KmlOffering(uuidObj, code, ops, contentList)
278       case "GeoTiffOffering" => GeoTiffOffering(uuidObj, code, ops, contentList)
279       case "SosOffering" => SosOffering(uuidObj, code, ops, contentList)
280       case "NetCdfOffering" => NetCdfOffering(uuidObj, code, ops, contentList)
281       case "HttpLinkOffering" => HttpLinkOffering(uuidObj, code, ops, contentList)
282       case "X3dOffering" => X3dOffering(uuidObj, code, ops, contentList)
283       case _ => throw new InvalidClassException(offeringType, s"Unknown Offering type $offeringType")
284     }

349   def createOwcOffering(owcOffering: OwcOffering): Option[OwcOffering] = {
350     db.withTransaction {
351       implicit connection => {
352 
353         val rowCount = SQL(
354           s"""
355           insert into $tableOwcOfferings values (
356             {uuid}, {offering_type}, {code}, {content}
357           )
358         """).on(
359           'uuid -> owcOffering.uuid.toString,
360           'offering_type -> owcOffering.getClass.getSimpleName,
361           'code -> owcOffering.code,
362           'content -> owcOffering.content.headOption
363         ).executeUpdate()
364 
365         owcOffering.operations.foreach {
366           owcOperation => {
367             if (findOwcOperationByUuid(owcOperation.uuid).isEmpty) {
368               createOwcOperation(owcOperation)
369             }

371             SQL(
372               s"""insert into $tableOwcOfferingsHasOwcOperations  values (
373                  |{owc_offerings_uuid}, {owc_operations_uuid}
374                  |)
375                """.stripMargin).on(
376               'owc_offerings_uuid -> owcOffering.uuid.toString,
377               'owc_operations_uuid -> owcOperation.uuid.toString
378             ).executeUpdate()
379           }
380         }
